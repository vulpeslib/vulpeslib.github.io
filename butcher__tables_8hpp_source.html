34template <typename T>
35bool checkButcherTable(const T& t) {
36 std::cout << "Checking table is well formed" << std::endl;
37 const double b_sum = std::accumulate(t.b.begin(), t.b.end(), 0.0);
38 const bool b_sum_1 = (std::abs(b_sum - 1.) < 1e-14);
39 if (!b_sum_1) {
40 std::cout << "ERROR: b_sum_1" << std::endl;
41 return false;
42 }
43 for (std::size_t i = 0; i < T::n_stages; ++i) {
44 const auto Arow = std::accumulate(t.A[i].begin(), t.A[i].end(), 0.0);
45 const bool A_row_sum_c = (std::abs(Arow - t.c[i]) < 1e-14);
46 if (!A_row_sum_c) {
47 std::cout << "ERROR: A_row_sum_c " << i << std::endl;
48 return false;
49 }
50 }
51 return true;
52}

54template <typename T, std::size_t N>
55constexpr std::array<T, N + 1> padStartArray(const std::array<T, N>& input) {
56 std::array<T, N + 1> result{};

72template <typename T, std::size_t N>
73constexpr std::array<std::array<T, N + 1>, N + 1> padTopLeftMat(
74 const std::array<std::array<T, N>, N>& input) {
75 std::array<std::array<T, N + 1>, N + 1> result{};
76 for (std::size_t i = 0; i < N; ++i) {
77 for (std::size_t j = 0; j < N; ++j) {
78 result[i + 1][j + 1] = input[i][j];
79 }
80 }
81 return result;
82}

84template <typename T, std::size_t N>
85constexpr std::array<std::array<T, N + 1>, N + 1> padTopRightMat(
86 const std::array<std::array<T, N>, N>& input) {
87 std::array<std::array<T, N + 1>, N + 1> result{};
88 for (std::size_t i = 0; i < N; ++i) {
89 for (std::size_t j = 0; j < N; ++j) {
90 result[i + 1][j] = input[i][j];
91 }
92 }
93 return result;
94}

96template <typename ButcherTableType>
97constexpr bool checkFSAL() {
98 return (ButcherTableType::c.back() == 1.0) &&
99 (ButcherTableType::A.back() == ButcherTableType::b);
