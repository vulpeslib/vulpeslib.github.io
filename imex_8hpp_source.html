109 void step() {
110 const std::size_t first_stage = prepareFSAL();
111 for (std::size_t i = first_stage; i < n_stages; ++i) {
112 const double stage_time = m_time + c[i] * m_step_size;
113 m_force_func(stage_time, m_force_eval);
114
115 // U_i = U_{n-1} + \sum a_{ij} h K_j + \sum \hat{a}_{ij} h \hat{K}_j
116 m_stage_sol.copy(m_solution);
117 if constexpr (ButcherTableType::has_explicit_part) {
118 for (std::size_t j = 0; j < i; ++j) {
119 m_stage_sol.add(m_step_size * A_ex[i][j], m_internal_stages_ex[j]);
120 }
121 }
122
123 // compute U* (in stage_sol; stage_sol now holds U_i)
124 // (M + h a_ii L) U* = h a_ii F + M U_i [implicit]
125 // M U* = M U_i [explicit] -> nothing to do
126 if constexpr (ButcherTableType::has_implicit_part) {
127 // U_i = U_{n-1} + \sum a_{ij} h K_j + \sum \hat{a}_{ij} h \hat{K}_j
128 for (std::size_t j = 0; j < i; ++j) {
129 m_stage_sol.add(m_step_size * A_im[i][j], m_internal_stages_im[j]);
130 }
131
132 if (A_im[i][i] != 0.0) {
133 // tmp = M U_i
134 m_mass_matrix.vmult(m_stage_sol, m_tmp);
135 // tmp += h a_ii F
136 m_tmp.add(m_step_size * A_im[i][i], m_force_eval);
137 // stage_operator = (M + h a_ii L)
138 m_stage_operator.copy(m_mass_matrix, SUBSET_NONZERO_PATTERN);
139 m_stage_operator.addScaledMatrix(
140 m_linear_op, m_step_size * A_im[i][i], SUBSET_NONZERO_PATTERN);
141 // Solve: (M + h a_ii L) U* = h a_ii F + M U_i
142 // TODO: this is actually a PETSc bug
143 // diff --git a/src/mat/impls/aij/seq/inode.c
144 // b/src/mat/impls/aij/seq/inode.c index 34eb9f1b7f7..0081e76c64b
145 // 100644
146 // --- a/src/mat/impls/aij/seq/inode.c
147 // +++ b/src/mat/impls/aij/seq/inode.c
148 // @@ -1385,7 +1385,7 @@ PetscErrorCode
149 // MatLUFactorNumeric_SeqAIJ_Inode(Mat B, Mat A, const MatFactorInfo
150 // nz = ai[r[i] + 1] - ai[r[i]];
151 // ajtmp = aj + ai[r[i]];
152 // v1 = aa + ai[r[i]];
153 // - v2 = aa + ai[r[i] + 1];
154 // + v2 = aa + ai[r[i + 1]];
155 // for (j = 0; j < nz; j++) {
156 // col = ics[ajtmp[j]];
157 // rtmp1[col] = v1[j];
158 la::SparseMatrix stage_op(m_stage_operator, MAT_COPY_VALUES);
159 m_solver.setOperators(stage_op);
160 m_solver.solve(m_tmp, m_stage_sol);
161 }
162 }
163
164 // compute the rhs
165 // rhs [explicit] = N(U_i)
166 if (m_non_linear_op) {
167 m_non_linear_op(m_stage_sol, stage_time, m_stage_rhs_ex);
168 } else {
169 m_stage_rhs_ex.zero();
170 }
171 if constexpr (ButcherTableType::has_implicit_part) {
172 // rhs [implicit] = L U_i
173 m_linear_op.vmult(m_stage_sol, m_stage_rhs_im);
174 // rhs [implicit] = F - L U_i
175 m_stage_rhs_im.addScaled(-1.0, m_force_eval);
176 // rhs [explicit] = -N(U_i)
177 m_stage_rhs_ex.scale(-1.0);
178 } else {
179 // rhs = N(U_i) + L U_i
180 m_linear_op.vmultAdd(m_stage_sol, m_stage_rhs_ex, m_stage_rhs_ex);
181 // rhs = F - N(U_i) - L U_i
182 m_stage_rhs_ex.addScaled(-1.0, m_force_eval);
183 }
184
185 // Solve for new internal stages explicit/implicit
186 if constexpr (ButcherTableType::has_explicit_part) {
187 m_solver.setOperators(m_mass_matrix);
188 m_solver.solve(m_stage_rhs_ex, m_internal_stages_ex[i]);
189 }
190 if constexpr (ButcherTableType::has_implicit_part) {
191 m_solver.setOperators(m_mass_matrix);
192 m_solver.solve(m_stage_rhs_im, m_internal_stages_im[i]);
193 }
194 }
195
196 // Update the solution
197 for (std::size_t i = 0; i < n_stages; ++i) {
198 if constexpr (ButcherTableType::has_explicit_part) {
199 m_solution.add(m_step_size * b_ex[i], m_internal_stages_ex[i]);
200 }
201 if constexpr (ButcherTableType::has_implicit_part) {
202 m_solution.add(m_step_size * b_im[i], m_internal_stages_im[i]);
203 }
204 }
205
206 // Update the time
207 m_step++;
208 m_time = m_initial_time + m_step_size * m_step;
209 }
210 auto& getSolver() { return m_solver; }
211 const auto& getSolver() const { return m_solver; }
212 const la::Vector& getSolution() const { return m_solution; };
213 real_t getTime() const { return m_time; };
214
215 protected:
216 void initialize() {
217 if constexpr (ButcherTableType::has_explicit_part) {
218 for (std::size_t i = 0; i < n_stages; ++i) {
219 m_internal_stages_ex.emplace_back(m_solution.getLocAndGlobSize());
220 }
221 }
222 if constexpr (ButcherTableType::has_implicit_part) {
223 for (std::size_t i = 0; i < n_stages; ++i) {
224 m_internal_stages_im.emplace_back(m_solution.getLocAndGlobSize());
225 }
226 }
227 m_solver.setType(la::LinearSolver::PrecondOnly);
228 m_solver.setPreconditioner(la::LinearSolver::LU);
229
230 if (ButcherTableType::has_implicit_part) {
231 // prepare the non-zero sparsity pattern
232 m_stage_operator.addScaledMatrix(
233 m_linear_op, std::numeric_limits<double>::infinity(),
234 DIFFERENT_NONZERO_PATTERN);
235 }
236 }
237
238 std::size_t prepareFSAL() {
239 std::size_t first_stage = 0;
240 if constexpr (FSAL) {
241 if (m_step > 0) {
242 first_stage = 1;
243 if constexpr (ButcherTableType::has_explicit_part) {
244 m_internal_stages_ex[0].copy(m_internal_stages_ex.back());
245 }
246 if constexpr (ButcherTableType::has_implicit_part) {
247 m_internal_stages_im[0].copy(m_internal_stages_im.back());
248 }
249 }
250 }
251 return first_stage;
252 }

258 std::size_t m_step = 0;
259 la::Vector m_solution;
260
261 // operators
262 const la::SparseMatrix& m_mass_matrix;

264 const NonlinearPartFunc m_non_linear_op;
265 const ForceFunc m_force_func;

267 la::LinearSolver m_solver;

269 // temporary objects

271 std::vector<la::Vector> m_internal_stages_im;
272 la::Vector m_force_eval;
273 la::Vector m_stage_sol;
274 la::Vector m_internal_stages_sum_im;
275 la::Vector m_stage_rhs_ex;
276 la::Vector m_stage_rhs_im;
277 la::Vector m_non_linear_stage_eval;
278
279 la::Vector m_tmp;
280 // (M + h a_ii L)
281 la::SparseMatrix m_stage_operator;
282
283 static constexpr auto& A_ex = ButcherTableType::A_ex;
284 static constexpr auto& b_ex = ButcherTableType::b_ex;
285 static constexpr auto& A_im = ButcherTableType::A_im;
286 static constexpr auto& b_im = ButcherTableType::b_im;
287 static constexpr auto& c = ButcherTableType::c;
288};
