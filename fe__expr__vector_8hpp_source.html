172 template <std::size_t J = 0, std::size_t I = 0>
173 decltype(auto) evaluate(
174 const FeValues<Dim>& fe_values,
175 std::size_t j,
176 std::size_t i,
177 std::size_t q) const {
178 if constexpr (J != I) {
179 // Off-diagonal block: test component I != trial component J.
180 // No coupling between distinct components in this expression.
181 return void();
182 } else if constexpr (IndexBegin <= I && I < IndexEnd) {
183 using TypeEvalFunc =
184 decltype(m_func.template evaluate<J, I>(fe_values, j, i, q));
185 if constexpr (std::is_same_v<TypeEvalFunc, void>) {
186 return void();
187 } else {
188 if constexpr (GradOnTest) {
189 // F(q) . grad(phi_i) * phi_j -- gradient on test index i
190 return m_func.template evaluate<J, I>(fe_values, j, i, q)
191 .dot(fe_values.getGradEval()(i, q)) *
192 fe_values.getShapeEval()(j, q);
193 } else {
194 // phi_i * F(q) . grad(phi_j) -- gradient on trial index j
195 return fe_values.getShapeEval()(i, q) *
196 m_func.template evaluate<J, I>(fe_values, j, i, q)
197 .dot(fe_values.getGradEval()(j, q));
198 }
199 }
200 } else {
201 // Component I outside [IndexBegin, IndexEnd): not active.
202 return void();
203 }
204 }
205
206 constexpr UpdateFlagsType computeUpdateFlags() const {
207 // Requires shape values (for u/v), gradients (for grad v or grad u), and
208 // whatever flags F needs (typically quadrature point coordinates).
209 return UpdateGradients | UpdateValues | m_func.computeUpdateFlags();
210 }
211
212 constexpr std::size_t computeOrder() const {
213 // Integrand: F(x) . grad(basis_p) * basis_q where one carries a gradient
214 // (order p-1) and the other the full basis (order p).
215 const std::size_t basis_order = m_fe_space.order();
216 const std::size_t grad_order = (basis_order > 0) ? (basis_order - 1) : 0;
217 return m_func.computeOrder() + basis_order + grad_order;
218 }

719 template <std::size_t J = 0, std::size_t I = 0>
720 decltype(auto) evaluate(
721 const FeValues<Dim>& fe_values,
722 std::size_t j,
723 std::size_t i,
724 std::size_t q) const {
725 if constexpr (J == TrialComp && I == TestComp) {
726 using TypeEvalCoeff =
727 decltype(m_coeff.template evaluate<J, I>(fe_values, j, i, q));
728 if constexpr (std::is_same_v<TypeEvalCoeff, void>) {
729 return void();
730 } else {
731 const auto coeff_val =
732 m_coeff.template evaluate<J, I>(fe_values, j, i, q);
733 const auto u_eval = m_u.template evaluate<J, I>(fe_values, j, i, q);
734 const auto v_eval = m_v.template evaluate<J, I>(fe_values, j, i, q);
735 return coeff_val * scalarFromBasis(u_eval) * scalarFromBasis(v_eval);
736 }
737 } else {
738 return void();
739 }
740 }
741
742 constexpr UpdateFlagsType computeUpdateFlags() const {
743 return m_coeff.computeUpdateFlags() | m_u.computeUpdateFlags() |
744 m_v.computeUpdateFlags();
745 }
746
747 constexpr std::size_t computeOrder() const {
748 return m_u.computeOrder() + m_v.computeOrder() + m_coeff.computeOrder();
749 }
