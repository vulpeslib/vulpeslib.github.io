147 template <std::size_t J = 0, std::size_t I = 0>
148 decltype(auto) evaluate(
149 const FeValues<Dim>& fe_values,
150 std::size_t j,
151 std::size_t i,
152 std::size_t q) const {
153 if constexpr (J != I) {
154 // Off-diagonal block: test component I != trial component J.
155 // No coupling between distinct components in this expression.
156 return void();
157 } else if constexpr (IndexBegin <= I && I < IndexEnd) {
158 // Diagonal block: F(q) . grad(phi_i) * phi_j.
159 using TypeEvalFunc =
160 decltype(m_func.template evaluate<J, I>(fe_values, j, i, q));
161 if constexpr (std::is_same_v<TypeEvalFunc, void>) {
162 return void();
163 } else {
164 return m_func.template evaluate<J, I>(fe_values, j, i, q)
165 .dot(fe_values.getGradEval()(i, q)) *
166 fe_values.getShapeEval()(j, q);
167 }
168 } else {
169 // Component I outside [IndexBegin, IndexEnd): not active.
170 return void();
171 }
172 }
173
174 constexpr UpdateFlagsType computeUpdateFlags() const {
175 // Requires shape values (for u), gradients (for grad v), and whatever
176 // flags F needs (typically quadrature point coordinates).
177 return UpdateGradients | UpdateValues | m_func.computeUpdateFlags();
178 }
179
180 constexpr std::size_t computeOrder() const {
181 // The integrand contains F(x) . grad(v_I) * u_I, so the quadrature-order
182 // estimate must include the function expression, the trial basis, and the
183 // test-gradient contribution. For a p-th order basis, grad(v_I) has order
184 // p - 1; clamp at 0 to avoid unsigned underflow for constant spaces.
185 const std::size_t u_order = m_u.computeOrder();
186 const std::size_t grad_v_order = (u_order > 0) ? (u_order - 1) : 0;
187 return m_func.computeOrder() + u_order + grad_v_order;
188 }

584 template <std::size_t J = 0, std::size_t I = 0>
585 decltype(auto) evaluate(
586 const FeValues<Dim>& fe_values,
587 std::size_t j,
588 std::size_t i,
589 std::size_t q) const {
590 if constexpr (J == TrialComp && I == TestComp) {
591 using TypeEvalCoeff =
592 decltype(m_coeff.template evaluate<J, I>(fe_values, j, i, q));
593 if constexpr (std::is_same_v<TypeEvalCoeff, void>) {
594 return void();
595 } else {
596 const auto coeff_val =

598 const auto u_eval = m_u.template evaluate<J, I>(fe_values, j, i, q);
599 const auto v_eval = m_v.template evaluate<J, I>(fe_values, j, i, q);
600 return coeff_val * scalarFromBasis(u_eval) * scalarFromBasis(v_eval);
601 }
602 } else {
603 return void();
604 }
605 }
