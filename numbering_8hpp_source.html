60template <typename T>
61std::vector<T> nodeDegrees(
62 const std::vector<T>& ind, const std::vector<T>& ptr, T num_rows) {
63 std::vector<T> degree(num_rows, 0);
64 for (T ii = 0; ii < num_rows; ++ii) {
65 degree[ii] = ptr[ii + 1] - ptr[ii];
66 for (T jj = ptr[ii]; jj < ptr[ii + 1]; ++jj) {
67 if (ind[jj] == ii) {
68 // Add one if the diagonal is in row ii
69 degree[ii] += 1;
70 break;
71 }
72 }
73 }
74 return degree;
75}
76} // namespace internal

88template <typename T>
89Permutation<T> computeReverseCuthillMckeeOrdering(
90 const datastructure::CrsMap<T, T>& sparsity_pattern, bool reversed) {
91 const std::vector<T>& ind = sparsity_pattern.data();
92 const std::vector<T>& ptr = sparsity_pattern.indices();
93 assert(ptr.size() > 0);
94 const T num_rows = ptr.size() - 1;
95 std::vector<T> order(num_rows, 0);
96 std::vector<T> degree = internal::nodeDegrees(ind, ptr, num_rows);
97
98 // std::sort indices based on degrees
99 std::vector<T> inds(num_rows);
100 std::iota(inds.begin(), inds.end(), 0);
101 std::sort(inds.begin(), inds.end(), [&degree](T i, T j) {
102 return degree[i] < degree[j];
103 });
104
105 std::vector<T> rev_inds(num_rows);
106 for (T i = 0; i < num_rows; ++i) {
107 rev_inds[inds[i]] = i;
108 }
109
110 const T max_degree = *std::max_element(degree.begin(), degree.end());
111 std::vector<T> temp_degrees(max_degree, 0);
112 T N = 0;
113
114 // Loop over possible disconnected components
115 for (T zz = 0; zz < num_rows; ++zz) {
116 if (inds[zz] != static_cast<T>(-1)) {
117 T seed = inds[zz];
118 order[N++] = seed;
119 inds[rev_inds[seed]] = static_cast<T>(-1);
120
121 T level_start = N - 1;
122 T level_end = N;
123
124 while (level_start < level_end) {
125 for (T ii = level_start; ii < level_end; ++ii) {
126 T i = order[ii];
127 T N_old = N;
128
129 // Add unvisited neighbors
130 for (T jj = ptr[i]; jj < ptr[i + 1]; ++jj) {
131 T j = ind[jj];
132 if (inds[rev_inds[j]] != static_cast<T>(-1)) {
133 inds[rev_inds[j]] = static_cast<T>(-1);
134 order[N++] = j;
135 }
136 }
137
138 // Prepare for insertion sort
139 T level_len = 0;
140 for (T kk = N_old; kk < N; ++kk) {
141 temp_degrees[level_len++] = degree[order[kk]];
142 }
143
144 // Perform insertion sort
145 for (T kk = 1; kk < level_len; ++kk) {
146 T temp = temp_degrees[kk];
147 T temp2 = order[N_old + kk];
148 T ll = kk;
149 while (ll > 0 && temp < temp_degrees[ll - 1]) {
150 temp_degrees[ll] = temp_degrees[ll - 1];
151 order[N_old + ll] = order[N_old + ll - 1];
152 --ll;
153 }
154 temp_degrees[ll] = temp;
155 order[N_old + ll] = temp2;
156 }
157 }
158
159 level_start = level_end;
160 level_end = N;
161 }
162 }
163 if (N == num_rows) {
164 break;
165 }
166 }
167
168 // Reverse the order to get the RCM ordering
169 if (!reversed) {
170 std::reverse(order.begin(), order.end());
171 }
172
173 // compute the direct permutation
174 std::vector<T> perm(order.size());
175 for (std::size_t c = 0; c < order.size(); ++c) {
176 perm[order[c]] = c;
177 }
178
179 return {perm, order};
180}

200template <typename T>
201void createGraph(
202 const datastructure::CrsMap<T, T>& sparsity_pattern,
203 boosttypes::Graph& graph,
204 boosttypes::GraphDegree& graph_degree) {
205 for (T row = 0; row < sparsity_pattern.size(); ++row) {
206 for (const auto col : sparsity_pattern[row]) {
207 boost::add_edge(row, col, graph);
208 }
209 }
210
211 boost::graph_traits<boosttypes::Graph>::vertex_iterator ui, ui_end;
212
213 graph_degree = get(boost::vertex_degree, graph);
214 for (boost::tie(ui, ui_end) = vertices(graph); ui != ui_end; ++ui) {
215 graph_degree[*ui] = degree(*ui, graph);
216 }
217}
218} // namespace internal

226template <typename T>
227Permutation<T> computeBoostKingOrdering(
228 const datastructure::CrsMap<T, T>& sparsity_pattern,
229 bool reversed_numbering) {
230 boosttypes::Graph graph(sparsity_pattern.size());
231 boosttypes::GraphDegree graph_degree;
232
233 internal::createGraph(sparsity_pattern, graph, graph_degree);
234
235 boosttypes::GraphIndex index_map = boost::get(boost::vertex_index, graph);
236
237 std::vector<boosttypes::Vertex> inv_perm(boost::num_vertices(graph));
238 std::vector<T> new_dof_indices(boost::num_vertices(graph));
239
240 if (reversed_numbering) {
241 boost::king_ordering(graph, inv_perm.begin());
242 } else {
243 boost::king_ordering(graph, inv_perm.rbegin());
244 }
245
246 for (boosttypes::size_type c = 0; c != inv_perm.size(); ++c) {
247 new_dof_indices[index_map[inv_perm[c]]] = c;
248 }
249
250 return {new_dof_indices, std::vector<T>(inv_perm.begin(), inv_perm.end())};
251}

260template <typename T>
261Permutation<T> computeBoostCuthillMckeeOrdering(
262 const datastructure::CrsMap<T, T>& sparsity_pattern,
263 bool reversed_numbering) {
264 boosttypes::Graph graph(sparsity_pattern.size());
265 boosttypes::GraphDegree graph_degree;
266
267 internal::createGraph(sparsity_pattern, graph, graph_degree);
268
269 boosttypes::GraphIndex index_map = boost::get(boost::vertex_index, graph);
270
271 std::vector<boosttypes::Vertex> inv_perm(boost::num_vertices(graph));
272 std::vector<T> new_dof_indices(boost::num_vertices(graph));
273
274 if (reversed_numbering) {
275 boost::cuthill_mckee_ordering(
276 graph, inv_perm.begin(), boost::get(boost::vertex_color, graph),
277 boost::make_degree_map(graph));
278 } else {
279 boost::cuthill_mckee_ordering(
280 graph, inv_perm.rbegin(), boost::get(boost::vertex_color, graph),
281 boost::make_degree_map(graph));
282 }
283
284 for (boosttypes::size_type c = 0; c != inv_perm.size(); ++c) {
285 new_dof_indices[index_map[inv_perm[c]]] = c;
286 }
287
288 return {new_dof_indices, std::vector<T>(inv_perm.begin(), inv_perm.end())};
