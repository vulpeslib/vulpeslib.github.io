39template <
40 typename MeshType,
41 typename FeExpressionTuple,
42 typename FeExpressionFaceTuple = std::tuple<>>
43class WeakFormIntegratorSimplexify : public WeakFormIntegratorBase<
44 MeshType,
45 FeExpressionTuple,
46 FeExpressionFaceTuple> {
47 public:
48 using WeakFormIntegratorBaseType = WeakFormIntegratorBase<
49 MeshType,
50 FeExpressionTuple,
51 FeExpressionFaceTuple>;
52 static inline constexpr std::size_t dim = MeshType::self_dim;
53 static inline constexpr uint32_t s_n_expr =
54 WeakFormIntegratorBaseType::s_n_expr;
55 static inline constexpr uint32_t s_n_face_expr =
56 WeakFormIntegratorBaseType::s_n_face_expr;
57
58 std::size_t computeMaxOrder() const {
59 std::size_t order = 1;
60 if constexpr (s_n_expr) {
61 order += std::apply(
62 [](const auto&... expr) {
63 return std::max({expr->computeOrder()...});
64 },
65 this->m_fe_expression);
66 }
67 if constexpr (s_n_face_expr) {
68 order += std::apply(
69 [](const auto&... expr) {
70 return std::max({expr->computeOrder()...});
71 },
72 this->m_fe_expression_face);
73 }
74 return order;
75 }
76
77 WeakFormIntegratorSimplexify(
78 const MeshType& mesh,
79 const dofs::DofHandler<MeshType>& dof_handler,
80 const FeExpressionTuple& fe_expression,
81 const FeExpressionFaceTuple& fe_expression_face = FeExpressionFaceTuple{})
82 : WeakFormIntegratorBaseType(
83 mesh, dof_handler, fe_expression, fe_expression_face) {
84 // the order is the max among the orders of the expressions
85 const std::size_t order = computeMaxOrder();
86 // cache quadrature rule
87 m_qrule_d_ref = qrule::QuadratureLibrary<dim, double>::getRule(
88 qrule::QuadratureElemType::Simplex, order);
89 m_qrule_d = *m_qrule_d_ref;
90 //
91 m_qrule_bndry_d_ref = qrule::QuadratureLibrary<dim - 1, double>::getRule(
92 qrule::QuadratureElemType::Simplex, order);
93 m_qrule_bndry_d.nodes.resize(m_qrule_bndry_d_ref->nodes.size());
94 }
95
96 void integrate(MeshType::index_t cell_id) override {
97 this->m_integrals.setZero();
98 const std::size_t n_q_nodes_simplex = m_qrule_d.nodes.size();
99 // WARNING: we assume all the expression we integrate have the same FeSpace
100 const auto& fe_space = std::get<0>(this->m_fe_expression)->feSpace();
101 const uint32_t n_dof_loc = fe_space.getBasis().size();
102
103 const bool need_neigh =
104 this->m_update_flags_face & fe::UpdateNeighborValues;
105 if (need_neigh) {
106 m_fe_values.createNeighValues();
107 }
108
109 // TODO: cache cell
110 const bool need_normal =
111 (this->m_update_flags | this->m_update_flags_face) & fe::UpdateNormals;
112 auto cell = geometry::Cell2D<double>{this->m_mesh, cell_id, need_normal};
113 const geometry::CartesianMapping<double, dim> mapping(cell.bbox());
114
115 // map cell to [-1, 1]^n
116 for (int64_t i = 0; i < cell.vtxs().cols(); ++i) {
117 mapping.transformInv_({&cell.vtxs()(0, i), dim});
118 }
119 geometry::Simplexify<dim> simplexification(cell, need_neigh);
120 const auto& simplex = simplexification.getSimplexesList();
121 const auto& vertexes = simplexification.getPointList();
122 // for each simplex
123 for (int32_t k = 0; k < simplexification.getNumSimplexes(); ++k) {
124 // build the transformation from the reference simplex
125 const Eigen::Vector<double, dim> traslation =
126 vertexes.col(simplex.coeff(0, k));
127 Eigen::Matrix<double, dim, dim> jacobian;
128 for (std::size_t d = 0; d < dim; ++d) {
129 jacobian.col(d) = vertexes.col(simplex.coeff(d + 1, k)) - traslation;
130 }
131 const double jacDet = std::abs(vmath::determinant(jacobian));
132 // map the q nodes to the simplex
133 for (std::size_t q = 0; q < n_q_nodes_simplex; ++q) {
134 m_qrule_d.nodes[q].weight = m_qrule_d_ref->nodes[q].weight * jacDet;
135 const Eigen::Map<const Eigen::Vector<double, dim>> pt_map_src(
136 m_qrule_d_ref->nodes[q].point.data());
137 Eigen::Map<Eigen::Vector<double, dim>> pt_map_dst(
138 m_qrule_d.nodes[q].point.data());
139 pt_map_dst = jacobian * pt_map_src + traslation;
140 }
141 // update evaluations and mappings
142 m_fe_values.evalOnReferenceFrame(fe_space.getBasisFast(), m_qrule_d);
143 m_fe_values.evalAndApplyMapping(
144 mapping, this->m_update_flags | fe::UpdateWeights);
145 // apply quadrature rule for each pair basis function (i, j)
146 for (uint32_t q = 0; q < n_q_nodes_simplex; ++q) {
147 for (uint32_t i = 0; i < n_dof_loc; ++i) {
148 for (uint32_t j = 0; j < n_dof_loc; ++j) {
149 uint32_t expr_id = 0;
150 // TODO: evaluate if the std::apply is better as the most inner
151 // loop (like now) or in the most outer loop
152 std::apply(
153 [this, &expr_id, i, j, q](auto&&... exprs) {
154 ((this->m_integrals(i, j, expr_id++) +=
155 exprs->evaluate(m_fe_values, i, j, q) *
156 m_fe_values.weights()[q]),
157 ...);
158 },
159 this->m_fe_expression);
160 }
161 }
162 }
163 }
164
165 if constexpr (s_n_face_expr) {
166 // loop on faces
167 this->m_face_integrals.setZero();
168 // TODO: resize only if needed
169 const auto nfaces = cell.nfaces();
170 const auto n_neigh = this->m_mesh.getDualMap()[cell_id].size();
171 this->m_face_integrals_neigh.resize(
172 n_dof_loc, n_dof_loc, s_n_face_expr, n_neigh);
173 this->m_face_integrals_neigh.setZero();
174 const std::size_t n_q_nodes_subsimplex = m_qrule_bndry_d.nodes.size();
175
176 for (int32_t face_id = 0; face_id < nfaces; ++face_id) {
177 Eigen::TensorMap<Eigen::Tensor<double, 3>> curr_face_integrals(
178 &this->m_face_integrals(0, 0, 0), n_dof_loc, n_dof_loc,
179 s_n_face_expr);
180
181 typename MeshType::index_t neigh_cell_id = -1;
182 // find to which neighbor cell this face belongs to
183 if (need_neigh) {
184 // find global face id
185 const auto glob_face_id =
186 this->m_mesh.getCellToFacetMap()[cell_id][face_id];
187 // find the two cells to which this face belongs to
188 const auto cell_ids = this->m_mesh.getFacetToCellMap()[glob_face_id];
189 // find the cell that is not cell_id
190 neigh_cell_id = cell_ids[(cell_ids[0] == cell_id)];
191 if (neigh_cell_id != static_cast<decltype(neigh_cell_id)>(-1)) {
192 // find the local_cell_id in dual map from global id
193 const auto neighs_begin =
194 this->m_mesh.getDualMap()[cell_id].begin();
195 const auto neighs_end = this->m_mesh.getDualMap()[cell_id].end();
196 const auto loc_neigh_cell_id = std::distance(
197 neighs_begin,
198 std::find(neighs_begin, neighs_end, neigh_cell_id));
199 // extract buffer where to write local matrix
200 curr_face_integrals = Eigen::TensorMap<Eigen::Tensor<double, 3>>(
201 &this->m_face_integrals_neigh(0, 0, 0, loc_neigh_cell_id),
202 n_dof_loc, n_dof_loc, s_n_face_expr);
203 } else {
204 std::cout
205 << "WARNING: querying neighbour of boundary cell, skipping!"
206 << std::endl;
207 continue;
208 }
209 }
210 // iterate face sub-simplifixation
211 const auto face_subsimplexes =
212 simplexification.getSubsimplexesListFacet(face_id);
213 const auto nsubsimplexes = face_subsimplexes.cols();
214 for (int64_t k = 0; k < nsubsimplexes; ++k) {
215 // build the transformation from the reference simplex
216 const Eigen::Vector<int32_t, dim> subsimplex =
217 face_subsimplexes.col(k);
218 Eigen::Vector<double, dim> traslation =
219 cell.vtxs().col(subsimplex[0]);
220 Eigen::Matrix<double, dim, dim - 1> jacobian;
221 for (std::size_t d = 0; d < dim - 1; ++d) {
222 jacobian.col(d) = cell.vtxs().col(subsimplex[d + 1]) - traslation;
223 }
224 // map the q nodes to the simplex
225 for (std::size_t q = 0; q < n_q_nodes_subsimplex; ++q) {
226 m_qrule_bndry_d.nodes[q].weight =
227 m_qrule_bndry_d_ref->nodes[q].weight;
228 const Eigen::Map<const Eigen::Vector<double, dim - 1>> pt_map_src(
229 m_qrule_bndry_d_ref->nodes[q].point.data());
230 Eigen::Map<Eigen::Vector<double, dim>> pt_map_dst(
231 m_qrule_bndry_d.nodes[q].point.data());
232 pt_map_dst = jacobian * pt_map_src + traslation;
233 }
234
235 // update evaluations and mappings
236 m_fe_values.evalOnReferenceFrame(
237 fe_space.getBasisFast(), m_qrule_bndry_d);
238 const auto normal = cell.normals().col(face_id);
239 m_fe_values.setFaceNormal(normal.normalized());
240 auto bnd_mapping = mapping;
241 bnd_mapping.setJacobianDet(normal.norm());
242 m_fe_values.evalAndApplyMapping(
243 bnd_mapping, this->m_update_flags_face | fe::UpdateWeights);
244 if (need_neigh) {
245 // compute bounding box neighbor
246 Eigen::Matrix<double, dim, 2> bbox_neigh;
247 bbox_neigh.col(0).setConstant(DBL_MAX);
248 bbox_neigh.col(1).setConstant(-DBL_MAX);
249 for (auto vtx_id : this->m_mesh.getCellToVtxMap()[neigh_cell_id]) {
250 for (std::size_t d = 0; d < dim; ++d) {
251 const double vd = this->m_mesh.getVertexes()[vtx_id][d];
252 bbox_neigh(d, 0) = std::min(bbox_neigh(d, 0), vd);
253 bbox_neigh(d, 1) = std::max(bbox_neigh(d, 1), vd);
254 }
255 }
256 // map reference quadrature nodes to the one in the bounding
257 // box inverse bbox -> neigh bbox
258 const geometry::CartesianMapping<double, dim> neigh_mapping(
259 bbox_neigh);
260 const Eigen::Array<double, dim, 1> compose_jacobian_bbox =
261 neigh_mapping.getJacobianInv().array() *
262 mapping.getJacobian().array();
263 const Eigen::Array<double, dim, 1> compose_translation_bbox =
264 neigh_mapping.getJacobianInv().array() *
265 mapping.getTranslation().array() +
266 neigh_mapping.getTranslationInv().array();
267 // map the q nodes to the simplex
268 jacobian.array().colwise() *= compose_jacobian_bbox;
269 traslation = compose_jacobian_bbox * traslation.array() +
270 compose_translation_bbox;
271 for (std::size_t q = 0; q < n_q_nodes_subsimplex; ++q) {
272 m_qrule_bndry_d.nodes[q].weight =
273 m_qrule_bndry_d_ref->nodes[q].weight;
274 const Eigen::Map<const Eigen::Vector<double, dim - 1>> pt_map_src(
275 m_qrule_bndry_d_ref->nodes[q].point.data());
276 Eigen::Map<Eigen::Vector<double, dim>> pt_map_dst(
277 m_qrule_bndry_d.nodes[q].point.data());
278 pt_map_dst = jacobian * pt_map_src + traslation;
279 }
280
281 // compute updated evaluation
282 m_fe_values.neighValues().evalOnReferenceFrame(
283 fe_space.getBasisFast(), m_qrule_bndry_d);
284 m_fe_values.neighValues().evalAndApplyMapping(
285 neigh_mapping, this->m_update_flags_face);
286 }
287 // apply quadrature rule for each pair basis function (i, j)
288 for (uint32_t q = 0; q < n_q_nodes_subsimplex; ++q) {
289 for (uint32_t i = 0; i < n_dof_loc; ++i) {
290 for (uint32_t j = 0; j < n_dof_loc; ++j) {
291 uint32_t expr_id = 0;
292 // TODO: evaluate if the std::apply is better as the most inner
293 // loop (like now) or in the most outer loop
294 std::apply(
295 [this, &curr_face_integrals, &expr_id, i, j,
296 q](auto&&... exprs) {
297 ((curr_face_integrals(i, j, expr_id++) +=
298 exprs->evaluate(m_fe_values, i, j, q) *
299 m_fe_values.weights()[q]),
300 ...);
301 },
302 this->m_fe_expression_face);
303 }
304 }
305 }
306 }
307 }
308 }
309 }
310
311 protected:
312 fe::FeValues<dim> m_fe_values;
313 std::shared_ptr<qrule::QuadratureRule<double, dim>> m_qrule_d_ref = nullptr;
314 qrule::QuadratureRule<double, dim> m_qrule_d;
315 std::shared_ptr<qrule::QuadratureRule<double, dim - 1>> m_qrule_bndry_d_ref =
316 nullptr;
317 qrule::QuadratureRule<double, dim> m_qrule_bndry_d;
318};
