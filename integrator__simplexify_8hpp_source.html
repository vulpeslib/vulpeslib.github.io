89 void integrate(index_t cell_id) override {
90 this->preallocate(cell_id);
91 auto& cell = this->getCell(this->m_cell_id, true);
92
93 // TODO: in 2D if we integrate only on boundary we do not need
94 // simplexification since the edges are already 1D simplexes
95 const bool need_bnd = s_n_face_expr || this->m_need_neigh_values;
96 Profiler::start("simplexification");
97 geometry::Simplexify<dim> simplexification(cell, need_bnd);
98 Profiler::stop("simplexification");
99 const auto& simplexes = simplexification.getSimplexesList();
100 const auto& vertexes = simplexification.getPointList();
101 // compute volume integrals
102 if constexpr (s_n_expr) {
103 // for each simplex
104 for (int32_t k = 0; k < simplexification.getNumSimplexes(); ++k) {
105 PROFILE(mapQuadraturePointsToSimplex(simplexes, vertexes, k));
106 // update evaluations of basis functions and mappings
107 this->m_fe_values.evalOnReferenceFrame(
108 this->m_fe_space.getBasisFast(), m_qrule_d);
109 this->m_fe_values.evalAndApplyMapping(
110 *(this->m_mapping), this->m_update_flags | fe::UpdateWeights);
111 PROFILE(integrateVolumeCell());
112 }
113 }
114 // compute boundary integrals
115 if constexpr (s_n_face_expr) {
116 // loop on faces
117 for (this->m_face_id = 0; this->m_face_id < cell.nfaces();
118 ++this->m_face_id) {
119 PROFILE(this->findCellNeigh());
120 const auto face_subsimplexes =
121 simplexification.getSubsimplexesListFacet(this->m_face_id);
122 const index_t nsubsimplexes = face_subsimplexes.cols();
123 const auto simplexification_pts = simplexification.getPointList();
124 // iterate face sub-simplifixation: triangles in 3D, a single line in 2D
125 for (index_t k = 0; k < nsubsimplexes; ++k) {
126 PROFILE(updateQptsAndMappingForThisAndNeigh(
127 face_subsimplexes, simplexification_pts, k));
128 PROFILE(integrateSurfaceSubcell());
129 }
130 }
131 }
132 this->applyBasisScaling();
133 }

135 protected:
136 void mapQuadraturePointsToSimplex(
137 const auto& simplexes, const auto& vertexes, int32_t k) {
138 const std::size_t n_q_nodes_simplex = m_qrule_d.nodes.size();
139 const Eigen::Vector<double, dim> traslation =
140 vertexes.col(simplexes.coeff(0, k));
141 Eigen::Matrix<double, dim, dim> jacobian;
142 for (std::size_t d = 0; d < dim; ++d) {
143 jacobian.col(d) = vertexes.col(simplexes.coeff(d + 1, k)) - traslation;
144 }
145 const double jac_det = std::abs(vmath::determinant(jacobian));
146 // map the q nodes to the simplex
147 for (std::size_t q = 0; q < n_q_nodes_simplex; ++q) {
148 m_qrule_d.nodes[q].weight = m_qrule_d_ref->nodes[q].weight * jac_det;
149 const Eigen::Map<const Eigen::Vector<double, dim>> pt_map_src(
150 m_qrule_d_ref->nodes[q].point.data());
151 Eigen::Map<Eigen::Vector<double, dim>> pt_map_dst(
152 m_qrule_d.nodes[q].point.data());
153 pt_map_dst = jacobian * pt_map_src + traslation;
154 }
155 }

157 void integrateVolumeCell() {
158 auto [bilinear_fe_expression, non_bilinear_fe_expression] =
159 internal::split_expression_tuple(this->m_fe_expression);
160 const std::size_t n_q_nodes_simplex = m_qrule_d.nodes.size();
161 const uint32_t n_dof_loc = this->m_fe_space.getBasis().size();
162 std::apply(
163 [this](auto&&... exprs) {
164 ((exprs->precompute(
165 this->m_fe_values, this->m_cell_manager.getCells())),
166 ...);
167 },
168 this->m_fe_expression);
169 // apply quadrature rule for each pair basis function (i, j)
170 for (uint32_t q = 0; q < n_q_nodes_simplex; ++q) {
171 const double Jdx = this->m_fe_values.weights()[q];
172 for (uint32_t i = 0; i < n_dof_loc; ++i) {
173 // TODO: it is worth it to make further local copies
174 // of data used in the expression to increase cache locality
175 // this should have ~10% speed-up
176 if constexpr (s_n_expr_bilinear) {
177 for (uint32_t j = 0; j < n_dof_loc; ++j) {
178 uint32_t expr_id = 0;
179 // TODO: evaluate if the std::apply is better as the most inner
180 // loop (like now) or in the most outer loop
181 std::apply(
182 [this, &expr_id, i, j, q, Jdx, n_dof_loc](auto&&... exprs) {
183 (([this, &expr_id, i, j, q, Jdx, n_dof_loc](auto&& expr) {
184 [this, expr_id, i, j, q, Jdx, expr,
185 n_dof_loc]<std::size_t... d>(std::index_sequence<d...>) {
186 ((this->m_integrals(
187 i * s_n_blocks + d % s_n_blocks,
188 j * s_n_blocks + d / s_n_blocks, expr_id) +=
189 expr->template evaluate<
190 d % s_n_blocks, d / s_n_blocks>(
191 this->m_fe_values, i, j, q) *
192 Jdx),
193 ...);
194 }(std::make_index_sequence<s_n_blocks * s_n_blocks>{});
195 expr_id++;
196 }(exprs)),
197 ...);
198 },
199 bilinear_fe_expression);
200 }
201 }
202 if constexpr (s_n_expr_linear) {
203 constexpr auto UNDEF = static_cast<std::size_t>(-1);
204 uint32_t expr_id = 0;
205 std::apply(
206 [this, &expr_id, i, q, n_dof_loc](auto&&... exprs) {
207 (([this, &expr_id, i, q, n_dof_loc](auto&& expr) {
208 [this, expr_id, i, q, expr,
209 n_dof_loc]<std::size_t... d>(std::index_sequence<d...>) {
210 ((this->m_rhs(i * s_n_blocks + d, expr_id) +=
211 expr->template evaluate<UNDEF, d>(
212 this->m_fe_values, -1, i, q) *
213 this->m_fe_values.weights()[q]),
214 ...);
215 }(std::make_index_sequence<s_n_blocks>{});
216 expr_id++;
217 }(exprs)),
218 ...);
219 },
220 non_bilinear_fe_expression);
221 }
222 }
223 }

226 void updateQptsAndMappingForThisAndNeigh(
227 const auto& face_subsimplexes,
228 const auto& simplexification_pts,
229 index_t k) {
230 auto& cell = this->getCell(this->m_cell_id);
231 const std::size_t n_q_nodes_subsimplex = m_qrule_bndry_d.nodes.size();
232 // build the transformation from the reference (sub)simplex to subsimplex
233 const Eigen::Vector<int32_t, dim> subsimplex = face_subsimplexes.col(k);
234 Eigen::Vector<double, dim> traslation =
235 simplexification_pts.col(subsimplex[0]);
236 Eigen::Matrix<double, dim, dim - 1> jacobian;
237 for (std::size_t d = 0; d < dim - 1; ++d) {
238 jacobian.col(d) =
239 simplexification_pts.col(subsimplex[d + 1]) - traslation;
240 }
241 // map the q nodes to the simplex
242 for (std::size_t q = 0; q < n_q_nodes_subsimplex; ++q) {
243 m_qrule_bndry_d.nodes[q].weight = m_qrule_bndry_d_ref->nodes[q].weight;
244 const Eigen::Map<const Eigen::Vector<double, dim - 1>> pt_map_src(
245 m_qrule_bndry_d_ref->nodes[q].point.data());
246 Eigen::Map<Eigen::Vector<double, dim>> pt_map_dst(
247 m_qrule_bndry_d.nodes[q].point.data());
248 pt_map_dst = jacobian * pt_map_src + traslation;
249 }
250
251 // update evaluations and mappings
252 this->m_fe_values.evalOnReferenceFrame(
253 this->m_fe_space.getBasisFast(), m_qrule_bndry_d);
254 // set the normal, needed to use the Nanson rule
255 const auto normal = cell.normals().col(this->m_face_id);
256 this->m_fe_values.setFaceNormal(normal);
257 auto bnd_mapping = *(this->m_mapping);
258 // transformation area of quadrature for sub-simplex
259 // the new area divided by the reference area i.e.
260 // the measure of orthogonal simplex in [0, 1]^(dim - 1)
261 constexpr double std_subsimp = 1. / vmath::factorial(dim - 1);
262 const double subsimplex_jdet = vmath::area(jacobian) / std_subsimp;
263 // nanson rule
264 // ||JF^-TN||, where F jacobian of ref-to-physic, J is |det(F)|,
265 // N unit outer in ref. By expoiting linearity and F is diagonal we have
266 // J/||Fn||, where n is the unit normal in physical space.
267 // Recall that normals are controvariant
268 const double jac_det = this->m_mapping->jacobianDet({});
269 const auto jac = this->m_mapping->getJacobian();
270 const double nanson = jac_det / jac.cwiseProduct(normal).norm();
271 // we just rescale quadrature weights accordingly
272 bnd_mapping.setJacobianDet(nanson * subsimplex_jdet);
273 this->m_fe_values.evalAndApplyMapping(
274 bnd_mapping, this->m_update_flags_face | fe::UpdateWeights);
275 // if we need the neigh we need the bounding box of the neigh cell
276 // to compute the transformation. This is the composition of three
277 // transformations since the cell is in reference configuration:
278 // 1. (sub)simplex-to-reference-subsimplex
279 // 2. bbox-transformation of the neigh
280 // 3. inverse-bbox-tranformation of this cell
281 if (this->m_need_neigh_values && this->m_cell_neigh_info.have_neigh) {
282 // map reference quadrature nodes to the one in the bounding
283 // box inverse bbox -> neigh bbox
284 const auto& neigh_mapping = *this->m_cell_manager.getMapping(
285 this->m_cell_neigh_info.neigh_cell_id);
286 const Eigen::Array<double, dim, 1> compose_jacobian_bbox =
287 neigh_mapping.getJacobianInv().array() *
288 this->m_mapping->getJacobian().array();
289 const Eigen::Array<double, dim, 1> compose_translation_bbox =
290 neigh_mapping.getJacobianInv().array() *
291 this->m_mapping->getTranslation().array() +
292 neigh_mapping.getTranslationInv().array();
293 // map the q nodes to the simplex
294 jacobian.array().colwise() *= compose_jacobian_bbox;
295 traslation =
296 compose_jacobian_bbox * traslation.array() + compose_translation_bbox;
297 for (std::size_t q = 0; q < n_q_nodes_subsimplex; ++q) {
298 m_qrule_bndry_d.nodes[q].weight = m_qrule_bndry_d_ref->nodes[q].weight;
299 const Eigen::Map<const Eigen::Vector<double, dim - 1>> pt_map_src(
300 m_qrule_bndry_d_ref->nodes[q].point.data());
301 Eigen::Map<Eigen::Vector<double, dim>> pt_map_dst(
302 m_qrule_bndry_d.nodes[q].point.data());
303 pt_map_dst = jacobian * pt_map_src + traslation;
304 }
305 // compute updated evaluation for the neigh
306 this->m_fe_values.neighValues().evalOnReferenceFrame(
307 this->m_fe_space.getBasisFast(), m_qrule_bndry_d);
308 this->m_fe_values.neighValues().evalAndApplyMapping(
309 neigh_mapping, this->m_update_flags_face);
310 }
311 }

313 void integrateSurfaceSubcell() {
314 const uint32_t n_dof_loc = this->m_fe_space.getBasis().size();
315 const std::size_t n_q_nodes_subsimplex = m_qrule_bndry_d.nodes.size();
316 // TODO: evaluate if the std::apply is better as the most
317 // inner loop or in the most outer loop
318 auto [bilinear_fe_expression, non_bilinear_fe_expression] =
319 internal::split_expression_tuple(this->m_fe_expression_face);
320 if constexpr (s_n_face_expr_bilinear) {
321 vutils::for_each_idxed(
322 bilinear_fe_expression, [this, n_q_nodes_subsimplex, n_dof_loc](
323 const auto& expr, Eigen::Index expr_id) {
324 Eigen::Map<Eigen::MatrixXd> curr_integral =
325 this->getCurrFaceIntegral(expr, expr_id);
326 if (!curr_integral.size()) {
327 return;
328 }
329
330 [&]<std::size_t... d>(std::index_sequence<d...>) {
331 (
332 [&]() {
333 for (uint32_t q = 0; q < n_q_nodes_subsimplex; ++q) {
334 const double dx = this->m_fe_values.weights()[q];
335 for (uint32_t i = 0; i < n_dof_loc; ++i) {
336 for (uint32_t j = 0; j < n_dof_loc; ++j) {
337 constexpr auto bi = d % s_n_blocks;
338 constexpr auto bj = d / s_n_blocks;
339 curr_integral(
340 i * s_n_blocks + bi, j * s_n_blocks + bj) +=
341 expr->template evaluate<bi, bj>(
342 this->m_fe_values, i, j, q) *
343 dx;
344 }
345 }
346 }
347 }(),
348 ...);
349 }(std::make_index_sequence<s_n_blocks * s_n_blocks>{});
350 });
351 }
352 if constexpr (s_n_face_expr_linear) {
353 constexpr auto UNDEF = static_cast<std::size_t>(-1);
354 vutils::for_each_idxed(
355 non_bilinear_fe_expression,
356 [this, n_q_nodes_subsimplex, n_dof_loc](
357 const auto& expr, Eigen::Index expr_id) {
358 // the fe expression might want to skip a face
359 // e.g. for Dirichlet boundary conditions
360 if (!expr->precompute(
361 this->m_fe_values, this->m_cell_manager.getCells())) {
362 return;
363 }
364
365 [&]<std::size_t... d>(std::index_sequence<d...>) {
366 (
367 [&]() {
368 for (uint32_t q = 0; q < n_q_nodes_subsimplex; ++q) {
369 const double dx = this->m_fe_values.weights()[q];
370 for (uint32_t j = 0; j < n_dof_loc; ++j) {
371 this->m_face_rhs(j * s_n_blocks + d, expr_id) +=
372 expr->template evaluate<UNDEF, d>(
373 this->m_fe_values, -1, j, q) *
374 dx;
375 }
376 }
377 }(),
378 ...);
379 }(std::make_index_sequence<s_n_blocks>{});
380 });
381 }
382 }
383
384 std::shared_ptr<qrule::QuadratureRule<double, dim>> m_qrule_d_ref = nullptr;
385 qrule::QuadratureRule<double, dim> m_qrule_d;
386 std::shared_ptr<qrule::QuadratureRule<double, dim - 1>> m_qrule_bndry_d_ref =
387 nullptr;
388 qrule::QuadratureRule<double, dim> m_qrule_bndry_d;
389};
