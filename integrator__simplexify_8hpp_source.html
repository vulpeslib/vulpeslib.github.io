282template <
283 typename MeshType,
284 typename FeExpressionTuple,
285 typename FeExpressionFaceTuple>
286void WeakFormIntegratorSimplexify<
287 MeshType,
288 FeExpressionTuple,
289 FeExpressionFaceTuple>::integrate(index_t cell_id) {
290 this->preallocate(cell_id);
291 auto& cell = this->getCell(this->m_cell_id, true);
292
293 const bool need_bnd = s_n_face_expr || this->m_need_neigh_values;
294 Profiler::start("integrate::simplexification");
295 // TODO: consider caching the simplexification for each cell
296 geometry::Simplexify<dim> simplexification(cell, need_bnd);
297 Profiler::stop("integrate::simplexification");
298 const auto& simplexes = simplexification.getSimplexesList();
299 const auto& vertexes = simplexification.getPointList();
300 // compute volume integrals
301 if constexpr (s_n_expr) {
302 // for each simplex
303 for (int32_t k = 0; k < simplexification.getNumSimplexes(); ++k) {
304 PROFILE(mapQuadraturePointsToSimplex(simplexes, vertexes, k));
305 // update evaluations of basis functions and mappings
306 this->m_fe_values.evalOnReferenceFrame(
307 this->m_fe_space.getBasisFast(), m_qrule_d);
308 this->m_fe_values.evalAndApplyMapping(
309 *(this->m_mapping), this->m_update_flags | fe::UpdateWeights);
310 PROFILE(integrateVolumeCell());
311 }
312 }
313 // compute boundary integrals
314 if constexpr (s_n_face_expr) {
315 // loop on faces
316 for (this->m_face_id = 0; this->m_face_id < cell.nfaces();
317 ++this->m_face_id) {
318 PROFILE(this->findCellNeigh());
319 const auto face_subsimplexes =
320 simplexification.getSubsimplexesListFacet(this->m_face_id);
321 const index_t nsubsimplexes = face_subsimplexes.cols();
322 const auto simplexification_pts = simplexification.getPointList();
323 // iterate face sub-simplifixation: triangles in 3D, a single line in 2D
324 for (index_t k = 0; k < nsubsimplexes; ++k) {
325 Profiler::start("integrate::updateQptsAndMappingForThisAndNeigh");
326 updateQptsAndMappingForThisAndNeigh(
327 face_subsimplexes, simplexification_pts, k);
328 Profiler::stop("integrate::updateQptsAndMappingForThisAndNeigh");
329 PROFILE(integrateSurfaceSubcell());
330 }
331 }
332 }
333 this->applyBasisScaling();
334}

337template <
338 typename MeshType,
339 typename FeExpressionTuple,
340 typename FeExpressionFaceTuple>
341template <typename SimplexMatrixType, typename VertexMatrixType>
342void WeakFormIntegratorSimplexify<
343 MeshType,
344 FeExpressionTuple,
345 FeExpressionFaceTuple>::
346 mapQuadraturePointsToSimplex(
347 const SimplexMatrixType& simplexes,
348 const VertexMatrixType& vertexes,
349 int32_t k) {
350 const std::size_t n_q_nodes_simplex = m_qrule_d.nodes.size();
351 const Eigen::Vector<double, dim> traslation =
352 vertexes.col(simplexes.coeff(0, k));
353 Eigen::Matrix<double, dim, dim> jacobian;
354 for (std::size_t d = 0; d < dim; ++d) {
355 jacobian.col(d) = vertexes.col(simplexes.coeff(d + 1, k)) - traslation;
356 }
357 const double jac_det = std::abs(vmath::determinant(jacobian));
358 // map the q nodes to the simplex
359 for (std::size_t q = 0; q < n_q_nodes_simplex; ++q) {
360 m_qrule_d.nodes[q].weight = m_qrule_d_ref->nodes[q].weight * jac_det;
361 const Eigen::Map<const Eigen::Vector<double, dim>> pt_map_src(
362 m_qrule_d_ref->nodes[q].point.data());
363 Eigen::Map<Eigen::Vector<double, dim>> pt_map_dst(
364 m_qrule_d.nodes[q].point.data());
365 pt_map_dst = jacobian * pt_map_src + traslation;
366 }
367}

370template <
371 typename MeshType,
372 typename FeExpressionTuple,
373 typename FeExpressionFaceTuple>
374void WeakFormIntegratorSimplexify<
375 MeshType,
376 FeExpressionTuple,
377 FeExpressionFaceTuple>::integrateVolumeCell() {
378 auto [bilinear_fe_expression, non_bilinear_fe_expression] =
379 internal::split_expression_tuple(this->m_fe_expression);
380 const std::size_t n_q_nodes_simplex = m_qrule_d.nodes.size();
381 const uint32_t n_dof_loc = this->m_fe_space.getBasis().size();
382 std::apply(
383 [this](auto&&... exprs) {
384 ((exprs->precompute(
385 this->m_fe_values, this->m_cell_manager.getCells())),
386 ...);
387 },
388 this->m_fe_expression);
389 // apply quadrature rule for each pair basis function (i, j)
390 for (uint32_t q = 0; q < n_q_nodes_simplex; ++q) {
391 const double Jdx = this->m_fe_values.weights()[q];
392 for (uint32_t i = 0; i < n_dof_loc; ++i) {
393 // TODO: it is worth it to make further local copies
394 // of data used in the expression to increase cache locality
395 // this should have ~10% speed-up
396 if constexpr (s_n_expr_bilinear) {
397 for (uint32_t j = 0; j < n_dof_loc; ++j) {
398 uint32_t expr_id = 0;
399 // TODO: evaluate if the std::apply is better as the most inner
400 // loop (like now) or in the most outer loop
401 std::apply(
402 [this, &expr_id, i, j, q, Jdx, n_dof_loc](auto&&... exprs) {
403 (([this, &expr_id, i, j, q, Jdx, n_dof_loc](auto&& expr) {
404 [this, expr_id, i, j, q, Jdx, expr,
405 n_dof_loc]<std::size_t... d>(std::index_sequence<d...>) {
406 ((this->m_integrals(
407 i * s_n_blocks + d % s_n_blocks,
408 j * s_n_blocks + d / s_n_blocks, expr_id) +=
409 expr->template evaluate<d % s_n_blocks, d / s_n_blocks>(
410 this->m_fe_values, i, j, q) *
411 Jdx),
412 ...);
413 }(std::make_index_sequence<s_n_blocks * s_n_blocks>{});
414 expr_id++;
415 }(exprs)),
416 ...);
417 },
418 bilinear_fe_expression);
419 }
420 }
421 if constexpr (s_n_expr_linear) {
422 constexpr auto UNDEF = static_cast<std::size_t>(-1);
423 uint32_t expr_id = 0;
424 std::apply(
425 [this, &expr_id, i, q, n_dof_loc](auto&&... exprs) {
426 (([this, &expr_id, i, q, n_dof_loc](auto&& expr) {
427 [this, expr_id, i, q, expr,
428 n_dof_loc]<std::size_t... d>(std::index_sequence<d...>) {
429 ((this->m_rhs(i * s_n_blocks + d, expr_id) +=
430 expr->template evaluate<UNDEF, d>(
431 this->m_fe_values, -1, i, q) *
432 this->m_fe_values.weights()[q]),
433 ...);
434 }(std::make_index_sequence<s_n_blocks>{});
435 expr_id++;
436 }(exprs)),
437 ...);
438 },
439 non_bilinear_fe_expression);
440 }
441 }
442 }
443}

446template <
447 typename MeshType,
448 typename FeExpressionTuple,
449 typename FeExpressionFaceTuple>
450template <typename FaceSubsimplexMatrix, typename SimplexPtsMatrix>
451void WeakFormIntegratorSimplexify<
452 MeshType,
453 FeExpressionTuple,
454 FeExpressionFaceTuple>::
455 updateQptsAndMappingForThisAndNeigh(
456 const FaceSubsimplexMatrix& face_subsimplexes,
457 const SimplexPtsMatrix& simplexification_pts,
458 index_t k) {
459 auto& cell = this->getCell(this->m_cell_id);
460 const std::size_t n_q_nodes_subsimplex = m_qrule_bndry_d.nodes.size();
461 // build the transformation from the reference (sub)simplex to subsimplex
462 const Eigen::Vector<int32_t, dim> subsimplex = face_subsimplexes.col(k);
463 Eigen::Vector<double, dim> traslation =
464 simplexification_pts.col(subsimplex[0]);
465 Eigen::Matrix<double, dim, dim - 1> jacobian;
466 for (std::size_t d = 0; d < dim - 1; ++d) {
467 jacobian.col(d) = simplexification_pts.col(subsimplex[d + 1]) - traslation;
468 }
469 // map the q nodes to the simplex
470 for (std::size_t q = 0; q < n_q_nodes_subsimplex; ++q) {
471 m_qrule_bndry_d.nodes[q].weight = m_qrule_bndry_d_ref->nodes[q].weight;
472 const Eigen::Map<const Eigen::Vector<double, dim - 1>> pt_map_src(
473 m_qrule_bndry_d_ref->nodes[q].point.data());
474 Eigen::Map<Eigen::Vector<double, dim>> pt_map_dst(
475 m_qrule_bndry_d.nodes[q].point.data());
476 pt_map_dst = jacobian * pt_map_src + traslation;
477 }
478
479 // update evaluations and mappings
480 this->m_fe_values.evalOnReferenceFrame(
481 this->m_fe_space.getBasisFast(), m_qrule_bndry_d);
482 // set the normal, needed to use the Nanson rule
483 const auto normal = cell.normals().col(this->m_face_id);
484 this->m_fe_values.setFaceNormal(normal);
485 auto bnd_mapping = *(this->m_mapping);
486 // transformation area of quadrature for sub-simplex
487 // the new area divided by the reference area i.e.
488 // the measure of orthogonal simplex in [0, 1]^(dim - 1)
489 constexpr double std_subsimp = 1. / vmath::factorial(dim - 1);
490 const double subsimplex_jdet = vmath::area(jacobian) / std_subsimp;
491 // nanson rule
492 // ||JF^-TN||, where F jacobian of ref-to-physic, J is |det(F)|,
493 // N unit outer in ref. By expoiting linearity and F is diagonal we have
494 // J/||Fn||, where n is the unit normal in physical space.
495 // Recall that normals are controvariant
496 const double jac_det = this->m_mapping->jacobianDet({});
497 const auto jac = this->m_mapping->getJacobian();
498 const double nanson = jac_det / jac.cwiseProduct(normal).norm();
499 // we just rescale quadrature weights accordingly
500 bnd_mapping.setJacobianDet(nanson * subsimplex_jdet);
501 this->m_fe_values.evalAndApplyMapping(
502 bnd_mapping, this->m_update_flags_face | fe::UpdateWeights);
503 // if we need the neigh we need the bounding box of the neigh cell
504 // to compute the transformation. This is the composition of three
505 // transformations since the cell is in reference configuration:
506 // 1. (sub)simplex-to-reference-subsimplex
507 // 2. bbox-transformation of the neigh
508 // 3. inverse-bbox-tranformation of this cell
509 if (this->m_need_neigh_values && this->m_cell_neigh_info.have_neigh) {
510 // map reference quadrature nodes to the one in the bounding
511 // box inverse bbox -> neigh bbox
512 const auto& neigh_mapping =
513 *this->m_cell_manager.getMapping(this->m_cell_neigh_info.neigh_cell_id);
514 const Eigen::Array<double, dim, 1> compose_jacobian_bbox =
515 neigh_mapping.getJacobianInv().array() *
516 this->m_mapping->getJacobian().array();
517 const Eigen::Array<double, dim, 1> compose_translation_bbox =
518 neigh_mapping.getJacobianInv().array() *
519 this->m_mapping->getTranslation().array() +
520 neigh_mapping.getTranslationInv().array();
521 // map the q nodes to the simplex
522 jacobian.array().colwise() *= compose_jacobian_bbox;
523 traslation =
524 compose_jacobian_bbox * traslation.array() + compose_translation_bbox;
525 for (std::size_t q = 0; q < n_q_nodes_subsimplex; ++q) {
526 m_qrule_bndry_d.nodes[q].weight = m_qrule_bndry_d_ref->nodes[q].weight;
527 const Eigen::Map<const Eigen::Vector<double, dim - 1>> pt_map_src(
528 m_qrule_bndry_d_ref->nodes[q].point.data());
529 Eigen::Map<Eigen::Vector<double, dim>> pt_map_dst(
530 m_qrule_bndry_d.nodes[q].point.data());
531 pt_map_dst = jacobian * pt_map_src + traslation;
532 }
533 // compute updated evaluation for the neigh
534 this->m_fe_values.neighValues().evalOnReferenceFrame(
535 this->m_fe_space.getBasisFast(), m_qrule_bndry_d);
536 this->m_fe_values.neighValues().evalAndApplyMapping(
537 neigh_mapping, this->m_update_flags_face);
538 }
539}

542template <
543 typename MeshType,
544 typename FeExpressionTuple,
545 typename FeExpressionFaceTuple>
546void WeakFormIntegratorSimplexify<
547 MeshType,
548 FeExpressionTuple,
549 FeExpressionFaceTuple>::integrateSurfaceSubcell() {
550 const uint32_t n_dof_loc = this->m_fe_space.getBasis().size();
551 const std::size_t n_q_nodes_subsimplex = m_qrule_bndry_d.nodes.size();
552 // TODO: evaluate if the std::apply is better as the most
553 // inner loop or in the most outer loop
554 auto [bilinear_fe_expression, non_bilinear_fe_expression] =
555 internal::split_expression_tuple(this->m_fe_expression_face);
556 if constexpr (s_n_face_expr_bilinear) {
557 vutils::for_each_idxed(
558 bilinear_fe_expression, [this, n_q_nodes_subsimplex, n_dof_loc](
559 const auto& expr, Eigen::Index expr_id) {
560 Eigen::Map<Eigen::MatrixXd> curr_integral =
561 this->getCurrFaceIntegral(expr, expr_id);
562 if (!curr_integral.size()) {
563 return;
564 }
565
566 [&]<std::size_t... d>(std::index_sequence<d...>) {
567 (
568 [&]() {
569 for (uint32_t q = 0; q < n_q_nodes_subsimplex; ++q) {
570 const double Jdx = this->m_fe_values.weights()[q];
571 for (uint32_t i = 0; i < n_dof_loc; ++i) {
572 for (uint32_t j = 0; j < n_dof_loc; ++j) {
573 constexpr auto bi = d % s_n_blocks;
574 constexpr auto bj = d / s_n_blocks;
575 curr_integral(
576 i * s_n_blocks + bi, j * s_n_blocks + bj) +=
577 expr->template evaluate<bi, bj>(
578 this->m_fe_values, i, j, q) *
579 Jdx;
580 }
581 }
582 }
583 }(),
584 ...);
585 }(std::make_index_sequence<s_n_blocks * s_n_blocks>{});
586 });
587 }
588 if constexpr (s_n_face_expr_linear) {
589 constexpr auto UNDEF = static_cast<std::size_t>(-1);
590 vutils::for_each_idxed(
591 non_bilinear_fe_expression,
592 [this, n_q_nodes_subsimplex, n_dof_loc](
593 const auto& expr, Eigen::Index expr_id) {
594 // the fe expression might want to skip a face
595 // e.g. for Dirichlet boundary conditions
596 if (!expr->precompute(
597 this->m_fe_values, this->m_cell_manager.getCells())) {
598 return;
599 }
600
601 [&]<std::size_t... d>(std::index_sequence<d...>) {
602 (
603 [&]() {
604 for (uint32_t q = 0; q < n_q_nodes_subsimplex; ++q) {
605 const double dx = this->m_fe_values.weights()[q];
606 for (uint32_t j = 0; j < n_dof_loc; ++j) {
607 this->m_face_rhs(j * s_n_blocks + d, expr_id) +=
608 expr->template evaluate<UNDEF, d>(
609 this->m_fe_values, -1, j, q) *
610 dx;
611 }
612 }
613 }(),
614 ...);
615 }(std::make_index_sequence<s_n_blocks>{});
616 });
617 }
