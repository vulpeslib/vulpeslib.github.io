284template <
285 typename MeshType,
286 typename FeExpressionTuple,
287 typename FeExpressionFaceTuple>
288void WeakFormIntegratorSimplexify<
289 MeshType,
290 FeExpressionTuple,
291 FeExpressionFaceTuple>::integrate(index_t cell_id) {
292 this->preallocate(cell_id);
293 auto& cell = this->getCell(this->m_cell_id, true);
294
295 const bool need_bnd = s_n_face_expr || this->m_need_neigh_values;
296 Profiler::start("integrate::simplexification");
297 // TODO: consider caching the simplexification for each cell
298 geometry::Simplexify<dim> simplexification(cell, need_bnd);
299 Profiler::stop("integrate::simplexification");
300 const auto& simplexes = simplexification.getSimplexesList();
301 const auto& vertexes = simplexification.getPointList();
302 // compute volume integrals
303 if constexpr (s_n_expr) {
304 // for each simplex
305 for (int32_t k = 0; k < simplexification.getNumSimplexes(); ++k) {
306 PROFILE(mapQuadraturePointsToSimplex(simplexes, vertexes, k));
307 // update evaluations of basis functions and mappings
308 this->m_fe_values.evalOnReferenceFrame(
309 this->m_fe_space.getBasisFast(), m_qrule_d);
310 this->m_fe_values.evalAndApplyMapping(
311 *(this->m_mapping), this->m_update_flags | fe::UpdateWeights);
312 PROFILE(integrateVolumeCell());
313 }
314 }
315 // compute boundary integrals
316 if constexpr (s_n_face_expr) {
317 // loop on faces
318 for (this->m_face_id = 0; this->m_face_id < cell.nfaces();
319 ++this->m_face_id) {
320 PROFILE(this->findCellNeigh());
321 const auto face_subsimplexes =
322 simplexification.getSubsimplexesListFacet(this->m_face_id);
323 const index_t nsubsimplexes = face_subsimplexes.cols();
324 const auto simplexification_pts = simplexification.getPointList();
325 // iterate face sub-simplifixation: triangles in 3D, a single line in 2D
326 for (index_t k = 0; k < nsubsimplexes; ++k) {
327 Profiler::start("integrate::updateQptsAndMappingForThisAndNeigh");
328 updateQptsAndMappingForThisAndNeigh(
329 face_subsimplexes, simplexification_pts, k);
330 Profiler::stop("integrate::updateQptsAndMappingForThisAndNeigh");
331 PROFILE(integrateSurfaceSubcell());
332 }
333 }
334 }
335 this->applyBasisScaling();
336}

338/* mapQuadraturePointsToSimplex definition */
339template <
340 typename MeshType,
341 typename FeExpressionTuple,
342 typename FeExpressionFaceTuple>
343template <typename SimplexMatrixType, typename VertexMatrixType>
344void WeakFormIntegratorSimplexify<
345 MeshType,
346 FeExpressionTuple,
347 FeExpressionFaceTuple>::
348 mapQuadraturePointsToSimplex(
349 const SimplexMatrixType& simplexes,
350 const VertexMatrixType& vertexes,
351 int32_t k) {
352 const std::size_t n_q_nodes_simplex = m_qrule_d.nodes.size();
353 const Eigen::Vector<double, dim> traslation =
354 vertexes.col(simplexes.coeff(0, k));
355 Eigen::Matrix<double, dim, dim> jacobian;
356 for (std::size_t d = 0; d < dim; ++d) {
357 jacobian.col(d) = vertexes.col(simplexes.coeff(d + 1, k)) - traslation;
358 }
359 const double jac_det = std::abs(vmath::determinant(jacobian));
360 // map the q nodes to the simplex
361 for (std::size_t q = 0; q < n_q_nodes_simplex; ++q) {
362 m_qrule_d.nodes[q].weight = m_qrule_d_ref->nodes[q].weight * jac_det;
363 const Eigen::Map<const Eigen::Vector<double, dim>> pt_map_src(
364 m_qrule_d_ref->nodes[q].point.data());
365 Eigen::Map<Eigen::Vector<double, dim>> pt_map_dst(
366 m_qrule_d.nodes[q].point.data());
367 pt_map_dst = jacobian * pt_map_src + traslation;
368 }
369}

371/* integrateVolumeCell definition */
372template <
373 typename MeshType,
374 typename FeExpressionTuple,
375 typename FeExpressionFaceTuple>
376void WeakFormIntegratorSimplexify<
377 MeshType,
378 FeExpressionTuple,
379 FeExpressionFaceTuple>::integrateVolumeCell() {
380 auto [bilinear_fe_expression, non_bilinear_fe_expression] =
381 internal::split_expression_tuple(this->m_fe_expression);
382 const std::size_t n_q_nodes_simplex = m_qrule_d.nodes.size();
383 const uint32_t n_dof_loc = this->m_fe_space.getBasis().size();
384 std::apply(
385 [this](auto&&... exprs) {
386 ((exprs->precompute(
387 this->m_fe_values, this->m_cell_manager.getCells())),
388 ...);
389 },
390 this->m_fe_expression);
391 // apply quadrature rule for each pair basis function (i, j)
392 for (uint32_t q = 0; q < n_q_nodes_simplex; ++q) {
393 const double Jdx = this->m_fe_values.weights()[q];
394 for (uint32_t i = 0; i < n_dof_loc; ++i) {
395 // TODO: it is worth it to make further local copies
396 // of data used in the expression to increase cache locality
397 // this should have ~10% speed-up
398 if constexpr (s_n_expr_bilinear) {
399 for (uint32_t j = 0; j < n_dof_loc; ++j) {
400 uint32_t expr_id = 0;
401 // TODO: evaluate if the std::apply is better as the most inner
402 // loop (like now) or in the most outer loop
403 std::apply(
404 [this, &expr_id, i, j, q, Jdx, n_dof_loc](auto&&... exprs) {
405 (([this, &expr_id, i, j, q, Jdx, n_dof_loc](auto&& expr) {
406 [this, expr_id, i, j, q, Jdx, expr,
407 n_dof_loc]<std::size_t... d>(std::index_sequence<d...>) {
408 ((this->m_integrals(
409 i * s_n_blocks + d % s_n_blocks,
410 j * s_n_blocks + d / s_n_blocks, expr_id) +=
411 expr->template evaluate<d % s_n_blocks, d / s_n_blocks>(
412 this->m_fe_values, i, j, q) *
413 Jdx),
414 ...);
415 }(std::make_index_sequence<s_n_blocks * s_n_blocks>{});
416 expr_id++;
417 }(exprs)),
418 ...);
419 },
420 bilinear_fe_expression);
421 }
422 }
423 if constexpr (s_n_expr_linear) {
424 constexpr auto UNDEF = static_cast<std::size_t>(-1);
425 uint32_t expr_id = 0;
426 std::apply(
427 [this, &expr_id, i, q, n_dof_loc](auto&&... exprs) {
428 (([this, &expr_id, i, q, n_dof_loc](auto&& expr) {
429 [this, expr_id, i, q, expr,
430 n_dof_loc]<std::size_t... d>(std::index_sequence<d...>) {
431 ((this->m_rhs(i * s_n_blocks + d, expr_id) +=
432 expr->template evaluate<UNDEF, d>(
433 this->m_fe_values, -1, i, q) *
434 this->m_fe_values.weights()[q]),
435 ...);
436 }(std::make_index_sequence<s_n_blocks>{});
437 expr_id++;
438 }(exprs)),
439 ...);
440 },
441 non_bilinear_fe_expression);
442 }
443 }
444 }
445}

448template <
449 typename MeshType,
450 typename FeExpressionTuple,
451 typename FeExpressionFaceTuple>
452template <typename FaceSubsimplexMatrix, typename SimplexPtsMatrix>
453void WeakFormIntegratorSimplexify<
454 MeshType,
455 FeExpressionTuple,
456 FeExpressionFaceTuple>::
457 updateQptsAndMappingForThisAndNeigh(
458 const FaceSubsimplexMatrix& face_subsimplexes,
459 const SimplexPtsMatrix& simplexification_pts,
460 index_t k) {
461 auto& cell = this->getCell(this->m_cell_id);
462 const std::size_t n_q_nodes_subsimplex = m_qrule_bndry_d.nodes.size();
463 // build the transformation from the reference (sub)simplex to subsimplex
464 const Eigen::Vector<int32_t, dim> subsimplex = face_subsimplexes.col(k);
465 Eigen::Vector<double, dim> traslation =
466 simplexification_pts.col(subsimplex[0]);
467 Eigen::Matrix<double, dim, dim - 1> jacobian;
468 for (std::size_t d = 0; d < dim - 1; ++d) {
469 jacobian.col(d) = simplexification_pts.col(subsimplex[d + 1]) - traslation;
470 }
471 // map the q nodes to the simplex
472 for (std::size_t q = 0; q < n_q_nodes_subsimplex; ++q) {
473 m_qrule_bndry_d.nodes[q].weight = m_qrule_bndry_d_ref->nodes[q].weight;
474 const Eigen::Map<const Eigen::Vector<double, dim - 1>> pt_map_src(
475 m_qrule_bndry_d_ref->nodes[q].point.data());
476 Eigen::Map<Eigen::Vector<double, dim>> pt_map_dst(
477 m_qrule_bndry_d.nodes[q].point.data());
478 pt_map_dst = jacobian * pt_map_src + traslation;
479 }
480
481 // update evaluations and mappings
482 this->m_fe_values.evalOnReferenceFrame(
483 this->m_fe_space.getBasisFast(), m_qrule_bndry_d);
484 // set the normal, needed to use the Nanson rule
485 const auto normal = cell.normals().col(this->m_face_id);
486 this->m_fe_values.setFaceNormal(normal);
487 auto bnd_mapping = *(this->m_mapping);
488 // transformation area of quadrature for sub-simplex
489 // the new area divided by the reference area i.e.
490 // the measure of orthogonal simplex in [0, 1]^(dim - 1)
491 constexpr double std_subsimp = 1. / vmath::factorial(dim - 1);
492 const double subsimplex_jdet = vmath::area(jacobian) / std_subsimp;
493 // nanson rule
494 // ||JF^-TN||, where F jacobian of ref-to-physic, J is |det(F)|,
495 // N unit outer in ref. By expoiting linearity and F is diagonal we have
496 // J/||Fn||, where n is the unit normal in physical space.
497 // Recall that normals are controvariant
498 const double jac_det = this->m_mapping->jacobianDet({});
499 const auto jac = this->m_mapping->getJacobian();
500 const double nanson = jac_det / jac.cwiseProduct(normal).norm();
501 // we just rescale quadrature weights accordingly
502 bnd_mapping.setJacobianDet(nanson * subsimplex_jdet);
503 this->m_fe_values.evalAndApplyMapping(
504 bnd_mapping, this->m_update_flags_face | fe::UpdateWeights);
505 // if we need the neigh we need the bounding box of the neigh cell
506 // to compute the transformation. This is the composition of three
507 // transformations since the cell is in reference configuration:
508 // 1. (sub)simplex-to-reference-subsimplex
509 // 2. bbox-transformation of the neigh
510 // 3. inverse-bbox-tranformation of this cell
511 // For periodic boundaries, a shift in physical coordinates is added
512 // before applying the neighbor's inverse transformation (step 3)
513 if (this->m_need_neigh_values && this->m_cell_neigh_info.have_neigh) {
514 // map reference quadrature nodes to the one in the bounding
515 // box inverse bbox -> neigh bbox
516 const auto& neigh_mapping =
517 *this->m_cell_manager.getMapping(this->m_cell_neigh_info.neigh_cell_id);
518
519 const Eigen::Array<double, dim, 1> compose_jacobian_bbox =
520 neigh_mapping.getJacobianInv().array() *
521 this->m_mapping->getJacobian().array();
522
523 // Composed translation: current->physical->neighbor with periodic shift
524 const Eigen::Array<double, dim, 1> compose_translation_bbox =
525 neigh_mapping.getJacobianInv().array() *
526 (this->m_mapping->getTranslation().array() +
527 this->m_cell_manager.getPeriodicShift(
528 this->m_cell_neigh_info.neigh_cell_id, this->m_cell_id)) +
529 neigh_mapping.getTranslationInv().array();
530
531 // map the q nodes to the simplex
532 jacobian.array().colwise() *= compose_jacobian_bbox;
533 traslation =
534 compose_jacobian_bbox * traslation.array() + compose_translation_bbox;
535 for (std::size_t q = 0; q < n_q_nodes_subsimplex; ++q) {
536 m_qrule_bndry_d.nodes[q].weight = m_qrule_bndry_d_ref->nodes[q].weight;
537 const Eigen::Map<const Eigen::Vector<double, dim - 1>> pt_map_src(
538 m_qrule_bndry_d_ref->nodes[q].point.data());
539 Eigen::Map<Eigen::Vector<double, dim>> pt_map_dst(
540 m_qrule_bndry_d.nodes[q].point.data());
541 pt_map_dst = jacobian * pt_map_src + traslation;
542 }
543 // compute updated evaluation for the neigh
544 this->m_fe_values.neighValues().evalOnReferenceFrame(
545 this->m_fe_space.getBasisFast(), m_qrule_bndry_d);
546 this->m_fe_values.neighValues().evalAndApplyMapping(
547 neigh_mapping, this->m_update_flags_face);
548 }
549}

551/* integrateSurfaceSubcell definition */
552template <
553 typename MeshType,
554 typename FeExpressionTuple,
555 typename FeExpressionFaceTuple>
556void WeakFormIntegratorSimplexify<
557 MeshType,
558 FeExpressionTuple,
559 FeExpressionFaceTuple>::integrateSurfaceSubcell() {
560 const uint32_t n_dof_loc = this->m_fe_space.getBasis().size();
561 const std::size_t n_q_nodes_subsimplex = m_qrule_bndry_d.nodes.size();
562 // TODO: evaluate if the std::apply is better as the most
563 // inner loop or in the most outer loop
564 auto [bilinear_fe_expression, non_bilinear_fe_expression] =
565 internal::split_expression_tuple(this->m_fe_expression_face);
566 if constexpr (s_n_face_expr_bilinear) {
567 vutils::for_each_idxed(
568 bilinear_fe_expression, [this, n_q_nodes_subsimplex, n_dof_loc](
569 const auto& expr, Eigen::Index expr_id) {
570 Eigen::Map<Eigen::MatrixXd> curr_integral =
571 this->getCurrFaceIntegral(expr, expr_id);
572 if (!curr_integral.size()) {
573 return;
574 }
575
576 [&]<std::size_t... d>(std::index_sequence<d...>) {
577 (
578 [&]() {
579 for (uint32_t q = 0; q < n_q_nodes_subsimplex; ++q) {
580 const double Jdx = this->m_fe_values.weights()[q];
581 for (uint32_t i = 0; i < n_dof_loc; ++i) {
582 for (uint32_t j = 0; j < n_dof_loc; ++j) {
583 constexpr auto bi = d % s_n_blocks;
584 constexpr auto bj = d / s_n_blocks;
585 curr_integral(
586 i * s_n_blocks + bi, j * s_n_blocks + bj) +=
587 expr->template evaluate<bi, bj>(
588 this->m_fe_values, i, j, q) *
589 Jdx;
590 }
591 }
592 }
593 }(),
594 ...);
595 }(std::make_index_sequence<s_n_blocks * s_n_blocks>{});
596 });
597 }
598 if constexpr (s_n_face_expr_linear) {
599 constexpr auto UNDEF = static_cast<std::size_t>(-1);
600 vutils::for_each_idxed(
601 non_bilinear_fe_expression,
602 [this, n_q_nodes_subsimplex, n_dof_loc](
603 const auto& expr, Eigen::Index expr_id) {
604 // the fe expression might want to skip a face
605 // e.g. for Dirichlet boundary conditions
606 if (!expr->precompute(
607 this->m_fe_values, this->m_cell_manager.getCells())) {
608 return;
609 }
610
611 [&]<std::size_t... d>(std::index_sequence<d...>) {
612 (
613 [&]() {
614 for (uint32_t q = 0; q < n_q_nodes_subsimplex; ++q) {
615 const double dx = this->m_fe_values.weights()[q];
616 for (uint32_t j = 0; j < n_dof_loc; ++j) {
617 this->m_face_rhs(j * s_n_blocks + d, expr_id) +=
618 expr->template evaluate<UNDEF, d>(
619 this->m_fe_values, -1, j, q) *
620 dx;
621 }
622 }
623 }(),
624 ...);
625 }(std::make_index_sequence<s_n_blocks>{});
626 });
627 }
628}

645template <typename MeshType, typename FeSpaceType, typename ExprType>
646class ScalarExpressionIntegrator {
647 public:
648 using index_t = typename MeshType::index_t;
649 static constexpr std::size_t Dim = MeshType::self_dim;
650
658 ScalarExpressionIntegrator(
659 const MeshType& mesh,
660 CellManager<MeshType, FeSpaceType>& cell_manager,
661 const ExprType& expression,
662 uint32_t quadrature_order = 5)
663 : m_mesh(mesh),
664 m_cell_manager(cell_manager),
665 m_expression(expression),
666 m_quadrature_order(quadrature_order) {
667 // Cache reference quadrature rule
668 m_qrule_ref = qrule::QuadratureLibrary<Dim, double>::getRule(
669 qrule::QuadratureElemType::Simplex, quadrature_order);
670 }
671
676 double integrate() {
677 double integral = 0.0;
678 const index_t n_cells = [&] {
679 if constexpr (MeshType::parallel()) {
680 return m_mesh.nLocEntity(Dim);
681 } else {
682 return m_mesh.nEntity(Dim);
683 }
684 }();
685
686 // Loop over all cells
687 for (index_t cell_id = 0; cell_id < n_cells; ++cell_id) {
688 integral += integrateCell(cell_id);
689 }
690
691 return integral;
692 }
693
694 private:
695 const MeshType& m_mesh;
696 CellManager<MeshType, FeSpaceType>& m_cell_manager;
697 const ExprType& m_expression;
698 uint32_t m_quadrature_order;
699 std::shared_ptr<qrule::QuadratureRule<double, Dim>> m_qrule_ref;
700
704 double integrateCell(index_t cell_id) {
705 double cell_integral = 0.0;
706
707 const auto& cell = m_cell_manager.getCell(cell_id);
708
709 // Simplexify cell (polygons -> simplices)
710 geometry::Simplexify<Dim> simplexification(
711 cell, false); // no boundary needed
712 const auto& simplexes = simplexification.getSimplexesList();
713 const auto& vertexes = simplexification.getPointList();
714
715 // Get mapping and FE space for this cell
716 const auto& mapping = *m_cell_manager.getMapping(cell_id);
717 const auto& fe_space = m_cell_manager.getFeSpace();
718
719 // Integrate over each simplex
720 for (int32_t k = 0; k < simplexification.getNumSimplexes(); ++k) {
721 cell_integral += integrateSimplexDirect(
722 cell_id, k, simplexes, vertexes, mapping, fe_space);
723 }
724
725 return cell_integral;
726 }
727
731 double integrateSimplexDirect(
732 index_t cell_id,
733 int32_t simplex_id,
734 const Eigen::MatrixXi& simplexes,
735 const Eigen::MatrixXd& vertexes,
736 const auto& mapping,
737 const auto& fe_space) {
738 double simplex_integral = 0.0;
739
740 qrule::QuadratureRule<double, Dim> qrule = *m_qrule_ref;
741
742 mapQPointsToPhysicalSimplex(qrule, simplexes, vertexes, simplex_id);
743
744 fe::FeValues<Dim> fe_values;
745 fe_values.setCellId(cell_id);
746 fe_values.evalOnReferenceFrame(fe_space.getBasisFast(), qrule);
747
748 // Compute update flags needed by the expression
749 // This ensures coordinate systems are properly synchronized for all
750 // sub-expressions
751 fe::UpdateFlagsType update_flags = m_expression.computeUpdateFlags();
752 fe_values.evalAndApplyMapping(mapping, update_flags | fe::UpdateWeights);
753
754 // Precompute expression
755 m_expression.precompute(fe_values, m_cell_manager.getCells());
756
757 // Accumulate integral - like integrateVolumeCell but WITHOUT basis loops
758 // For pure scalar integration: template <0,0> for scalar component,
759 // but pass -1 for both trial and test basis indices (no basis functions)
760 const std::size_t n_q_nodes = qrule.nodes.size();
761 for (std::size_t q = 0; q < n_q_nodes; ++q) {
762 const double Jdx = fe_values.weights()[q];
763 // Scalar expression: no trial function (-1), no test function (-1)
764 double expr_value =
765 m_expression.template evaluate<0, 0>(fe_values, -1, -1, q);
766 simplex_integral += expr_value * Jdx;
767 }
768
769 return simplex_integral;
770 }
771
777 void mapQPointsToPhysicalSimplex(
778 qrule::QuadratureRule<double, Dim>& qrule,
779 const Eigen::MatrixXi& simplexes,
780 const Eigen::MatrixXd& vertexes,
781 int32_t k) {
782 const std::size_t n_q_nodes = qrule.nodes.size();
783
784 // Extract simplex vertex positions
785 const Eigen::Vector<double, Dim> translation =
786 vertexes.col(simplexes.coeff(0, k));
787 Eigen::Matrix<double, Dim, Dim> jacobian;
788 for (std::size_t d = 0; d < Dim; ++d) {
789 jacobian.col(d) = vertexes.col(simplexes.coeff(d + 1, k)) - translation;
790 }
791 const double jac_det = std::abs(vmath::determinant(jacobian));
792
793 // Map quadrature points and scale weights by Jacobian determinant
794 for (std::size_t q = 0; q < n_q_nodes; ++q) {
795 qrule.nodes[q].weight = m_qrule_ref->nodes[q].weight * jac_det;
796 const Eigen::Map<const Eigen::Vector<double, Dim>> pt_map_src(
797 m_qrule_ref->nodes[q].point.data());
798 Eigen::Map<Eigen::Vector<double, Dim>> pt_map_dst(
799 qrule.nodes[q].point.data());
800 pt_map_dst = jacobian * pt_map_src + translation;
801 }
802 }
803};
