63template <int32_t Dim>
64std::pair<double, double> updateVoronoiCentroid(
65 const std::array<Point<double, Dim>, 2>& bbox,
66 std::vector<Point<double, Dim>>& points,
67 int32_t n_inside_seeds) {
68 double err = 0.0;
69 double total_area = 0.0;
70
71 voronoi<Dim>(bbox, points, [&](auto& cell, auto id) {
72 double area_and_centroid[Dim + 1];
73 cell.centroid(area_and_centroid);
74 Eigen::Map<Point<double, Dim>> centroid(area_and_centroid + 1);
75 points[id] += centroid;
76 // the point may be outside due to the adjustment
77 // however we are interested in an estimate of the error
78 double area = area_and_centroid[0];
79 if (id < n_inside_seeds) {
80 err += centroid.cwiseAbs2().sum() * area * area;
81 total_area += area;
82 }
83 });
84
85 const double total_err =
86 std::sqrt(err) * points.size() / std::pow(total_area, 1.5);
87 return {total_err, total_area};
88}

90template <typename FloatType, typename IndexType, IndexType Dim, typename D>
91Point<FloatType, Dim> computeGradient(
92 const Point<FloatType, Dim> x,
93 const DistanceExpression<double, Dim, D>& distanceFunction,
94 IndexType idx,
95 FloatType finite_difference_h) {
96 Point<FloatType, Dim> gradient;
97 for (size_t d = 0; d < Dim; ++d) {
98 Point<double, Dim> x_plus = x;
99 Point<double, Dim> x_minus = x;
100 x_plus[d] += finite_difference_h;
101 x_minus[d] -= finite_difference_h;
102 // TODO: trunc computations of distance at j-th index
103 const auto d_x_plus = computeDistance(x_plus, distanceFunction);
104 const auto d_x_minus = computeDistance(x_minus, distanceFunction);
105 if (idx == -1) {
106 gradient[d] = d_x_plus.back() - d_x_minus.back();
107
108 } else {
109 gradient[d] = d_x_plus[idx] - d_x_minus[idx];
110 }
111 gradient[d] /= 2.0 * finite_difference_h;
112 }
113 return gradient;
114}

116template <int32_t Dim, typename D>
117Mesh2D<double, int32_t> computeVoronoiMesh2D(
118 const std::array<Point<double, 2>, 2>& bbox,
119 const std::vector<Point<double, 2>>& points,
120 const DistanceExpression<double, Dim, D>& distanceFunction) {
121 // the square of the euclidean distance
122 constexpr double same_vertex_distance_tol = 1e-30;
123
124 datastructure::CrsMapUncompressed<int32_t> vertices_idxs;
125 std::vector<Point<double, 2>> vertices;
126 std::vector<bool> visited(points.size(), false);
127 std::vector<int32_t> neighs_buffer(128);
128 std::vector<double> vertices_buffer(128);
129
130 voronoi2D(
131 bbox, points, [&](const voro::voronoicell_neighbor_2d& cell, auto id) {
132 cell.neighbors(neighs_buffer);
133 cell.ordered_vertices(points[id][0], points[id][1], vertices_buffer);
134 const size_t n_cell_vtxs = vertices_buffer.size() / 2;
135
136 // if the cell is internal to the domain
137 if (computeDistance(points[id], distanceFunction).back() <= 0.0) {
138 // for each vertex of this cell
139 for (size_t vid = 0; vid < n_cell_vtxs; ++vid) {
140 bool collision_found = false;
141 // for each cell in the neighbourhood
142 for (size_t cidn = 0;
143 cidn < neighs_buffer.size() && !collision_found; ++cidn) {
144 const int32_t neigh = neighs_buffer[cidn];
145 // if neigh < 0, means the cell is touching the boundary bbox
146 if (neigh >= 0 && visited[neigh]) {
147 const size_t n_neigh_vtxs = vertices_idxs.getRow(neigh).size();
148 for (size_t vidn = 0; vidn < n_neigh_vtxs; ++vidn) {
149 const Point<double, 2> vtx1 =
150 vertices[vertices_idxs.getRow(neigh)[vidn]];
151 const Eigen::Map<const Point<double, 2>> vtx2(
152 &vertices_buffer[2 * vid]);
153 const double dist_sqrd = (vtx1 - vtx2).cwiseAbs2().sum();
154 const double magn_sqrd =
155 vtx1.cwiseAbs2().sum() + vtx2.cwiseAbs2().sum();
156 if (dist_sqrd / magn_sqrd < same_vertex_distance_tol) {
157 collision_found = true;
158 vertices_idxs.getRow(id).push_back(
159 vertices_idxs.getRow(neigh)[vidn]);
160 break;
161 }
162 }
163 }
164 }
165
166 if (!collision_found) {
167 vertices.emplace_back(
168 vertices_buffer[2 * vid], vertices_buffer[2 * vid + 1]);
169 vertices_idxs.getRow(id).push_back(vertices.size() - 1);
170 }
171 }
172 visited[id] = true;
173 }
174 });
175 return Mesh2D<double, int32_t>(vertices, vertices_idxs);
176}

178template <int32_t Dim, typename D>
179Mesh3D<double, int32_t> computeVoronoiMesh3D(
180 const std::array<Point<double, 3>, 2>& bbox,
181 const std::vector<Point<double, 3>>& points,
182 const DistanceExpression<double, Dim, D>& distanceFunction) {
183 // the square of the euclidean distance
184 constexpr double same_vertex_distance_tol = 1e-30;
185
186 datastructure::CrsMapUncompressed<int32_t> vertices_idxs;
187 std::vector<Point<double, 3>> vertices;
188 std::vector<bool> visited(points.size(), false);
189 std::vector<int32_t> neighs_buffer(128);
190 std::vector<double> vertices_buffer(128);
191 std::vector<std::vector<int32_t>> faces_staggered;
192 std::vector<int32_t> face_sizes(128);
193 std::vector<int32_t> face_vertices(512);
194
195 voronoi3D(bbox, points, [&](auto& cell, auto id) {
196 const auto dist = computeDistance(points[id], distanceFunction).back();
197
198 if (dist <= 0.0) {
199 cell.vertices(
200 points[id][0], points[id][1], points[id][2], vertices_buffer);
201 assert(vertices_buffer.size() % 3 == 0);
202 cell.neighbors(neighs_buffer);
203 // for each vertex of this cell
204 const size_t n_cell_vtxs = vertices_buffer.size() / 3;
205 for (size_t vid = 0; vid < n_cell_vtxs; ++vid) {
206 bool collision_found = false;
207 // for each cell in the neighbourhood
208 const size_t n_neighs = neighs_buffer.size();
209 for (size_t cidn = 0; cidn < n_neighs && !collision_found; ++cidn) {
210 const int32_t neigh = neighs_buffer[cidn];
211 // if neigh < 0, means the cell is touching the boundary bbox
212 if (neigh >= 0 && visited[neigh]) {
213 const size_t n_neigh_vtxs = vertices_idxs.getRow(neigh).size();
214 for (size_t vidn = 0; vidn < n_neigh_vtxs; ++vidn) {
215 const Point<double, 3> vtx1 =
216 vertices[vertices_idxs.getRow(neigh)[vidn]];
217 const Eigen::Map<const Point<double, 3>> vtx2(
218 &vertices_buffer[3 * vid]);
219 const double dist_sqrd = (vtx1 - vtx2).cwiseAbs2().sum();
220 const double magn_sqrd =
221 vtx1.cwiseAbs2().sum() + vtx2.cwiseAbs2().sum();
222 if (dist_sqrd / magn_sqrd < same_vertex_distance_tol) {
223 collision_found = true;
224 vertices_idxs.getRow(id).push_back(
225 vertices_idxs.getRow(neigh)[vidn]);
226 break;
227 }
228 }
229 }
230 }
231
232 if (!collision_found) {
233 vertices.emplace_back(
234 vertices_buffer[3 * vid], vertices_buffer[3 * vid + 1],
235 vertices_buffer[3 * vid + 2]);
236 vertices_idxs.getRow(id).push_back(vertices.size() - 1);
237 }
238 }
239 visited[id] = true;
240
241 // Get face-vertices connectivity
242 cell.face_vertices(face_vertices, face_sizes);
243
244 faces_staggered.emplace_back();
245 size_t count = 0;
246 // for each face
247 for (size_t j = 0; j < face_sizes.size(); ++j) {
248 for (int32_t k = 0; k < face_sizes[j] + 1; ++k) {
249 assert(count < face_vertices.size());
250 if (k == 0) {
251 assert(face_vertices[count] == face_sizes[j]);
252 faces_staggered.back().push_back(face_vertices[count++]);
253 } else {
254 faces_staggered.back().push_back(
255 vertices_idxs.getRow(id)[face_vertices[count++]]);
256 }
257 }
258 }
259 assert(count == face_vertices.size());
260 }
261 });
262 return Mesh3D<double, int32_t>{vertices, vertices_idxs, faces_staggered};
263}

265template <typename FloatType, typename IndexType, IndexType Dim>
266void polyMesherAdjustFixedPoints(
267 const std::vector<Point<FloatType, Dim>>& fixed_pts,
268 const std::vector<IndexType>& near_boundary_idxs,
269 std::vector<Point<FloatType, Dim>>& seed_pts) {
270 if (fixed_pts.empty()) {
271 return;
272 }
273
274 for (const auto& fix_pt : fixed_pts) {
275 // find the 2 * dim (TODO: or 2 + Dim?) closest points to `fix_pt`
276 std::array<std::pair<IndexType, FloatType>, 2 * Dim> closest_seeds;
277 for (size_t d = 0; d < closest_seeds.size(); ++d) {
278 closest_seeds[d] = {-1, std::numeric_limits<FloatType>::max()};
279 }
280 // TODO: here we could be much faster with a data stucture like a kdtree
281 // however this should have complexity sqrt(n_seeds) which is reasonable
282 for (const IndexType idx : near_boundary_idxs) {
283 const FloatType dist = (seed_pts[idx] - fix_pt).norm();
284 auto max_it = std::max_element(
285 closest_seeds.begin(), closest_seeds.end(),
286 [](const auto& lhs, const auto& rhs) {
287 return lhs.second < rhs.second;
288 });
289 if (dist < max_it->second) {
290 max_it->second = dist;
291 max_it->first = idx;
292 }
293 }
294
295 // TODO: is this needed? we could move the points on any circumference
296 std::sort(
297 closest_seeds.begin(), closest_seeds.end(),
298 [](const auto& lhs, const auto& rhs) {
299 return lhs.second < rhs.second;
300 });
301
302 if (closest_seeds.back().first == -1) {
303 std::cout << "WARNING: could not find enough points near the fixed point "
304 << std::endl;
305 } else {
306 // fix the tesselation points
307 for (size_t i = 1; i < closest_seeds.size(); ++i) {
308 Point<FloatType, Dim> normal =
309 seed_pts[closest_seeds[i].first] - fix_pt;
310 seed_pts[closest_seeds[i].first] -=
311 normal * (1.0 - closest_seeds[0].second / closest_seeds[i].second);
312 // we are moving the points on the circle centered in `fix_pt`, check
313 // the distance from center is always the same
314 assert(
315 std::abs(
316 (fix_pt - seed_pts[closest_seeds[i].first]).norm() -
317 (fix_pt - seed_pts[closest_seeds[0].first]).norm()) <
318 2 * std::numeric_limits<FloatType>::epsilon());
319 }
320 }
321 }
322}

324template <typename FloatType, typename IndexType, IndexType Dim, typename D>
325std::vector<IndexType> polyMesherReflect(
326 const DistanceExpression<FloatType, Dim, D>& distanceFunction,
327 FloatType alpha,
328 const polymesher::PolyMesherParams<FloatType>& params,
329 std::vector<Point<FloatType, Dim>>& seed_pts) {
330 std::vector<IndexType> near_boundary_idxs;
331
332 const size_t n_seed_pts = seed_pts.size();
333 size_t near_boundary = 0;
334 size_t reflected_with_success = 0;
335 for (size_t i = 0; i < n_seed_pts; ++i) {
336 const Point<FloatType, Dim> seed_pt_i = seed_pts[i];
337 const auto d_i = computeDistance(seed_pt_i, distanceFunction);
338 assert(d_i.back() < 0.0);
339 if (-d_i.back() < alpha) {
340 near_boundary++;
341 near_boundary_idxs.push_back(i);
342 bool reflection_inserted = false;
343 // loop subdomains
344 for (size_t j = 0; j < d_i.size() - 1; ++j) {
345 // distance of point i from boundary j (might not be positive!)
346 const FloatType d_seed_j = d_i[j];
347 const FloatType d_seed_j_abs = std::abs(d_seed_j);
348 if (d_seed_j_abs < alpha) {
349 // compute reflection with finite difference
350 // R(x) = x - 2d(x)\nabla d(x)
351 const Point<FloatType, Dim> gradient = computeGradient(
352 seed_pt_i, distanceFunction, (IndexType)j,
353 params.finite_difference_h);
354 const Point<FloatType, Dim> rflc_pt =
355 seed_pts[i] - 2.0 * d_seed_j * gradient;
356 const auto pt_d = computeDistance(rflc_pt, distanceFunction);
357 // distance is positive (is outside) and necessary for this subdomain
358 if (pt_d.back() > 0 &&
359 pt_d.back() > params.reflection_relax * d_seed_j_abs) {
360 near_boundary_idxs.push_back(seed_pts.size());
361 seed_pts.push_back(rflc_pt);
362 reflection_inserted = true;
363 }
364 }
365 }
366 reflected_with_success += reflection_inserted;
367 }
368 }
369
370 if (reflected_with_success < near_boundary) {
371 std::cout << "WARNING: could not insert some reflections, succeeded: "
372 << reflected_with_success << " / " << near_boundary << std::endl;
373 }
374
375 return near_boundary_idxs;
376}

378template <typename FloatType, typename IndexType, IndexType Dim, typename D>
379void polyMesherClipToBoundary(
380 Mesh2D<FloatType, IndexType>& mesh,
381 FloatType alpha,
382 const DistanceExpression<FloatType, Dim, D>& distanceFunction,
383 const polymesher::PolyMesherParams<FloatType>& params) {
384 size_t possible_clipping = 0;
385 size_t successfull_clipping = 0;
386 for (auto& vtx : mesh.getVertexes()) {
387 const auto distance_from_bnd =
388 computeDistance(vtx, distanceFunction).back();
389 // if close enough to the boundary we try to clip the point to it
390 if (std::abs(distance_from_bnd) < params.clipping_relax * alpha) {
391 possible_clipping++;
392 const Point<FloatType, Dim> gradient = computeGradient(
393 vtx, distanceFunction, -1, params.finite_difference_h);
394 const Point<FloatType, Dim> new_vtx = vtx - distance_from_bnd * gradient;
395 // check the adjustment improved the position
396 // (gradient computaitons can be ill)
397 const auto new_distance_from_bnd =
398 computeDistance(new_vtx, distanceFunction).back();
399 if (std::abs(new_distance_from_bnd) < std::abs(distance_from_bnd)) {
400 vtx = new_vtx;
401 successfull_clipping++;
402 }
403 }
404 }
405 std::cout << "possible_clipping: " << possible_clipping
406 << " successfull_clipping: " << successfull_clipping << std::endl;
407}

409template <typename FloatType, typename IndexType>
410FloatType polyMesherComputeEdgeAngle(
411 const Mesh2D<FloatType, IndexType>& mesh,
412 IndexType cell_id,
413 const std::array<IndexType, 2>& edge) {
414 if (cell_id == -1) {
415 return std::numeric_limits<FloatType>::max();
416 }
417 const auto vtx_ids = mesh.getCellToVtxMap().getRow(cell_id);
418 const auto& pts = mesh.getVertexes();
419
420 FloatType meanvy = 0.0;
421 FloatType meanvx = 0.0;
422 for (const auto& vid : vtx_ids) {
423 meanvx += pts[vid][0];
424 meanvy += pts[vid][1];
425 }
426 meanvx /= vtx_ids.size();
427 meanvy /= vtx_ids.size();
428
429 constexpr FloatType pi2 = 2 * vmath::pi<FloatType>();
430 FloatType angle =
431 std::atan2(pts[edge[1]][1] - meanvy, pts[edge[1]][0] - meanvx) -
432 std::atan2(pts[edge[0]][1] - meanvy, pts[edge[0]][0] - meanvx);
433
434 // numerical stability
435 constexpr auto min_angle = 1e-8;
436 angle += min_angle;
437 angle = std::fmod(angle + pi2, pi2);
438 assert((angle >= 0) && (angle <= vmath::pi<FloatType>()));
439 return std::abs(angle) * vtx_ids.size() / pi2;
440}

442template <typename FloatType, typename IndexType, IndexType Dim, typename D>
443void polyMesherCollapseEdges(
444 Mesh2D<FloatType, IndexType>& mesh,
445 const DistanceExpression<FloatType, Dim, D>& distanceFunction,
446 FloatType edge_collapse_tol) {
447 std::vector<IndexType> small_edges;
448 const IndexType n_edges = mesh.getFacetToCellMap().size();
449 for (IndexType i = 0; i < n_edges; ++i) {
450 const auto edge = mesh.getFacetToVtxMap().getRow(i);
451 const auto& cells = mesh.getFacetToCellMap()[i];
452 // if the edge does not belong to any triangle
453 const auto& map = mesh.getCellToVtxMap();
454 if ((cells[0] == -1 || map.getRow(cells[0]).size() > 3) &&
455 (cells[1] == -1 || map.getRow(cells[1]).size() > 3)) {
456 //
457 const FloatType angle_normalized_size_0 =
458 polyMesherComputeEdgeAngle(mesh, cells[0], {edge[0], edge[1]});
459 const FloatType angle_normalized_size_1 =
460 polyMesherComputeEdgeAngle(mesh, cells[1], {edge[1], edge[0]});
461 // if the edge is too small for one of the two cells
462 if ((angle_normalized_size_0 < edge_collapse_tol) ||
463 (angle_normalized_size_1 < edge_collapse_tol)) {
464 small_edges.push_back(i);
465 }
466 }
467 }
468
469 // we remove only the edges that do not form a path (are "alone")
470 // we build the vertex to edge map to check for this ^
471 // TODO: remove this constraint, or at least pick the smallest edge of
472 // the path and remove it
473 std::unordered_map<IndexType, uint64_t> vertex_to_edges;
474 for (size_t i = 0; i < small_edges.size(); ++i) {
475 const auto edge_id = small_edges[i];
476 const auto edge_vtx = mesh.getFacetToVtxMap().getRow(edge_id);
477 // assume that the edges are no more than 2^32
478 // we store both the count of edges and the edge id (which is overwritten if
479 // we have more than one edge, but it is fine since we skip them)
480 vertex_to_edges[edge_vtx[0]] += (1ULL << 32) + uint64_t(edge_id);
481 vertex_to_edges[edge_vtx[1]] += (1ULL << 32) + uint64_t(edge_id);
482 }
483
484 std::unordered_map<IndexType, IndexType> vtxs_to_remove_in_edge;
485 for (const auto& [vertex_id, edge_count_and_id] : vertex_to_edges) {
486 // if the vertex belongs to only one edge, then we increase the number of
487 // vertices that can be removed in that edge
488 if ((edge_count_and_id >> 32) == 1) {
489 vtxs_to_remove_in_edge[edge_count_and_id & 0xFFFFFFFFULL]++;
490 }
491 }
492
493 std::vector<std::array<IndexType, 2>> vtxs_to_remove;
494 auto new_cell_to_vtx = mesh.getCellToVtxMap();
495 for (const auto& [edge_id, vtx_count] : vtxs_to_remove_in_edge) {
496 // if we can remove either vertex
497 if (vtx_count == 2) {
498 // check which vertex is closest to the boundary and we keep it
499 // for point inside the domain which we keep does not matter
500 // but near the boundary this allows to better approximate the border
501 const IndexType v1_id = mesh.getFacetToVtxMap()(edge_id, 0);
502 const IndexType v2_id = mesh.getFacetToVtxMap()(edge_id, 1);
503 const auto& pt1 = mesh.getVertexes()[v1_id];
504 const auto& pt2 = mesh.getVertexes()[v2_id];
505 const auto d1 = computeDistance(pt1, distanceFunction).back();
506 const auto d2 = computeDistance(pt2, distanceFunction).back();
507 // the first element in the array is the one to remove, then we have the
508 // other: we need this information later
509 if (std::abs(d1) < std::abs(d2)) {
510 vtxs_to_remove.push_back({v2_id, v1_id});
511 } else {
512 vtxs_to_remove.push_back({v1_id, v2_id});
513 }
514 // remove the edge for the affected cells
515 for (const auto& cell : mesh.getFacetToCellMap()[edge_id]) {
516 if (cell != -1) {
517 for (IndexType& vtx_id : new_cell_to_vtx.getRow(cell)) {
518 if (vtx_id == vtxs_to_remove.back()[0]) {
519 vtx_id = -1;
520 }
521 }
522 }
523 }
524 }
525 }
526
527 // most important bit (sign bit for signed integer, largest 2 power for
528 // unsigned)
529 constexpr IndexType mask = IndexType(1)
530 << IndexType(sizeof(IndexType) * 8 - 1);
531 // remove points
532 std::sort(
533 vtxs_to_remove.begin(), vtxs_to_remove.end(),
534 [](const auto& lhs, const auto& rhs) { return lhs[0] < rhs[0]; });
535 // dummy to avoid `j` problem in the merge
536 vtxs_to_remove.push_back({-1, -1});
537 std::vector<Point<FloatType, 2>> new_vxts;
538 std::vector<IndexType> vtx_old_new_table;
539 IndexType j = 0; // vtx to remove idx
540 const IndexType n_vtxs = mesh.getVertexes().size();
541 for (IndexType i = 0; i < n_vtxs; ++i) {
542 if (vtxs_to_remove[j][0] == i) {
543 // We need to substitute this vertex with its correspoinding in the edge.
544 // The mapping is incomplete, so we signal with a mask that it must be
545 // recomputed
546 vtx_old_new_table.push_back(vtxs_to_remove[j][1] | mask);
547 j++;
548 continue;
549 }
550 vtx_old_new_table.push_back(new_vxts.size());
551 new_vxts.push_back(mesh.getVertexes()[i]);
552 }
553 assert(
554 new_vxts.size() == mesh.getVertexes().size() - vtxs_to_remove.size() + 1);
555
556 // recompute cell to vtx map removing -1 from `new_cell_to_vtx`
557 std::vector<IndexType> crs_indexes = {0};
558 std::vector<IndexType> crs_data;
559 for (IndexType i = 0; i < new_cell_to_vtx.size(); ++i) {
560 const auto row = new_cell_to_vtx.getRow(i);
561 for (const auto elem : row) {
562 if (elem != -1) {
563 const auto new_elem = vtx_old_new_table[elem];
564 // requires recomputations since the vertex was removed
565 if (mask & new_elem) {
566 crs_data.push_back(vtx_old_new_table[new_elem & (~mask)]);
567 } else {
568 crs_data.push_back(new_elem);
569 }
570 }
571 }
572 crs_indexes.push_back(crs_data.size());

584enum RandomEngine : std::uint8_t { Pseudo, AdditiveRecurrence };

586template <
587 typename FloatType,
588 typename IndexType,
589 IndexType Dim,
590 typename D,
591 typename = std::enable_if_t<(Dim == 2) || (Dim == 3)>>
592std::conditional_t<
593 Dim == 2,
594 Mesh2D<FloatType, IndexType>,
595 Mesh3D<FloatType, IndexType>>
596mesh(
597 const DistanceExpression<FloatType, Dim, D>& distanceFunction,
598 size_t n_elems,
599 const std::vector<Point<FloatType, Dim>>& fixed_pts = {},
600 const PolyMesherParams<FloatType>& params = {},
601 size_t seed = 0) {
602 const auto bbox = distanceFunction.bbox();
603
604 std::mt19937 gen(seed);
605 std::uniform_real_distribution<FloatType> dis(0.0, 1.0);
606
607 const RandomEngine engine_type = RandomEngine::AdditiveRecurrence;
608 const auto gen_random_pt = [&](size_t iter) {
609 std::array<FloatType, Dim> pt;
610 switch (engine_type) {
611 case RandomEngine::Pseudo: {
612 for (size_t d = 0; d < Dim; ++d) {
613 pt[d] = dis(gen);
614 }
615 break;
616 }
617 case RandomEngine::AdditiveRecurrence: {
618 pt = random::additiveRecurrence<Dim>(iter, iter + 1)[0];
619 break;
620 }
621 default: {
622 assert(false);
623 }
624 }
625 return pt;
626 };
627
628 std::vector<Point<FloatType, Dim>> seed_pts(n_elems);
629 size_t found = 0;
630 size_t count = 0;
631 while (found < n_elems) {
632 // random point in [0, 1]^Dim
633 const auto raw_random_pt = gen_random_pt(count++);
634 Eigen::Map<const Point<FloatType, Dim>> random_pt(raw_random_pt.data());
635 // rescale in the bounding box
636 const Point<FloatType, Dim> pt =
637 bbox[0].array() + random_pt.array() * (bbox[1] - bbox[0]).array();
638 // accept if internal
639 const auto distance = computeDistance(pt, distanceFunction);
640 if (distance.back() < 0) {
641 seed_pts[found++] = pt;
642 }
643 }
644
645 return mesh(seed_pts, distanceFunction, fixed_pts, params);
646}

648template <
649 typename FloatType,
650 typename IndexType,
651 IndexType Dim,
652 typename D,
653 typename = std::enable_if_t<(Dim == 2) || (Dim == 3)>>
654std::conditional_t<
655 Dim == 2,
656 Mesh2D<FloatType, IndexType>,
657 Mesh3D<FloatType, IndexType>>
658mesh(
659 const std::vector<Point<FloatType, Dim>>& seed_pts,
660 const DistanceExpression<FloatType, Dim, D>& distanceFunction,
661 const std::vector<Point<FloatType, Dim>>& fixed_pts = {},
662 const PolyMesherParams<FloatType>& params = {}) {
663 const auto bbox = distanceFunction.bbox();
664 // increase the bounding box of the half size of the box in each direction
665 // the final size is thus two times the original
666 const std::array<Point<FloatType, Dim>, 2> extended_bbox = {
667 bbox[0].array() - 0.5 * (bbox[1] - bbox[0]).array(),
668 bbox[1].array() + 0.5 * (bbox[1] - bbox[0]).array()};
669 FloatType area = (bbox[1] - bbox[0]).prod();
670
671 FloatType err = std::numeric_limits<FloatType>::infinity();
672 std::vector<Point<FloatType, Dim>> center_pts = seed_pts;
673 for (size_t it = 0; it < params.max_iter; ++it) {
674 const size_t n_elems = center_pts.size();
675 // average relaxed width of an element
676 const FloatType alpha =
677 params.area_relax * std::pow(area / n_elems, 1.0 / Dim);
678 const auto near_boundary_idxs = internal::polyMesherReflect(
679 distanceFunction, alpha, params, center_pts);
680 internal::polyMesherAdjustFixedPoints(
681 fixed_pts, near_boundary_idxs, center_pts);
682
683 // we break the loop before updating the centroid since we are
684 // re-computing the voronoi, and this task needs updated fixed points
685 // and reflected points
686 if (err < params.toll || it == (params.max_iter - 1)) {
687 break;
688 }
689
690 std::tie(err, area) = internal::updateVoronoiCentroid<Dim>(
691 extended_bbox, center_pts, n_elems);
692
693 // Select only the cells inside the domain
694 std::vector<Point<FloatType, Dim>> new_center_pts;
695 for (const auto& pt : center_pts) {
696 if (computeDistance(pt, distanceFunction).back() <= 0.) {
697 new_center_pts.push_back(pt);
698 }
699 }
700 center_pts = new_center_pts;
701
702 std::cout << "Polymesher: " << err << " " << it << std::endl;
703 }
704
705 // Increase mesh quality by collapsing and clipping
706 if constexpr (Dim == 2) {
707 auto tmp_mesh = internal::computeVoronoiMesh2D(
708 extended_bbox, center_pts, distanceFunction);
709 internal::polyMesherCollapseEdges<FloatType, IndexType>(
710 tmp_mesh, distanceFunction, params.collapse_edge_toll);
711 const auto alpha = std::pow(area / center_pts.size(), 1.0 / Dim);
712 // TODO: do this before collapse to remove possible degenerate edges
713 internal::polyMesherClipToBoundary<FloatType, IndexType, Dim>(
714 tmp_mesh, alpha, distanceFunction, params);
715 return tmp_mesh;
716 } else if constexpr (Dim == 3) {
717 auto tmp_mesh = internal::computeVoronoiMesh3D(
718 extended_bbox, center_pts, distanceFunction);
719 return tmp_mesh;
720 } else {
721 static_assert(Dim * 0, "Not Implemented");
722 }
723}
