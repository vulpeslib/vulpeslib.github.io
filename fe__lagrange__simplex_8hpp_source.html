45template <std::size_t Dim>
46struct LagrangianNodesPolicy {
47 using real_t = mpq_class;
48 static std::vector<Point<real_t, Dim>> computeNodes(std::size_t n_pts) {
49 const auto n_pts1 = real_t(n_pts - 1);
50 std::vector<Point<real_t, Dim>> linspace;
51 for (std::size_t j = 0; j < vmath::pow(n_pts, Dim); ++j) {
52 Point<real_t, Dim> pt;
53 std::size_t idx = j;
54 std::size_t idx_sum = 0;
55 for (std::size_t d = 0; d < Dim; ++d) {
56 const auto idx_d = idx % n_pts;
57 idx_sum += idx_d;
58 pt[d] = real_t(idx_d) / n_pts1;
59 idx /= n_pts;
60 }
61 if (idx_sum > (n_pts - 1)) {
62 continue;
63 }
64 linspace.push_back(pt);
65 }
66 return linspace;
67 }
68};

95template <std::size_t Dim>
96struct WarpAndBlendNodesPolicy {
97 using real_t = mpf_class;
98 static std::vector<Point<real_t, Dim>> computeNodes(std::size_t n_pts) {
99 constexpr double k_tol = 2 * std::numeric_limits<double>::epsilon();
100 // we are upcasting the precision because we are taking this approximate
101 // nodes as exact. Indeed, the nodes univocally define the Lagrangian basis,
102 // thus, we take them as our ground truth even if (potentially) a better
103 // basis with smaller Lebesque constant could be found (by optimizing alpha)
104 std::vector<Point<real_t, Dim>> nodes;
105 const Eigen::Index order = static_cast<Eigen::Index>(n_pts - 1);
106 if constexpr (Dim == 1) {
107 const auto qrule_gl = qrule::gmpGaussLobattoQuadrature(n_pts, 256);
108 for (std::size_t i = 0; i < n_pts; ++i) {
109 nodes.emplace_back(qrule_gl.nodes[n_pts - i - 1].point[0]);
110 }
111 } else if constexpr (Dim == 2) {
112 const auto nodes_d = xy2rs(nodes2D(order));
113 for (Eigen::Index i = 0; i < nodes_d.cols(); ++i) {
114 real_t x = nodes_d(0, i);
115 real_t y = nodes_d(1, i);
116 // clip to border
117 if (x < k_tol) {
118 x = 0.0;
119 }
120 if (y < k_tol) {
121 y = 0.0;
122 }
123 if (x + y > 1.0 - k_tol) {
124 x /= (x + y);
125 y /= (x + y);
126 }
127 nodes.push_back(Point<real_t, Dim>{x, y});
128 }
129 } else if constexpr (Dim == 3) {
130 const auto nodes_d = xyz2rst(nodes3D(order));
131 for (Eigen::Index i = 0; i < nodes_d.cols(); ++i) {
132 // clip to border
133 const Point<real_t, Dim> node = nodes_d.col(i).template cast<real_t>();
134 Point<real_t, Dim> clipped_node = node.unaryExpr(
135 [](const real_t& v) { return v < k_tol ? real_t(0.0) : v; });
136 if (clipped_node.sum() > 1.0 - k_tol) {
137 clipped_node = clipped_node.array() / clipped_node.sum();
138 }
139 nodes.push_back(clipped_node);
140 }
141 } else {
142 static_assert(
143 Dim * 0 && "WarpAndBlendNodesPolicy not implemented for this Dim.");
144 }
145 return nodes;
146 }
147};

160template <std::size_t Dim, typename DofPolicy, typename NodePolicy>
161class FeSpaceLagrangeP : public FeSpaceBase<
162 Dim,
163 internal::real_t,
164 DofPolicy,
165 FeSpaceLagrangeP<Dim, DofPolicy, NodePolicy>> {
166 public:
167 using Polynomial = polynomial::NdPolynomial<internal::real_t, Dim>;
168 explicit FeSpaceLagrangeP(std::size_t order)
169 : FeSpaceBase<
170 Dim,
171 internal::real_t,
172 DofPolicy,
173 FeSpaceLagrangeP<Dim, DofPolicy, NodePolicy>>(order) {
174 const auto nodes = NodePolicy::computeNodes(order + 1);
175 const auto coef = internal::computeLagrangianPolyCoef<Dim>(nodes);
176 m_polys.reserve(coef.cols());
177 for (int64_t basis_id = 0; basis_id < coef.cols(); ++basis_id) {
178 // create a new polynomial with rows coefficients
179 m_polys.emplace_back(coef.rows());
180 for (int64_t mono_id = 0; mono_id < coef.rows(); ++mono_id) {
181 // get the exponents of the current monomial
182 const std::array<std::size_t, Dim> exps =
183 monomial::getMonomial<Dim, std::size_t>(mono_id);
184 // set the coefficient of the polynomial (dimension indipendent)
185 std::apply(
186 [&](auto... args) {
187 m_polys.back().coef(args...) =
188 vmath::to<internal::real_t>(coef(mono_id, basis_id));
189 },
190 exps);
191 }
192 }
193
194 // save nodes
195 for (const auto& pt : nodes) {
196 m_nodes.push_back(pt.template cast<internal::real_t>());
197 }
198 }
199
200 const auto& getNodes() const { return m_nodes; }
201 const auto& getBasis() const { return m_polys; }
202 inline static constexpr FeType elemType() { return FeType::Simplex; };
203
204 template <typename CellType>
205 inline static constexpr auto getMapping(const CellType& cell) {
206 // TODO: compute simplex mapping
207 return 0;
208 };
209
210 protected:
211 std::vector<Polynomial> m_polys;
212 std::vector<Point<internal::real_t, Dim>> m_nodes;
213};
