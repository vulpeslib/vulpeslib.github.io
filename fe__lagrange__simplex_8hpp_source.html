47template <std::size_t Dim>
48struct LagrangianNodesPolicy {
49 using real_t = mpq_class;
50 static std::vector<Point<real_t, Dim>> computeNodes(std::size_t n_pts) {
51 const auto n_pts1 = real_t(n_pts - 1);
52 std::vector<Point<real_t, Dim>> linspace;
53 for (std::size_t j = 0; j < vmath::pow(n_pts, Dim); ++j) {
54 Point<real_t, Dim> pt;
55 std::size_t idx = j;
56 std::size_t idx_sum = 0;
57 for (std::size_t d = 0; d < Dim; ++d) {
58 const auto idx_d = idx % n_pts;
59 idx_sum += idx_d;
60 pt[d] = real_t(idx_d) / n_pts1;
61 idx /= n_pts;
62 }
63 if (idx_sum > (n_pts - 1)) {
64 continue;
65 }
66 linspace.push_back(pt);
67 }
68 return linspace;
69 }
70};

89template <std::size_t Dim>
90struct WarpAndBlendNodesPolicy {
91 using real_t = mpf_class;
92 static std::vector<Point<real_t, Dim>> computeNodes(std::size_t n_pts) {
93 constexpr double k_tol = 2 * std::numeric_limits<double>::epsilon();
94 // we are upcasting the precision because we are taking this approximate
95 // nodes as exact. Indeed, the nodes univocally define the Lagrangian basis,
96 // thus, we take them as our ground truth even if (potentially) a better
97 // basis with smaller Lebesque constant could be found (by optimizing alpha)
98 std::vector<Point<real_t, Dim>> nodes;
99 const Eigen::Index order = static_cast<Eigen::Index>(n_pts - 1);
100 if constexpr (Dim == 1) {
101 const auto qrule_gl = qrule::gmpGaussLobattoQuadrature(n_pts, 256);
102 for (std::size_t i = 0; i < n_pts; ++i) {
103 nodes.emplace_back(qrule_gl.nodes[n_pts - i - 1].point[0]);
104 }
105 } else if constexpr (Dim == 2) {
106 const auto nodes_d = xy2rs(nodes2D(order));
107 for (Eigen::Index i = 0; i < nodes_d.cols(); ++i) {
108 real_t x = nodes_d(0, i);
109 real_t y = nodes_d(1, i);
110 // clip to border
111 if (x < k_tol) {
112 x = 0.0;
113 }
114 if (y < k_tol) {
115 y = 0.0;
116 }
117 if (x + y > 1.0 - k_tol) {
118 x /= (x + y);
119 y /= (x + y);
120 }
121 nodes.push_back(Point<real_t, Dim>{x, y});
122 }
123 } else if constexpr (Dim == 3) {
124 const auto nodes_d = xyz2rst(nodes3D(order));
125 for (Eigen::Index i = 0; i < nodes_d.cols(); ++i) {
126 // clip to border
127 const Point<real_t, Dim> node = nodes_d.col(i).template cast<real_t>();
128 Point<real_t, Dim> clipped_node = node.unaryExpr(
129 [](const real_t& v) { return v < k_tol ? real_t(0.0) : v; });
130 if (clipped_node.sum() > 1.0 - k_tol) {
131 clipped_node = clipped_node.array() / clipped_node.sum();
132 }
133 nodes.push_back(clipped_node);
134 }
135 } else {
136 static_assert(
137 Dim * 0 && "WarpAndBlendNodesPolicy not implemented for this Dim.");
138 }
139 return nodes;
140 }
141};

144template <std::size_t Dim, typename DofPolicy, typename NodePolicy>
145class FeSpaceLagrangeP : public FeSpaceBase<
146 Dim,
147 internal::real_t,
148 DofPolicy,
149 FeSpaceLagrangeP<Dim, DofPolicy, NodePolicy>> {
150 public:
151 using Polynomial = polynomial::NdPolynomial<internal::real_t, Dim>;
152 explicit FeSpaceLagrangeP(std::size_t order)
153 : FeSpaceBase<
154 Dim,
155 internal::real_t,
156 DofPolicy,
157 FeSpaceLagrangeP<Dim, DofPolicy, NodePolicy>>(order) {
158 const auto nodes = NodePolicy::computeNodes(order + 1);
159 const auto coef = internal::computeLagrangianPolyCoef<Dim>(nodes);
160 m_polys.reserve(coef.cols());
161 for (int64_t basis_id = 0; basis_id < coef.cols(); ++basis_id) {
162 // create a new polynomial with rows coefficients
163 m_polys.emplace_back(coef.rows());
164 for (int64_t mono_id = 0; mono_id < coef.rows(); ++mono_id) {
165 // get the exponents of the current monomial
166 const std::array<std::size_t, Dim> exps =
167 monomial::getMonomial<Dim, std::size_t>(mono_id);
168 // set the coefficient of the polynomial (dimension indipendent)
169 std::apply(
170 [&](auto... args) {
171 m_polys.back().coef(args...) =
172 vmath::to<internal::real_t>(coef(mono_id, basis_id));
173 },
174 exps);
175 }
176 }
177
178 // save nodes
179 for (const auto& pt : nodes) {
180 m_nodes.push_back(pt.template cast<internal::real_t>());
181 }
182 }
183
184 const auto& getNodes() const { return m_nodes; }
185 const auto& getBasis() const { return m_polys; }
186 inline static constexpr FeType elemType() { //
187 return FeType::Simplex;
188 };
189
190 template <typename CellType>
191 inline static constexpr auto getMapping(const CellType& cell) {
192 return 0;
193 };
194
195 protected:
196 std::vector<Polynomial> m_polys;
197 std::vector<Point<internal::real_t, Dim>> m_nodes;
198};
