100template <typename V, typename U, std::size_t Dim>
101QuadratureRule<V, Dim> castRule(const QuadratureRule<U, Dim>& rule_src) {
102 QuadratureRule<V, Dim> rule_dst;
103 rule_dst.degree = rule_src.degree;
104 rule_dst.nodes.reserve(rule_src.nodes.size());
105 for (const auto& node : rule_src.nodes) {
106 rule_dst.nodes.emplace_back();
107 rule_dst.nodes.back().weight = vmath::to<V>(node.weight);
108 for (std::size_t d = 0; d < Dim; ++d) {
109 rule_dst.nodes.back().point[d] = vmath::to<V>(node.point[d]);
110 }
111 }
112 return rule_dst;
113}
