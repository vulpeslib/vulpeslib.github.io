145 void derivative_() {
146 if (m_coefficients.size() == 1) {
147 m_coefficients[0] = T(0);
148 return;
149 }
150 for (std::size_t i = 1; i < m_coefficients.size(); ++i) {

192 void primitive_(const T& ordinate = T(0), const T& target_value = T(0)) {
193 static_assert(
194 !std::is_integral_v<T>,
195 "Primitive is not well defined for integral polynomial types");
196 m_coefficients.push_back(m_coefficients.back() / m_coefficients.size());
197 for (std::size_t i = m_coefficients.size() - 2; i > 0; --i) {
198 m_coefficients[i] = m_coefficients[i - 1] / T(i);
199 }
200 const auto ordinate_value = ordinate == T(0) ? T(0) : evaluate(ordinate);
201 m_coefficients[0] = target_value - ordinate_value;
202 }

222 T evaluate(const T& x) const {
223 if (m_coefficients.size() == 1) {
224 return m_coefficients[0];
225 }
226 T result = m_coefficients.back();
227 for (std::size_t i = m_coefficients.size() - 2; i > 0; --i) {
228 result = result * x + m_coefficients[i];
229 }
230 result = result * x + m_coefficients[0];
231 return result;
232 }

325 Polynomial<T>& operator*=(const Polynomial<T>& other) {
326 std::vector<T> result(
327 m_coefficients.size() + other.m_coefficients.size() - 1, T(0));
328
329 for (std::size_t i = 0; i < m_coefficients.size(); ++i) {
330 for (std::size_t j = 0; j < other.m_coefficients.size(); ++j) {
331 result[i + j] += m_coefficients[i] * other.m_coefficients[j];
332 }
333 }
334
335 m_coefficients = result;
336 trim();
337 return *this;
338 }

383 static void divide(
384 const Polynomial<T>& divisor,
385 Polynomial<T>& quotient,
386 Polynomial<T>& remainder) {
387 static_assert(
388 !std::is_integral_v<T>,
389 "Division is not well defined for integral polynomial types");
390 assert(divisor.m_coefficients.size() > 0);
391
392 while (remainder.degree() >= divisor.degree() &&
393 remainder.m_coefficients.size() > 0) {
394 const std::size_t degree_diff = remainder.degree() - divisor.degree();
395 T leading_coeff =
396 remainder.m_coefficients.back() / divisor.m_coefficients.back();
397
398 std::vector<T> term(degree_diff + 1, T(0));
399 term.back() = leading_coeff;
400
401 Polynomial<T> term_poly(term);
402 quotient += term_poly;
403 remainder -= (term_poly * divisor);
404 remainder.trim();
405 }
406 }

547template <typename T>
548std::vector<Polynomial<T>> buildLegendrePolynomials(std::size_t deg) {
549 std::vector<Polynomial<T>> legendre_polynomials;
550 const Polynomial<T> one(std::vector<T>{1});
551 legendre_polynomials.push_back(one);
552 if (deg == 0) {
553 return legendre_polynomials;
554 }
555 legendre_polynomials.push_back(Polynomial<T>({0, 1}));
556
557 const Polynomial<T> X({0, 1});
558 for (std::size_t n = 1; n < deg + 1; n++) {
559 const auto& Pn_1 = legendre_polynomials[n - 1];
560 const auto& Pn = legendre_polynomials[n];
561 T coef1 = 2 * n + 1;
562 T coef2 = n;
563 legendre_polynomials.push_back(coef1 * X * Pn - coef2 * Pn_1);
564 legendre_polynomials.back() /= T(n + 1);
565 }
566 return legendre_polynomials;
567}
