52template <typename T, bool IsSpaceP = true>
53class Polynomial2D : public PolynomialBase<T, 2, Polynomial2D<T, IsSpaceP>> {

56 explicit Polynomial2D(std::size_t size) : m_coefficients(size) {}

59 constexpr std::size_t order() const { return m_coefficients.dim() - 1; }

75 Polynomial2D<T, IsSpaceP> derivative(std::size_t component) const {
76 assert(component < 2);
77 Polynomial2D<T, IsSpaceP> result(dim());
78 if (component == 0) {
79 for (std::size_t i = 1; i < dim(); ++i) {
80 const auto jmax = dim() - IsSpaceP * i;
81 for (std::size_t j = 0; j < jmax; ++j) {
82 result.coef(i - 1, j) = coef(i, j) * T(i);
83 }
84 }
85 } else {
86 for (std::size_t i = 0; i < dim(); ++i) {
87 const auto jmax = dim() - IsSpaceP * i;
88 for (std::size_t j = 1; j < jmax; ++j) {
89 result.coef(i, j - 1) = coef(i, j) * T(j);
90 }
91 }
92 }
93 return result;
94 }

116 T evaluate(const T* x) const {
117 // WARNING: this is not thread safe
118 static std::vector<T> partial_eval;
119 if (partial_eval.size() < dim()) {
120 partial_eval.resize(dim());
121 }
122
123 // Step 1: Compute Q_i(y0) for each i using Horner's method in y
124 for (std::size_t i = 0; i < dim(); ++i) {
125 T Q_i = T(0);
126 const auto jmax = dim() - IsSpaceP * i;
127 for (std::size_t j = jmax - 1; j > 0; --j) {
128 Q_i = Q_i * x[1] + m_coefficients(i, j);
129 }
130 partial_eval[i] = Q_i * x[1] + m_coefficients(i, 0);
131 }
132
133 // Step 2: Compute P(x0, y0) using Horner's method in x
134 T eval = T(0);
135 for (std::size_t i = dim() - 1; i > 0; --i) {
136 eval = eval * x[0] + partial_eval[i];
137 }
138
139 return eval * x[0] + partial_eval[0];
140 }

142 T integrate(
143 const Eigen::Map<Eigen::Vector<T, -1>>& monomial_integrals) const {
144 assert(m_coefficients.size() <= monomial_integrals.size());
145 T integral = 0.0;
146 for (std::size_t i = 0; i < m_coefficients.size(); ++i) {
147 integral += m_coefficients.data()[i] * monomial_integrals[i];
148 }
149 return integral;
150 }

196 Polynomial2D<T, IsSpaceP>& operator*=(
197 const Polynomial2D<T, IsSpaceP>& other) {
198 std::size_t new_dim = dim() + other.dim();
199 decltype(m_coefficients) new_coeffs(new_dim);
200
201 for (std::size_t i = 0; i < dim(); ++i) {
202 const auto jmax = dim() - IsSpaceP * i;
203 for (std::size_t j = 0; j < jmax; ++j) {
204 for (std::size_t k = 0; k < other.dim(); ++k) {
205 const auto lmax = other.dim() - IsSpaceP * k;
206 for (std::size_t l = 0; l < lmax; ++l) {
207 new_coeffs(i + k, j + l) += coef(i, j) * other.coef(k, l);
208 }
209 }
210 }
211 }
212
213 m_coefficients = std::move(new_coeffs);
214 return *this;
215 }

237 friend std::ostream& operator<<(std::ostream& os, const Polynomial2D& poly) {
238 for (std::size_t i = 0; i < poly.dim(); ++i) {
239 const auto jmax = poly.dim() - IsSpaceP * i;
240 for (std::size_t j = 0; j < jmax; ++j) {
241 if (poly.m_coefficients(i, j) != T(0)) {
242 if (i > 0 || j > 0) {
243 os << " + ";
244 }
245 os << poly.m_coefficients(i, j);
246 if (i > 0) {
247 os << "x^" << i;
248 }
249 if (j > 0) {
250 os << "y^" << j;
251 }
252 }
253 }
254 }
255 return os;

264 template <bool Add>
265 Polynomial2D<T, IsSpaceP>& add_(const Polynomial2D<T, IsSpaceP>& other) {
266 if (dim() < other.dim()) {
267 auto result_coef = other.m_coefficients;
268 if constexpr (!Add) {
269 // Negate coefficients for subtraction
270 std::transform(
271 result_coef.begin(), result_coef.end(), result_coef.begin(),
272 [](const auto& val) { return -val; });
273 }
274 for (std::size_t i = 0; i < dim(); ++i) {
275 const auto jmax = dim() - IsSpaceP * i;
276 for (std::size_t j = 0; j < jmax; ++j) {
277 result_coef(i, j) += coef(i, j);
278 }
279 }
280 m_coefficients = std::move(result_coef);
281 } else {
282 for (std::size_t i = 0; i < other.dim(); ++i) {
283 const auto jmax = other.dim() - IsSpaceP * i;
284 for (std::size_t j = 0; j < jmax; ++j) {
285 if constexpr (Add) {
286 m_coefficients(i, j) += other.coef(i, j);
287 } else {
288 m_coefficients(i, j) -= other.coef(i, j);
289 }
290 }
291 }
292 }
293 return *this;
294 }
