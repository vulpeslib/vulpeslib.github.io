32template <std::size_t Dim, typename E>
33class FeExpressionUnaryMinus
34 : public FeExpressionBase<Dim, FeExpressionUnaryMinus<Dim, E>> {
35 public:
36 static constexpr bool is_leaf = false;
37 static constexpr uint8_t is_bilinear = E::is_bilinear;
38
39 explicit constexpr FeExpressionUnaryMinus(const E& expr) : m_expr(expr) {};
40
41 template <std::size_t J = 0, std::size_t I = 0>
42 decltype(auto) evaluate(
43 const FeValues<Dim>& fe_values,
44 std::size_t j,
45 std::size_t i,
46 std::size_t q) const {
47 using TypeEval =
48 decltype(m_expr.template evaluate<J, I>(fe_values, j, i, q));
49 if constexpr (std::is_same_v<TypeEval, void>) {
50 return void();
51 } else {
52 return -m_expr.template evaluate<J, I>(fe_values, j, i, q);
53 }
54 }
55
56 constexpr UpdateFlagsType computeUpdateFlags() const {
57 return m_expr.computeUpdateFlags();
58 }
59 constexpr std::size_t computeOrder() const { return m_expr.computeOrder(); }
60
61 decltype(auto) evaluatePoly(
62 const FeValues<Dim>& fe_values,
63 std::size_t j,
64 std::size_t i,
65 std::size_t bi,
66 std::size_t bj) const {
67 return m_expr.evaluatePoly(fe_values, j, i, bi, bj);
68 }
69 decltype(auto) evaluatePolyCoef(
70 const auto& mapping,
71 const auto& fe_values,
72 std::size_t bi,
73 std::size_t bj) const {
74 using MatType = std::decay_t<decltype(m_expr.evaluatePolyCoef(
75 mapping, fe_values, bi, bj))>;
76 return MatType{-m_expr.evaluatePolyCoef(mapping, fe_values, bi, bj)};
77 }
78 constexpr bool precompute(
79 const FeValues<Dim>& fe_values, const CellsType<Dim>& cells) const {
80 return m_expr.precompute(fe_values, cells);
81 }
82
83 protected:
84 typename std::conditional_t<E::is_leaf, const E&, const E> m_expr;
85};

125template <std::size_t Dim, typename E>
126class FeExpressionFieldComponent
127 : public FeExpressionBase<Dim, FeExpressionFieldComponent<Dim, E>> {
128 private:
129 template <typename T, typename = void>
130 struct FsTypeOf {
131 using type = void;
132 };
133 template <typename T>
134 struct FsTypeOf<T, std::void_t<typename T::FeSpaceType>> {
135 using type = typename T::FeSpaceType;
136 };
137
138 public:
139 using OutputType = double;
140 using FeSpaceType = typename FsTypeOf<E>::type;
141 static constexpr bool is_leaf = false;
142 static constexpr uint8_t is_bilinear = E::is_bilinear;
143 // Forward the block range of E unchanged: selecting a field block does not
144 // change which test/trial fields the expression spans, so nblocks (the
145 // number of rhs integrator iterations) stays equal to E's.
146 static constexpr std::size_t k_index_begin = E::k_index_begin;
147 static constexpr std::size_t k_index_end = E::k_index_end;
148
149 explicit constexpr FeExpressionFieldComponent(const E& expr)
150 : m_expr(expr) {};
151
152 // Forward feSpace() when E is a test/trial expression (has feSpace()).
153 // SFINAE excludes this for IsNothing expressions (no feSpace() on E).
154 template <
155 typename U = E,
156 typename = std::void_t<decltype(std::declval<const U&>().feSpace())>>
157 constexpr decltype(auto) feSpace() const {
158 return m_expr.feSpace();
159 }
160
161 template <std::size_t J = 0, std::size_t I = 0>
162 decltype(auto) evaluate(
163 const FeValues<Dim>& fe_values,
164 std::size_t j,
165 std::size_t i,
166 std::size_t q) const {
167 // Two-index Eigen access (row, col): the test/trial value is an
168 // EigenVulpes::OneComponentVector with a compile-time active index and no
169 // LinearAccessBit, so single-index operator[] is ill-formed. (row, 0) also
170 // works for plain Eigen column vectors, so this stays general over E.
171 if constexpr (is_bilinear == IsBilinear) {
172 return m_expr.template evaluate<J, I>(fe_values, j, i, q)(J, I);
173 } else if constexpr (is_bilinear == IsLinearTest) {
174 return m_expr.template evaluate<J, I>(fe_values, j, i, q)(I, 0);
175 } else if constexpr (is_bilinear == IsLinearTrial) {
176 return m_expr.template evaluate<J, I>(fe_values, j, i, q)(J, 0);
177 } else {
178 static_assert(
179 Dim * 0 &&
180 "fe::field_component requires a test/trial expression (IsLinearTest, "
181 "IsLinearTrial, or IsBilinear). For spatial component extraction "
182 "from "
183 "a coefficient use fe::spatial_component<K>.");
184 }
185 }
186
187 constexpr UpdateFlagsType computeUpdateFlags() const {
188 return m_expr.computeUpdateFlags();
189 }
190 constexpr std::size_t computeOrder() const { return m_expr.computeOrder(); }
191
192 constexpr bool precompute(
193 const FeValues<Dim>& fe_values, const CellsType<Dim>& cells) const {
194 return m_expr.precompute(fe_values, cells);
195 }
196
197 protected:
198 typename std::conditional_t<E::is_leaf, const E&, const E> m_expr;
199};

232template <std::size_t K, std::size_t Dim, typename E>
233class FeExpressionSpatialComponent
234 : public FeExpressionBase<Dim, FeExpressionSpatialComponent<K, Dim, E>> {
235 private:

238 using type = void;
239 };

242 using type = typename T::FeSpaceType;
243 };
244
245 public:
246 using OutputType = double;
247 using FeSpaceType = typename FsTypeOf<E>::type;

250 // Forward the block range of E unchanged: extracting a spatial direction
251 // does not move the expression to a different test/trial field block.

263 constexpr decltype(auto) feSpace() const {
264 return m_expr.feSpace();
265 }
266
267 template <std::size_t J = 0, std::size_t I = 0>
268 decltype(auto) evaluate(
269 const FeValues<Dim>& fe_values,
270 std::size_t j,
271 std::size_t i,
272 std::size_t q) const {
273 using EvalType =

291 protected:
292 std::conditional_t<E::is_leaf, const E&, const E> m_expr;
293};
