31template <std::size_t Dim, typename E>
32class FeExpressionUnaryMinus
33 : public FeExpressionBase<Dim, FeExpressionUnaryMinus<Dim, E>> {
34 public:
35 static constexpr bool is_leaf = false;
36 static constexpr uint8_t is_bilinear = E::is_bilinear;
37
38 explicit constexpr FeExpressionUnaryMinus(const E& expr) : m_expr(expr) {};
39
40 template <std::size_t I = 0, std::size_t J = 0>
41 decltype(auto) evaluate(
42 const FeValues<Dim>& fe_values,
43 std::size_t i,
44 std::size_t j,
45 std::size_t q) const {
46 return -m_expr.template evaluate<I, J>(fe_values, i, j, q);
47 }
48
49 constexpr UpdateFlagsType computeUpdateFlags() const {
50 return m_expr.computeUpdateFlags();
51 }
52 constexpr std::size_t computeOrder() const { return m_expr.computeOrder(); }
53
54 decltype(auto) evaluatePoly(
55 const FeValues<Dim>& fe_values,
56 std::size_t i,
57 std::size_t j,
58 std::size_t bi,
59 std::size_t bj) const {
60 return m_expr.evaluatePoly(fe_values, i, j, bi, bj);
61 }
62 decltype(auto) evaluatePolyCoef(
63 const auto& mapping,
64 const auto& fe_values,
65 std::size_t bi,
66 std::size_t bj) const {
67 using MatType = std::decay_t<decltype(m_expr.evaluatePolyCoef(
68 mapping, fe_values, bi, bj))>;

70 }
71 constexpr bool precompute(
72 const FeValues<Dim>& fe_values, const CellsType<Dim>& cells) const {
73 return m_expr.precompute(fe_values, cells);
74 }

76 protected:
77 typename std::conditional_t<E::is_leaf, const E&, const E> m_expr;
78};
