489template <typename Mesh>
490std::vector<SCOTCH_Num> ParallelMesh<Mesh>::partitionWithParmetis() {
491 using real_t = float;
492 const auto rank = mpi::rank();
493 const auto size = mpi::size();
494 // how many vertices on each proc
495 std::vector<SCOTCH_Num> vtxdist(size + 1);
496 // crs map in Metis format
497 std::vector<SCOTCH_Num> xadj;
498 std::vector<SCOTCH_Num> adjncy;
499
500 if (rank == 0) {
501 const auto& dual = m_rank0_mesh->getDualMap();
502
503 // how many vertices on each proc
504 vtxdist[0] = 0;
505 for (int i = 0; i < size; ++i) {
506 vtxdist[i + 1] =
507 vtxdist[i] + dual.size() / size + (i < (dual.size() % size));
508 }
509 assert(vtxdist.back() == dual.size());
510
511 // build local crs map for others
512 for (int p = 1; p < size; ++p) {
513 xadj.resize(vtxdist[p + 1] - vtxdist[p] + 1);
514 const std::size_t adjncy_size =
515 dual.indices()[vtxdist[p + 1]] - dual.indices()[vtxdist[p]];
516 const auto offset = dual.indices()[vtxdist[p]];
517 for (std::size_t j = 0; j < xadj.size(); ++j) {
518 xadj[j] = dual.indices()[j + vtxdist[p]] - offset;
519 }
520 mpi::send(xadj, p);
521 mpi::send(adjncy_size, p);
522 mpi::send(&dual.data()[offset], adjncy_size, p);
523 }
524
525 // build local crs map for self
526 xadj.assign(
527 dual.indices().begin(), dual.indices().begin() + vtxdist[1] + 1);
528 adjncy.assign(
529 dual.data().begin(), dual.data().begin() + dual.indices()[vtxdist[1]]);
530 } else {
531 mpi::recv(xadj, 0);
532 adjncy.resize(mpi::recv<std::size_t>(0));
533 mpi::recv(adjncy.data(), static_cast<int32_t>(adjncy.size()), 0);
534 }
535 mpi::bcast(vtxdist, 0);
536
537 // ParMETIS parameters
538 SCOTCH_Num ncon = 1;
539 SCOTCH_Num wgtflag = 0;
540 SCOTCH_Num numflag = 0;
541 SCOTCH_Num nparts = size;
542 real_t ubvec = 1.05;
543 const SCOTCH_Num random_seed = 0;
544 SCOTCH_Num options[3] = {1, 0, random_seed};
545 std::vector<real_t> tpwgts(nparts);
546 for (int i = 0; i < size; ++i) {
547 tpwgts[i] = 1. / static_cast<double>(nparts);
548 }
549
550 // result buffer
551 std::vector<SCOTCH_Num> part(vtxdist[rank + 1] - vtxdist[rank]);
552
553 std::cout << "Partition the graph;" << std::endl;
554 // Partition the graph
555 auto comm = MPI_COMM_WORLD;
556 ParMETIS_V3_PartKway(
557 vtxdist.data(), xadj.data(), adjncy.data(), nullptr, nullptr, &wgtflag,
558 &numflag, &ncon, &nparts, tpwgts.data(), &ubvec, options, &m_edgecut,
559 part.data(), &comm);
560 return part;
561}

563template <typename Mesh>
564void ParallelMesh<Mesh>::reconstructPartitioningAndScatterDual(
565 std::vector<index_t>& part) {
566 const auto size = mpi::size();
567 const auto& dual = m_rank0_mesh->getDualMap();
568 std::vector<index_t> all_parts;
569 if (mpi::root()) {
570 std::cout << "Partitioning completed. Edge cuts: " << m_edgecut
571 << std::endl;
572 // std::cout << "rank 0 | part: " << part << std::endl;
573 all_parts = part;
574 for (int p = 1; p < size; ++p) {
575 mpi::recv(part, p);
576 // std::cout << "rank " << p << " | part: " << part << std::endl;
577 all_parts.insert(all_parts.end(), part.begin(), part.end());
578 }
579
580 // we re-order the partition to minimize the bandwidth
581 // of the processors block
582 // In practice if two partitions are connected they should
583 // belong also to procs that are close together
584 Eigen::MatrixXi dense_proc_adj = Eigen::MatrixXi::Zero(size, size);
585 for (index_t cell_id = 0; cell_id < dual.size(); ++cell_id) {
586 const auto src_cell_proc = all_parts[cell_id];
587 for (index_t neigh_cell_id : dual[cell_id]) {
588 const auto dst_cell_proc = all_parts[neigh_cell_id];
589 dense_proc_adj(src_cell_proc, dst_cell_proc) = 1;
590 }
591 }
592 datastructure::CrsMapUncompressed<index_t> uncomp_proc_adj;
593 uncomp_proc_adj.resize(size);
594 for (Eigen::Index i = 0; i < dense_proc_adj.rows(); ++i) {
595 for (Eigen::Index j = 0; j < dense_proc_adj.cols(); ++j) {
596 if (dense_proc_adj(i, j)) {
597 uncomp_proc_adj[i].push_back(j);
598 }
599 }
600 }
601 const auto proc_adj = uncomp_proc_adj.template compress<index_t>();
602 const auto [perm, perm_inv] =
603 dofs::computeReverseCuthillMckeeOrdering(proc_adj, false);
604 for (std::size_t i = 0; i < all_parts.size(); ++i) {
605 all_parts[i] = perm[all_parts[i]];
606 }
607
608 for (std::size_t cell_id = 0; cell_id < all_parts.size(); ++cell_id) {
609 m_all_own_and_ghost_cell_id[all_parts[cell_id]].push_back(cell_id);
610 }
611
612 for (int p = 0; p < size; ++p) {
613 m_all_own_size[p] = m_all_own_and_ghost_cell_id[p].size();
614 }
615
616 // send dual map of owned cells
617 for (int p = 1; p < size; ++p) {
618 mpi::send(m_all_own_and_ghost_cell_id[p], p);
619 mpi::send(dual.slice(m_all_own_and_ghost_cell_id[p]), p);
620 }
621 m_local_numbering[self_dim] = m_all_own_and_ghost_cell_id[0];
622 this->getDualMap() = dual.slice(m_all_own_and_ghost_cell_id[0]);
623 } else {
624 mpi::send(part, 0);
625 mpi::recv(m_local_numbering[self_dim], 0);
626 mpi::recv(this->getDualMap(), 0);
627 }
628}

630template <typename Mesh>
631void ParallelMesh<Mesh>::buildGhostCellsFromWeakDual() {
632 const auto size = mpi::size();
633 SetType ghost_cell_set(size);
634 for (index_t loc_cid = 0; loc_cid < m_weak_dual.size(); ++loc_cid) {
635 for (auto glob_neigh_id : m_weak_dual[loc_cid]) {
636 auto it = std::lower_bound(
637 m_local_numbering[self_dim].begin(),
638 m_local_numbering[self_dim].end(), glob_neigh_id);
639 if ((it == m_local_numbering[self_dim].end()) || (*it != glob_neigh_id)) {
640 ghost_cell_set.insert(glob_neigh_id);
641 }
642 }
643 }
644 m_own_enty_size[self_dim] = m_local_numbering[self_dim].size();
645 m_ghost_enty_size[self_dim] = ghost_cell_set.size();
646 m_local_numbering[self_dim].insert(
647 m_local_numbering[self_dim].end(), ghost_cell_set.begin(),
648 ghost_cell_set.end());
649}

651template <typename Mesh>
652void ParallelMesh<Mesh>::collectGhostCellsInRoot() {
653 const auto size = mpi::size();
654 if (mpi::root()) {
655 // recive ghost from procs
656 m_all_own_and_ghost_cell_id[0] = m_local_numbering[self_dim];
657 m_all_ghost_size[0] =
658 m_all_own_and_ghost_cell_id[0].size() - m_all_own_size[0];
659 for (int p = 1; p < size; ++p) {
660 m_all_ghost_size[p] = mpi::recv<index_t>(p);
661 m_all_own_and_ghost_cell_id[p].resize(
662 m_all_own_and_ghost_cell_id[p].size() + m_all_ghost_size[p]);
663 mpi::recv(
664 &m_all_own_and_ghost_cell_id[p][m_all_own_size[p]],
665 m_all_ghost_size[p], p);
666 }
667 } else {
668 mpi::send(m_ghost_enty_size[self_dim], 0);
669 mpi::send(
670 &m_local_numbering[self_dim][m_own_enty_size[self_dim]],
671 m_ghost_enty_size[self_dim], 0, MPI_COMM_WORLD);
672 }
673}

675template <typename Mesh>
676void ParallelMesh<Mesh>::buildGhostCells(std::vector<index_t>& part) {
677 const auto size = mpi::size();
678 m_all_own_and_ghost_cell_id.resize(size);
679 m_all_own_size.resize(size);
680 m_all_ghost_size.resize(size);
681 // send partitions to root and compute ghost
682 // reconstruct partitioning and send dual map owning slice
683 PROFILE_INTERVAL(reconstructPartitioningAndScatterDual(part));
684
685 // In the ghost cells we want also the weakly connected cells
686 // that is the cells that share also only a vertex
687 // Indeed, we need this information to compute the sparsity pattern
688 PROFILE_INTERVAL(buildWeakDual());
689 PROFILE_INTERVAL(buildGhostCellsFromWeakDual());
690 PROFILE_INTERVAL(collectGhostCellsInRoot());
691}

693template <typename Mesh>
694void ParallelMesh<Mesh>::buildLocalCellToEntityMaps() {
695 const auto size = mpi::size();
696 // build and send local maps
697 if (mpi::root()) {
698 for (int p = 1; p < size; ++p) {
699 mpi::send(
700 m_rank0_mesh->getDualMap().slice(m_all_own_and_ghost_cell_id[p]), p);
701 const auto cell_to_vtx_loc =
702 m_rank0_mesh->getCellToVtxMap().slice(m_all_own_and_ghost_cell_id[p]);
703 mpi::send(cell_to_vtx_loc, p);
704 const auto cell_to_facet_loc = m_rank0_mesh->getCellToFacetMap().slice(
705 m_all_own_and_ghost_cell_id[p]);
706 mpi::send(cell_to_facet_loc, p);
707 if constexpr (self_dim == 3) {
708 const auto cell_to_edge_loc = m_rank0_mesh->getCellToEdgeMap().slice(
709 m_all_own_and_ghost_cell_id[p]);
710 mpi::send(cell_to_edge_loc, p);
711 }
712 }
713 // build for self
714 this->getDualMap() =
715 m_rank0_mesh->getDualMap().slice(m_all_own_and_ghost_cell_id[0]);
716 this->getCellToVtxMap() =
717 m_rank0_mesh->getCellToVtxMap().slice(m_all_own_and_ghost_cell_id[0]);
718 this->getCellToFacetMap() =
719 m_rank0_mesh->getCellToFacetMap().slice(m_all_own_and_ghost_cell_id[0]);
720 if constexpr (self_dim == 3) {
721 this->getCellToEdgeMap() = m_rank0_mesh->getCellToEdgeMap().slice(
722 m_all_own_and_ghost_cell_id[0]);
723 }
724 } else {
725 mpi::recv(this->getDualMap(), 0);
726 mpi::recv(this->getCellToVtxMap(), 0);
727 mpi::recv(this->getCellToFacetMap(), 0);
728 if constexpr (self_dim == 3) {
729 mpi::recv(this->getCellToEdgeMap(), 0);
730 }
731 }

734template <typename Mesh>
735void ParallelMesh<Mesh>::buildWeakDual() {
736 const auto size = mpi::size();
737 // scatter the cell to vtx map
738 if (mpi::root()) {
739 for (int p = 0; p < size; ++p) {
740 const auto cell_to_vtx_loc =
741 m_rank0_mesh->getCellToVtxMap().slice(m_all_own_and_ghost_cell_id[p]);
742 if (p > 0) {
743 mpi::send(cell_to_vtx_loc, p);
744 } else {
745 this->getCellToVtxMap() = cell_to_vtx_loc;
746 }
747 }
748 } else {
749 mpi::recv(this->getCellToVtxMap(), 0);
750 }
751 // find unique local vertices
752 const auto& vtx_loc = this->getCellToVtxMap().data();
753 SetType need_vtxs_set(vtx_loc.begin(), vtx_loc.end());
754 std::vector<index_t> need_vtxs(need_vtxs_set.begin(), need_vtxs_set.end());
755 // send to root the unique vertices and scatter the vtx to cell map
756 if (mpi::root()) {
757 std::vector<index_t> need_vtxs_tmp;
758 for (int p = 1; p < size; ++p) {
759 mpi::recv(need_vtxs_tmp, p);
760 mpi::send(m_rank0_mesh->getVtxToCellMap().slice(need_vtxs_tmp), p);
761 }
762 this->getVtxToCellMap() = m_rank0_mesh->getVtxToCellMap().slice(need_vtxs);
763 } else {
764 mpi::send(need_vtxs, 0);
765 mpi::recv(this->getVtxToCellMap(), 0);
766 }
767 // invert vtx mapping
768 boost::unordered_flat_map<index_t, index_t> vtx_glob_to_loc;
769 for (std::size_t i = 0; i < need_vtxs.size(); ++i) {
770 vtx_glob_to_loc[need_vtxs[i]] = i;
771 }
772 // compute weak dual
773 const auto n_loc_cell = this->getCellToVtxMap().size();
774 datastructure::CrsMapUncompressed<index_t, 32> weak_dual_uncompressed;
775 weak_dual_uncompressed.resize(n_loc_cell);
776 for (index_t loc_cell_id = 0; loc_cell_id < n_loc_cell; ++loc_cell_id) {
777 SetType weak_neigh;
778 for (auto glob_vtx_id : this->getCellToVtxMap()[loc_cell_id]) {
779 const auto loc_vtx_id = vtx_glob_to_loc[glob_vtx_id];
780 for (auto glob_neigh_cell : this->getVtxToCellMap()[loc_vtx_id]) {
781 weak_neigh.insert(glob_neigh_cell);
782 }
783 }
784 weak_dual_uncompressed[loc_cell_id].assign(
785 weak_neigh.begin(), weak_neigh.end());
786 }
787 m_weak_dual = weak_dual_uncompressed.template compress<index_t>();
788}

790template <typename Mesh>
791void ParallelMesh<Mesh>::buildLocalRenumbering() {
792 SetType vtx_idxs, facet_idxs, edge_idxs;
793
794 // compute the owning part
795 const auto n_owned_cells = m_own_enty_size[self_dim];
796 for (index_t cell_id = 0; cell_id < n_owned_cells; ++cell_id) {
797 for (const auto vtx_id : this->getCellToVtxMap()[cell_id]) {
798 vtx_idxs.insert(vtx_id);
799 }
800 for (const auto facet_id : this->getCellToFacetMap()[cell_id]) {
801 facet_idxs.insert(facet_id);
802 }
803 if constexpr (self_dim == 3) {
804 for (const auto edge_id : this->getCellToEdgeMap()[cell_id]) {
805 edge_idxs.insert(edge_id);
806 }
807 }
808 }
809 // build local to global as vector
810 m_local_numbering[0].assign(vtx_idxs.begin(), vtx_idxs.end());
811 m_local_numbering[self_dim - 1].assign(facet_idxs.begin(), facet_idxs.end());
812 if constexpr (self_dim == 3) {
813 m_local_numbering[self_dim - 2].assign(edge_idxs.begin(), edge_idxs.end());
814 }
815 for (std::size_t d = 0; d < self_dim; ++d) {
816 m_own_enty_size[d] = m_local_numbering[d].size();
817 }
818 // compute the non-owning part
819 const auto n_ghost_cells = m_ghost_enty_size[self_dim];
820 const auto n_total_cells = n_ghost_cells + n_owned_cells;
821 for (index_t cell_id = n_owned_cells; cell_id < n_total_cells; ++cell_id) {
822 for (const auto vtx_id : this->getCellToVtxMap()[cell_id]) {
823 const auto [it, inserted] = vtx_idxs.insert(vtx_id);
824 if (inserted) {
825 m_local_numbering[0].push_back(vtx_id);
826 }
827 }
828 for (const auto facet_id : this->getCellToFacetMap()[cell_id]) {
829 const auto [it, inserted] = facet_idxs.insert(facet_id);
830 if (inserted) {
831 m_local_numbering[self_dim - 1].push_back(facet_id);
832 }
833 }
834 if constexpr (self_dim == 3) {
835 for (const auto edge_id : this->getCellToEdgeMap()[cell_id]) {
836 const auto [it, inserted] = edge_idxs.insert(edge_id);
837 if (inserted) {
838 m_local_numbering[self_dim - 2].push_back(edge_id);
839 }
840 }
841 }
842 }
843 for (std::size_t d = 0; d < self_dim; ++d) {
844 m_ghost_enty_size[d] = m_local_numbering[d].size() - m_own_enty_size[d];
845 }
846
847 // build global to local by inversion
848 for (std::size_t d = 0; d < self_dim + 1; ++d) {
849 const auto& local_map = m_local_numbering[d];
850 for (std::size_t i = 0; i < local_map.size(); ++i) {
851 m_global_numbering[d][local_map[i]] = i;
852 }
853 }

856template <typename Mesh>
857void ParallelMesh<Mesh>::buildLocalEntityToCellMaps() {
858 const auto size = mpi::size();
859 // build local maps in root and distribute them
860 if (mpi::root()) {
861 {
862 std::array<std::vector<index_t>, self_dim> local_numbering;
863 for (int p = 1; p < size; ++p) {
864 // recv local to global numbering
865 for (std::size_t d = 0; d < self_dim; ++d) {
866 mpi::recv(local_numbering[d], p);
867 }
868
869 // build and send vtxs and local maps (F2C F2V V2C [E2C])
870 const auto vtx_loc =
871 vutils::slice(m_rank0_mesh->getVertexes(), local_numbering[0]);
872 mpi::send(vtx_loc, p);
873 const auto facet_to_cell_loc = vutils::slice(
874 m_rank0_mesh->getFacetToCellMap(), local_numbering[self_dim - 1]);
875 mpi::send(facet_to_cell_loc, p);
876 const auto facet_to_vtx_loc = m_rank0_mesh->getFacetToVtxMap().slice(
877 local_numbering[self_dim - 1]);
878 mpi::send(facet_to_vtx_loc, p);
879 mpi::send(m_rank0_mesh->getVtxToCellMap().slice(local_numbering[0]), p);
880 if constexpr (self_dim == 3) {
881 const auto edge_to_cell_loc = m_rank0_mesh->getEdgeToCellMap().slice(
882 local_numbering[self_dim - 2]);
883 mpi::send(edge_to_cell_loc, p);
884
885 const auto edge_to_vtx_loc = vutils::slice(
886 m_rank0_mesh->getEdgeToVtxMap(), local_numbering[self_dim - 2]);
887 mpi::send(edge_to_vtx_loc, p);
888 }
889 }
890 }
891 // build for self
892 this->getVertexes() =
893 vutils::slice(m_rank0_mesh->getVertexes(), m_local_numbering[0]);
894 this->getFacetToCellMap() = vutils::slice(
895 m_rank0_mesh->getFacetToCellMap(), m_local_numbering[self_dim - 1]);
896 this->getFacetToVtxMap() =
897 m_rank0_mesh->getFacetToVtxMap().slice(m_local_numbering[self_dim - 1]);
898 this->getVtxToCellMap() =
899 m_rank0_mesh->getVtxToCellMap().slice(m_local_numbering[0]);
900 if constexpr (self_dim == 3) {
901 this->getEdgeToCellMap() = m_rank0_mesh->getEdgeToCellMap().slice(
902 m_local_numbering[self_dim - 2]);
903 this->getEdgeToVtxMap() = vutils::slice(
904 m_rank0_mesh->getEdgeToVtxMap(), m_local_numbering[self_dim - 2]);
905 }
906 } else {
907 // send local numbering to root
908 for (std::size_t d = 0; d < self_dim; ++d) {
909 mpi::send(m_local_numbering[d], 0);
910 }
911 // recv vertexes and local maps (F2C F2V V2C [E2C])
912 mpi::recv(this->getVertexes(), 0);
913 mpi::recv(this->getFacetToCellMap(), 0);
914 mpi::recv(this->getFacetToVtxMap(), 0);
915 mpi::recv(this->getVtxToCellMap(), 0);
916 if constexpr (self_dim == 3) {
917 mpi::recv(this->getEdgeToCellMap(), 0);
918 mpi::recv(this->getEdgeToVtxMap(), 0);
919 }
920 }
921 removeNotOwnedFromLocalEntityToCellMaps();
922}

924template <typename Mesh>
925void ParallelMesh<Mesh>::removeNotOwnedFromLocalEntityToCellMaps() {
926 // in the maps we may have references to cells that do not belong to
927 // the proc that owns that map.
928 // `owned_table` has size of # of global cells, which scales badly
929 // (linear with the number of proc). However this should not be a
930 // large issue since it is a temporary object that is allocated
931 // in a staggered way by the processors, since they enter in this
932 // function just after rank 0 has finished computing the local map
933 std::vector<bool> owned_table(m_global_entity_size.back(), false);
934 for (const auto global_cell_id : m_local_numbering[self_dim]) {
935 owned_table[global_cell_id] = true;
936 }
937 // Dual
938 assert(
939 this->getDualMap().size() ==
940 m_ghost_enty_size[self_dim] + m_own_enty_size[self_dim]);
941 for (auto& glob_cell_id : this->getDualMap().data()) {
942 if (!owned_table[glob_cell_id]) {
943 glob_cell_id = -1;
944 }
945 }
946 this->getDualMap().remove(-1);
947 // F2C
948 const auto n_loc_faces = this->getFacetToCellMap().size();
949 for (std::size_t loc_face_id = 0; loc_face_id < n_loc_faces; ++loc_face_id) {
950 for (auto& glob_cell_id : this->getFacetToCellMap()[loc_face_id]) {
951 if (glob_cell_id == -1) {
952 continue;
953 }
954 if (!owned_table[glob_cell_id]) {
955 glob_cell_id = -1;
956 }
957 }
958 }
959 // V2C
960 for (auto& glob_cell_id : this->getVtxToCellMap().data()) {
961 if (!owned_table[glob_cell_id]) {
962 glob_cell_id = -1;
963 }
964 }
965 this->getVtxToCellMap().remove(-1);
966 // E2C
967 if constexpr (self_dim == 3) {
968 for (auto& glob_cell_id : this->getEdgeToCellMap().data()) {
969 if (!owned_table[glob_cell_id]) {
970 glob_cell_id = -1;
971 }
972 }
973 this->getEdgeToCellMap().remove(-1);
974 }
975}

989 m_dual_graph_glob = this->m_dual_graph;

991 m_cell_to_vtx_glob = this->m_cell_to_vtx;

1000 fid = m_global_numbering[self_dim - 1].at(fid);
1001 }
1002 for (auto& vid : this->m_cell_to_vtx.data()) {
1003 vid = m_global_numbering[0].at(vid);
1004 }

1010 cid0cid1[1] = m_global_numbering[self_dim].at(cid0cid1[1]);
1011 }
1012 }
1013 for (auto& vid : this->m_facet_to_vtx.data()) {
1014 vid = m_global_numbering[0].at(vid);
1015 }
1016 for (auto& cid : this->m_vtx_to_cell.data()) {
1017 cid = m_global_numbering[self_dim].at(cid);
1018 }
1019
1020 if constexpr (self_dim == 3) {
1021 m_edge_to_vtx_glob = this->m_edge_to_vtx;
1022 m_cell_to_edge_glob = this->m_cell_to_edge;
1023 m_edge_to_cell_glob = this->m_edge_to_cell;
1024

1027 v1v2[1] = m_global_numbering[0].at(v1v2[1]);
1028 }
1029 for (auto& eid : this->m_cell_to_edge.data()) {

1031 }
1032 for (auto& cid : this->m_edge_to_cell.data()) {
1033 cid = m_global_numbering[self_dim].at(cid);

1035 }
