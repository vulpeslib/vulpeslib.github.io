227template <typename Mesh>
228std::vector<::idx_t> ParallelMesh<Mesh>::partitionWithParmetis() {
229 const auto rank = mpi::rank();
230 const auto size = mpi::size();
231 // how many vertices on each proc
232 std::vector<::idx_t> vtxdist(size + 1);
233 // crs map in Metis format
234 std::vector<::idx_t> xadj;
235 std::vector<::idx_t> adjncy;
236
237 if (rank == 0) {
238 const auto& dual = m_rank0_mesh->getDualMap();
239
240 // how many vertices on each proc
241 vtxdist[0] = 0;
242 for (int i = 0; i < size; ++i) {
243 vtxdist[i + 1] =
244 vtxdist[i] + dual.size() / size + (i < (dual.size() % size));
245 }
246 assert(vtxdist.back() == dual.size());
247
248 // build local crs map for others
249 for (int p = 1; p < size; ++p) {
250 xadj.resize(vtxdist[p + 1] - vtxdist[p] + 1);
251 const std::size_t adjncy_size =
252 dual.indices()[vtxdist[p + 1]] - dual.indices()[vtxdist[p]];
253 const auto offset = dual.indices()[vtxdist[p]];
254 for (std::size_t j = 0; j < xadj.size(); ++j) {
255 xadj[j] = dual.indices()[j + vtxdist[p]] - offset;
256 }
257 mpi::send(xadj, p);
258 mpi::send(adjncy_size, p);
259 mpi::send(&dual.data()[offset], adjncy_size, p);
260 }
261
262 // build local crs map for self
263 xadj.assign(
264 dual.indices().begin(), dual.indices().begin() + vtxdist[1] + 1);
265 adjncy.assign(
266 dual.data().begin(), dual.data().begin() + dual.indices()[vtxdist[1]]);
267 } else {
268 mpi::recv(xadj, 0);
269 adjncy.resize(mpi::recv<std::size_t>(0));
270 mpi::recv(adjncy.data(), static_cast<int32_t>(adjncy.size()), 0);
271 }
272 mpi::bcast(vtxdist, 0);
273
274 // ParMETIS parameters
275 idx_t ncon = 1;
276 idx_t wgtflag = 0;
277 idx_t numflag = 0;
278 idx_t nparts = size;
279 real_t ubvec = 1.05;
280 const idx_t random_seed = 0;
281 idx_t options[3] = {1, 0, random_seed};
282 std::vector<real_t> tpwgts(nparts);
283 for (int i = 0; i < size; ++i) {
284 tpwgts[i] = 1. / static_cast<double>(nparts);
285 }
286
287 // result buffer
288 std::vector<idx_t> part(vtxdist[rank + 1] - vtxdist[rank]);
289
290 std::cout << "Partition the graph;" << std::endl;
291 // Partition the graph
292 auto comm = MPI_COMM_WORLD;
293 ParMETIS_V3_PartKway(
294 vtxdist.data(), xadj.data(), adjncy.data(), nullptr, nullptr, &wgtflag,
295 &numflag, &ncon, &nparts, tpwgts.data(), &ubvec, options, &m_edgecut,
296 part.data(), &comm);
297 return part;
298}

300template <typename Mesh>
301void ParallelMesh<Mesh>::buildGhostCells(std::vector<index_t>& part) {
302 const auto size = mpi::size();
303 m_all_own_and_ghost_cell_id.resize(size);
304 m_all_own_size.resize(size);
305 m_all_ghost_size.resize(size);
306 // send partitions to root and compute ghost
307 const auto& dual = m_rank0_mesh->getDualMap();
308 std::vector<index_t> all_parts;
309
310 // reconstruct partitioning and send dual map owning slice
311 if (mpi::root()) {
312 std::cout << "Partitioning completed. Edge cuts: " << m_edgecut
313 << std::endl;
314 // std::cout << "rank 0 | part: " << part << std::endl;
315 all_parts = part;
316 for (int p = 1; p < size; ++p) {
317 mpi::recv(part, p);
318 // std::cout << "rank " << p << " | part: " << part << std::endl;
319 all_parts.insert(all_parts.end(), part.begin(), part.end());
320 }
321
322 // we re-order the partition to minimize the bandwidth
323 // of the processors block
324 // In practice if two partitions are connected they should
325 // belong also to procs that are close together
326 Eigen::MatrixXi dense_proc_adj = Eigen::MatrixXi::Zero(size, size);
327 for (index_t cell_id = 0; cell_id < dual.size(); ++cell_id) {
328 const auto src_cell_proc = all_parts[cell_id];
329 for (index_t neigh_cell_id : dual[cell_id]) {
330 const auto dst_cell_proc = all_parts[neigh_cell_id];
331 dense_proc_adj(src_cell_proc, dst_cell_proc) = 1;
332 }
333 }
334 datastructure::CrsMapUncompressed<index_t> uncomp_proc_adj;
335 uncomp_proc_adj.resize(size);
336 for (Eigen::Index i = 0; i < dense_proc_adj.rows(); ++i) {
337 for (Eigen::Index j = 0; j < dense_proc_adj.cols(); ++j) {
338 if (dense_proc_adj(i, j)) {
339 uncomp_proc_adj[i].push_back(j);
340 }
341 }
342 }
343 const auto proc_adj = uncomp_proc_adj.template compress<index_t>();
344 const auto [perm, perm_inv] =
345 dofs::computeReverseCuthillMckeeOrdering(proc_adj, false);
346 for (std::size_t i = 0; i < all_parts.size(); ++i) {
347 all_parts[i] = perm[all_parts[i]];
348 }
349
350 for (std::size_t cell_id = 0; cell_id < all_parts.size(); ++cell_id) {
351 m_all_own_and_ghost_cell_id[all_parts[cell_id]].push_back(cell_id);
352 }
353
354 for (int p = 0; p < size; ++p) {
355 m_all_own_size[p] = m_all_own_and_ghost_cell_id[p].size();
356 }
357
358 // send dual map of owned cells
359 for (int p = 1; p < size; ++p) {
360 mpi::send(m_all_own_and_ghost_cell_id[p], p);
361 mpi::send(dual.slice(m_all_own_and_ghost_cell_id[p]), p);
362 }
363 m_local_numbering[self_dim] = m_all_own_and_ghost_cell_id[0];
364 this->getDualMap() = dual.slice(m_all_own_and_ghost_cell_id[0]);
365 } else {
366 mpi::send(part, 0);
367 mpi::recv(m_local_numbering[self_dim], 0);
368 mpi::recv(this->getDualMap(), 0);
369 }
370
371 // In the ghost cells we want also the weakly connected cells
372 // that is the cells that share also only a vertex
373 // Indeed, we need this information to compute the sparsity pattern
374 buildWeakDual();
375
376 // build ghost cells
377 SetType ghost_cell_set(size);
378 for (index_t loc_cid = 0; loc_cid < m_weak_dual.size(); ++loc_cid) {
379 for (auto glob_neigh_id : m_weak_dual[loc_cid]) {
380 auto it = std::lower_bound(
381 m_local_numbering[self_dim].begin(),
382 m_local_numbering[self_dim].end(), glob_neigh_id);
383 if ((it == m_local_numbering[self_dim].end()) || (*it != glob_neigh_id)) {
384 ghost_cell_set.insert(glob_neigh_id);
385 }
386 }
387 }
388 m_own_enty_size[self_dim] = m_local_numbering[self_dim].size();
389 m_ghost_enty_size[self_dim] = ghost_cell_set.size();
390 m_local_numbering[self_dim].insert(
391 m_local_numbering[self_dim].end(), ghost_cell_set.begin(),
392 ghost_cell_set.end());
393
394 // collect ghost cells in root
395 if (mpi::root()) {
396 // recive ghost from procs
397 m_all_own_and_ghost_cell_id[0] = m_local_numbering[self_dim];
398 m_all_ghost_size[0] =
399 m_all_own_and_ghost_cell_id[0].size() - m_all_own_size[0];
400 for (int p = 1; p < size; ++p) {
401 m_all_ghost_size[p] = mpi::recv<index_t>(p);
402 m_all_own_and_ghost_cell_id[p].resize(
403 m_all_own_and_ghost_cell_id[p].size() + m_all_ghost_size[p]);
404 mpi::recv(
405 &m_all_own_and_ghost_cell_id[p][m_all_own_size[p]],
406 m_all_ghost_size[p], p);
407 }
408 } else {
409 mpi::send(m_ghost_enty_size[self_dim], 0);
410 mpi::send(
411 &m_local_numbering[self_dim][m_own_enty_size[self_dim]],
412 m_ghost_enty_size[self_dim], 0, MPI_COMM_WORLD);
413 }
414}

416template <typename Mesh>
417void ParallelMesh<Mesh>::buildLocalCellToEntityMaps() {
418 const auto size = mpi::size();
419 // build and send local maps
420 if (mpi::root()) {
421 for (int p = 1; p < size; ++p) {
422 mpi::send(
423 m_rank0_mesh->getDualMap().slice(m_all_own_and_ghost_cell_id[p]), p);
424 const auto cell_to_vtx_loc =
425 m_rank0_mesh->getCellToVtxMap().slice(m_all_own_and_ghost_cell_id[p]);
426 mpi::send(cell_to_vtx_loc, p);
427 const auto cell_to_facet_loc = m_rank0_mesh->getCellToFacetMap().slice(
428 m_all_own_and_ghost_cell_id[p]);
429 mpi::send(cell_to_facet_loc, p);
430 if constexpr (self_dim == 3) {
431 const auto cell_to_edge_loc = m_rank0_mesh->getCellToEdgeMap().slice(
432 m_all_own_and_ghost_cell_id[p]);
433 mpi::send(cell_to_edge_loc, p);
434 }
435 }
436 // build for self
437 this->getDualMap() =
438 m_rank0_mesh->getDualMap().slice(m_all_own_and_ghost_cell_id[0]);
439 this->getCellToVtxMap() =
440 m_rank0_mesh->getCellToVtxMap().slice(m_all_own_and_ghost_cell_id[0]);
441 this->getCellToFacetMap() =
442 m_rank0_mesh->getCellToFacetMap().slice(m_all_own_and_ghost_cell_id[0]);
443 if constexpr (self_dim == 3) {
444 this->getCellToEdgeMap() = m_rank0_mesh->getCellToEdgeMap().slice(
445 m_all_own_and_ghost_cell_id[0]);
446 }
447 } else {
448 mpi::recv(this->getDualMap(), 0);
449 mpi::recv(this->getCellToVtxMap(), 0);
450 mpi::recv(this->getCellToFacetMap(), 0);
451 if constexpr (self_dim == 3) {
452 mpi::recv(this->getCellToEdgeMap(), 0);
453 }
454 }
455}

457template <typename Mesh>
458void ParallelMesh<Mesh>::buildWeakDual() {
459 const auto size = mpi::size();
460 // scatter the cell to vtx map
461 if (mpi::root()) {
462 for (int p = 0; p < size; ++p) {
463 const auto cell_to_vtx_loc =
464 m_rank0_mesh->getCellToVtxMap().slice(m_all_own_and_ghost_cell_id[p]);
465 if (p > 0) {
466 mpi::send(cell_to_vtx_loc, p);
467 } else {
468 this->getCellToVtxMap() = cell_to_vtx_loc;
469 }
470 }
471 } else {
472 mpi::recv(this->getCellToVtxMap(), 0);
473 }
474 // find unique local vertices
475 const auto& vtx_loc = this->getCellToVtxMap().data();
476 SetType need_vtxs_set(vtx_loc.begin(), vtx_loc.end());
477 std::vector<index_t> need_vtxs(need_vtxs_set.begin(), need_vtxs_set.end());
478 // send to root the unique vertices and scatter the vtx to cell map
479 if (mpi::root()) {
480 std::vector<index_t> need_vtxs_tmp;
481 for (int p = 1; p < size; ++p) {
482 mpi::recv(need_vtxs_tmp, p);
483 mpi::send(m_rank0_mesh->getVtxToCellMap().slice(need_vtxs_tmp), p);
484 }
485 this->getVtxToCellMap() = m_rank0_mesh->getVtxToCellMap().slice(need_vtxs);
486 } else {
487 mpi::send(need_vtxs, 0);
488 mpi::recv(this->getVtxToCellMap(), 0);
489 }
490 // invert vtx mapping
491 boost::unordered_flat_map<index_t, index_t> vtx_glob_to_loc;
492 for (std::size_t i = 0; i < need_vtxs.size(); ++i) {
493 vtx_glob_to_loc[need_vtxs[i]] = i;
494 }
495 // compute weak dual
496 const auto n_loc_cell = this->getCellToVtxMap().size();
497 datastructure::CrsMapUncompressed<index_t, 32> weak_dual_uncompressed;
498 weak_dual_uncompressed.resize(n_loc_cell);
499 for (index_t loc_cell_id = 0; loc_cell_id < n_loc_cell; ++loc_cell_id) {
500 SetType weak_neigh;
501 for (auto glob_vtx_id : this->getCellToVtxMap()[loc_cell_id]) {
502 const auto loc_vtx_id = vtx_glob_to_loc[glob_vtx_id];
503 for (auto glob_neigh_cell : this->getVtxToCellMap()[loc_vtx_id]) {
504 weak_neigh.insert(glob_neigh_cell);
505 }
506 }
507 weak_dual_uncompressed[loc_cell_id].assign(
508 weak_neigh.begin(), weak_neigh.end());
509 }
510 m_weak_dual = weak_dual_uncompressed.template compress<index_t>();
511}

513template <typename Mesh>
514void ParallelMesh<Mesh>::buildLocalRenumbering() {
515 SetType vtx_idxs, facet_idxs, edge_idxs;
516
517 // compute the owning part
518 const auto n_owned_cells = m_own_enty_size[self_dim];
519 for (index_t cell_id = 0; cell_id < n_owned_cells; ++cell_id) {
520 for (const auto vtx_id : this->getCellToVtxMap()[cell_id]) {
521 vtx_idxs.insert(vtx_id);
522 }
523 for (const auto facet_id : this->getCellToFacetMap()[cell_id]) {
524 facet_idxs.insert(facet_id);
525 }
526 if constexpr (self_dim == 3) {
527 for (const auto edge_id : this->getCellToEdgeMap()[cell_id]) {
528 edge_idxs.insert(edge_id);
529 }
530 }
531 }
532 // build local to global as vector
533 m_local_numbering[0].assign(vtx_idxs.begin(), vtx_idxs.end());
534 m_local_numbering[self_dim - 1].assign(facet_idxs.begin(), facet_idxs.end());
535 if constexpr (self_dim == 3) {
536 m_local_numbering[self_dim - 2].assign(edge_idxs.begin(), edge_idxs.end());
537 }
538 for (std::size_t d = 0; d < self_dim; ++d) {
539 m_own_enty_size[d] = m_local_numbering[d].size();
540 }
541 // compute the non-owning part
542 const auto n_ghost_cells = m_ghost_enty_size[self_dim];
543 const auto n_total_cells = n_ghost_cells + n_owned_cells;
544 for (index_t cell_id = n_owned_cells; cell_id < n_total_cells; ++cell_id) {
545 for (const auto vtx_id : this->getCellToVtxMap()[cell_id]) {
546 const auto [it, inserted] = vtx_idxs.insert(vtx_id);
547 if (inserted) {
548 m_local_numbering[0].push_back(vtx_id);
549 }
550 }
551 for (const auto facet_id : this->getCellToFacetMap()[cell_id]) {
552 const auto [it, inserted] = facet_idxs.insert(facet_id);
553 if (inserted) {
554 m_local_numbering[self_dim - 1].push_back(facet_id);
555 }
556 }
557 if constexpr (self_dim == 3) {
558 for (const auto edge_id : this->getCellToEdgeMap()[cell_id]) {
559 const auto [it, inserted] = edge_idxs.insert(edge_id);
560 if (inserted) {
561 m_local_numbering[self_dim - 2].push_back(edge_id);
562 }
563 }
564 }
565 }
566 for (std::size_t d = 0; d < self_dim; ++d) {
567 m_ghost_enty_size[d] = m_local_numbering[d].size() - m_own_enty_size[d];
568 }
569
570 // build global to local by inversion
571 for (std::size_t d = 0; d < self_dim + 1; ++d) {
572 const auto& local_map = m_local_numbering[d];
573 for (std::size_t i = 0; i < local_map.size(); ++i) {
574 m_global_numbering[d][local_map[i]] = i;
575 }
576 }
577}

579template <typename Mesh>
580void ParallelMesh<Mesh>::buildLocalEntityToCellMaps() {
581 const auto size = mpi::size();
582 // build local maps in root and distribute them
583 if (mpi::root()) {
584 {
585 std::array<std::vector<index_t>, self_dim> local_numbering;
586 for (int p = 1; p < size; ++p) {
587 // recv local to global numbering
588 for (std::size_t d = 0; d < self_dim; ++d) {
589 mpi::recv(local_numbering[d], p);
590 }
591
592 // build and send vtxs and local maps (F2C F2V V2C [E2C])
593 const auto vtx_loc =
594 vutils::slice(m_rank0_mesh->getVertexes(), local_numbering[0]);
595 mpi::send(vtx_loc, p);
596 const auto facet_to_cell_loc = vutils::slice(
597 m_rank0_mesh->getFacetToCellMap(), local_numbering[self_dim - 1]);
598 mpi::send(facet_to_cell_loc, p);
599 const auto facet_to_vtx_loc = m_rank0_mesh->getFacetToVtxMap().slice(
600 local_numbering[self_dim - 1]);
601 mpi::send(facet_to_vtx_loc, p);
602 mpi::send(m_rank0_mesh->getVtxToCellMap().slice(local_numbering[0]), p);
603 if constexpr (self_dim == 3) {
604 const auto edge_to_cell_loc = m_rank0_mesh->getEdgeToCellMap().slice(
605 local_numbering[self_dim - 2]);
606 mpi::send(edge_to_cell_loc, p);
607 }
608 }
609 }
610 // build for self
611 this->getVertexes() =
612 vutils::slice(m_rank0_mesh->getVertexes(), m_local_numbering[0]);
613 this->getFacetToCellMap() = vutils::slice(
614 m_rank0_mesh->getFacetToCellMap(), m_local_numbering[self_dim - 1]);
615 this->getFacetToVtxMap() =
616 m_rank0_mesh->getFacetToVtxMap().slice(m_local_numbering[self_dim - 1]);
617 this->getVtxToCellMap() =
618 m_rank0_mesh->getVtxToCellMap().slice(m_local_numbering[0]);
619 if constexpr (self_dim == 3) {
620 this->getEdgeToCellMap() = m_rank0_mesh->getEdgeToCellMap().slice(
621 m_local_numbering[self_dim - 2]);
622 }
623 } else {
624 // send local numbering to root
625 for (std::size_t d = 0; d < self_dim; ++d) {
626 mpi::send(m_local_numbering[d], 0);
627 }
628 // recv vertexes and local maps (F2C F2V V2C [E2C])
629 mpi::recv(this->getVertexes(), 0);
630 mpi::recv(this->getFacetToCellMap(), 0);
631 mpi::recv(this->getFacetToVtxMap(), 0);
632 mpi::recv(this->getVtxToCellMap(), 0);
633 if constexpr (self_dim == 3) {
634 mpi::recv(this->getEdgeToCellMap(), 0);
635 }
636 }
637 removeNotOwnedFromLocalEntityToCellMaps();
638}

640template <typename Mesh>
641void ParallelMesh<Mesh>::removeNotOwnedFromLocalEntityToCellMaps() {
642 // in the maps we may have references to cells that do not belong to
643 // the proc that owns that map.
644 // `owned_table` has size of # of global cells, which scales badly
645 // (linear with the number of proc). However this should not be a
646 // large issue since it is a temporary object that is allocated
647 // in a staggered way by the processors, since they enter in this
648 // function just after rank 0 has finished computing the local map
649 std::vector<bool> owned_table(m_global_entity_size.back(), false);
650 for (const auto global_cell_id : m_local_numbering[self_dim]) {
651 owned_table[global_cell_id] = true;
652 }
653 // Dual
654 assert(
655 this->getDualMap().size() ==
656 m_ghost_enty_size[self_dim] + m_own_enty_size[self_dim]);
657 for (auto& glob_cell_id : this->getDualMap().data()) {
658 if (!owned_table[glob_cell_id]) {
659 glob_cell_id = -1;
660 }
661 }
662 this->getDualMap().remove(-1);
663 // F2C
664 const auto n_loc_faces = this->getFacetToCellMap().size();
665 for (std::size_t loc_face_id = 0; loc_face_id < n_loc_faces; ++loc_face_id) {
666 for (auto& glob_cell_id : this->getFacetToCellMap()[loc_face_id]) {
667 if (glob_cell_id == -1) {
668 continue;
669 }
670 if (!owned_table[glob_cell_id]) {
671 glob_cell_id = -1;
672 }
673 }
674 }
675 // V2C
676 for (auto& glob_cell_id : this->getVtxToCellMap().data()) {
677 if (!owned_table[glob_cell_id]) {
678 glob_cell_id = -1;
679 }
680 }
681 this->getVtxToCellMap().remove(-1);
682 // E2C
683 if constexpr (self_dim == 3) {
684 for (auto& glob_cell_id : this->getEdgeToCellMap().data()) {
685 if (!owned_table[glob_cell_id]) {
686 glob_cell_id = -1;
687 }
688 }
689 this->getEdgeToCellMap().remove(-1);
690 }
691}
