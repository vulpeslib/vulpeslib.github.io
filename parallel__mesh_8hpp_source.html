503template <typename Mesh>
504std::vector<::idx_t> ParallelMesh<Mesh>::partitionWithParmetis() {
505 const auto rank = mpi::rank();
506 const auto size = mpi::size();
507 // how many vertices on each proc
508 std::vector<::idx_t> vtxdist(size + 1);
509 // crs map in Metis format
510 std::vector<::idx_t> xadj;
511 std::vector<::idx_t> adjncy;
512
513 if (rank == 0) {
514 const auto& dual = m_rank0_mesh->getDualMap();
515
516 // how many vertices on each proc
517 vtxdist[0] = 0;
518 for (int i = 0; i < size; ++i) {
519 vtxdist[i + 1] =
520 vtxdist[i] + dual.size() / size + (i < (dual.size() % size));
521 }
522 assert(vtxdist.back() == dual.size());
523
524 // build local crs map for others
525 for (int p = 1; p < size; ++p) {
526 xadj.resize(vtxdist[p + 1] - vtxdist[p] + 1);
527 const std::size_t adjncy_size =
528 dual.indices()[vtxdist[p + 1]] - dual.indices()[vtxdist[p]];
529 const auto offset = dual.indices()[vtxdist[p]];
530 for (std::size_t j = 0; j < xadj.size(); ++j) {
531 xadj[j] = dual.indices()[j + vtxdist[p]] - offset;
532 }
533 mpi::send(xadj, p);
534 mpi::send(adjncy_size, p);
535 mpi::send(&dual.data()[offset], adjncy_size, p);
536 }
537
538 // build local crs map for self
539 xadj.assign(
540 dual.indices().begin(), dual.indices().begin() + vtxdist[1] + 1);
541 adjncy.assign(
542 dual.data().begin(), dual.data().begin() + dual.indices()[vtxdist[1]]);
543 } else {
544 mpi::recv(xadj, 0);
545 adjncy.resize(mpi::recv<std::size_t>(0));
546 mpi::recv(adjncy.data(), static_cast<int32_t>(adjncy.size()), 0);
547 }
548 mpi::bcast(vtxdist, 0);
549
550 // ParMETIS parameters
551 idx_t ncon = 1;
552 idx_t wgtflag = 0;
553 idx_t numflag = 0;
554 idx_t nparts = size;
555 real_t ubvec = 1.05;
556 const idx_t random_seed = 0;
557 idx_t options[3] = {1, 0, random_seed};
558 std::vector<real_t> tpwgts(nparts);
559 for (int i = 0; i < size; ++i) {
560 tpwgts[i] = 1. / static_cast<double>(nparts);
561 }
562
563 // result buffer
564 std::vector<idx_t> part(vtxdist[rank + 1] - vtxdist[rank]);
565
566 std::cout << "Partition the graph;" << std::endl;
567 // Partition the graph
568 auto comm = MPI_COMM_WORLD;
569 ParMETIS_V3_PartKway(
570 vtxdist.data(), xadj.data(), adjncy.data(), nullptr, nullptr, &wgtflag,
571 &numflag, &ncon, &nparts, tpwgts.data(), &ubvec, options, &m_edgecut,
572 part.data(), &comm);
573 return part;
574}

576template <typename Mesh>
577void ParallelMesh<Mesh>::reconstructPartitioningAndScatterDual(
578 std::vector<index_t>& part) {
579 const auto size = mpi::size();
580 const auto& dual = m_rank0_mesh->getDualMap();
581 std::vector<index_t> all_parts;
582 if (mpi::root()) {
583 std::cout << "Partitioning completed. Edge cuts: " << m_edgecut
584 << std::endl;
585 // std::cout << "rank 0 | part: " << part << std::endl;
586 all_parts = part;
587 for (int p = 1; p < size; ++p) {
588 mpi::recv(part, p);
589 // std::cout << "rank " << p << " | part: " << part << std::endl;
590 all_parts.insert(all_parts.end(), part.begin(), part.end());
591 }
592
593 // we re-order the partition to minimize the bandwidth
594 // of the processors block
595 // In practice if two partitions are connected they should
596 // belong also to procs that are close together
597 Eigen::MatrixXi dense_proc_adj = Eigen::MatrixXi::Zero(size, size);
598 for (index_t cell_id = 0; cell_id < dual.size(); ++cell_id) {
599 const auto src_cell_proc = all_parts[cell_id];
600 for (index_t neigh_cell_id : dual[cell_id]) {
601 const auto dst_cell_proc = all_parts[neigh_cell_id];
602 dense_proc_adj(src_cell_proc, dst_cell_proc) = 1;
603 }
604 }
605 datastructure::CrsMapUncompressed<index_t> uncomp_proc_adj;
606 uncomp_proc_adj.resize(size);
607 for (Eigen::Index i = 0; i < dense_proc_adj.rows(); ++i) {
608 for (Eigen::Index j = 0; j < dense_proc_adj.cols(); ++j) {
609 if (dense_proc_adj(i, j)) {
610 uncomp_proc_adj[i].push_back(j);
611 }
612 }
613 }
614 const auto proc_adj = uncomp_proc_adj.template compress<index_t>();
615 const auto [perm, perm_inv] =
616 dofs::computeReverseCuthillMckeeOrdering(proc_adj, false);
617 for (std::size_t i = 0; i < all_parts.size(); ++i) {
618 all_parts[i] = perm[all_parts[i]];
619 }
620
621 for (std::size_t cell_id = 0; cell_id < all_parts.size(); ++cell_id) {
622 m_all_own_and_ghost_cell_id[all_parts[cell_id]].push_back(cell_id);
623 }
624
625 for (int p = 0; p < size; ++p) {
626 m_all_own_size[p] = m_all_own_and_ghost_cell_id[p].size();
627 }
628
629 // send dual map of owned cells
630 for (int p = 1; p < size; ++p) {
631 mpi::send(m_all_own_and_ghost_cell_id[p], p);
632 mpi::send(dual.slice(m_all_own_and_ghost_cell_id[p]), p);
633 }
634 m_local_numbering[self_dim] = m_all_own_and_ghost_cell_id[0];
635 this->getDualMap() = dual.slice(m_all_own_and_ghost_cell_id[0]);
636 } else {
637 mpi::send(part, 0);
638 mpi::recv(m_local_numbering[self_dim], 0);
639 mpi::recv(this->getDualMap(), 0);
640 }
641}

643template <typename Mesh>
644void ParallelMesh<Mesh>::buildGhostCellsFromWeakDual() {
645 const auto size = mpi::size();
646 SetType ghost_cell_set(size);
647 for (index_t loc_cid = 0; loc_cid < m_weak_dual.size(); ++loc_cid) {
648 for (auto glob_neigh_id : m_weak_dual[loc_cid]) {
649 auto it = std::lower_bound(
650 m_local_numbering[self_dim].begin(),
651 m_local_numbering[self_dim].end(), glob_neigh_id);
652 if ((it == m_local_numbering[self_dim].end()) || (*it != glob_neigh_id)) {
653 ghost_cell_set.insert(glob_neigh_id);
654 }
655 }
656 }
657 m_own_enty_size[self_dim] = m_local_numbering[self_dim].size();
658 m_ghost_enty_size[self_dim] = ghost_cell_set.size();
659 m_local_numbering[self_dim].insert(
660 m_local_numbering[self_dim].end(), ghost_cell_set.begin(),
661 ghost_cell_set.end());
662}

664template <typename Mesh>
665void ParallelMesh<Mesh>::collectGhostCellsInRoot() {
666 const auto size = mpi::size();
667 if (mpi::root()) {
668 // recive ghost from procs
669 m_all_own_and_ghost_cell_id[0] = m_local_numbering[self_dim];
670 m_all_ghost_size[0] =
671 m_all_own_and_ghost_cell_id[0].size() - m_all_own_size[0];
672 for (int p = 1; p < size; ++p) {
673 m_all_ghost_size[p] = mpi::recv<index_t>(p);
674 m_all_own_and_ghost_cell_id[p].resize(
675 m_all_own_and_ghost_cell_id[p].size() + m_all_ghost_size[p]);
676 mpi::recv(
677 &m_all_own_and_ghost_cell_id[p][m_all_own_size[p]],
678 m_all_ghost_size[p], p);
679 }
680 } else {
681 mpi::send(m_ghost_enty_size[self_dim], 0);
682 mpi::send(
683 &m_local_numbering[self_dim][m_own_enty_size[self_dim]],
684 m_ghost_enty_size[self_dim], 0, MPI_COMM_WORLD);
685 }
686}

688template <typename Mesh>
689void ParallelMesh<Mesh>::buildGhostCells(std::vector<index_t>& part) {
690 const auto size = mpi::size();
691 m_all_own_and_ghost_cell_id.resize(size);
692 m_all_own_size.resize(size);
693 m_all_ghost_size.resize(size);
694 // send partitions to root and compute ghost
695 // reconstruct partitioning and send dual map owning slice
696 PROFILE_INTERVAL(reconstructPartitioningAndScatterDual(part));
697
698 // In the ghost cells we want also the weakly connected cells
699 // that is the cells that share also only a vertex
700 // Indeed, we need this information to compute the sparsity pattern
701 PROFILE_INTERVAL(buildWeakDual());
702 PROFILE_INTERVAL(buildGhostCellsFromWeakDual());
703 PROFILE_INTERVAL(collectGhostCellsInRoot());
704}

706template <typename Mesh>
707void ParallelMesh<Mesh>::buildLocalCellToEntityMaps() {
708 const auto size = mpi::size();
709 // build and send local maps
710 if (mpi::root()) {
711 for (int p = 1; p < size; ++p) {
712 mpi::send(
713 m_rank0_mesh->getDualMap().slice(m_all_own_and_ghost_cell_id[p]), p);
714 const auto cell_to_vtx_loc =
715 m_rank0_mesh->getCellToVtxMap().slice(m_all_own_and_ghost_cell_id[p]);
716 mpi::send(cell_to_vtx_loc, p);
717 const auto cell_to_facet_loc = m_rank0_mesh->getCellToFacetMap().slice(
718 m_all_own_and_ghost_cell_id[p]);
719 mpi::send(cell_to_facet_loc, p);
720 if constexpr (self_dim == 3) {
721 const auto cell_to_edge_loc = m_rank0_mesh->getCellToEdgeMap().slice(
722 m_all_own_and_ghost_cell_id[p]);
723 mpi::send(cell_to_edge_loc, p);
724 }
725 }
726 // build for self
727 this->getDualMap() =
728 m_rank0_mesh->getDualMap().slice(m_all_own_and_ghost_cell_id[0]);
729 this->getCellToVtxMap() =
730 m_rank0_mesh->getCellToVtxMap().slice(m_all_own_and_ghost_cell_id[0]);
731 this->getCellToFacetMap() =
732 m_rank0_mesh->getCellToFacetMap().slice(m_all_own_and_ghost_cell_id[0]);
733 if constexpr (self_dim == 3) {
734 this->getCellToEdgeMap() = m_rank0_mesh->getCellToEdgeMap().slice(
735 m_all_own_and_ghost_cell_id[0]);
736 }
737 } else {
738 mpi::recv(this->getDualMap(), 0);
739 mpi::recv(this->getCellToVtxMap(), 0);
740 mpi::recv(this->getCellToFacetMap(), 0);
741 if constexpr (self_dim == 3) {
742 mpi::recv(this->getCellToEdgeMap(), 0);
743 }
744 }

747template <typename Mesh>
748void ParallelMesh<Mesh>::buildWeakDual() {
749 const auto size = mpi::size();
750 // scatter the cell to vtx map
751 if (mpi::root()) {
752 for (int p = 0; p < size; ++p) {
753 const auto cell_to_vtx_loc =
754 m_rank0_mesh->getCellToVtxMap().slice(m_all_own_and_ghost_cell_id[p]);
755 if (p > 0) {
756 mpi::send(cell_to_vtx_loc, p);
757 } else {
758 this->getCellToVtxMap() = cell_to_vtx_loc;
759 }
760 }
761 } else {
762 mpi::recv(this->getCellToVtxMap(), 0);
763 }
764 // find unique local vertices
765 const auto& vtx_loc = this->getCellToVtxMap().data();
766 SetType need_vtxs_set(vtx_loc.begin(), vtx_loc.end());
767 std::vector<index_t> need_vtxs(need_vtxs_set.begin(), need_vtxs_set.end());
768 // send to root the unique vertices and scatter the vtx to cell map
769 if (mpi::root()) {
770 std::vector<index_t> need_vtxs_tmp;
771 for (int p = 1; p < size; ++p) {
772 mpi::recv(need_vtxs_tmp, p);
773 mpi::send(m_rank0_mesh->getVtxToCellMap().slice(need_vtxs_tmp), p);
774 }
775 this->getVtxToCellMap() = m_rank0_mesh->getVtxToCellMap().slice(need_vtxs);
776 } else {
777 mpi::send(need_vtxs, 0);
778 mpi::recv(this->getVtxToCellMap(), 0);
779 }
780 // invert vtx mapping
781 boost::unordered_flat_map<index_t, index_t> vtx_glob_to_loc;
782 for (std::size_t i = 0; i < need_vtxs.size(); ++i) {
783 vtx_glob_to_loc[need_vtxs[i]] = i;
784 }
785 // compute weak dual
786 const auto n_loc_cell = this->getCellToVtxMap().size();
787 datastructure::CrsMapUncompressed<index_t, 32> weak_dual_uncompressed;
788 weak_dual_uncompressed.resize(n_loc_cell);
789 for (index_t loc_cell_id = 0; loc_cell_id < n_loc_cell; ++loc_cell_id) {
790 SetType weak_neigh;
791 for (auto glob_vtx_id : this->getCellToVtxMap()[loc_cell_id]) {
792 const auto loc_vtx_id = vtx_glob_to_loc[glob_vtx_id];
793 for (auto glob_neigh_cell : this->getVtxToCellMap()[loc_vtx_id]) {
794 weak_neigh.insert(glob_neigh_cell);
795 }
796 }
797 weak_dual_uncompressed[loc_cell_id].assign(
798 weak_neigh.begin(), weak_neigh.end());
799 }
800 m_weak_dual = weak_dual_uncompressed.template compress<index_t>();
801}

803template <typename Mesh>
804void ParallelMesh<Mesh>::buildLocalRenumbering() {
805 SetType vtx_idxs, facet_idxs, edge_idxs;
806
807 // compute the owning part
808 const auto n_owned_cells = m_own_enty_size[self_dim];
809 for (index_t cell_id = 0; cell_id < n_owned_cells; ++cell_id) {
810 for (const auto vtx_id : this->getCellToVtxMap()[cell_id]) {
811 vtx_idxs.insert(vtx_id);
812 }
813 for (const auto facet_id : this->getCellToFacetMap()[cell_id]) {
814 facet_idxs.insert(facet_id);
815 }
816 if constexpr (self_dim == 3) {
817 for (const auto edge_id : this->getCellToEdgeMap()[cell_id]) {
818 edge_idxs.insert(edge_id);
819 }
820 }
821 }
822 // build local to global as vector
823 m_local_numbering[0].assign(vtx_idxs.begin(), vtx_idxs.end());
824 m_local_numbering[self_dim - 1].assign(facet_idxs.begin(), facet_idxs.end());
825 if constexpr (self_dim == 3) {
826 m_local_numbering[self_dim - 2].assign(edge_idxs.begin(), edge_idxs.end());
827 }
828 for (std::size_t d = 0; d < self_dim; ++d) {
829 m_own_enty_size[d] = m_local_numbering[d].size();
830 }
831 // compute the non-owning part
832 const auto n_ghost_cells = m_ghost_enty_size[self_dim];
833 const auto n_total_cells = n_ghost_cells + n_owned_cells;
834 for (index_t cell_id = n_owned_cells; cell_id < n_total_cells; ++cell_id) {
835 for (const auto vtx_id : this->getCellToVtxMap()[cell_id]) {
836 const auto [it, inserted] = vtx_idxs.insert(vtx_id);
837 if (inserted) {
838 m_local_numbering[0].push_back(vtx_id);
839 }
840 }
841 for (const auto facet_id : this->getCellToFacetMap()[cell_id]) {
842 const auto [it, inserted] = facet_idxs.insert(facet_id);
843 if (inserted) {
844 m_local_numbering[self_dim - 1].push_back(facet_id);
845 }
846 }
847 if constexpr (self_dim == 3) {
848 for (const auto edge_id : this->getCellToEdgeMap()[cell_id]) {
849 const auto [it, inserted] = edge_idxs.insert(edge_id);
850 if (inserted) {
851 m_local_numbering[self_dim - 2].push_back(edge_id);
852 }
853 }
854 }
855 }
856 for (std::size_t d = 0; d < self_dim; ++d) {
857 m_ghost_enty_size[d] = m_local_numbering[d].size() - m_own_enty_size[d];
858 }
859
860 // build global to local by inversion
861 for (std::size_t d = 0; d < self_dim + 1; ++d) {
862 const auto& local_map = m_local_numbering[d];
863 for (std::size_t i = 0; i < local_map.size(); ++i) {
864 m_global_numbering[d][local_map[i]] = i;
865 }
866 }
867}

869template <typename Mesh>
870void ParallelMesh<Mesh>::buildLocalEntityToCellMaps() {
871 const auto size = mpi::size();
872 // build local maps in root and distribute them
873 if (mpi::root()) {
874 {
875 std::array<std::vector<index_t>, self_dim> local_numbering;
876 for (int p = 1; p < size; ++p) {
877 // recv local to global numbering
878 for (std::size_t d = 0; d < self_dim; ++d) {
879 mpi::recv(local_numbering[d], p);
880 }
881
882 // build and send vtxs and local maps (F2C F2V V2C [E2C])
883 const auto vtx_loc =
884 vutils::slice(m_rank0_mesh->getVertexes(), local_numbering[0]);
885 mpi::send(vtx_loc, p);
886 const auto facet_to_cell_loc = vutils::slice(
887 m_rank0_mesh->getFacetToCellMap(), local_numbering[self_dim - 1]);
888 mpi::send(facet_to_cell_loc, p);
889 const auto facet_to_vtx_loc = m_rank0_mesh->getFacetToVtxMap().slice(
890 local_numbering[self_dim - 1]);
891 mpi::send(facet_to_vtx_loc, p);
892 mpi::send(m_rank0_mesh->getVtxToCellMap().slice(local_numbering[0]), p);
893 if constexpr (self_dim == 3) {
894 const auto edge_to_cell_loc = m_rank0_mesh->getEdgeToCellMap().slice(
895 local_numbering[self_dim - 2]);
896 mpi::send(edge_to_cell_loc, p);
897
898 const auto edge_to_vtx_loc = vutils::slice(
899 m_rank0_mesh->getEdgeToVtxMap(), local_numbering[self_dim - 2]);
900 mpi::send(edge_to_vtx_loc, p);
901 }
902 }
903 }
904 // build for self
905 this->getVertexes() =
906 vutils::slice(m_rank0_mesh->getVertexes(), m_local_numbering[0]);
907 this->getFacetToCellMap() = vutils::slice(
908 m_rank0_mesh->getFacetToCellMap(), m_local_numbering[self_dim - 1]);
909 this->getFacetToVtxMap() =
910 m_rank0_mesh->getFacetToVtxMap().slice(m_local_numbering[self_dim - 1]);
911 this->getVtxToCellMap() =
912 m_rank0_mesh->getVtxToCellMap().slice(m_local_numbering[0]);
913 if constexpr (self_dim == 3) {
914 this->getEdgeToCellMap() = m_rank0_mesh->getEdgeToCellMap().slice(
915 m_local_numbering[self_dim - 2]);
916 this->getEdgeToVtxMap() = vutils::slice(
917 m_rank0_mesh->getEdgeToVtxMap(), m_local_numbering[self_dim - 2]);
918 }
919 } else {
920 // send local numbering to root
921 for (std::size_t d = 0; d < self_dim; ++d) {
922 mpi::send(m_local_numbering[d], 0);
923 }
924 // recv vertexes and local maps (F2C F2V V2C [E2C])
925 mpi::recv(this->getVertexes(), 0);
926 mpi::recv(this->getFacetToCellMap(), 0);
927 mpi::recv(this->getFacetToVtxMap(), 0);
928 mpi::recv(this->getVtxToCellMap(), 0);
929 if constexpr (self_dim == 3) {
930 mpi::recv(this->getEdgeToCellMap(), 0);
931 mpi::recv(this->getEdgeToVtxMap(), 0);
932 }
933 }
934 removeNotOwnedFromLocalEntityToCellMaps();
935}

937template <typename Mesh>
938void ParallelMesh<Mesh>::removeNotOwnedFromLocalEntityToCellMaps() {
939 // in the maps we may have references to cells that do not belong to
940 // the proc that owns that map.
941 // `owned_table` has size of # of global cells, which scales badly
942 // (linear with the number of proc). However this should not be a
943 // large issue since it is a temporary object that is allocated
944 // in a staggered way by the processors, since they enter in this
945 // function just after rank 0 has finished computing the local map
946 std::vector<bool> owned_table(m_global_entity_size.back(), false);
947 for (const auto global_cell_id : m_local_numbering[self_dim]) {
948 owned_table[global_cell_id] = true;
949 }
950 // Dual
951 assert(
952 this->getDualMap().size() ==
953 m_ghost_enty_size[self_dim] + m_own_enty_size[self_dim]);
954 for (auto& glob_cell_id : this->getDualMap().data()) {
955 if (!owned_table[glob_cell_id]) {
956 glob_cell_id = -1;
957 }
958 }
959 this->getDualMap().remove(-1);
960 // F2C
961 const auto n_loc_faces = this->getFacetToCellMap().size();
962 for (std::size_t loc_face_id = 0; loc_face_id < n_loc_faces; ++loc_face_id) {
963 for (auto& glob_cell_id : this->getFacetToCellMap()[loc_face_id]) {
964 if (glob_cell_id == -1) {
965 continue;
966 }
967 if (!owned_table[glob_cell_id]) {
968 glob_cell_id = -1;
969 }
970 }
971 }
972 // V2C
973 for (auto& glob_cell_id : this->getVtxToCellMap().data()) {
974 if (!owned_table[glob_cell_id]) {
975 glob_cell_id = -1;
976 }
977 }
978 this->getVtxToCellMap().remove(-1);
979 // E2C
980 if constexpr (self_dim == 3) {
981 for (auto& glob_cell_id : this->getEdgeToCellMap().data()) {
982 if (!owned_table[glob_cell_id]) {
983 glob_cell_id = -1;
984 }
985 }
986 this->getEdgeToCellMap().remove(-1);
987 }
988}

1006 m_facet_to_vtx_glob = this->m_facet_to_vtx;
1007 m_vtx_to_cell_glob = this->m_vtx_to_cell;
1008

1010 cid = m_global_numbering[self_dim].at(cid);
1011 }
1012 for (auto& fid : this->m_cell_to_facet.data()) {
1013 fid = m_global_numbering[self_dim - 1].at(fid);
1014 }
1015 for (auto& vid : this->m_cell_to_vtx.data()) {
1016 vid = m_global_numbering[0].at(vid);
1017 }
1018 for (auto& cid0cid1 : this->m_facet_to_cell) {

1023 cid0cid1[1] = m_global_numbering[self_dim].at(cid0cid1[1]);
1024 }
1025 }
1026 for (auto& vid : this->m_facet_to_vtx.data()) {
1027 vid = m_global_numbering[0].at(vid);
1028 }
1029 for (auto& cid : this->m_vtx_to_cell.data()) {
1030 cid = m_global_numbering[self_dim].at(cid);
1031 }
1032
1033 if constexpr (self_dim == 3) {
1034 m_edge_to_vtx_glob = this->m_edge_to_vtx;
1035 m_cell_to_edge_glob = this->m_cell_to_edge;
1036 m_edge_to_cell_glob = this->m_edge_to_cell;
1037
1038 for (auto& v1v2 : this->m_edge_to_vtx) {
1039 v1v2[0] = m_global_numbering[0].at(v1v2[0]);
1040 v1v2[1] = m_global_numbering[0].at(v1v2[1]);
1041 }
1042 for (auto& eid : this->m_cell_to_edge.data()) {
1043 eid = m_global_numbering[1].at(eid);
1044 }
1045 for (auto& cid : this->m_edge_to_cell.data()) {
1046 cid = m_global_numbering[self_dim].at(cid);
1047 }
1048 }
