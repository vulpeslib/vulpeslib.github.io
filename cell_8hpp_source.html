37template <typename FloatType>
38class Cell2D {
39 public:
40 Cell2D() = default;
41 template <typename IndexType>
42 Cell2D(
43 const Mesh2D<FloatType, IndexType>& mesh,
44 IndexType cell_id,
45 bool compute_normals = false) {
46 const auto cell_to_vtxs = mesh.getCellToVtxMap()[cell_id];
47 m_vtxs.resize(2, cell_to_vtxs.size());
48 for (int64_t loc_id = 0; loc_id < m_vtxs.cols(); ++loc_id) {
49 m_vtxs.col(loc_id) = mesh.getVertexes()[cell_to_vtxs[loc_id]];
50 }
51 computeBbox();
52 if (compute_normals) {
53 computeNormals();
54 }
55 }
56 void computeBbox() {
57 m_bbox.col(0) = m_vtxs.rowwise().minCoeff();
58 m_bbox.col(1) = m_vtxs.rowwise().maxCoeff();
59 }
60 void computeNormals() {
61 m_normals.resize(m_vtxs.rows(), m_vtxs.cols());
62 const int64_t nvtxs = m_vtxs.cols();
63 for (int64_t i = 0; i < nvtxs; ++i) {
64 Eigen::Vector<FloatType, 2> v12 =
65 m_vtxs.col(i) - m_vtxs.col((i + 1) % nvtxs);
66 m_normals(0, i) = -v12(1);
67 m_normals(1, i) = v12(0);
68 }
69 }
70
71 const auto& vtxs() const { return m_vtxs; }
72 auto& vtxs() { return m_vtxs; }
73 const auto& bbox() const { return m_bbox; }
74 const auto& normals() const { return m_normals; }
75 int64_t nfaces() const { return m_vtxs.cols(); }
76
77 protected:
78 Eigen::Matrix<FloatType, 2, -1, Eigen::ColMajor> m_vtxs;
79 Eigen::Matrix<FloatType, 2, -1, Eigen::ColMajor> m_normals;
80 Eigen::Matrix<FloatType, 2, 2> m_bbox;
81};

83template <typename FloatType, typename IndexType>
84class Cell3D {
85 public:
86 Cell3D(
87 const datastructure::CrsMapUncompressed<IndexType>& faces,
88 const std::vector<Point<FloatType, 3>>& coords,
89 bool compute_normals = false,
90 bool build_local_maps = true,
91 bool build_inverse_maps = true,
92 bool find_edges_to_compute = true)
93 : m_faces(faces.template compress<IndexType>()) {
94 m_vtxs.resize(3, coords.size());
95 for (std::size_t i = 0; i < coords.size(); ++i) {
96 m_vtxs.col(i) = coords[i];
97 }
98 init(
99 compute_normals, build_local_maps, build_inverse_maps,
100 find_edges_to_compute);
101 }
102
103 Cell3D(
104 const Mesh3D<FloatType, IndexType>& mesh,
105 IndexType cell_id,
106 bool compute_normals = false,
107 bool build_local_maps = true,
108 bool build_inverse_maps = true,
109 bool find_edges_to_compute = true) {
110 // copy coordinates
111 std::unordered_map<IndexType, IndexType> vtx_global_to_local;
112 std::size_t nvtxs = 0;
113 m_vtxs.resize(3, mesh.getCellToVtxMap()[cell_id].size());
114 for (const auto vtx_id : mesh.getCellToVtxMap()[cell_id]) {
115 vtx_global_to_local[vtx_id] = nvtxs;
116 m_vtxs.col(nvtxs++) = mesh.getVertexes()[vtx_id];
117 }
118 // copy faces
119 datastructure::CrsMapUncompressed<IndexType> uncompressed_faces;
120 for (const auto face_id : mesh.getCellToFacetMap()[cell_id]) {
121 uncompressed_faces.push_back({});
122 for (const auto vtx_id : mesh.getFacetToVtxMap()[face_id]) {
123 uncompressed_faces[uncompressed_faces.size() - 1].push_back(
124 vtx_global_to_local[vtx_id]);
125 }
126 }
127 m_faces = uncompressed_faces.template compress<IndexType>();
128 //
129 init(
130 compute_normals, build_local_maps, build_inverse_maps,
131 find_edges_to_compute);
132 }
133
134 void computeBbox() {
135 m_bbox.col(0) = m_vtxs.rowwise().minCoeff();
136 m_bbox.col(1) = m_vtxs.rowwise().maxCoeff();
137 }
138
139 void init(
140 bool compute_normals,
141 bool build_local_maps,
142 bool build_inverse_maps,
143 bool find_edges_to_compute) {
144 computeBbox();
145 if (compute_normals) {
146 computeNormals();
147 }
148 if (build_local_maps) {
149 buildEdgeToFace();
150 }
151 if (build_inverse_maps) {
152 buildInverseMaps();
153 }
154 if (find_edges_to_compute) {
155 findEdgesToCompute();
156 }
157 }
158
159 void computeNormals() {
160 m_normals.resize(3, m_faces.size());
161 const std::size_t nfaces = m_faces.size();
162 for (std::size_t fi = 0; fi < nfaces; ++fi) {
163 const auto p1 = m_vtxs.col(m_faces[fi][0]);
164 const auto p2 = m_vtxs.col(m_faces[fi][1]);
165 const auto p3 = m_vtxs.col(m_faces[fi][2]);
166
167 const Point<FloatType, 3> v1 = p2 - p1;
168 const Point<FloatType, 3> v2 = p3 - p1;
169
170 const Point<FloatType, 3> normal_vector = v1.cross(v2);
171 m_normals.col(fi) = normal_vector;
172
173#ifndef NDEBUG
174 // compare with SVD
175 Eigen::Vector<FloatType, 3> centroid = m_vtxs.rowwise().mean();
176 Eigen::Matrix<FloatType, 3, -1> centered_vtxs =
177 m_vtxs.colwise() - centroid;
178 Eigen::JacobiSVD<decltype(centered_vtxs)> svd(
179 centered_vtxs, Eigen::ComputeThinU | Eigen::ComputeThinV);
180 Eigen::Vector<FloatType, 3> normal = svd.matrixV().col(2).normalized();
181
182 assert((m_normals.col(fi).normalized() - normal).norm() < 1e-12);
183#endif
184 }
185 }

211 const auto& bbox() const { return m_bbox; }

213 protected:
214 void buildEdgeToFace();
215 void buildInverseMaps();
216 void findEdgesToCompute();
217
218 bool m_local_map_built = false;
219 bool m_inverse_map_built = false;
220 bool m_edges_to_compute_built = false;
221
222 // list of indices defining a face (order must point towards outer normal)
223 datastructure::CrsMap<IndexType, IndexType> m_faces;
224 // pairs of indices defining and edge
225 std::vector<std::array<IndexType, 2>> m_edges;
226 // list of physical coordinates
227 Eigen::Matrix<FloatType, 3, -1> m_vtxs;
228 Eigen::Matrix<FloatType, 3, 2> m_bbox;
229 Eigen::Matrix<FloatType, 3, -1> m_normals;
230 // map each edge to the two faces (indexes) to which the edge belongs to
231 // the order is important, because the two edges are traversed in opposed
232 // ways in order to point towards the outer normal, the first face is the
233 // one where the edges are traversed in increasing order of the indices
234 std::vector<std::array<IndexType, 2>> m_edges_to_faces;
235 // origin of the coordinates system for each face as the index of a vertex
236 std::vector<IndexType> m_face_xF_idx;
237 // list of indices of the edges that necessitates the integrals to be computed
238 std::vector<IndexType> m_edges_to_compute;
239
240 datastructure::CrsMap<IndexType, IndexType> m_vtx_to_faces;
241 datastructure::CrsMap<IndexType, IndexType> m_vtx_to_edge;
242 datastructure::CrsMap<IndexType, IndexType> m_face_to_edge;
243};

245template <typename FloatType, typename IndexType>
246void Cell3D<FloatType, IndexType>::buildInverseMaps() {
247 if (m_inverse_map_built) {
248 return;
249 }
250 buildEdgeToFace();
251 datastructure::CrsMapUncompressed<IndexType> uncompressed_vtx_to_faces;
252 uncompressed_vtx_to_faces.resize(m_vtxs.cols());
253 for (IndexType face_id = 0; face_id < m_faces.size(); ++face_id) {
254 for (IndexType vtx_id : m_faces[face_id]) {
255 uncompressed_vtx_to_faces[vtx_id].push_back(face_id);
256 }
257 }
258 m_vtx_to_faces = uncompressed_vtx_to_faces.template compress<IndexType>();
259 //
260 datastructure::CrsMapUncompressed<IndexType> uncompressed_vtx_to_edge;
261 uncompressed_vtx_to_edge.resize(m_vtxs.cols());
262 for (size_t edge_id = 0; edge_id < m_edges.size(); ++edge_id) {
263 for (IndexType vtx_id : m_edges[edge_id]) {
264 uncompressed_vtx_to_edge[vtx_id].push_back(edge_id);
265 }
266 }
267 m_vtx_to_edge = uncompressed_vtx_to_edge.template compress<IndexType>();
268 //
269 datastructure::CrsMapUncompressed<IndexType> uncompressed_face_to_edge;
270 uncompressed_face_to_edge.resize(m_faces.size());
271 for (size_t edge_id = 0; edge_id < m_edges.size(); ++edge_id) {
272 for (IndexType face_id : m_edges_to_faces[edge_id]) {
273 uncompressed_face_to_edge[face_id].push_back(edge_id);
274 }
275 }
276 m_face_to_edge = uncompressed_face_to_edge.template compress<IndexType>();
277 m_inverse_map_built = true;
278}

280template <typename FloatType, typename IndexType>
281void Cell3D<FloatType, IndexType>::buildEdgeToFace() {
282 if (m_local_map_built) {
283 return;
284 }
285 // edges_to_faces
286 std::vector<std::tuple<std::array<IndexType, 2>, IndexType, bool>>
287 face_edge_flat_map;
288 for (IndexType fi = 0; fi < m_faces.size(); ++fi) {
289 for (IndexType i = 0; i < m_faces[fi].size(); ++i) {
290 size_t i1 = (i + 1) % m_faces[fi].size();
291 bool ordered = false;
292 if (m_faces[fi][i] < m_faces[fi][i1]) {
293 m_edges.push_back({m_faces[fi][i], m_faces[fi][i1]});
294 ordered = true;
295 } else {
296 m_edges.push_back({m_faces[fi][i1], m_faces[fi][i]});
297 ordered = false;
298 }
299 face_edge_flat_map.push_back({m_edges.back(), fi, ordered});
300 }
301 }
302
303 std::set<std::array<IndexType, 2>> edges_unq(m_edges.begin(), m_edges.end());
304 m_edges.assign(edges_unq.begin(), edges_unq.end());
305
306 m_edges_to_faces.resize(
307 m_edges.size(), std::array<IndexType, 2>{IndexType(-1), IndexType(-1)});
308 for (size_t i = 0; i < face_edge_flat_map.size(); ++i) {
309 size_t edge_idx = 0;
310 while (m_edges[edge_idx] != std::get<0>(face_edge_flat_map[i])) {
311 edge_idx++;
312 }
313 const bool ordered = std::get<2>(face_edge_flat_map[i]);
314 m_edges_to_faces[edge_idx][1 - static_cast<int>(ordered)] =
315 std::get<1>(face_edge_flat_map[i]);
316 }
317 m_local_map_built = true;
318}

320template <typename FloatType, typename IndexType>
321void Cell3D<FloatType, IndexType>::findEdgesToCompute() {
322 if (m_edges_to_compute_built) {
323 return;
324 }
325 buildInverseMaps();
326 // we have to find the center for each face, keep track of which faces have
327 // been set
328 std::vector<bool> face_mask(m_faces.size(), true);
329 IndexType remaining_faces = m_faces.size();
330 m_face_xF_idx.resize(m_faces.size(), -1);
331 // buffer to count how many faces are attached to each vertex
332 std::vector<int32_t> vtx_edges_count(m_vtxs.cols());
333 // at each iteration we pick a new center xF for some faces as the vertex of
334 // the polyhedron that has more edges (still) attached to it
335 // status 2 (undecided), status 1 (must compute), status 0 (do not compute)
336 // this heuristic may not give the best outcome but should reduce
337 // significantly the cost of the quadrature-free algo using the procedure of
338 // "Quadrature-free polytopic discontinuous Galerkin methods for transport
339 // problems" TJ Radley, P Houston, ME Hubbard
340 std::vector<int8_t> edges_to_compute_status(m_edges.size(), 2);
341 while (remaining_faces) {
342 std::fill(vtx_edges_count.begin(), vtx_edges_count.end(), 0);
343 for (size_t ei = 0; ei < m_edges.size(); ++ei) {
344 vtx_edges_count[m_edges[ei][0]] += edges_to_compute_status[ei] / 2;
345 vtx_edges_count[m_edges[ei][1]] += edges_to_compute_status[ei] / 2;
346 }
347 // find the vertex with most (undecided) edges attached
348 const auto max_it =
349 std::max_element(vtx_edges_count.begin(), vtx_edges_count.end());
350 // all edges are assigned, nothing left to do
351 if (*max_it == 0) {
352 // give a default center to the unassigned faces
353 for (size_t i = 0; i < m_face_xF_idx.size(); ++i) {
354 if (m_face_xF_idx[i] == static_cast<IndexType>(-1)) {
355 m_face_xF_idx[i] = m_faces[i][0];
356 }
357 }
358 break;
359 }
360 const size_t idx = std::distance(vtx_edges_count.begin(), max_it);
361 assert(idx < vtx_edges_count.size());
362 // remove from the edges_to_compute the edges that have this vertex as an
363 // end
364 for (IndexType fi : m_vtx_to_faces[idx]) {
365 // update center only if not set yet
366 m_face_xF_idx[fi] =
367 (m_face_xF_idx[fi] == static_cast<IndexType>(-1)) * idx;
368 remaining_faces -= face_mask[fi];
369 face_mask[fi] = false;
370 }
371 // update edge status
372 // - if idx \in edge_id and undecided -> may not compute (status 0)
373 for (IndexType edge_id : m_vtx_to_edge[idx]) {
374 edges_to_compute_status[edge_id] = edges_to_compute_status[edge_id] % 2;
375 }
376 // - else if idx \in face and undecided -> must compute (status 1)
377 for (IndexType face_id : m_vtx_to_faces[idx]) {
378 for (IndexType edge_id : m_face_to_edge[face_id]) {
379 const auto status = edges_to_compute_status[edge_id];
380 edges_to_compute_status[edge_id] =
381 (status & int8_t(1)) + (status >> int8_t(1));
382 }
383 }
384 }
385 for (std::size_t ei = 0; ei < edges_to_compute_status.size(); ++ei) {
386 assert(edges_to_compute_status[ei] != 2);
387 if (edges_to_compute_status[ei]) {
388 m_edges_to_compute.push_back(ei);
389 }
390 }
391 m_edges_to_compute_built = true;
392}
