149template <typename FloatType, typename IndexType>
150class Cell3D {
151 public:
152 Cell3D() = default;
153
154 Cell3D(
155 const datastructure::CrsMapUncompressed<IndexType>& faces,
156 const std::vector<Point<FloatType, 3>>& coords,
157 CellInitOptionsType cell_init_options = CellInitAll);
158
159 Cell3D(
160 const Mesh3D<FloatType, IndexType>& mesh,
161 IndexType cell_id,
162 CellInitOptionsType cell_init_options = CellInitVtxOnly);
163
164 void computeBbox() {
165 m_bbox.col(0) = m_vtxs.rowwise().minCoeff();
166 m_bbox.col(1) = m_vtxs.rowwise().maxCoeff();
167 }
168
169 void init(CellInitOptionsType cell_init_options);
170
171 void computeCellDiameter();
172
173 void computeMeasure(bool compute_barycenter);
174 void computeNormals();
175
176 void computeInscribedDiameter();
177 void computeFaceDiameter();
178 void computeFaceKb();
179
180 const auto& vtxs() const { return m_vtxs; }
181 auto& vtxs() { return m_vtxs; }
182 const auto& faces() const { return m_faces; }
183 Eigen::Index nfaces() const { return m_faces.size(); }
184 Eigen::Index nedges() const {
185 assert(m_local_map_built);
186 return m_edges.size();
187 }
188 const auto& normals() const { return m_normals; }
189 const auto& edges() const {
190 assert(m_local_map_built);
191 return m_edges;
192 }
193 const auto& getEdgeToFaces() const {
194 assert(m_local_map_built);
195 return m_edges_to_faces;
196 }
197 const auto& getFaceCenterIdx() const {
198 assert(m_edges_to_compute_built);
199 return m_face_xF_idx;
200 }
201 const auto& getEdgesToCompute() const {
202 assert(m_edges_to_compute_built);
203 return m_edges_to_compute;
204 }
205 const auto& getVtxToFace() const {
206 assert(m_inverse_map_built);
207 return m_vtx_to_faces;
208 }
209 const auto& bbox() const { return m_bbox; }
210 const auto& barycenter() const { return m_barycenter; }
211 auto diameter() const { return m_diameter; }
212 auto measure() const { return m_measure; }
213 auto frontierMeasure() const { return m_frontier_measure; }
214 const auto& faceDiam() const { return m_face_diameter; }
215 const auto& faceKb() const { return m_face_kb; }
216
217 protected:
218 void buildEdgeToFace();
219 void buildInverseMaps();
220 void findEdgesToCompute();
221
222 bool m_local_map_built = false;
223 bool m_inverse_map_built = false;
224 bool m_edges_to_compute_built = false;
225
226 // list of indices defining a face (order must point towards outer normal)
227 datastructure::CrsMap<IndexType, IndexType> m_faces;
228 // pairs of indices defining and edge
229 std::vector<std::array<IndexType, 2>> m_edges;
230 // list of physical coordinates
231 Eigen::Matrix<FloatType, 3, -1> m_vtxs;
232 Eigen::Matrix<FloatType, 3, 2> m_bbox;
233 Eigen::Matrix<FloatType, 3, -1> m_normals;
234 // map each edge to the two faces (indexes) to which the edge belongs to
235 // the order is important, because the two edges are traversed in opposed
236 // ways in order to point towards the outer normal, the first face is the
237 // one where the edges are traversed in increasing order of the indices
238 std::vector<std::array<IndexType, 2>> m_edges_to_faces;
239 // origin of the coordinates system for each face as the index of a vertex
240 std::vector<IndexType> m_face_xF_idx;
241 // list of indices of the edges that necessitates the integrals to be
242 // computed
243 std::vector<IndexType> m_edges_to_compute;
244
245 datastructure::CrsMap<IndexType, IndexType> m_vtx_to_faces;
246 datastructure::CrsMap<IndexType, IndexType> m_vtx_to_edge;
247 datastructure::CrsMap<IndexType, IndexType> m_face_to_edge;
248
249 FloatType m_diameter = 0.0;
250 FloatType m_inscribed_diameter = 0.0;
251 FloatType m_measure = 0.0;
252 FloatType m_frontier_measure = 0.0;
253 Eigen::Vector<FloatType, -1> m_face_measures;
254 Eigen::Vector<FloatType, 3> m_barycenter;
255
256 Eigen::Vector<FloatType, -1> m_face_diameter;
257 Eigen::Vector<FloatType, -1> m_face_kb;
258};
