78enum CellInitOptions : uint32_t {
79 CellInitVtxOnly = 0ULL,
80 CellInitNormals = (1ULL << 0ULL),
81 CellInitLocalMaps = (1ULL << 1ULL),
82 CellInitInverseMaps = (1ULL << 2ULL),
83 CellInitEdgeToCompute = (1ULL << 3ULL),
84 CellInitFaces = (1ULL << 4ULL),
85 CellInitBBox = (1ULL << 5ULL),
86 CellInitDiameter = (1ULL << 6ULL),
87 CellInitMeasure = (1ULL << 7ULL),
88 CellInitBarycenter = (1ULL << 8ULL),
89 CellInitFaceDiameter = (1ULL << 9ULL),
90 CellInitFaceKb = (1ULL << 10ULL),
91 CellInitAll = 0xFFFFFFFFUL,
92};
93
94using CellInitOptionsType = std::underlying_type_t<CellInitOptions>;

96template <typename FloatType, typename IndexType = int32_t>
97class Cell2D {
98 public:
99 Cell2D() = default;
100 Cell2D(
101 const Mesh2D<FloatType, IndexType>& mesh,
102 IndexType cell_id,
103 CellInitOptionsType cell_init_options = CellInitVtxOnly) {
104 const auto cell_to_vtxs = mesh.getCellToVtxMap()[cell_id];
105 m_vtxs.resize(2, cell_to_vtxs.size());
106 for (Eigen::Index loc_id = 0; loc_id < m_vtxs.cols(); ++loc_id) {
107 m_vtxs.col(loc_id) = mesh.getVertexes()[cell_to_vtxs[loc_id]];
108 }
109 if (cell_init_options & CellInitBBox) {
110 computeBbox();
111 }
112 if (cell_init_options & CellInitNormals) {
113 computeNormals();
114 }
115 if (cell_init_options & CellInitDiameter) {
116 computeCellDiameter();
117 }
118 if (cell_init_options & (CellInitMeasure | CellInitBarycenter)) {
119 computeMeasure(cell_init_options & CellInitBarycenter);
120 }
121 if (cell_init_options & CellInitFaceDiameter) {
122 computeFaceDiameter();
123 }
124 if (cell_init_options & CellInitFaceKb) {
125 computeFaceKb();
126 }
127 }
128 void computeBbox() {
129 m_bbox.col(0) = m_vtxs.rowwise().minCoeff();
130 m_bbox.col(1) = m_vtxs.rowwise().maxCoeff();
131 }
132 void computeNormals() {
133 m_normals.resize(m_vtxs.rows(), m_vtxs.cols());
134 const Eigen::Index nvtxs = m_vtxs.cols();
135 for (Eigen::Index i = 0; i < nvtxs; ++i) {
136 const Point<FloatType, 2> v12 =
137 m_vtxs.col(i) - m_vtxs.col((i + 1) % nvtxs);
138 m_normals(0, i) = -v12(1);
139 m_normals(1, i) = v12(0);
140 }
141 }
142 void computeCellDiameter() {
143 for (Eigen::Index i = 0; i < m_vtxs.cols(); ++i) {
144 for (Eigen::Index j = 0; j < i; ++j) {
145 const FloatType dij = (m_vtxs.col(i) - m_vtxs.col(j)).squaredNorm();
146 m_diameter = std::max(m_diameter, dij);
147 }
148 }
149 m_diameter = std::sqrt(m_diameter);
150 }
151 void computeFaceDiameter() {
152 if (m_normals.size()) {
153 m_face_diameter = m_normals.colwise().norm();
154 assert(m_face_diameter.size() == m_normals.cols());
155 } else {
156 m_face_diameter.resize(m_vtxs.rows());
157 const Eigen::Index nvtxs = m_vtxs.cols();
158 for (Eigen::Index i = 0; i < nvtxs; ++i) {
159 m_face_diameter(i) =
160 (m_vtxs.col(i) - m_vtxs.col((i + 1) % nvtxs)).norm();
161 }
162 }
163 }
164 void computeFaceKb() {
165 const Eigen::Index nvtxs = m_vtxs.cols();
166 m_face_kb = Eigen::Vector<FloatType, -1>::Zero(nvtxs);
167 for (Eigen::Index fi = 0; fi < nvtxs; ++fi) {
168 const Point<FloatType, 2> p1 = m_vtxs.col(fi);
169 const Point<FloatType, 2> p2 = m_vtxs.col((fi + 1) % nvtxs);
170 for (Eigen::Index vi = 1; vi < nvtxs - 1; ++vi) {
171 const Point<FloatType, 2> p3 = m_vtxs.col((fi + vi + 1) % nvtxs);
172 const FloatType simplex_area =
173 0.5 * std::abs(
174 p1[0] * (p2[1] - p3[1]) + p2[0] * (p3[1] - p1[1]) +
175 p3[0] * (p1[1] - p2[1]));
176 m_face_kb(fi) = std::max(m_face_kb(fi), simplex_area);
177 }
178 }
179 }
180
181 void computeMeasure(bool compute_barycenter) {
182 const auto [area_ints, edges_ints] =
183 qfree::integrateMonomialsQFree2D<true, FloatType, IndexType>(
184 *this, IndexType(compute_barycenter));
185 m_measure = area_ints(0, 0);
186 assert(m_measure > 0);
187 m_face_measures.resize(m_vtxs.cols());
188 for (IndexType i = 0; i < m_vtxs.cols(); ++i) {
189 m_face_measures[i] = edges_ints(0, 0, i);
190 assert(m_face_measures[i] > 0);
191 }
192 if (compute_barycenter) {
193 m_barycenter[0] = area_ints(1, 0) / m_measure;
194 m_barycenter[1] = area_ints(0, 1) / m_measure;
195 }
196 }
197
198 const auto& vtxs() const { return m_vtxs; }
199 auto& vtxs() { return m_vtxs; }
200 const auto& bbox() const { return m_bbox; }
201 const auto& normals() const { return m_normals; }
202 Eigen::Index nfaces() const { return m_vtxs.cols(); }
203 Eigen::Index nedges() const { return m_vtxs.cols(); }
204 const auto& barycenter() const { return m_barycenter; }
205 auto diameter() const { return m_diameter; }
206 auto measure() const { return m_measure; }
207 const auto& faceDiam() const { return m_face_diameter; }
208 const auto& faceKb() const { return m_face_kb; }
209
210 protected:
211 Eigen::Matrix<FloatType, 2, -1, Eigen::ColMajor> m_vtxs;
212 Eigen::Matrix<FloatType, 2, -1, Eigen::ColMajor> m_normals;
213 Eigen::Matrix<FloatType, 2, 2> m_bbox;
214 FloatType m_diameter = 0.0;
215 FloatType m_measure = 0.0;
216 Eigen::Vector<FloatType, -1> m_face_measures;
217 Eigen::Vector<FloatType, 2> m_barycenter;
218
219 Eigen::Vector<FloatType, -1> m_face_diameter;
220 Eigen::Vector<FloatType, -1> m_face_kb;
221};

223template <typename FloatType, typename IndexType>
224class Cell3D {
225 public:
226 Cell3D() = default;
227
228 Cell3D(
229 const datastructure::CrsMapUncompressed<IndexType>& faces,
230 const std::vector<Point<FloatType, 3>>& coords,
231 CellInitOptionsType cell_init_options = CellInitAll)
232 : m_faces(faces.template compress<IndexType>()) {
233 m_vtxs.resize(3, coords.size());
234 for (std::size_t i = 0; i < coords.size(); ++i) {
235 m_vtxs.col(i) = coords[i];
236 }
237 init(cell_init_options);
238 }
239
240 Cell3D(
241 const Mesh3D<FloatType, IndexType>& mesh,
242 IndexType cell_id,
243 CellInitOptionsType cell_init_options = CellInitVtxOnly) {
244 // copy coordinates
245 boost::unordered_flat_map<IndexType, IndexType> vtx_global_to_local;
246 std::size_t nvtxs = 0;
247 m_vtxs.resize(3, mesh.getCellToVtxMap()[cell_id].size());
248 for (const auto vtx_id : mesh.getCellToVtxMap()[cell_id]) {
249 if (cell_init_options & CellInitFaces) {
250 vtx_global_to_local[vtx_id] = nvtxs;
251 }
252 m_vtxs.col(nvtxs) = mesh.getVertexes()[vtx_id];
253 nvtxs++;
254 }
255 // copy faces
256 if (cell_init_options & CellInitFaces) {
257 datastructure::CrsMapUncompressed<IndexType> uncompressed_faces;
258 for (const auto face_id : mesh.getCellToFacetMap()[cell_id]) {
259 const auto fw_cell = mesh.getFacetToCellMap()[face_id][0];
260 const auto vtx_begin = mesh.getFacetToVtxMap()[face_id].begin();
261 const auto vtx_end = mesh.getFacetToVtxMap()[face_id].end();
262 if (fw_cell == cell_id) {
263 uncompressed_faces.emplace_back(vtx_begin, vtx_end);
264 } else {
265 uncompressed_faces.emplace_back(vtx_end, vtx_begin);
266 }
267 auto& last_face = uncompressed_faces[uncompressed_faces.size() - 1];
268 for (auto it = last_face.begin(); it < last_face.end(); ++it) {
269 *it = vtx_global_to_local[*it];
270 }
271 }
272 m_faces = uncompressed_faces.template compress<IndexType>();
273 }
274 //
275 init(cell_init_options);
276 }
277
278 void computeBbox() {
279 m_bbox.col(0) = m_vtxs.rowwise().minCoeff();
280 m_bbox.col(1) = m_vtxs.rowwise().maxCoeff();
281 }
282
283 void init(CellInitOptionsType cell_init_options) {
284 if (cell_init_options & (CellInitMeasure | CellInitBarycenter)) {
285 cell_init_options |= CellInitLocalMaps;
286 }
287 //
288 if (cell_init_options & CellInitFaceKb) {
289 cell_init_options |= CellInitNormals;
290 }
291 if (cell_init_options & CellInitBBox) {
292 computeBbox();
293 }
294 if (cell_init_options & CellInitNormals) {
295 computeNormals();
296 }
297 if (cell_init_options & CellInitLocalMaps) {
298 buildEdgeToFace();
299 }
300 if (cell_init_options & CellInitInverseMaps) {
301 buildInverseMaps();
302 }
303 if (cell_init_options & CellInitEdgeToCompute) {
304 findEdgesToCompute();
305 }
306 if (cell_init_options & CellInitDiameter) {
307 computeCellDiameter();
308 }
309 if (cell_init_options & (CellInitMeasure | CellInitBarycenter)) {
310 computeMeasure(cell_init_options & CellInitBarycenter);
311 }
312 if (cell_init_options & CellInitFaceDiameter) {
313 computeFaceDiameter();
314 }
315 if (cell_init_options & CellInitFaceKb) {
316 computeFaceKb();
317 }
318 }
319
320 void computeCellDiameter() {
321 for (Eigen::Index i = 0; i < m_vtxs.cols(); ++i) {
322 for (Eigen::Index j = 0; j < i; ++j) {
323 const FloatType dij = (m_vtxs.col(i) - m_vtxs.col(j)).squaredNorm();
324 m_diameter = std::max(m_diameter, dij);
325 }
326 }
327 m_diameter = std::sqrt(m_diameter);
328 }
329
330 void computeMeasure(bool compute_barycenter) {
331 const auto [vol_ints, faces_ints] =
332 qfree::integrateMonomialsQFree3D<true, false, FloatType, IndexType>(
333 *this, IndexType(compute_barycenter));
334 m_measure = vol_ints(0, 0, 0);
335 assert(m_measure > 0);
336 m_face_measures.resize(nfaces());
337 for (Eigen::Index i = 0; i < nfaces(); ++i) {
338 m_face_measures[i] = faces_ints(0, 0, 0, i);
339 assert(m_face_measures[i] > 0);
340 }
341 if (compute_barycenter) {
342 m_barycenter[0] = vol_ints(1, 0, 0) / m_measure;
343 m_barycenter[1] = vol_ints(0, 1, 0) / m_measure;
344 m_barycenter[2] = vol_ints(0, 0, 1) / m_measure;
345 }
346 }
347
348 void computeNormals() {
349 m_normals.resize(3, m_faces.size());
350 const std::size_t nfaces = m_faces.size();
351 for (std::size_t fi = 0; fi < nfaces; ++fi) {
352 const auto p1 = m_vtxs.col(m_faces[fi][0]);
353 const auto p2 = m_vtxs.col(m_faces[fi][1]);
354 const auto p3 = m_vtxs.col(m_faces[fi][2]);
355
356 const Point<FloatType, 3> v1 = p2 - p1;
357 const Point<FloatType, 3> v2 = p3 - p1;
358 m_normals.col(fi) = v1.cross(v2);
359
360#ifndef NDEBUG
361 // compare with SVD
362 Eigen::Map<const Eigen::Vector<IndexType, -1>> face_ids(
363 m_faces[fi].data(), m_faces[fi].size());
364 const Eigen::Matrix<FloatType, 3, -1> face_vtxs =
365 m_vtxs(Eigen::placeholders::all, face_ids);
366 Eigen::Vector<FloatType, 3> centroid = face_vtxs.rowwise().mean();
367 Eigen::Matrix<FloatType, 3, -1> centered_vtxs =
368 face_vtxs.colwise() - centroid;
369 Eigen::JacobiSVD<decltype(centered_vtxs)> svd(
370 centered_vtxs, Eigen::ComputeThinU | Eigen::ComputeThinV);
371 Eigen::Vector<FloatType, 3> normal_svd =
372 svd.matrixU().col(2).normalized();
373 Eigen::Vector<FloatType, 3> normal = m_normals.col(fi).normalized();
374 assert(
375 ((normal_svd - normal).norm() < 1e-12) ||
376 ((normal_svd + normal).norm() < 1e-12));
377#endif
378 }
379 }
380
381 void computeFaceDiameter() {
382 m_face_diameter = decltype(m_face_diameter)::Zero(nfaces());
383 for (IndexType fi = 0; fi < m_faces.size(); ++fi) {
384 for (IndexType i = 0; i < m_faces[fi].size(); ++i) {
385 const Point<FloatType, 3> p0 = m_vtxs.col(m_faces[fi][i]);
386 for (IndexType j = i + 1; j < m_faces[fi].size(); ++j) {
387 const FloatType d = (p0 - m_vtxs.col(m_faces[fi][j])).squaredNorm();
388 m_face_diameter(fi) = std::max(m_face_diameter(fi), d);
389 }
390 m_face_diameter(fi) = std::sqrt(m_face_diameter(fi));
391 }
392 }
393 }
394 void computeFaceKb() {
395 // TODO: the following is wrong if the cell is concave
396 m_face_kb = decltype(m_face_kb)::Zero(nfaces());
397 for (IndexType fi = 0; fi < m_faces.size(); ++fi) {
398 // maximum distance between this face and any vertex
399 FloatType max_dist = 0.0;
400 // find the plane as equation ax + by + cz + d = 0
401 const Point<FloatType, 3> abc = m_normals.col(fi).normalized();
402 const FloatType d = -abc.dot(m_vtxs.col(m_faces[fi][0]));
403 for (IndexType vi = 0; vi < m_vtxs.cols(); ++vi) {
404 const auto dist = std::abs(abc.dot(m_vtxs.col(vi)) + d);
405 max_dist = std::max(max_dist, dist);
406 }
407 assert((m_diameter == 0) || (max_dist <= m_diameter * (1.0 + 1e-12)));
408 // maximum triangle area in the face
409 FloatType max_tria_area = 0.0;
410 const auto nvtxs = m_faces[fi].size();
411 for (Eigen::Index i = 0; i < nvtxs; ++i) {
412 const Point<FloatType, 3> p1 = m_vtxs.col(m_faces[fi][i]);
413 for (Eigen::Index j = i + 1; j < nvtxs; ++j) {
414 const Point<FloatType, 3> v1 = m_vtxs.col(m_faces[fi][j]) - p1;
415 for (Eigen::Index k = j + 1; k < nvtxs; ++k) {
416 const Point<FloatType, 3> v2 = m_vtxs.col(m_faces[fi][k]) - p1;
417 const FloatType area = 0.25 * v1.cross(v2).squaredNorm();
418 max_tria_area = std::max(max_tria_area, area);
419 }
420 }
421 }
422 max_tria_area = std::sqrt(max_tria_area);
423 assert(
424 (m_face_measures.size() == 0) ||
425 (max_tria_area <= m_face_measures(fi) * (1.0 + 1e-10)));
426 // simplex area
427 m_face_kb(fi) = max_dist * max_tria_area / 3.0;
428 }
429 }

476 // list of indices defining a face (order must point towards outer normal)
477 datastructure::CrsMap<IndexType, IndexType> m_faces;
478 // pairs of indices defining and edge
479 std::vector<std::array<IndexType, 2>> m_edges;
480 // list of physical coordinates
481 Eigen::Matrix<FloatType, 3, -1> m_vtxs;

483 Eigen::Matrix<FloatType, 3, -1> m_normals;
484 // map each edge to the two faces (indexes) to which the edge belongs to

487 // one where the edges are traversed in increasing order of the indices
488 std::vector<std::array<IndexType, 2>> m_edges_to_faces;
489 // origin of the coordinates system for each face as the index of a vertex
490 std::vector<IndexType> m_face_xF_idx;
491 // list of indices of the edges that necessitates the integrals to be
492 // computed
493 std::vector<IndexType> m_edges_to_compute;

498

500 FloatType m_measure = 0.0;
501 Eigen::Vector<FloatType, -1> m_face_measures;
502 Eigen::Vector<FloatType, 3> m_barycenter;
503

505 Eigen::Vector<FloatType, -1> m_face_kb;

508template <typename FloatType, typename IndexType>
509void Cell3D<FloatType, IndexType>::buildInverseMaps() {
510 if (m_inverse_map_built) {
511 return;
512 }
513 buildEdgeToFace();
514 datastructure::CrsMapUncompressed<IndexType> uncompressed_vtx_to_faces;
515 uncompressed_vtx_to_faces.resize(m_vtxs.cols());
516 for (IndexType face_id = 0; face_id < m_faces.size(); ++face_id) {
517 for (IndexType vtx_id : m_faces[face_id]) {
518 uncompressed_vtx_to_faces[vtx_id].push_back(face_id);
519 }
520 }
521 m_vtx_to_faces = uncompressed_vtx_to_faces.template compress<IndexType>();
522 //
523 datastructure::CrsMapUncompressed<IndexType> uncompressed_vtx_to_edge;
524 uncompressed_vtx_to_edge.resize(m_vtxs.cols());
525 for (size_t edge_id = 0; edge_id < m_edges.size(); ++edge_id) {
526 for (IndexType vtx_id : m_edges[edge_id]) {
527 uncompressed_vtx_to_edge[vtx_id].push_back(edge_id);
528 }
529 }
530 m_vtx_to_edge = uncompressed_vtx_to_edge.template compress<IndexType>();
531 //
532 datastructure::CrsMapUncompressed<IndexType> uncompressed_face_to_edge;
533 uncompressed_face_to_edge.resize(m_faces.size());
534 for (size_t edge_id = 0; edge_id < m_edges.size(); ++edge_id) {
535 for (IndexType face_id : m_edges_to_faces[edge_id]) {
536 uncompressed_face_to_edge[face_id].push_back(edge_id);
537 }
538 }
539 m_face_to_edge = uncompressed_face_to_edge.template compress<IndexType>();
540 m_inverse_map_built = true;
541}

543template <typename FloatType, typename IndexType>
544void Cell3D<FloatType, IndexType>::buildEdgeToFace() {
545 if (m_local_map_built) {
546 return;
547 }
548 // edges_to_faces
549 std::vector<std::tuple<std::array<IndexType, 2>, IndexType, bool>>
550 face_edge_flat_map;
551 for (IndexType fi = 0; fi < m_faces.size(); ++fi) {
552 for (IndexType i = 0; i < m_faces[fi].size(); ++i) {
553 size_t i1 = (i + 1) % m_faces[fi].size();
554 bool ordered = false;
555 if (m_faces[fi][i] < m_faces[fi][i1]) {
556 m_edges.push_back({m_faces[fi][i], m_faces[fi][i1]});
557 ordered = true;
558 } else {
559 m_edges.push_back({m_faces[fi][i1], m_faces[fi][i]});
560 ordered = false;
561 }
562 face_edge_flat_map.push_back({m_edges.back(), fi, ordered});
563 }
564 }
565
566 std::set<std::array<IndexType, 2>> edges_unq(m_edges.begin(), m_edges.end());
567 m_edges.assign(edges_unq.begin(), edges_unq.end());
568
569 m_edges_to_faces.resize(m_edges.size(), {IndexType(-1), IndexType(-1)});
570 for (size_t i = 0; i < face_edge_flat_map.size(); ++i) {
571 size_t edge_idx = 0;
572 while (m_edges[edge_idx] != std::get<0>(face_edge_flat_map[i])) {
573 edge_idx++;
574 }
575 const bool ordered = std::get<2>(face_edge_flat_map[i]);
576 m_edges_to_faces[edge_idx][1 - static_cast<int>(ordered)] =
577 std::get<1>(face_edge_flat_map[i]);
578 }
579#ifndef NDEBUG
580 for (const auto& edge : m_edges_to_faces) {
581 assert(edge[0] != IndexType(-1));
582 assert(edge[1] != IndexType(-1));
583 }
584#endif
585 m_local_map_built = true;
586}

588template <typename FloatType, typename IndexType>
589void Cell3D<FloatType, IndexType>::findEdgesToCompute() {
590 if (m_edges_to_compute_built) {
591 return;
592 }
593 buildInverseMaps();
594 // we have to find the center for each face, keep track of which faces have
595 // been set
596 std::vector<bool> face_mask(m_faces.size(), true);
597 IndexType remaining_faces = m_faces.size();
598 m_face_xF_idx.resize(m_faces.size(), -1);
599 // buffer to count how many faces are attached to each vertex
600 std::vector<int32_t> vtx_edges_count(m_vtxs.cols());
601 // at each iteration we pick a new center xF for some faces as the vertex of
602 // the polyhedron that has more edges (still) attached to it
603 // status 2 (undecided), status 1 (must compute), status 0 (do not compute)
604 // this heuristic may not give the best outcome but should reduce
605 // significantly the cost of the quadrature-free algo using the procedure of
606 // "Quadrature-free polytopic discontinuous Galerkin methods for transport
607 // problems" TJ Radley, P Houston, ME Hubbard
608 std::vector<int8_t> edges_to_compute_status(m_edges.size(), 2);
609 while (remaining_faces) {
610 std::fill(vtx_edges_count.begin(), vtx_edges_count.end(), 0);
611 for (size_t ei = 0; ei < m_edges.size(); ++ei) {
612 vtx_edges_count[m_edges[ei][0]] += edges_to_compute_status[ei] / 2;
613 vtx_edges_count[m_edges[ei][1]] += edges_to_compute_status[ei] / 2;
614 }
615 // find the vertex with most (undecided) edges attached
616 const auto max_it =
617 std::max_element(vtx_edges_count.begin(), vtx_edges_count.end());
618 // all edges are assigned, nothing left to do
619 if (*max_it == 0) {
620 // give a default center to the unassigned faces
621 for (size_t i = 0; i < m_face_xF_idx.size(); ++i) {
622 if (m_face_xF_idx[i] == static_cast<IndexType>(-1)) {
623 m_face_xF_idx[i] = m_faces[i][0];
624 }
625 }
626 break;
627 }
628 const size_t idx = std::distance(vtx_edges_count.begin(), max_it);
629 assert(idx < vtx_edges_count.size());
630 // remove from the edges_to_compute the edges that have this vertex as an
631 // end
632 for (IndexType fi : m_vtx_to_faces[idx]) {
633 // update center only if not set yet
634 m_face_xF_idx[fi] =
635 (m_face_xF_idx[fi] == static_cast<IndexType>(-1)) * idx;
636 remaining_faces -= face_mask[fi];
637 face_mask[fi] = false;
638 }
639 // update edge status
640 // - if idx \in edge_id and undecided -> may not compute (status 0)
641 for (IndexType edge_id : m_vtx_to_edge[idx]) {
642 edges_to_compute_status[edge_id] = edges_to_compute_status[edge_id] % 2;
643 }
644 // - else if idx \in face and undecided -> must compute (status 1)
645 for (IndexType face_id : m_vtx_to_faces[idx]) {
646 for (IndexType edge_id : m_face_to_edge[face_id]) {
647 const auto status = edges_to_compute_status[edge_id];
648 edges_to_compute_status[edge_id] =
649 (status & int8_t(1)) + (status >> int8_t(1));
650 }
651 }
652 }
653 for (std::size_t ei = 0; ei < edges_to_compute_status.size(); ++ei) {
654 assert(edges_to_compute_status[ei] != 2);
655 if (edges_to_compute_status[ei]) {
656 m_edges_to_compute.push_back(ei);
657 }
658 }
659 m_edges_to_compute_built = true;
660}
