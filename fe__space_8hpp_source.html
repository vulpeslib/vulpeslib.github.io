197inline double conditionNumber(const Eigen::MatrixXd& A) {
198 Eigen::JacobiSVD<Eigen::MatrixXd> svd(A);
199 const auto& singularValues = svd.singularValues();
200 const double condNum =
201 singularValues(0) / singularValues(singularValues.size() - 1);
202 return condNum;
203}

208template <typename FloatType, std::size_t Dim>
209Eigen::Matrix<FloatType, -1, -1> buildVandermonde(
210 const std::vector<Point<FloatType, Dim>>& points) {
211 const auto n = points.size();
212
213 Eigen::Matrix<FloatType, -1, -1> V = decltype(V)::Ones(n, n);
214 for (std::size_t p = 0; p < n; ++p) {
215 const auto& point = points[p];
216 for (std::size_t l = 1; l < n; ++l) {
217 const auto exps = monomial::getMonomial<Dim, std::size_t>(l);
218 for (std::size_t d = 0; d < Dim; ++d) {
219 V(p, l) *= vmath::pow(point[d], exps[d]);
220 }
221 }
222 }
223
224 return V;
225}

230template <typename FloatType>
231Eigen::Matrix<FloatType, -1, -1> computeInverse(
232 const Eigen::Matrix<FloatType, -1, -1>& matrix) {
233 using MatrixType = Eigen::Matrix<FloatType, -1, -1>;
234 // if we have exact rational representation, minimize number of operations
235 if constexpr (std::is_same_v<FloatType, mpq_class>) {
236 Eigen::PartialPivLU<MatrixType> lu_decomp(matrix);
237 return lu_decomp.inverse();
238 } else {
239 Eigen::FullPivLU<MatrixType> lu_decomp(matrix);
240 return lu_decomp.inverse();
241 }
242}

248template <std::size_t Dim, typename FloatType>
249Eigen::Matrix<FloatType, -1, -1> computeLagrangianPolyCoef(
250 const std::vector<Point<FloatType, Dim>>& points) {
251 const auto V = buildVandermonde<FloatType, Dim>(points);
252 return computeInverse(V);
253}
