175inline double conditionNumber(const Eigen::MatrixXd& A) {
176 Eigen::JacobiSVD<Eigen::MatrixXd> svd(A);
177 const auto& singularValues = svd.singularValues();
178 const double condNum =
179 singularValues(0) / singularValues(singularValues.size() - 1);
180 return condNum;
181}

186template <typename FloatType, std::size_t Dim>
187Eigen::Matrix<FloatType, -1, -1> buildVandermonde(
188 const std::vector<Point<FloatType, Dim>>& points) {
189 const auto n = points.size();
190
191 Eigen::Matrix<FloatType, -1, -1> V = decltype(V)::Ones(n, n);
192 for (std::size_t p = 0; p < n; ++p) {
193 const auto& point = points[p];
194 for (std::size_t l = 1; l < n; ++l) {
195 const auto exps = monomial::getMonomial<Dim, std::size_t>(l);
196 for (std::size_t d = 0; d < Dim; ++d) {
197 V(p, l) *= vmath::pow(point[d], exps[d]);
198 }
199 }
200 }
201
202 return V;
203}

208template <typename FloatType>
209Eigen::Matrix<FloatType, -1, -1> computeInverse(
210 const Eigen::Matrix<FloatType, -1, -1>& matrix) {
211 using MatrixType = Eigen::Matrix<FloatType, -1, -1>;
212 // if we have exact rational representation, minimize number of operations
213 if constexpr (std::is_same_v<FloatType, mpq_class>) {
214 Eigen::PartialPivLU<MatrixType> lu_decomp(matrix);
215 return lu_decomp.inverse();
216 } else {
217 Eigen::FullPivLU<MatrixType> lu_decomp(matrix);
218 return lu_decomp.inverse();
219 }
220}

226template <std::size_t Dim, typename FloatType>
227Eigen::Matrix<FloatType, -1, -1> computeLagrangianPolyCoef(
228 const std::vector<Point<FloatType, Dim>>& points) {
229 const auto V = buildVandermonde<FloatType, Dim>(points);
230 return computeInverse(V);
231}
