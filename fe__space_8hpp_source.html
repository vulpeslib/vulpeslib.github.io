194inline double conditionNumber(const Eigen::MatrixXd& A) {
195 Eigen::JacobiSVD<Eigen::MatrixXd> svd(A);
196 const auto& singularValues = svd.singularValues();
197 const double condNum =
198 singularValues(0) / singularValues(singularValues.size() - 1);
199 return condNum;
200}

205template <typename FloatType, std::size_t Dim>
206Eigen::Matrix<FloatType, -1, -1> buildVandermonde(
207 const std::vector<Point<FloatType, Dim>>& points) {
208 const auto n = points.size();
209
210 Eigen::Matrix<FloatType, -1, -1> V = decltype(V)::Ones(n, n);
211 for (std::size_t p = 0; p < n; ++p) {
212 const auto& point = points[p];
213 for (std::size_t l = 1; l < n; ++l) {
214 const auto exps = monomial::getMonomial<Dim, std::size_t>(l);
215 for (std::size_t d = 0; d < Dim; ++d) {
216 V(p, l) *= vmath::pow(point[d], exps[d]);
217 }
218 }
219 }
220
221 return V;
222}

227template <typename FloatType>
228Eigen::Matrix<FloatType, -1, -1> computeInverse(
229 const Eigen::Matrix<FloatType, -1, -1>& matrix) {
230 using MatrixType = Eigen::Matrix<FloatType, -1, -1>;
231 // if we have exact rational representation, minimize number of operations
232 if constexpr (std::is_same_v<FloatType, mpq_class>) {
233 Eigen::PartialPivLU<MatrixType> lu_decomp(matrix);
234 return lu_decomp.inverse();
235 } else {
236 Eigen::FullPivLU<MatrixType> lu_decomp(matrix);
237 return lu_decomp.inverse();
238 }
239}

245template <std::size_t Dim, typename FloatType>
246Eigen::Matrix<FloatType, -1, -1> computeLagrangianPolyCoef(
247 const std::vector<Point<FloatType, Dim>>& points) {
248 const auto V = buildVandermonde<FloatType, Dim>(points);
249 return computeInverse(V);
250}
