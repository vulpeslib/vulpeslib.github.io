36template <typename MeshType, typename FeSpaceType>
37Eigen::MatrixXd projectSolution(
38 const dofs::DofHandler<MeshType>& dof_handler,
39 const assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
40 const la::Vector& solution) {
41 using index_t = MeshType::index_t;
42 using Point = MeshType::SelfPoint;
43
44 const auto& mesh = cell_manager.getMesh();
45 const auto n_cells = mesh.numCells();
46 const auto& vtxs = mesh.getVertexes();
47 const auto& fe_space = cell_manager.getFeSpace();
48 const auto& basis = fe_space.getBasisFast();
49
50 // TODO: manage MPI
51 const auto bs = solution.getBlockSize();
52 const auto sol_data = solution.toEigenMap();
53
54 Eigen::MatrixXd solution_on_vtxs = Eigen::MatrixXd::Zero(bs, vtxs.size());
55 for (index_t cell_id = 0; cell_id < n_cells; ++cell_id) {
56 const auto& cell = cell_manager.getCells()[cell_id];
57 const auto dof_ids = dof_handler.getCellToDof()[cell_id];
58 const auto cell_vtxs = mesh.getCellToVtxMap()[cell_id];
59 for (index_t loc_vtx_id = 0; loc_vtx_id < cell_vtxs.size(); ++loc_vtx_id) {
60 const Point ref_pt = cell->vtxs().col(loc_vtx_id);
61 for (std::size_t fn_id = 0; fn_id < basis.size(); ++fn_id) {
62 const double fn_eval = basis[fn_id].evaluate(ref_pt);
63 for (PetscInt b = 0; b < bs; ++b) {
64 const double coef = sol_data[dof_ids[fn_id] * bs + b];
65 solution_on_vtxs(b, cell_vtxs[loc_vtx_id]) += coef * fn_eval;
66 }
67 }
68 }
69 }
70 // if the space is discontinuous we average on the vertices
71 if (fe_space.getNumDofsForSubdimEntity(0) == 0) {
72 for (std::size_t vtx_id = 0; vtx_id < vtxs.size(); ++vtx_id) {
73 for (PetscInt b = 0; b < bs; ++b) {
74 solution_on_vtxs(b, vtx_id) /= mesh.getVtxToCellMap()[vtx_id].size();
75 }
76 }
77 }
78 return solution_on_vtxs;
79}

81template <typename MeshType, typename FeSpaceType>
82datastructure::CrsMap<double, typename MeshType::index_t>
83projectSolutionSubmesh(
84 const geometry::MeshExporterSubmesh<MeshType>& mesh_exporter,
85 const dofs::DofHandler<MeshType>& dof_handler,
86 const assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
87 const la::Vector& solution) {
88 using index_t = MeshType::index_t;
89 using Point = MeshType::SelfPoint;
90
91 const auto& mesh = cell_manager.getMesh();
92 const auto n_cells = mesh.numCells();
93 const auto& fe_space = cell_manager.getFeSpace();
94 const auto& basis = fe_space.getBasisFast();
95 const auto& vtk_grids = mesh_exporter.getVtkGrids();
96
97 // TODO: manage MPI
98 const auto bs = solution.getBlockSize();
99 const auto sol_data = solution.toEigenMap();
100
101 datastructure::CrsMap<double, index_t> solution_on_vtxs;
102 solution_on_vtxs.indices().push_back(0);
103 std::vector<double> cell_vtx_evals;
104 assert(static_cast<index_t>(vtk_grids.size()) == n_cells);
105 for (index_t cell_id = 0; cell_id < n_cells; ++cell_id) {
106 const auto dof_ids = dof_handler.getCellToDof()[cell_id];
107 auto vtk_grid = vtk_grids[cell_id];
108 vtkPoints* points = vtk_grid->GetPoints();
109 const auto n_loc_vtxs = points->GetNumberOfPoints();
110 cell_vtx_evals.resize(n_loc_vtxs * bs);
111 std::fill(cell_vtx_evals.begin(), cell_vtx_evals.end(), 0.0);
112 for (index_t loc_vtx_id = 0; loc_vtx_id < n_loc_vtxs; ++loc_vtx_id) {
113 static_assert(
114 Point::RowsAtCompileTime > 0 && Point::RowsAtCompileTime <= 3);
115 double point_buffer[3];
116 points->GetPoint(loc_vtx_id, point_buffer);
117 Eigen::Map<Point> ref_pt(point_buffer);
118 cell_manager.getMapping(cell_id)->transformInv_(ref_pt);
119 for (std::size_t fn_id = 0; fn_id < basis.size(); ++fn_id) {
120 const double fn_eval = basis[fn_id].evaluate(ref_pt);
121 for (PetscInt b = 0; b < bs; ++b) {
122 const double coef = sol_data[dof_ids[fn_id] * bs + b];
123 cell_vtx_evals[loc_vtx_id * bs + b] += coef * fn_eval;
124 }
125 }
126 }
127 solution_on_vtxs.pushBack(cell_vtx_evals);
128 }
129 return solution_on_vtxs;
130}
