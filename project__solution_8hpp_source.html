40template <typename MeshType, typename FeSpaceType>
41Eigen::MatrixXd projectSolution(
42 const dofs::DofHandler<MeshType>& dof_handler,
43 const assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
44 const la::Vector& solution) {
45 using index_t = MeshType::index_t;
46 using Point = MeshType::SelfPoint;
47
48 const auto& mesh = cell_manager.getMesh();
49 const auto n_cells = mesh.numCells();
50 const auto& vtxs = mesh.getVertexes();
51 const auto& fe_space = cell_manager.getFeSpace();
52 const auto& basis = fe_space.getBasisFast();
53
54 // TODO: manage MPI
55 const auto bs = solution.getBlockSize();
56 const auto sol_data = solution.toEigenMap();
57
58 Eigen::MatrixXd solution_on_vtxs = Eigen::MatrixXd::Zero(bs, vtxs.size());
59 for (index_t cell_id = 0; cell_id < n_cells; ++cell_id) {
60 const auto& cell = cell_manager.getCells()[cell_id];
61 const auto dof_ids = dof_handler.getCellToDof()[cell_id];
62 const auto cell_vtxs = mesh.getCellToVtxMap()[cell_id];
63 for (index_t loc_vtx_id = 0; loc_vtx_id < cell_vtxs.size(); ++loc_vtx_id) {
64 const Point ref_pt = cell->vtxs().col(loc_vtx_id);
65 for (std::size_t fn_id = 0; fn_id < basis.size(); ++fn_id) {
66 const double fn_eval = basis[fn_id].evaluate(ref_pt);
67 for (PetscInt b = 0; b < bs; ++b) {
68 const double coef = sol_data[dof_ids[fn_id] * bs + b];
69 solution_on_vtxs(b, cell_vtxs[loc_vtx_id]) += coef * fn_eval;
70 }
71 }
72 }
73 }
74 // if the space is discontinuous we average on the vertices
75 if (fe_space.getNumDofsForSubdimEntity(0) == 0) {
76 for (std::size_t vtx_id = 0; vtx_id < vtxs.size(); ++vtx_id) {
77 for (PetscInt b = 0; b < bs; ++b) {
78 solution_on_vtxs(b, vtx_id) /= mesh.getVtxToCellMap()[vtx_id].size();
79 }
80 }
81 }
82 return solution_on_vtxs;
83}

90template <typename MeshType, typename FeSpaceType>
91datastructure::CrsMap<double, typename MeshType::index_t>
92projectSolutionSubmesh(
93 const geometry::MeshExporterSubmesh<MeshType>& mesh_exporter,
94 const dofs::DofHandler<MeshType>& dof_handler,
95 const assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
96 const la::Vector& solution) {
97 using index_t = MeshType::index_t;
98 using Point = MeshType::SelfPoint;
99
100 const auto& mesh = cell_manager.getMesh();
101 const auto n_cells = mesh.numCells();
102 const auto& fe_space = cell_manager.getFeSpace();
103 const auto& basis = fe_space.getBasisFast();
104 const auto& vtk_grids = mesh_exporter.getVtkGrids();
105
106 // TODO: manage MPI
107 const auto bs = solution.getBlockSize();
108 const auto sol_data = solution.toEigenMap();
109
110 datastructure::CrsMap<double, index_t> solution_on_vtxs;
111 solution_on_vtxs.indices().push_back(0);
112 std::vector<double> cell_vtx_evals;
113 assert(static_cast<index_t>(vtk_grids.size()) == n_cells);
114 for (index_t cell_id = 0; cell_id < n_cells; ++cell_id) {
115 const auto dof_ids = dof_handler.getCellToDof()[cell_id];
116 auto vtk_grid = vtk_grids[cell_id];
117 vtkPoints* points = vtk_grid->GetPoints();
118 const auto n_loc_vtxs = points->GetNumberOfPoints();
119 cell_vtx_evals.resize(n_loc_vtxs * bs);
120 std::fill(cell_vtx_evals.begin(), cell_vtx_evals.end(), 0.0);
121 for (index_t loc_vtx_id = 0; loc_vtx_id < n_loc_vtxs; ++loc_vtx_id) {
122 static_assert(
123 Point::RowsAtCompileTime > 0 && Point::RowsAtCompileTime <= 3);
124 double point_buffer[3];
125 points->GetPoint(loc_vtx_id, point_buffer);
126 Eigen::Map<Point> ref_pt(point_buffer);
127 cell_manager.getMapping(cell_id)->transformInv_(ref_pt);
128 for (std::size_t fn_id = 0; fn_id < basis.size(); ++fn_id) {
129 const double fn_eval = basis[fn_id].evaluate(ref_pt);
130 for (PetscInt b = 0; b < bs; ++b) {
131 const double coef = sol_data[dof_ids[fn_id] * bs + b];
132 cell_vtx_evals[loc_vtx_id * bs + b] += coef * fn_eval;
133 }
134 }
135 }
136 solution_on_vtxs.pushBack(cell_vtx_evals);
137 }
138 return solution_on_vtxs;
139}
