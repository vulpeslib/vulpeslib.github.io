122template <std::size_t Dim, typename FloatType>
123QuadratureLibrary<Dim, FloatType>::QRuleRef
124QuadratureLibrary<Dim, FloatType>::getRule(
125 QuadratureElemType elem_type, std::size_t order) {
126 // rule storage
127 static std::map<QuadratureKey, QRuleRef> s_quadrature_library;
128 // lazy build of the rules
129 if constexpr (Dim >= 1 && Dim <= 3) {
130 const auto key = QuadratureKey{elem_type, order};
131 const auto it = s_quadrature_library.find(key);
132 if (it != s_quadrature_library.cend()) {
133 return it->second;
134 } else {
135 const QRuleRef rule =
136 QuadratureLibrary<Dim, FloatType>::makeRule(elem_type, order);
137 s_quadrature_library[key] = rule;
138 return rule;
139 }
140 } else {
141 static_assert(
142 Dim * 0, "Not implemented quadrature rules for Dim > 3 or Dim == 0");
143 }
144};

146template <std::size_t Dim, typename FloatType>
147QuadratureLibrary<Dim, FloatType>::QRuleRef
148QuadratureLibrary<Dim, FloatType>::makeRule(
149 QuadratureElemType elem_type, std::size_t order) {
150 QuadratureRule<mpf_class, Dim> qrule_mpf;
151 switch (elem_type) {
152 case TensorLegendre: {
153 QuadratureRule<mpf_class, 1> qrule_mpf_1d =
154 gmpGaussLegendreQuadrature(order, s_gmp_precision);
155 qrule_mpf = buildTensorQuadrature<Dim>(qrule_mpf_1d);
156 break;
157 }
158 case TensorLobatto: {
159 QuadratureRule<mpf_class, 1> qrule_mpf_1d =
160 gmpGaussLobattoQuadrature(order, s_gmp_precision);
161 qrule_mpf = buildTensorQuadrature<Dim>(qrule_mpf_1d);
162 break;
163 }
164 case Simplex: {
165 assert(Dim > 0);
166 if constexpr (Dim == 1) {
167 qrule_mpf = gmpGaussLegendreQuadrature(order, s_gmp_precision);
168 } else {
169 bool tabled = false;
170 if constexpr (Dim == 2) {
171 if (order <= num_triangle_tabled_rules) {
172 qrule_mpf = getTriangleTabledQuadrature(order);
173 tabled = true;
174 }
175 }
176 if constexpr (Dim == 3) {
177 if (order <= num_tetrahedron_tabled_rules) {
178 qrule_mpf = getTetrahedronTabledQuadrature(order);
179 tabled = true;
180 }
181 }
182 if (!tabled) {
183 const QuadratureRule<mpf_class, Dim + 1> qrule_bary_mpf =
184 mpfGrundammMoellerQuadrature<Dim>(order, s_gmp_precision);
185 qrule_mpf = baricentrycToCartesian(qrule_bary_mpf);
186 }
187 }
188 break;
189 }
190 default: {
191 assert(false);
192 return nullptr;
193 }
194 }
195 // rescale in [0, 1]
196 for (auto& pt : qrule_mpf.nodes) {
197 pt.weight /= (1 << Dim);
198 for (std::size_t d = 0; d < Dim; ++d) {
199 pt.point[d] = pt.point[d] * 0.5 + 0.5;
200 }
201 }
202 return std::make_shared<QRule>(castRule<float_t>(qrule_mpf));
203}
