41template <std::size_t Dim, typename DofPolicy>
42class FeSpaceModalQ : public FeSpaceBase<
43 Dim,
44 internal::real_t,
45 DofPolicy,
46 FeSpaceModalQ<Dim, DofPolicy>> {
47 public:
48 inline static constexpr bool needBasisScaling() { return Dim == 3; }
49
50 using Polynomial =
51 polynomial::SparsePolynomial<Dim, uint8_t, internal::real_t, false>;
52 using PolynomialFast =
53 polynomial::SparsePolynomial<Dim, uint8_t, double, false>;
54 explicit FeSpaceModalQ(std::size_t order)
55 : FeSpaceBase<
56 Dim,
57 internal::real_t,
58 DofPolicy,
59 FeSpaceModalQ<Dim, DofPolicy>>(order) {
60 using Polynomial1D = polynomial::Polynomial<internal::real_t>;
61
62 const auto n_basis_func = vmath::binomial(order + Dim, Dim);
63 assert(
64 (Dim == 1 && n_basis_func == (order + 1)) ||
65 (Dim == 2 && n_basis_func == (order + 1) * (order + 2) / 2) ||
66 (Dim == 3 &&
67 n_basis_func == (order + 1) * (order + 2) * (order + 3) / 6));
68
69 const auto legendre_q =
70 polynomial::buildLegendrePolynomials<mpq_class>(order);
71 // pass from exact representation to approximate one
72 std::vector<Polynomial1D> legendre(legendre_q.size());
73 std::transform(
74 legendre_q.begin(), legendre_q.end(), legendre.begin(),
75 [](const auto& p) { return p.template to<internal::real_t>(); });
76
77 // loop on each element of the cartesian product of 1D polynomials
78 // such that the sum of the 1D degrees is smaller than size
79 for (std::size_t i = 0; i < n_basis_func; ++i) {
80 const std::array<std::size_t, Dim> cart_idx =
81 monomial::getMonomial<Dim, std::size_t>(i);
82 std::array<const Polynomial1D*, Dim> comps_1d;
83 // `1.0 / vmath::pow(2.0, Dim)` has exact double representation
84 internal::real_t normalization = 1.0 / vmath::pow(2.0, Dim);
85 for (std::size_t d = 0; d < Dim; ++d) {
86 comps_1d[d] = &legendre[cart_idx[d]];
87 normalization *= internal::real_t(2 * cart_idx[d] + 1);
88 }
89 // add the cartesian product of elements indicated by
90 // the cartesian index `cart_idx`
91 m_polys.emplace_back(comps_1d);
92 if (Dim == 2) {
93 // we match Lymph implementation but unit norm is not really needed
94 // since it does not hold on the polygon but only on the bbox
95 m_polys.back() *= internal::real_t(::sqrt(normalization));
96 }
97 }
98
99 for (const auto& p : m_polys) {
100 m_polys_fast.push_back(p.template to<double>());
101 }
102 }
103
104 // constraint on cell shape (simplex, tensor, general)
105 inline static constexpr FeType elemType() { //
106 return FeType::Poly;
107 };
108
109 template <typename CellType>
110 inline static constexpr auto getMapping(const CellType& cell) {
111 return geometry::CartesianMapping<double, Dim>{cell.bbox()};
112 };
113
114 inline static constexpr geometry::CellInitOptionsType cellInitOptions() {
115 return geometry::CellInitFaces | geometry::CellInitBBox |
116 geometry::CellInitNormals | geometry::CellInitMeasure |
117 geometry::CellInitFaceDiameter | geometry::CellInitFaceKb |
118 geometry::CellInitDiameter;
119 };
120
121 const auto& getBasis() const { return m_polys; }
122 const auto& getBasisFast() const { return m_polys_fast; }
123
124 protected:
125 std::vector<Polynomial> m_polys;
126 std::vector<PolynomialFast> m_polys_fast;
127};
