36template <std::size_t Dim, typename DofPolicy>
37class FeSpaceModalQ : public FeSpaceBase<
38 Dim,
39 internal::real_t,
40 DofPolicy,
41 FeSpaceModalQ<Dim, DofPolicy>> {
42 public:
43 inline static constexpr bool needBasisScaling() { return Dim == 3; }
44
45 using Polynomial =
46 polynomial::SparsePolynomial<Dim, uint8_t, internal::real_t, false>;
47 using PolynomialFast =
48 polynomial::SparsePolynomial<Dim, uint8_t, double, false>;
49 explicit FeSpaceModalQ(std::size_t order)
50 : FeSpaceBase<
51 Dim,
52 internal::real_t,
53 DofPolicy,
54 FeSpaceModalQ<Dim, DofPolicy>>(order) {
55 using Polynomial1D = polynomial::Polynomial<internal::real_t>;
56
57 const auto n_basis_func = vmath::binomial(order + Dim, Dim);
58 assert(
59 (Dim == 1 && n_basis_func == (order + 1)) ||
60 (Dim == 2 && n_basis_func == (order + 1) * (order + 2) / 2) ||
61 (Dim == 3 &&
62 n_basis_func == (order + 1) * (order + 2) * (order + 3) / 6));
63
64 const auto legendre_q =
65 polynomial::buildLegendrePolynomials<mpq_class>(order);
66 // pass from exact representation to approximate one
67 std::vector<Polynomial1D> legendre(legendre_q.size());
68 std::transform(
69 legendre_q.begin(), legendre_q.end(), legendre.begin(),
70 [](const auto& p) { return p.template to<internal::real_t>(); });
71
72 // loop on each element of the cartesian product of 1D polynomials
73 // such that the sum of the 1D degrees is smaller than size
74 for (std::size_t i = 0; i < n_basis_func; ++i) {
75 const std::array<std::size_t, Dim> cart_idx =
76 monomial::getMonomial<Dim, std::size_t>(i);
77 std::array<const Polynomial1D*, Dim> comps_1d;
78 // `1.0 / vmath::pow(2.0, Dim)` has exact double representation
79 internal::real_t normalization = 1.0 / vmath::pow(2.0, Dim);
80 for (std::size_t d = 0; d < Dim; ++d) {
81 comps_1d[d] = &legendre[cart_idx[d]];
82 normalization *= internal::real_t(2 * cart_idx[d] + 1);
83 }
84 // add the cartesian product of elements indicated by
85 // the cartesian index `cart_idx`
86 m_polys.emplace_back(comps_1d);
87 if (Dim == 2) {
88 // we match Lymph implementation but unit norm is not really needed
89 // since it does not hold on the polygon but only on the bbox
90 m_polys.back() *= internal::real_t(::sqrt(normalization));
91 }
92 }
93
94 for (const auto& p : m_polys) {
95 m_polys_fast.push_back(p.template to<double>());
96 }
97 }
98
99 // constraint on cell shape (simplex, tensor, general)
100 inline static constexpr FeType elemType() { //
101 return FeType::Poly;
102 };
103
104 template <typename CellType>
105 inline static constexpr auto getMapping(const CellType& cell) {
106 return geometry::CartesianMapping<double, Dim>{cell.bbox()};
107 };
108
109 inline static constexpr geometry::CellInitOptionsType cellInitOptions() {
110 return geometry::CellInitFaces | geometry::CellInitBBox |
111 geometry::CellInitNormals | geometry::CellInitMeasure |
112 geometry::CellInitFaceDiameter | geometry::CellInitFaceKb |
113 geometry::CellInitDiameter;
114 };
115
116 const auto& getBasis() const { return m_polys; }
117 const auto& getBasisFast() const { return m_polys_fast; }
118
119 protected:
120 std::vector<Polynomial> m_polys;
121 std::vector<PolynomialFast> m_polys_fast;
122};
