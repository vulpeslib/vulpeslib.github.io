224 MutableRowView getRow(IndexType row_idx) {

335 auto slice(const std::vector<IndexType>& subindices) const {
336 std::decay_t<decltype(*this)> slice_map({}, {});
337 if constexpr (RowSize == IndexType(-1)) {
338 slice_map.m_indices.push_back(0);
339 for (std::size_t i = 0; i < subindices.size(); ++i) {
340 const auto idx = subindices[i];
341 assert(idx + 1 < static_cast<IndexType>(m_indices.size()));
342 const IndexType row_size = m_indices[idx + 1] - m_indices[idx];
343 slice_map.m_indices.push_back(row_size + slice_map.m_indices.back());
344 for (IndexType j = 0; j < row_size; ++j) {
345 slice_map.m_data.push_back(m_data[m_indices[idx] + j]);
346 }
347 }
348 } else {
349 for (std::size_t i = 0; i < subindices.size(); ++i) {
350 const auto idx = subindices[i];
351 for (IndexType j = 0; j < RowSize; ++j) {
352 slice_map.m_data.push_back(m_data[idx + j]);
353 }
354 }
355 }
356 return slice_map;
357 }

367 auto permute(
368 const std::vector<IndexType>& perm,
369 const std::vector<IndexType>& perm_inv) const {
370 assert(perm.size() == perm_inv.size());
371 assert(perm.size() == size());
372 std::decay_t<decltype(*this)> slice_map({}, {});
373 if constexpr (RowSize == IndexType(-1)) {
374 slice_map.m_indices.push_back(0);
375 for (std::size_t i = 0; i < perm_inv.size(); ++i) {
376 const auto idx = perm_inv[i];
377 assert(idx + 1 < m_indices.size());
378 const IndexType row_size = m_indices[idx + 1] - m_indices[idx];
379 slice_map.m_indices.push_back(row_size + slice_map.m_indices.back());
380 for (IndexType j = 0; j < row_size; ++j) {
381 slice_map.m_data.push_back(perm[m_data[m_indices[idx] + j]]);
382 }
383 }
384 } else {
385 static_assert(RowSize * 0 && "NotImplemented");
386 }
387 return slice_map;
388 }

396 void remove(const DataType& value_to_remove) {
397 if (m_indices.size() < 1) {
398 return;
399 }
400 size_t curr = 0;
401 size_t row_start = 0;
402 for (size_t row = 0; row < m_indices.size() - 1; ++row) {
403 size_t row_end = m_indices[row + 1];
404 for (size_t i = row_start; i < row_end; ++i) {
405 if (m_data[i] != value_to_remove) {
406 m_data[curr++] = m_data[i];
407 }
408 }
409 m_indices[row + 1] = curr;
410 row_start = row_end;
411 }
412 m_data.resize(curr);
413 }

422 const auto& indices() const { return m_indices; }
426 auto& data() { return m_data; }
430 auto& indices() { return m_indices; }

432 protected:
434 std::vector<DataType> m_data;
436 std::vector<IndexType> m_indices;

489template <typename T, std::size_t N>
490class SmallVector {
491 public:
492 using iterator = T*;
493 using const_iterator = const T*;
494
505 iterator begin() {
506 if (std::holds_alternative<StaticStorage>(m_storage)) {
507 return std::get<StaticStorage>(m_storage).data();
508 }
509 return std::get<DynamicStorage>(m_storage).data();
510 }
511
521 iterator end() {
522 if (std::holds_alternative<StaticStorage>(m_storage)) {
523 return std::get<StaticStorage>(m_storage).data() + m_size;
524 }
525 return std::get<DynamicStorage>(m_storage).data() + m_size;
526 }
527
538 const_iterator begin() const {
539 if (std::holds_alternative<StaticStorage>(m_storage)) {
540 return std::get<StaticStorage>(m_storage).data();
541 }
542 return std::get<DynamicStorage>(m_storage).data();
543 }
544
555 const_iterator end() const {
556 if (std::holds_alternative<StaticStorage>(m_storage)) {
557 return std::get<StaticStorage>(m_storage).data() + m_size;
558 }
559 return std::get<DynamicStorage>(m_storage).data() + m_size;
560 }
561
563 SmallVector() : m_size(0) {}
564
570 SmallVector(std::initializer_list<T> init_list) : m_size(init_list.size()) {
571 if (m_size <= N) {
572 std::copy(
573 init_list.begin(), init_list.end(),
574 std::get<StaticStorage>(m_storage).begin());
575 } else {
576 m_storage = DynamicStorage(init_list);
577 }
578 }
579
585 void push_back(const T& value) {
586 if (m_size < N) {
587 std::get<StaticStorage>(m_storage)[m_size] = value;
588 } else {
589 // Switch to dynamic storage if we haven't yet
590 if (std::holds_alternative<StaticStorage>(m_storage)) {
591 // Move elements from static storage to dynamic vector
592 std::vector<T> dynamic_storage;
593 dynamic_storage.reserve(m_size + 1);
594 for (std::size_t i = 0; i < m_size; ++i) {
595 dynamic_storage.push_back(std::get<StaticStorage>(m_storage)[i]);
596 }
597 m_storage = std::move(dynamic_storage);
598 }
599 std::get<DynamicStorage>(m_storage).push_back(value);
600 }
601 ++m_size;
602 }
603
614 template <typename... Args>
615 void emplace_back(Args&&... args) {
616 if (m_size < N) {
617 std::get<StaticStorage>(m_storage)[m_size] = T{args...};
618 } else {
619 // Switch to dynamic storage if we haven't yet
620 if (std::holds_alternative<StaticStorage>(m_storage)) {
621 // Move elements from static storage to dynamic vector
622 std::vector<T> dynamic_storage;
623 dynamic_storage.reserve(m_size + 1);
624 for (std::size_t i = 0; i < m_size; ++i) {
625 dynamic_storage.push_back(std::get<StaticStorage>(m_storage)[i]);
626 }
627 m_storage = std::move(dynamic_storage);
628 }
629 std::get<DynamicStorage>(m_storage).emplace_back(
630 std::forward<Args>(args)...);
631 }
632 ++m_size;
633 }
634
642 template <class InputIt>
643 void assign(InputIt first, InputIt last) {
644 m_size = std::distance(first, last);
645 if (m_size <= N) {
646 m_storage.template emplace<StaticStorage>();
647 std::copy(first, last, std::get<StaticStorage>(m_storage).begin());
648 } else {
649 m_storage.template emplace<DynamicStorage>(first, last);
650 }
651 }
652
659 T& operator[](std::size_t index) {
660 if (std::holds_alternative<StaticStorage>(m_storage)) {
661 return std::get<StaticStorage>(m_storage)[index];
662 }
663 return std::get<DynamicStorage>(m_storage)[index];
664 }
665
672 const T& operator[](std::size_t index) const {
673 if (std::holds_alternative<StaticStorage>(m_storage)) {
674 return std::get<StaticStorage>(m_storage)[index];
675 }
676 return std::get<DynamicStorage>(m_storage)[index];
677 }
678
684 constexpr std::size_t size() const { return m_size; }
685
686 private:
687 using StaticStorage = std::array<T, N>;
688 using DynamicStorage = std::vector<T>;
689
691 std::variant<StaticStorage, DynamicStorage> m_storage;
692 std::size_t m_size;
693};
