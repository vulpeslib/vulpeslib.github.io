235 MutableRowView getRow(IndexType row_idx) {
236 if constexpr (RowSize == IndexType(-1)) {
237 assert(row_idx < (IndexType)m_indices.size() - 1);
238 assert(
239 (m_indices[row_idx] < (IndexType)m_data.size()) ||
240 (m_indices[row_idx] == 0 && m_data.size() == 0));
241 assert(m_indices[row_idx + 1] <= (IndexType)m_data.size());
242 return MutableRowView(
243 &m_data[m_indices[row_idx]],
244 m_indices[row_idx + 1] - m_indices[row_idx]);
245 } else {

346 auto slice(const std::vector<IndexType>& subindices) const {
347 std::decay_t<decltype(*this)> slice_map({}, {});
348 if constexpr (RowSize == IndexType(-1)) {
349 slice_map.m_indices.push_back(0);
350 for (std::size_t i = 0; i < subindices.size(); ++i) {
351 const auto idx = subindices[i];
352 assert(idx + 1 < static_cast<IndexType>(m_indices.size()));
353 const IndexType row_size = m_indices[idx + 1] - m_indices[idx];
354 slice_map.m_indices.push_back(row_size + slice_map.m_indices.back());
355 for (IndexType j = 0; j < row_size; ++j) {
356 slice_map.m_data.push_back(m_data[m_indices[idx] + j]);
357 }
358 }
359 } else {
360 slice_map.m_data.resize(m_data.size());
361 for (std::size_t i = 0; i < subindices.size(); ++i) {
362 const auto idx = subindices[i];
363 for (IndexType j = 0; j < RowSize; ++j) {
364 slice_map.m_data[i * RowSize + j] = m_data[idx * RowSize + j];
365 }
366 }
367 }
368 return slice_map;
369 }

379 auto permute(
380 const std::vector<IndexType>& perm,
381 const std::vector<IndexType>& perm_inv) const {
382 assert(perm.size() == perm_inv.size());
383 assert(perm.size() == size());
384 std::decay_t<decltype(*this)> slice_map({}, {});
385 if constexpr (RowSize == IndexType(-1)) {
386 slice_map.m_indices.push_back(0);
387 for (std::size_t i = 0; i < perm_inv.size(); ++i) {
388 const auto idx = perm_inv[i];
389 assert(idx + 1 < m_indices.size());
390 const IndexType row_size = m_indices[idx + 1] - m_indices[idx];
391 slice_map.m_indices.push_back(row_size + slice_map.m_indices.back());
392 for (IndexType j = 0; j < row_size; ++j) {
393 slice_map.m_data.push_back(perm[m_data[m_indices[idx] + j]]);
394 }
395 }
396 } else {
397 static_assert(RowSize * 0 && "NotImplemented");
398 }
399 return slice_map;
400 }

408 void remove(const DataType& value_to_remove) {
409 if (m_indices.size() < 1) {
410 return;
411 }
412 size_t curr = 0;
413 size_t row_start = 0;
414 for (size_t row = 0; row < m_indices.size() - 1; ++row) {
415 size_t row_end = m_indices[row + 1];
416 for (size_t i = row_start; i < row_end; ++i) {
417 if (m_data[i] != value_to_remove) {
418 m_data[curr++] = m_data[i];
419 }
420 }
421 m_indices[row + 1] = curr;
422 row_start = row_end;
423 }
424 m_data.resize(curr);
425 }
426
427 void pushBack(const std::vector<DataType>& row) {
428 m_data.reserve(m_data.size() + row.size());
429 for (const auto& e : row) {
430 m_data.push_back(e);
431 }
432 m_indices.push_back(m_data.size());
433 }
434
438 const auto& data() const { return m_data; }
442 const auto& indices() const { return m_indices; }
446 auto& data() { return m_data; }
450 auto& indices() { return m_indices; }
451
452 protected:
454 std::vector<DataType> m_data;
456 std::vector<IndexType> m_indices;
457};

509template <typename T, std::size_t N>
510class SmallVector {
511 public:
512 using iterator = T*;
513 using const_iterator = const T*;
514
525 iterator begin() {
526 if (std::holds_alternative<StaticStorage>(m_storage)) {
527 return std::get<StaticStorage>(m_storage).data();
528 }
529 return std::get<DynamicStorage>(m_storage).data();
530 }
531
541 iterator end() {
542 if (std::holds_alternative<StaticStorage>(m_storage)) {
543 return std::get<StaticStorage>(m_storage).data() + m_size;
544 }
545 return std::get<DynamicStorage>(m_storage).data() + m_size;
546 }
547
558 const_iterator begin() const {
559 if (std::holds_alternative<StaticStorage>(m_storage)) {
560 return std::get<StaticStorage>(m_storage).data();
561 }
562 return std::get<DynamicStorage>(m_storage).data();
563 }
564
575 const_iterator end() const {
576 if (std::holds_alternative<StaticStorage>(m_storage)) {
577 return std::get<StaticStorage>(m_storage).data() + m_size;
578 }
579 return std::get<DynamicStorage>(m_storage).data() + m_size;
580 }
581
583 SmallVector() : m_size(0) {}
584
590 SmallVector(std::initializer_list<T> init_list) : m_size(init_list.size()) {
591 if (m_size <= N) {
592 std::copy(
593 init_list.begin(), init_list.end(),
594 std::get<StaticStorage>(m_storage).begin());
595 } else {
596 m_storage = DynamicStorage(init_list);
597 }
598 }
599
600 SmallVector(const T* begin, const T* end) {
601 if (end < begin) {
602 m_size = begin - end;
603 if (m_size <= N) {
604 std::copy(
605 std::make_reverse_iterator(begin), std::make_reverse_iterator(end),
606 std::get<StaticStorage>(m_storage).begin());
607 } else {
608 m_storage = DynamicStorage(
609 std::make_reverse_iterator(begin), std::make_reverse_iterator(end));
610 }
611 } else {
612 m_size = end - begin;
613 if (m_size <= N) {
614 std::copy(begin, end, std::get<StaticStorage>(m_storage).begin());
615 } else {
616 m_storage = DynamicStorage(begin, end);
617 }
618 }
619 }
620
626 void push_back(const T& value) {
627 if (m_size < N) {
628 std::get<StaticStorage>(m_storage)[m_size] = value;
629 } else {
630 // Switch to dynamic storage if we haven't yet
631 if (std::holds_alternative<StaticStorage>(m_storage)) {
632 // Move elements from static storage to dynamic vector
633 std::vector<T> dynamic_storage;
634 dynamic_storage.reserve(m_size + 1);
635 for (std::size_t i = 0; i < m_size; ++i) {
636 dynamic_storage.push_back(std::get<StaticStorage>(m_storage)[i]);
637 }
638 m_storage = std::move(dynamic_storage);
639 }
640 std::get<DynamicStorage>(m_storage).push_back(value);
641 }
642 ++m_size;
643 }
644
655 template <typename... Args>
656 void emplace_back(Args&&... args) {
657 if (m_size < N) {
658 std::get<StaticStorage>(m_storage)[m_size] = T{args...};
659 } else {
660 // Switch to dynamic storage if we haven't yet
661 if (std::holds_alternative<StaticStorage>(m_storage)) {
662 // Move elements from static storage to dynamic vector
663 std::vector<T> dynamic_storage;
664 dynamic_storage.reserve(m_size + 1);
665 for (std::size_t i = 0; i < m_size; ++i) {
666 dynamic_storage.push_back(std::get<StaticStorage>(m_storage)[i]);
667 }
668 m_storage = std::move(dynamic_storage);
669 }
670 std::get<DynamicStorage>(m_storage).emplace_back(
671 std::forward<Args>(args)...);
672 }
673 ++m_size;
674 }
675
683 template <class InputIt>
684 void assign(InputIt first, InputIt last) {
685 m_size = std::distance(first, last);
686 if (m_size <= N) {
687 m_storage.template emplace<StaticStorage>();
688 std::copy(first, last, std::get<StaticStorage>(m_storage).begin());
689 } else {
690 m_storage.template emplace<DynamicStorage>(first, last);
691 }
692 }
693
700 T& operator[](std::size_t index) {
701 assert(index < m_size);
702 if (std::holds_alternative<StaticStorage>(m_storage)) {
703 return std::get<StaticStorage>(m_storage)[index];
704 }
705 return std::get<DynamicStorage>(m_storage)[index];
706 }
707
714 const T& operator[](std::size_t index) const {
715 assert(index < m_size);
716 if (std::holds_alternative<StaticStorage>(m_storage)) {
717 return std::get<StaticStorage>(m_storage)[index];
718 }
719 return std::get<DynamicStorage>(m_storage)[index];
720 }
721
727 constexpr std::size_t size() const { return m_size; }
728
729 private:
730 using StaticStorage = std::array<T, N>;
731 using DynamicStorage = std::vector<T>;
732
734 std::variant<StaticStorage, DynamicStorage> m_storage;
735 std::size_t m_size;
736};

865 template <typename IndexType>
866 CrsMap<DataType, IndexType> compress() const {
867 std::vector<DataType> data;
868 std::vector<IndexType> indices = {0};
869 for (size_t i = 0; i < m_data.size(); ++i) {
870 indices.push_back(indices.back() + m_data[i].size());
871 for (const auto& e : m_data[i]) {
872 data.push_back(e);
873 }
