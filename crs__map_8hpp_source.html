234 MutableRowView getRow(IndexType row_idx) {

345 auto slice(const std::vector<IndexType>& subindices) const {
346 std::decay_t<decltype(*this)> slice_map({}, {});
347 if constexpr (RowSize == IndexType(-1)) {
348 slice_map.m_indices.push_back(0);
349 for (std::size_t i = 0; i < subindices.size(); ++i) {
350 const auto idx = subindices[i];
351 assert(idx + 1 < static_cast<IndexType>(m_indices.size()));
352 const IndexType row_size = m_indices[idx + 1] - m_indices[idx];
353 slice_map.m_indices.push_back(row_size + slice_map.m_indices.back());
354 for (IndexType j = 0; j < row_size; ++j) {
355 slice_map.m_data.push_back(m_data[m_indices[idx] + j]);
356 }
357 }
358 } else {
359 slice_map.m_data.resize(RowSize * subindices.size());
360 for (std::size_t i = 0; i < subindices.size(); ++i) {
361 const auto idx = subindices[i];
362 for (IndexType j = 0; j < RowSize; ++j) {
363 slice_map.m_data[i * RowSize + j] = m_data[idx * RowSize + j];
364 }
365 }
366 }
367 return slice_map;
368 }

378 auto permute(
379 const std::vector<IndexType>& perm,
380 const std::vector<IndexType>& perm_inv) const {
381 assert(perm.size() == perm_inv.size());
382 assert(perm.size() == static_cast<std::size_t>(size()));
383 std::decay_t<decltype(*this)> slice_map({}, {});
384 if constexpr (RowSize == IndexType(-1)) {
385 slice_map.m_indices.push_back(0);
386 for (std::size_t i = 0; i < perm_inv.size(); ++i) {
387 const auto idx = perm_inv[i];
388 assert(static_cast<std::size_t>(idx + 1) < m_indices.size());
389 const IndexType row_size = m_indices[idx + 1] - m_indices[idx];
390 slice_map.m_indices.push_back(row_size + slice_map.m_indices.back());
391 for (IndexType j = 0; j < row_size; ++j) {
392 slice_map.m_data.push_back(perm[m_data[m_indices[idx] + j]]);
393 }
394 }
395 } else {
396 static_assert(RowSize * 0 && "NotImplemented");
397 }
398 return slice_map;
399 }

407 void remove(const DataType& value_to_remove) {
408 if (m_indices.size() < 1) {
409 return;
410 }
411 size_t curr = 0;
412 size_t row_start = 0;
413 for (size_t row = 0; row < m_indices.size() - 1; ++row) {
414 size_t row_end = m_indices[row + 1];
415 for (size_t i = row_start; i < row_end; ++i) {
416 if (m_data[i] != value_to_remove) {
417 m_data[curr++] = m_data[i];
418 }
419 }
420 m_indices[row + 1] = curr;
421 row_start = row_end;
422 }
423 m_data.resize(curr);
424 }
425
426 void pushBack(const std::vector<DataType>& row) {
427 m_data.reserve(m_data.size() + row.size());
428 for (const auto& e : row) {
429 m_data.push_back(e);
430 }
431 m_indices.push_back(m_data.size());
432 }

437 const auto& data() const { return m_data; }
441 const auto& indices() const { return m_indices; }
445 auto& data() { return m_data; }
449 auto& indices() { return m_indices; }
450
451 protected:
453 std::vector<DataType> m_data;
455 std::vector<IndexType> m_indices;

508template <typename T, std::size_t N>
509class SmallVector {
510 public:
511 using iterator = T*;
512 using const_iterator = const T*;
513
524 iterator begin() {
525 if (std::holds_alternative<StaticStorage>(m_storage)) {
526 return std::get<StaticStorage>(m_storage).data();
527 }
528 return std::get<DynamicStorage>(m_storage).data();
529 }
530
540 iterator end() {
541 if (std::holds_alternative<StaticStorage>(m_storage)) {
542 return std::get<StaticStorage>(m_storage).data() + m_size;
543 }
544 return std::get<DynamicStorage>(m_storage).data() + m_size;
545 }
546
557 const_iterator begin() const {
558 if (std::holds_alternative<StaticStorage>(m_storage)) {
559 return std::get<StaticStorage>(m_storage).data();
560 }
561 return std::get<DynamicStorage>(m_storage).data();
562 }
563
574 const_iterator end() const {
575 if (std::holds_alternative<StaticStorage>(m_storage)) {
576 return std::get<StaticStorage>(m_storage).data() + m_size;
577 }
578 return std::get<DynamicStorage>(m_storage).data() + m_size;
579 }
580
582 SmallVector() : m_size(0) {}
583
589 SmallVector(std::initializer_list<T> init_list) : m_size(init_list.size()) {
590 if (m_size <= N) {
591 std::copy(
592 init_list.begin(), init_list.end(),
593 std::get<StaticStorage>(m_storage).begin());
594 } else {
595 m_storage = DynamicStorage(init_list);
596 }
597 }
598
599 SmallVector(const T* begin, const T* end) {
600 if (end < begin) {
601 m_size = begin - end;
602 if (m_size <= N) {
603 std::copy(
604 std::make_reverse_iterator(begin), std::make_reverse_iterator(end),
605 std::get<StaticStorage>(m_storage).begin());
606 } else {
607 m_storage = DynamicStorage(
608 std::make_reverse_iterator(begin), std::make_reverse_iterator(end));
609 }
610 } else {
611 m_size = end - begin;
612 if (m_size <= N) {
613 std::copy(begin, end, std::get<StaticStorage>(m_storage).begin());
614 } else {
615 m_storage = DynamicStorage(begin, end);
616 }
617 }
618 }
619
625 void push_back(const T& value) {
626 if (m_size < N) {
627 std::get<StaticStorage>(m_storage)[m_size] = value;
628 } else {
629 // Switch to dynamic storage if we haven't yet
630 if (std::holds_alternative<StaticStorage>(m_storage)) {
631 // Move elements from static storage to dynamic vector
632 std::vector<T> dynamic_storage;
633 dynamic_storage.reserve(m_size + 1);
634 for (std::size_t i = 0; i < m_size; ++i) {
635 dynamic_storage.push_back(std::get<StaticStorage>(m_storage)[i]);
636 }
637 m_storage = std::move(dynamic_storage);
638 }
639 std::get<DynamicStorage>(m_storage).push_back(value);
640 }
641 ++m_size;
642 }
643
654 template <typename... Args>
655 void emplace_back(Args&&... args) {
656 if (m_size < N) {
657 std::get<StaticStorage>(m_storage)[m_size] = T{args...};
658 } else {
659 // Switch to dynamic storage if we haven't yet
660 if (std::holds_alternative<StaticStorage>(m_storage)) {
661 // Move elements from static storage to dynamic vector
662 std::vector<T> dynamic_storage;
663 dynamic_storage.reserve(m_size + 1);
664 for (std::size_t i = 0; i < m_size; ++i) {
665 dynamic_storage.push_back(std::get<StaticStorage>(m_storage)[i]);
666 }
667 m_storage = std::move(dynamic_storage);
668 }
669 std::get<DynamicStorage>(m_storage).emplace_back(
670 std::forward<Args>(args)...);
671 }
672 ++m_size;
673 }
674
682 template <class InputIt>
683 void assign(InputIt first, InputIt last) {
684 m_size = std::distance(first, last);
685 if (m_size <= N) {
686 m_storage.template emplace<StaticStorage>();
687 std::copy(first, last, std::get<StaticStorage>(m_storage).begin());
688 } else {
689 m_storage.template emplace<DynamicStorage>(first, last);
690 }
691 }
692
699 T& operator[](std::size_t index) {
700 assert(index < m_size);
701 if (std::holds_alternative<StaticStorage>(m_storage)) {
702 return std::get<StaticStorage>(m_storage)[index];
703 }
704 return std::get<DynamicStorage>(m_storage)[index];
705 }
706
713 const T& operator[](std::size_t index) const {
714 assert(index < m_size);
715 if (std::holds_alternative<StaticStorage>(m_storage)) {
716 return std::get<StaticStorage>(m_storage)[index];
717 }
718 return std::get<DynamicStorage>(m_storage)[index];
719 }
720
726 constexpr std::size_t size() const { return m_size; }
727
728 private:
729 using StaticStorage = std::array<T, N>;
730 using DynamicStorage = std::vector<T>;
731
733 std::variant<StaticStorage, DynamicStorage> m_storage;
734 std::size_t m_size;
735};
