195 template <double PNorm, bool Periodic = false>
196 std::pair<Eigen::MatrixXd, Eigen::Matrix<int64_t, -1, -1>> queryKnn(
197 const double* points,
198 int64_t n_points,
199 int64_t k,
200 double eps = 0,
201 double distance_upper_bound = std::numeric_limits<double>::max()) const {
202 assert(m_is_built);
203 // each column is a point
204 Eigen::MatrixXd distances(k, n_points);
205 Eigen::Matrix<int64_t, -1, -1> indices(k, n_points);
206
207 std::vector<int64_t> ks(k);
208 std::iota(ks.begin(), ks.end(), 1);
209
210 scipy::query_knn<Dim, PNorm, Periodic>(
211 &m_tree, distances.data(), indices.data(), points, n_points, ks.data(),
212 k, k, eps, distance_upper_bound);
213
214 return {distances, indices};
215 }

368 void buildRun(bool median, bool compact) {
369 double maxs[Dim];
370 double mins[Dim];
371
372 m_tree.raw_data = m_raw_data;
373 m_tree.tree_buffer = &m_tree_buffer;
374 m_tree.leafsize = m_leafsize;
375 m_tree.n = m_raw_data_size / Dim;
376 m_tree.raw_maxes = maxs;
377 m_tree.raw_mins = mins;
378 m_tree.raw_indices = m_indices.data();
379 scipy::build_ckdtree(
380 &m_tree, 0, m_tree.n, m_tree.raw_maxes, m_tree.raw_mins, median,
381 compact);
382 m_tree.ctree = m_tree_buffer.data();
383
384 compututeBounds();
385 m_tree.raw_maxes = m_maxs.data();
386 m_tree.raw_mins = m_mins.data();
387 postInitTraverse(m_tree.ctree, m_tree.ctree);
388 m_is_built = true;
389 }

403 void postInitTraverse(scipy::ckdtreenode* root, scipy::ckdtreenode* node) {
404 if (node->split_dim == -1) {
405 node->less = nullptr;
406 node->greater = nullptr;
407 } else {
408 // NOLINTBEGIN(cppcoreguidelines-pro-type-reinterpret-cast)
409 using int_t = scipy::ckdtree_intp_t;
410 node->less = root + reinterpret_cast<int_t>(node->less);
411 node->greater = root + reinterpret_cast<int_t>(node->greater);
412 // NOLINTEND(cppcoreguidelines-pro-type-reinterpret-cast)
413 postInitTraverse(root, node->less);
414 postInitTraverse(root, node->greater);
415 }
416 }

424 void compututeBounds() {
425 if (m_raw_data_size) {
426 for (int64_t i = 0; i < Dim; ++i) {
427 m_maxs[i] = m_raw_data[i];
428 m_mins[i] = m_raw_data[i];
429 }
430 const int64_t size = m_raw_data_size / Dim;
431 for (int64_t i = 1; i < size; ++i) {
432 for (int64_t d = 0; d < Dim; ++d) {
433 m_maxs[d] = std::max(m_maxs[d], m_raw_data[i * Dim + d]);
434 m_mins[d] = std::min(m_mins[d], m_raw_data[i * Dim + d]);
435 }
436 }
437 }
438 }

454
455 // Maximum values in each dimension.
456 std::array<double, Dim> m_maxs{{-std::numeric_limits<double>::max()}};
457 // Minimum values in each dimension.
458 std::array<double, Dim> m_mins{{std::numeric_limits<double>::max()}};
459 // Bounding box size for periodic boundary conditions.
460 std::array<double, 2 * Dim> m_boxsize = {0};
461 // The underlying SciPy KDTree structure.
462 scipy::ckdtree<Dim> m_tree;
