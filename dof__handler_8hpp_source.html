437template <typename MeshType>
438template <
439 std::size_t Dim,
440 typename FloatType,
441 typename DofPolicy,
442 typename FeSpace>
443void DofHandler<MeshType>::buildCellToCellDof(
444 const fe::FeSpaceBase<Dim, FloatType, DofPolicy, FeSpace>& fe_space) {
445 // copy the sparsity pattern from the one of `m_cell_to_dof`
446 m_cell_to_cell_dof = m_cell_to_dof;
447 // cell_id is the local id in MPI numeration
448 for (index_t cell_id = 0; cell_id < m_cell_to_cell_dof.size(); ++cell_id) {
449 for (auto& dof_id : m_cell_to_cell_dof[cell_id]) {
450 // expand information of dof_id by expoiting global (geometric)
451 // enumeration
452 index_t global_dof_id = dof_id;
453 if constexpr (MeshType::parallel()) {
454 global_dof_id = m_local_to_global_dof[dof_id];
455 }
456 index_t enty_dim = 0;
457 while (!(global_dof_id < m_glob_cum_ndof_per_entity[enty_dim + 1])) {
458 enty_dim++;
459 }
460 const index_t entity_dof_id =
461 global_dof_id - m_glob_cum_ndof_per_entity[enty_dim];
462 const index_t n_dofs_per_entity =
463 fe_space.getNumDofsForSubdimEntity(enty_dim);
464 const index_t owning_enty_id = entity_dof_id / n_dofs_per_entity;
465 const index_t enty_multiplicity = entity_dof_id % n_dofs_per_entity;
466 // pass the expanded information to the fe space and update dof_id
467 // from global to local numbering (cell based, not MPI partition based)
468 dof_id = fe_space.dofToCellDof(
469 m_mesh, cell_id, owning_enty_id, enty_dim, enty_multiplicity);
470 }
471 }

474template <typename MeshType>
475template <
476 std::size_t Dim,
477 typename FloatType,
478 typename DofPolicy,
479 typename FeSpace>
480void DofHandler<MeshType>::computeNumDofs(
481 const fe::FeSpaceBase<Dim, FloatType, DofPolicy, FeSpace>& fe_space) {
482 for (std::size_t d = 0; d < FeSpace::dim + 1; ++d) {
483 const auto n_dofs_per_entity = fe_space.getNumDofsForSubdimEntity(d);
484 const auto n_glob_entities = m_mesh.nEntity(d);
485 auto n_loc_entities = m_mesh.nLocEntity(d);
486 if constexpr (MeshType::parallel()) {
487 n_loc_entities += m_mesh.nGhostEntity(d);
488 }
489 m_glob_cum_ndof_per_entity[d + 1] =
490 m_glob_cum_ndof_per_entity[d] + n_dofs_per_entity * n_glob_entities;
491 m_loc_cum_ndof_per_entity[d + 1] =
492 m_loc_cum_ndof_per_entity[d] + n_dofs_per_entity * n_loc_entities;
493 }
494}

496template <typename MeshType>
497template <
498 std::size_t SubentityDim,
499 std::size_t Dim,
500 typename FloatType,
501 typename DofPolicy,
502 typename FeSpace>
503void DofHandler<MeshType>::distributeSubentityDofs(
504 const fe::FeSpaceBase<Dim, FloatType, DofPolicy, FeSpace>& fe_space,
505 auto& dof_to_cell) {
506 const index_t n_dofs_per_entity =
507 fe_space.getNumDofsForSubdimEntity(SubentityDim);
508 if (n_dofs_per_entity > 0) {
509 index_t n_loc_entity = m_mesh.nLocEntity(SubentityDim);
510 if constexpr (MeshType::parallel()) {
511 n_loc_entity += m_mesh.nGhostEntity(SubentityDim);
512 }
513 // for each entity in the mesh, may be vertex, edge, facet, cell
514 for (index_t enty_id = 0; enty_id < n_loc_entity; ++enty_id) {
515 // for each dof that belongs to this entity
516 for (index_t j = 0; j < n_dofs_per_entity; ++j) {
517 // compute global dof id
518 index_t global_enty_id = enty_id;
519 if constexpr (MeshType::parallel()) {
520 global_enty_id = m_mesh.localToGlobal(SubentityDim)[enty_id];
521 }
522 const auto glob_dof_id = m_glob_cum_ndof_per_entity[SubentityDim] +
523 global_enty_id * n_dofs_per_entity + j;
524 auto loc_dof_id = glob_dof_id;
525 if constexpr (MeshType::parallel()) {
526 loc_dof_id = m_local_to_global_dof.size();
527 m_local_to_global_dof.push_back(glob_dof_id);
528 m_global_to_local_dof[glob_dof_id] = loc_dof_id;
529 }
530 // put the `dof_id` in the `dof_to_cell map`
531 // different behavior if we are in a cell or not
532 if constexpr (SubentityDim == Dim) {
533 // we are in a cell, we know to which cell this dof belongs to
534 dof_to_cell[loc_dof_id].push_back(global_enty_id);
535 // in parallel we must keep track of owned and shared dofs
536 if constexpr (MeshType::parallel()) {
537 // we own the cell dof if we own the cell
538 if (enty_id < m_mesh.nLocEntity(SubentityDim)) {
539 m_owned_dofs.push_back(glob_dof_id);
540 m_local_owned_dofs.push_back(loc_dof_id);
541 } else {
542 m_shared_dofs.push_back(glob_dof_id);
543 m_local_shared_dofs.push_back(loc_dof_id);
544 }
545 }
546 } else {
547 const auto& enty_to_cell_map =
548 m_mesh.template getEntityToCellGlobalMap<SubentityDim>();
549 // loop over all the cells to which this entity belongs to
550 for (const auto cell_id : enty_to_cell_map[enty_id]) {
551 // facet mappings always have two elements, one of which may be -1
552 // in this case we skip it
553 if constexpr (SubentityDim == Dim - 1) {
554 if (cell_id == static_cast<decltype(cell_id)>(-1)) {
555 continue;
556 }
557 }
558 dof_to_cell[loc_dof_id].push_back(cell_id);
559 // in parallel we must keep track of owned and shared dofs
560 if constexpr (MeshType::parallel()) {
561 // if the dof belongs to an entity that is on a ghost cell
562 // than it is a shared dof
563 const auto local_cell_id = m_mesh.globalToLocal(Dim).at(cell_id);
564 if ((local_cell_id >= m_mesh.nLocEntity(Dim)) &&
565 (m_shared_dofs.empty() ||
566 m_shared_dofs.back() != glob_dof_id)) {
567 m_shared_dofs.push_back(glob_dof_id);
568 m_local_shared_dofs.push_back(loc_dof_id);
569 }
570 }
571 }
572 // if the dof is not shared, than it is owned
573 if constexpr (MeshType::parallel()) {
574 if (!(m_shared_dofs.size() &&
575 (m_shared_dofs.back() == glob_dof_id))) {
576 m_owned_dofs.push_back(glob_dof_id);
577 m_local_owned_dofs.push_back(loc_dof_id);
578 }
579 }
580 }
581 }
582 }
583 }
584}

607template <typename MeshType>
608void DofHandler<MeshType>::sortSharedDofs() {
609 std::vector<index_t> idxs(m_shared_dofs.size());
610 std::iota(idxs.begin(), idxs.end(), 0);
611 std::sort(idxs.begin(), idxs.end(), [this](auto a, auto b) {
612 return m_shared_dofs[a] < m_shared_dofs[b];
613 });
614 // apply sorting permutation
615 std::vector<index_t> tmp(m_shared_dofs.size());
616 for (std::size_t i = 0; i < idxs.size(); ++i) {
617 tmp[i] = m_shared_dofs[idxs[i]];
618 }
619 std::swap(tmp, m_shared_dofs);
620 for (std::size_t i = 0; i < idxs.size(); ++i) {
621 tmp[i] = m_local_shared_dofs[idxs[i]];
622 }
623 std::swap(tmp, m_local_shared_dofs);
624}

626template <typename MeshType>
627void DofHandler<MeshType>::invertDofToCellMappingAndCompress(
628 auto& dof_to_cell) {
629 constexpr std::size_t static_buffer_size =
630 vmath::pow((index_t)7, MeshType::self_dim);
631 datastructure::CrsMapUncompressed<index_t, static_buffer_size> cell_to_dof;
632 auto n_loc_cell = m_mesh.nLocEntity(MeshType::self_dim);
633 if constexpr (MeshType::parallel()) {
634 n_loc_cell += m_mesh.nGhostEntity(MeshType::self_dim);
635 }
636 cell_to_dof.resize(n_loc_cell);
637 for (std::size_t dof_id = 0; dof_id < dof_to_cell.size(); ++dof_id) {
638 // we exploit the fact that the dofs are sorted later
639 // when we compute the sparsity pattern
640 // TODO: is this already sorted if in serial?
641 std::sort(dof_to_cell[dof_id].begin(), dof_to_cell[dof_id].end());
642 for (const auto global_cell_id : dof_to_cell[dof_id]) {
643 auto loc_cell_id = global_cell_id;
644 if constexpr (MeshType::parallel()) {
645 loc_cell_id =
646 m_mesh.globalToLocal(MeshType::self_dim).at(global_cell_id);
647 }
648 cell_to_dof[loc_cell_id].push_back(dof_id);
649 }
650 }
651
652 m_dof_to_cell = dof_to_cell.template compress<index_t>();
653 m_cell_to_dof = cell_to_dof.template compress<index_t>();
654}

656template <typename MeshType>
657void DofHandler<MeshType>::computeOwnedCellDofsAndBndDofs() {
658 const auto max_dof_idx =
659 m_cell_to_dof.indices()[m_mesh.nLocEntity(MeshType::self_dim)];
660 const auto& data = m_cell_to_dof.data();
661 boost::unordered_flat_set<index_t> owned_cells_dof(
662 data.begin(), data.begin() + max_dof_idx);
663 m_owned_cells_dofs.assign(owned_cells_dof.begin(), owned_cells_dof.end());
664
665 // find which dofs lie on the partition
666 // i.e. they are both shared and belong to owned cells
667 for (auto loc_dof_id : m_owned_cells_dofs) {
668 const auto glob_dof_id = m_local_to_global_dof[loc_dof_id];
669 const auto it = std::lower_bound(
670 m_shared_dofs.begin(), m_shared_dofs.end(), glob_dof_id);
671 if ((it != m_shared_dofs.end()) && (*it == glob_dof_id)) {
672 m_partition_bnd_dofs.push_back(glob_dof_id);
673 }
674 }
675 std::sort(m_partition_bnd_dofs.begin(), m_partition_bnd_dofs.end());
676}

678template <typename MeshType>
679template <
680 std::size_t Dim,
681 typename FloatType,
682 typename DofPolicy,
683 typename FeSpace>
684void DofHandler<MeshType>::distributeDofs(
685 const fe::FeSpaceBase<Dim, FloatType, DofPolicy, FeSpace>& fe_space) {
686 // smallest row size that make sense
687 // namely we fill all the space we have in the `SmallVector`
688 constexpr std::size_t static_row_size =
689 sizeof(std::vector<index_t>) / sizeof(index_t);
690 datastructure::CrsMapUncompressed<index_t, static_row_size> dof_to_cell;
691 static_assert(FeSpace::dim == MeshType::self_dim);
692
693 computeNumDofs(fe_space);
694 dof_to_cell.resize(m_loc_cum_ndof_per_entity[FeSpace::dim + 1]);
695 PROFILE_INTERVAL(distributeAllSubentityDofs(fe_space, dof_to_cell));
696
697 if (MeshType::parallel()) {
698 // we sort them because it is easier for rank 0 to use them
699 // when computing the global renumbering
700 PROFILE_INTERVAL(sortSharedDofs());
701 }
702
703 PROFILE_INTERVAL(invertDofToCellMappingAndCompress(dof_to_cell));
704
705 PROFILE_INTERVAL(buildCellToCellDof(fe_space));
706
707 if constexpr (MeshType::parallel()) {
708 PROFILE_INTERVAL(computeOwnedCellDofsAndBndDofs());
709 }
710
711 PROFILE_INTERVAL(computeSparsityPattern<FeSpace>());
712
713 if constexpr (MeshType::parallel()) {
714 m_all_dof_offset.resize(mpi::size() + 1, 0);
715 const auto total_dofs = m_glob_cum_ndof_per_entity.back();
716 for (int p = 0; p < mpi::size(); ++p) {
717 const auto reminder = (p < (total_dofs % mpi::size())) ? 1 : 0;
718 m_all_dof_offset[p + 1] =
719 m_all_dof_offset[p] + total_dofs / mpi::size() + reminder;
720 }
721 }
722}

724template <typename MeshType>
725template <bool DofInteractionDistance>
726std::pair<typename MeshType::index_t, typename MeshType::index_t>
727DofHandler<MeshType>::computeSparsityForDof(
728 typename MeshType::index_t dof_id,
729 typename MeshType::index_t* buffer1,
730 typename MeshType::index_t* buffer2) {
731 static std::array<std::size_t, 2> buffers_size = {0, 0};
732 static std::size_t active_buffer = 0;
733 buffers_size = {0, 0};
734 active_buffer = 0;
735 std::array<index_t*, 2> buffers = {buffer1, buffer2};
736
737 // merge all the dofs that belong to this cell (`dofs_view`) and the
738 // others (`buffer[active_buffer]`) into the inactive buffer (`buffer[1 -
739 // active_buffer]`). We assume the dofs are sorted and we remove
740 // duplicates
741 const auto updateBuffer = [&](auto dofs_view) {
742 const index_t* f1 = dofs_view.data();
743 const index_t* l1 = f1 + dofs_view.size();
744 const index_t* f2 = buffers[active_buffer];
745 const index_t* l2 = f2 + buffers_size[active_buffer];
746 index_t* o = buffers[1 - active_buffer];
747 // WARNING: this is not thread-safe
748 buffers_size[1 - active_buffer] =
749 std::distance(o, std::set_union(f1, l1, f2, l2, o));
750 // at each iteration we swap the source buffer with all the dofs and
751 // the destination buffer where we make the union, to avoid an extra
752 // copy
753 buffers_size[active_buffer] = 0;
754 active_buffer = 1 - active_buffer;
755 };
756
757 // for each cell to which this dof belongs to
758 for (index_t global_cell_id : m_dof_to_cell[dof_id]) {
759 auto local_cell_id = global_cell_id;
760 if constexpr (MeshType::parallel()) {
761 local_cell_id =
762 m_mesh.globalToLocal(MeshType::self_dim).at(global_cell_id);
763 }
764 updateBuffer(m_cell_to_dof[local_cell_id]);
765 // if there is interaction with also the neighbours
766 // e.g. if we are in DG each dof belongs to one cell but there
767 // are face couplings
768 if constexpr (DofInteractionDistance) {
769 for (index_t loc_neigh_cell_id : m_mesh.getDualMap()[local_cell_id]) {
770 updateBuffer(m_cell_to_dof[loc_neigh_cell_id]);
771 }
772 }
773 }
774
775 return {active_buffer, buffers_size[active_buffer]};
776}

778template <typename MeshType>
779template <typename FeSpace>
780void DofHandler<MeshType>::computeSparsityPattern() {
781 static constexpr std::size_t n_max_dofs_per_row = 1048576;
782 static std::array<std::array<index_t, n_max_dofs_per_row>, 2> buffers;
783
784 // Probe a good static size for the uncompressed map.
785 // We compute the sparsity pattern for some random dof ids
786 // and check which is the number of nnz at that row.
787 // We then ask for a static size that is greater than the average
788 constexpr index_t probe_size = 1000;
789 index_t nnz = 16;
790 if (m_dof_to_cell.size() > probe_size) {
791 nnz = 0;
792 std::mt19937 gen(0);
793 std::uniform_int_distribution<uint64_t> dis(0, m_dof_to_cell.size() - 1);
794 for (index_t i = 0; i < probe_size; ++i) {
795 const auto random_dof_id = dis(gen);
796 const auto [active_buffer, buffer_size] =
797 computeSparsityForDof<FeSpace::dofsInteractionDistance()>(
798 random_dof_id, buffers[0].data(), buffers[1].data());
799 nnz += buffer_size;
800 }
801 nnz /= probe_size;
802 }
803 std::cout << "Sparsity pattern probing suggested size: " << nnz << std::endl;
804 auto uncompressed_sparsity_pattern =
805 datastructure::getUncompressedCrsMapWithStaticStorageGreaterThan<index_t>(
806 nnz);
807 std::visit(
808 [&](auto& map) { map.resize(m_dof_to_cell.size()); },
809 uncompressed_sparsity_pattern);
810
811 // compute the sparsity pattern for all the dofs
812 for (index_t dof_id = 0; dof_id < m_dof_to_cell.size(); ++dof_id) {
813 const auto [active_buffer, buffer_size] =
814 computeSparsityForDof<FeSpace::dofsInteractionDistance()>(
815 dof_id, buffers[0].data(), buffers[1].data());
816 const auto nnz_dofid = buffer_size;
817 // copy the sparsity pattern from the static buffer to the uncompressed map
818 std::visit(
819 [dof_id, nnz_dofid, active_buffer](auto& map) {
820 map[dof_id].assign(
821 buffers[active_buffer].data(),
822 buffers[active_buffer].data() + nnz_dofid);
823 },
824 uncompressed_sparsity_pattern);
825 }
826
827 // compress the sparsity pattern
828 m_sparsity_pattern = std::visit(
829 [](const auto& map) { return map.template compress<index_t>(); },
830 uncompressed_sparsity_pattern);
831}

833template <typename MeshType>
834void DofHandler<MeshType>::renumberMpiSubdomainWise() {
835 std::cout << "Call: renumberMpiSubdomainWise" << std::endl;
836 m_has_renumbering = true;
837 const auto size = mpi::size();
838 // collect the size of owned dofs and the list of shared dofs
839 std::vector<std::size_t> cum_dof_offset(size + 1, 0);
840 m_all_shared_dof.resize(size);
841 m_all_partition_bnd_dofs.resize(size);
842
843 if (mpi::root()) {
844 cum_dof_offset[1] = m_owned_dofs.size();
845 m_all_shared_dof[0] = m_shared_dofs;
846 m_all_partition_bnd_dofs[0] = m_partition_bnd_dofs;
847 for (int p = 1; p < size; ++p) {
848 const auto owned_dof_p = mpi::recv<std::size_t>(p);
849 cum_dof_offset[p + 1] = cum_dof_offset[p] + owned_dof_p;
850 mpi::recv(m_all_shared_dof[p], p);
851 mpi::recv(m_all_partition_bnd_dofs[p], p);
852 }
853 } else {
854 mpi::send(m_owned_dofs.size(), 0);
855 mpi::send(m_shared_dofs, 0);
856 mpi::send(m_partition_bnd_dofs, 0);
857 }
858 // root renumbers the shared dofs and sends them
859 std::vector<index_t> shared_renumbering;
860 m_dof_offset = 0;
861 if (mpi::root()) {
862 // all the shared dofs without duplicates
863 std::vector<index_t> unq_shared_dof;
864 // list of boundary shared dofs "belonging" to proc p
865 // that is a bnd dof is owned by the proc with smallest rank that shares it
866 std::vector<index_t> bnd_dof_belonging_to_p;
867 // global-global old to new numbering
868 m_all_old_to_new_numbering.resize(m_glob_cum_ndof_per_entity.back(), -1);
869 std::vector<index_t> tmp_buffer;
870
871 for (int p = 0; p < size - 1; ++p) {
872 // to be numbered, they belong to p
873 std::set_difference(
874 m_all_partition_bnd_dofs[p].begin(),
875 m_all_partition_bnd_dofs[p].end(), unq_shared_dof.begin(),
876 unq_shared_dof.end(), std::back_inserter(bnd_dof_belonging_to_p));
877 // compute the dof offset taking into account the shared between 0 and p
878 for (int i = p; i < size; ++i) {
879 cum_dof_offset[i + 1] += bnd_dof_belonging_to_p.size();
880 }
881 mpi::send(cum_dof_offset[p + 1], p + 1);
882 const auto offset_without_shared =
883 cum_dof_offset[p + 1] - bnd_dof_belonging_to_p.size();
884 for (std::size_t i = 0; i < bnd_dof_belonging_to_p.size(); ++i) {
885 m_all_old_to_new_numbering[bnd_dof_belonging_to_p[i]] =
886 offset_without_shared + i;
887 }
888 // update unq_shared_dof
889 std::set_union(
890 unq_shared_dof.begin(), unq_shared_dof.end(),
891 m_all_partition_bnd_dofs[p].begin(),
892 m_all_partition_bnd_dofs[p].end(), std::back_inserter(tmp_buffer));
893 std::swap(unq_shared_dof, tmp_buffer);
894 bnd_dof_belonging_to_p.resize(0);
895 tmp_buffer.resize(0);
896 }
897
898 // renumber owning according to offset in sequence for self
899 m_old_to_new_numbering.resize(m_local_to_global_dof.size());
900 for (std::size_t i = 0; i < m_owned_dofs.size(); ++i) {
901 m_old_to_new_numbering[m_global_to_local_dof.at(m_owned_dofs[i])] = i;
902 m_all_old_to_new_numbering[m_owned_dofs[i]] = i;
903 }
904
905 // recive enumeration from all
906 std::vector<index_t> tmp_own_global_old;
907 std::vector<index_t> tmp_own_global_new;
908 for (int p = 1; p < size; ++p) {
909 mpi::recv(tmp_own_global_old, p);
910 mpi::recv(tmp_own_global_new, p);
911 for (std::size_t i = 0; i < tmp_own_global_old.size(); ++i) {
912 m_all_old_to_new_numbering[tmp_own_global_old[i]] =
913 tmp_own_global_new[i];
914 }
915 }
916
917 // find the enumeration for each proc
918 for (int p = 1; p < size; ++p) {
919 shared_renumbering.resize(m_all_shared_dof[p].size());
920 for (std::size_t i = 0; i < m_all_shared_dof[p].size(); ++i) {
921 shared_renumbering[i] =
922 m_all_old_to_new_numbering[m_all_shared_dof[p][i]];
923 }
924 mpi::send(shared_renumbering, p);
925 }
926 // find the enumeration for self
927 shared_renumbering.resize(m_all_shared_dof[0].size());
928 for (std::size_t i = 0; i < m_all_shared_dof[0].size(); ++i) {
929 shared_renumbering[i] =
930 m_all_old_to_new_numbering[m_all_shared_dof[0][i]];
931 }
932 } else {
933 m_dof_offset = mpi::recv<std::size_t>(0);
934 // renumber according to offset in sequence
935 std::vector<index_t> own_global_old;
936 std::vector<index_t> own_global_new;
937 m_old_to_new_numbering.resize(m_local_to_global_dof.size());
938 for (std::size_t i = 0; i < m_owned_dofs.size(); ++i) {
939 const auto glob_own_id = m_owned_dofs[i];
940 const auto loc_own_id = m_global_to_local_dof.at(glob_own_id);
941 const index_t new_glob_id = i + m_dof_offset;
942 m_old_to_new_numbering[loc_own_id] = new_glob_id;
943 own_global_old.push_back(glob_own_id);
944 own_global_new.push_back(new_glob_id);
945 }
946 mpi::send(own_global_old, 0);
947 mpi::send(own_global_new, 0);
948 mpi::recv(shared_renumbering, 0);
949 }
950 // broadcast the offset
951 m_all_dof_offset.resize(mpi::size() + 1);
952 m_all_dof_offset.back() = m_glob_cum_ndof_per_entity.back();
953 for (int p = 0; p < mpi::size(); ++p) {
954 m_all_dof_offset[p] = m_dof_offset;
955 mpi::bcast(m_all_dof_offset[p], p);
956 }
957 // apply the shared dof renumbering
958 assert(shared_renumbering.size() == m_shared_dofs.size());
959 for (std::size_t i = 0; i < m_shared_dofs.size(); ++i) {
960 m_old_to_new_numbering[m_local_shared_dofs[i]] = shared_renumbering[i];
961 }
962 // invert mapping
963 m_new_to_old_numbering.clear();
964 for (std::size_t i = 0; i < m_old_to_new_numbering.size(); ++i) {
965 m_new_to_old_numbering[m_old_to_new_numbering[i]] = i;
966 }
967}

969template <typename MeshType>
970void DofHandler<MeshType>::applyRenumbering(
971 const RenumberingFunctor<typename MeshType::index_t>& renumbering_func,
972 bool reversed) {
973 m_has_renumbering = true;
974 assert(m_old_to_new_numbering.size() == m_new_to_old_numbering.size());
975 using SparsityPatternType = decltype(m_sparsity_pattern);
976 SparsityPatternType* sparsity_pattern_ptr = &m_sparsity_pattern;
977 std::shared_ptr<SparsityPatternType> owned_sparsity_pattern;
978 if (MeshType::parallel()) {
979 owned_sparsity_pattern =
980 std::make_shared<SparsityPatternType>(computeOwnedSparsityPattern());
981 sparsity_pattern_ptr = owned_sparsity_pattern.get();
982 }
983 const auto [perm, perm_inv] =
984 renumbering_func(*sparsity_pattern_ptr, reversed);
985 assert(perm.size() == perm_inv.size());
986 // apply perm and perm_inv to mappings
987 if constexpr (MeshType::parallel()) {
988 const auto size = mpi::size();
989 // update the old to new for owned dofs
990 std::vector<index_t> new_owned_dof_id(m_owned_dofs.size());
991 {
992 std::vector<index_t> tmp = m_old_to_new_numbering;
993 for (std::size_t i = 0; i < perm.size(); ++i) {
994 tmp[m_local_owned_dofs[i]] =
995 m_old_to_new_numbering[m_local_owned_dofs[perm[i]]];
996 new_owned_dof_id[i] =
997 m_old_to_new_numbering[m_local_owned_dofs[perm[i]]];
998 }
999 m_old_to_new_numbering = std::move(tmp);
1000 }
1001
1002 std::vector<index_t> new_shared_numbering;
1003 if (mpi::root()) {
1004 // get the new numbering of the dofs owned by each proc
1005 std::vector<index_t> owned_dofs_p = m_owned_dofs;
1006 for (int p = 0; p < size; ++p) {
1007 if (p > 0) {
1008 mpi::recv(owned_dofs_p, p);
1009 mpi::recv(new_owned_dof_id, p);
1010 }
1011 assert(new_owned_dof_id.size() == owned_dofs_p.size());
1012 for (std::size_t i = 0; i < new_owned_dof_id.size(); ++i) {
1013 m_all_old_to_new_numbering[owned_dofs_p[i]] = new_owned_dof_id[i];
1014 }
1015 }
1016
1017 // send new numbering of shared
1018 for (int p = 1; p < size; ++p) {
1019 new_shared_numbering.resize(m_all_shared_dof[p].size());
1020 for (std::size_t i = 0; i < m_all_shared_dof[p].size(); ++i) {
1021 new_shared_numbering[i] =
1022 m_all_old_to_new_numbering[m_all_shared_dof[p][i]];
1023 }
1024 mpi::send(new_shared_numbering, p);
1025 }
1026 // do the same for self
1027 new_shared_numbering.resize(m_shared_dofs.size());
1028 for (std::size_t i = 0; i < m_shared_dofs.size(); ++i) {
1029 new_shared_numbering[i] = m_all_old_to_new_numbering[m_shared_dofs[i]];
1030 }
1031 } else {
1032 mpi::send(m_owned_dofs, 0);
1033 // new glob ids for m_owned_dofs
1034 mpi::send(new_owned_dof_id, 0);
1035 // new enumeration for shared
1036 mpi::recv(new_shared_numbering, 0);
1037 }
1038 // apply new enumeration for shared
1039 assert(m_local_shared_dofs.size() == new_shared_numbering.size());
1040 for (std::size_t i = 0; i < m_local_shared_dofs.size(); ++i) {
1041 m_old_to_new_numbering[m_local_shared_dofs[i]] = new_shared_numbering[i];
1042 }
1043 // invert the old_to_new
1044 m_new_to_old_numbering.clear();
1045 for (std::size_t i = 0; i < m_old_to_new_numbering.size(); ++i) {
1046 m_new_to_old_numbering[m_old_to_new_numbering[i]] = i;
1047 }
1048 } else {
1049 if (m_old_to_new_numbering.size() != perm.size()) {
1050 m_old_to_new_numbering = std::move(perm);
1051 m_new_to_old_numbering = std::move(perm_inv);
1052 } else {
1053 for (std::size_t i = 0; i < m_old_to_new_numbering.size(); ++i) {
1054 m_old_to_new_numbering[i] = perm[m_old_to_new_numbering[i]];
1055 m_new_to_old_numbering[i] = perm_inv[m_new_to_old_numbering[i]];
1056 }
1057 }
1058 }

1061template <typename MeshType>
1062auto DofHandler<MeshType>::computeOwnedSparsityPattern() {
1063 if (MeshType::parallel()) {
1064 // build inverse map (from the local id of an owned dof to the owned id)
1065 const auto n_local_dofs = m_owned_dofs.size() + m_shared_dofs.size();
1066 std::vector<index_t> loc_owned_dofs_inv(n_local_dofs, -1);
1067 for (std::size_t i = 0; i < m_local_owned_dofs.size(); ++i) {
1068 loc_owned_dofs_inv[m_local_owned_dofs[i]] = i;
1069 }
1070 //
1071 std::decay_t<decltype(m_sparsity_pattern)> owned_map({}, {});
1072 const auto& indices = m_sparsity_pattern.indices();
1073 const auto& data = m_sparsity_pattern.data();
1074 owned_map.indices().push_back(0);
1075 for (const auto own_loc_dof_id : m_local_owned_dofs) {
1076 const auto row_start = indices[own_loc_dof_id];
1077 const index_t row_size = indices[own_loc_dof_id + 1] - row_start;
1078 index_t real_size = 0;
1079 for (index_t j = 0; j < row_size; ++j) {
1080 const index_t new_glob_id = data[row_start + j];
1081 const index_t loc_id = m_new_to_old_numbering[new_glob_id];
1082 const index_t old_glob_id = m_local_to_global_dof[loc_id];
1083 const auto it = std::lower_bound(
1084 m_shared_dofs.begin(), m_shared_dofs.end(), old_glob_id);
1085 // if it is not shared, than it is owned
1086 // shared are many less, thus it is quicker than to search in owned
1087 if ((it == m_shared_dofs.end()) || (*it != old_glob_id)) {
1088 owned_map.data().push_back(loc_owned_dofs_inv[loc_id]);
1089 real_size++;
1090 }
1091 }
1092 owned_map.indices().push_back(real_size + owned_map.indices().back());
1093 }
1094 return owned_map;
1095 } else {
1096 return m_sparsity_pattern;
1097 }
1098}

1100template <typename MeshType>
1101void DofHandler<MeshType>::finalize() {
1102 if constexpr (MeshType::parallel()) {
1103 // the parallel renumbering maintains the local numeration of rows
1104 if (m_has_renumbering) {
1105 for (auto& col : m_sparsity_pattern.data()) {
1106 col = m_old_to_new_numbering[col];
1107 }
1108 } else {
1109 for (auto& col : m_sparsity_pattern.data()) {
1110 col = m_local_to_global_dof[col];
1111 }
1112 }
1113 } else {
1114 if (m_has_renumbering) {
1115 m_sparsity_pattern = m_sparsity_pattern.permute(
1116 m_old_to_new_numbering, m_new_to_old_numbering);
1117 }
1118 }
1119
1120 // apply the cell-to-celldof mapping (to account for basis function numbering
1121 // in non-discontinous elements)
1122 assert(m_cell_to_cell_dof.size() == m_cell_to_dof.size());
1123 std::vector<index_t> tmp_buffer;
1124 for (index_t cell_id = 0; cell_id < m_cell_to_dof.size(); ++cell_id) {
1125 assert(m_cell_to_dof[cell_id].size() == m_cell_to_cell_dof[cell_id].size());
1126 const auto row_size = m_cell_to_dof[cell_id].size();
1127 if (tmp_buffer.size() < static_cast<std::size_t>(row_size)) {
1128 tmp_buffer.resize(row_size);
1129 }
1130 for (index_t i = 0; i < row_size; ++i) {
1131 tmp_buffer[m_cell_to_cell_dof[cell_id][i]] = m_cell_to_dof[cell_id][i];
1132 }
1133 std::copy(
1134 tmp_buffer.begin(), tmp_buffer.begin() + row_size,
1135 m_cell_to_dof[cell_id].begin());
1136 }
1137
1138 // apply the local to global to cell-to-dof
1139 if (m_has_renumbering) {
1140 for (auto& loc_dof : m_cell_to_dof.data()) {
1141 loc_dof = m_old_to_new_numbering[loc_dof];
1142 }
1143 } else {
1144 if constexpr (MeshType::parallel()) {
1145 for (auto& loc_dof : m_cell_to_dof.data()) {
1146 loc_dof = m_local_to_global_dof[loc_dof];
1147 }
1148 }
1149 }

1152template <typename MeshType>
1153auto DofHandler<MeshType>::gatherSparsityPatternRank0() const {
1154 using SpPatternType = datastructure::CrsMap<index_t, index_t>;
1155 using LocToNewMap = std::vector<index_t>;
1156 // recv from ranks the sp-pattern and the loc-to-new
1157 std::vector<SpPatternType> sp_patterns(mpi::size());
1158 std::vector<LocToNewMap> loc_to_new_maps(mpi::size());
1159 std::vector<std::vector<index_t>> all_owned_dofs(mpi::size());
1160 const auto& map =
1161 m_has_renumbering ? m_old_to_new_numbering : m_local_to_global_dof;
1162 if (mpi::root()) {
1163 sp_patterns[0] = m_sparsity_pattern;
1164 loc_to_new_maps[0] = map;
1165 all_owned_dofs[0] = m_owned_cells_dofs;
1166 for (int p = 1; p < mpi::size(); ++p) {
1167 mpi::recv(sp_patterns[p], p);
1168 mpi::recv(loc_to_new_maps[p], p);
1169 mpi::recv(all_owned_dofs[p], p);
1170 }
1171 } else {
1172 mpi::send(m_sparsity_pattern, 0);
1173 mpi::send(map, 0);
1174 mpi::send(m_owned_cells_dofs, 0);
1175 }
1176
1177 // for each proc; for each row; if empty -> add; else -> check eq;
1178 datastructure::CrsMapUncompressed<index_t, 128> global_sp_pattern;
1179 if (mpi::root()) {
1180 for (int p = 0; p < mpi::size(); ++p) {
1181 for (index_t loc_dof_id = 0; loc_dof_id < sp_patterns[p].size();
1182 ++loc_dof_id) {
1183 const auto glob_dof_id = loc_to_new_maps[p][loc_dof_id];
1184 auto& glob_row = global_sp_pattern.getRow(glob_dof_id);
1185 const auto loc_row = sp_patterns[p][loc_dof_id];
1186 std::sort(loc_row.begin(), loc_row.end());
1187
1188 // the sparsity pattern computed is complete (correct) only
1189 // if the dof belongs to a cell the proc owns.
1190 if (std::find(
1191 all_owned_dofs[p].begin(), all_owned_dofs[p].end(),
1192 loc_dof_id) != all_owned_dofs[p].end()) {
1193 if (glob_row.size()) {
1194 assert(glob_row.size() == static_cast<std::size_t>(loc_row.size()));
1195 assert(
1196 std::equal(glob_row.begin(), glob_row.end(), loc_row.begin()));
1197 } else {
1198 glob_row.assign(loc_row.begin(), loc_row.end());
1199 }
1200 }
1201 }
1202 }
1203 }
1204 return global_sp_pattern.template compress<index_t>();
1205}
