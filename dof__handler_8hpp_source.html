47template <typename FloatType>
48struct Statistics {
49 FloatType mean;
50 FloatType median;
51 FloatType min;
52 FloatType max;
53 FloatType std_dev;
54};
55
56template <typename FloatType>
57std::ostream& operator<<(std::ostream& os, const Statistics<FloatType>& stats) {
58 os << "Min: " << stats.min << "\n"
59 << "Median: " << stats.median << "\n"
60 << "Max: " << stats.max << "\n"
61 << "Mean: " << stats.mean << "\n"
62 << "StdDev: " << stats.std_dev << "\n";
63 return os;
64}

66template <typename FloatType>
67Statistics<FloatType> calculateStatistics(std::vector<FloatType> data) {
68 constexpr FloatType nan = std::numeric_limits<FloatType>::quiet_NaN();
69 Statistics<FloatType> stats = {
70 .mean = nan, .median = nan, .min = nan, .max = nan, .std_dev = nan};
71
72 if (data.empty()) {
73 return stats;
74 }
75
76 FloatType sum = 0.0;
77 stats.min = data[0];
78 stats.max = data[0];
79
80 // Calculate mean, min, and max in one pass
81 for (const FloatType& value : data) {
82 sum += value;
83 if (value < stats.min) {
84 stats.min = value;
85 }
86 if (value > stats.max) {
87 stats.max = value;
88 }
89 }
90
91 stats.mean = sum / data.size();
92
93 // Find the median (need copy to change data order)
94 size_t size = data.size();
95 std::nth_element(data.begin(), data.begin() + size / 2, data.end());
96
97 if (size % 2 == 0) {
98 std::nth_element(data.begin(), data.begin() + size / 2 - 1, data.end());
99 stats.median = (data[size / 2] + data[size / 2 - 1]) / 2.0;
100 } else {
101 stats.median = data[size / 2];
102 }
103
104 // Calculate standard deviation
105 FloatType variance_sum = 0.0;
106 for (const FloatType& value : data) {
107 variance_sum += (value - stats.mean) * (value - stats.mean);
108 }
109 stats.std_dev = std::sqrt(variance_sum / data.size());
110
111 return stats;
112}

435template <
436 std::size_t Dim,
437 typename FloatType,
438 typename DofPolicy,
439 typename FeSpace>
440void DofHandler<MeshType>::buildCellToCellDof(
441 const fe::FeSpaceBase<Dim, FloatType, DofPolicy, FeSpace>& fe_space) {
442 m_cell_to_cell_dof = m_cell_to_dof;
443 // cell_id is the local id in MPI numeration
444 for (index_t cell_id = 0; cell_id < m_cell_to_cell_dof.size(); ++cell_id) {
445 for (auto& dof_id : m_cell_to_cell_dof[cell_id]) {
446 // expand information of dof_id by expoiting global (geometric)
447 // enumeration
448 index_t enty_dim = 0;
449 while (!(dof_id < m_glob_cum_ndof_per_entity[enty_dim + 1])) {
450 enty_dim++;
451 assert(
452 enty_dim < static_cast<index_t>(m_glob_cum_ndof_per_entity.size()));
453 }
454 const index_t entity_dof_id =
455 dof_id - m_glob_cum_ndof_per_entity[enty_dim];
456 const index_t n_dofs_per_entity =
457 fe_space.getNumDofsForSubdimEntity(enty_dim);
458 const index_t owning_enty_id = entity_dof_id / n_dofs_per_entity;
459 const index_t enty_multiplicity = entity_dof_id % n_dofs_per_entity;
460 // pass the expanded information to the fe space and update dof_id
461 // from global to local numbering (cell based, not MPI partition based)
462 dof_id = fe_space.dofToCellDof(
463 m_mesh, cell_id, owning_enty_id, enty_dim, enty_multiplicity);
464 }
465 }

469template <
470 std::size_t Dim,
471 typename FloatType,
472 typename DofPolicy,
473 typename FeSpace>
474void DofHandler<MeshType>::computeNumDofs(
475 const fe::FeSpaceBase<Dim, FloatType, DofPolicy, FeSpace>& fe_space) {
476 for (std::size_t d = 0; d < FeSpace::dim + 1; ++d) {
477 const auto n_dofs_per_entity = fe_space.getNumDofsForSubdimEntity(d);
478 const auto n_glob_entities = m_mesh.nEntity(d);
479 auto n_loc_entities = m_mesh.nLocEntity(d);
480 if constexpr (MeshType::parallel()) {
481 n_loc_entities += m_mesh.nGhostEntity(d);
482 }
483 m_glob_cum_ndof_per_entity[d + 1] =
484 m_glob_cum_ndof_per_entity[d] + n_dofs_per_entity * n_glob_entities;
485 m_loc_cum_ndof_per_entity[d + 1] =
486 m_loc_cum_ndof_per_entity[d] + n_dofs_per_entity * n_loc_entities;
487 }
488}

490template <typename MeshType>
491template <
492 std::size_t SubentityDim,
493 std::size_t Dim,
494 typename FloatType,
495 typename DofPolicy,
496 typename FeSpace>
497void DofHandler<MeshType>::distributeSubentityDofs(
498 const fe::FeSpaceBase<Dim, FloatType, DofPolicy, FeSpace>& fe_space,
499 auto& dof_to_cell) {
500 const index_t n_dofs_per_entity =
501 fe_space.getNumDofsForSubdimEntity(SubentityDim);
502 if (n_dofs_per_entity > 0) {
503 index_t n_loc_entity = m_mesh.nLocEntity(SubentityDim);
504 if constexpr (MeshType::parallel()) {
505 n_loc_entity += m_mesh.nGhostEntity(SubentityDim);
506 }
507 // for each entity in the mesh, may be vertex, edge, facet, cell
508 for (index_t enty_id = 0; enty_id < n_loc_entity; ++enty_id) {
509 // for each dof that belongs to this entity
510 for (index_t j = 0; j < n_dofs_per_entity; ++j) {
511 // compute global dof id
512 index_t global_enty_id = enty_id;
513 if constexpr (MeshType::parallel()) {
514 global_enty_id = m_mesh.localToGlobal(SubentityDim)[enty_id];
515 }
516 const auto glob_dof_id = m_glob_cum_ndof_per_entity[SubentityDim] +
517 global_enty_id * n_dofs_per_entity + j;
518 auto loc_dof_id = glob_dof_id;
519 if constexpr (MeshType::parallel()) {
520 loc_dof_id = m_local_to_global_dof.size();
521 m_local_to_global_dof.push_back(glob_dof_id);
522 m_global_to_local_dof[glob_dof_id] = loc_dof_id;
523 }
524 // put the `dof_id` in the `dof_to_cell map`
525 // different behavior if we are in a cell or not
526 if constexpr (SubentityDim == Dim) {
527 // we are in a cell, we know to which cell this dof belongs to
528 dof_to_cell[loc_dof_id].push_back(global_enty_id);
529 // in parallel we must keep track of owned and shared dofs
530 if constexpr (MeshType::parallel()) {
531 // we own the cell dof if we own the cell
532 if (enty_id < m_mesh.nLocEntity(SubentityDim)) {
533 m_owned_dofs.push_back(glob_dof_id);
534 m_local_owned_dofs.push_back(loc_dof_id);
535 } else {
536 m_shared_dofs.push_back(glob_dof_id);
537 m_local_shared_dofs.push_back(loc_dof_id);
538 }
539 }
540 } else {
541 const auto& enty_to_cell_map =
542 m_mesh.template getEntityToCellMap<SubentityDim>();
543 // loop over all the cells to which this entity belongs to
544 for (const auto cell_id : enty_to_cell_map[enty_id]) {
545 // facet mappings always have two elements, one of which may be -1
546 // in this case we skip it
547 if constexpr (SubentityDim == Dim - 1) {
548 if (cell_id == static_cast<decltype(cell_id)>(-1)) {
549 continue;
550 }
551 }
552 dof_to_cell[loc_dof_id].push_back(cell_id);
553 // in parallel we must keep track of owned and shared dofs
554 if constexpr (MeshType::parallel()) {
555 // if the dof belongs to an entity that is on a ghost cell
556 // than it is a shared dof
557 const auto local_cell_id = m_mesh.globalToLocal(Dim).at(cell_id);
558 if ((local_cell_id >= m_mesh.nLocEntity(Dim)) &&
559 (m_shared_dofs.empty() ||
560 m_shared_dofs.back() != glob_dof_id)) {
561 m_shared_dofs.push_back(glob_dof_id);
562 m_local_shared_dofs.push_back(loc_dof_id);
563 }
564 }
565 }
566 // if the dof is not shared, than it is owned
567 if constexpr (MeshType::parallel()) {
568 if (!(m_shared_dofs.size() &&
569 (m_shared_dofs.back() == glob_dof_id))) {
570 m_owned_dofs.push_back(glob_dof_id);
571 m_local_owned_dofs.push_back(loc_dof_id);
572 }
573 }
574 }
575 }
576 }
577 }
578}

601template <typename MeshType>
602void DofHandler<MeshType>::sortSharedDofs() {
603 std::vector<index_t> idxs(m_shared_dofs.size());
604 std::iota(idxs.begin(), idxs.end(), 0);
605 std::sort(idxs.begin(), idxs.end(), [this](auto a, auto b) {
606 return m_shared_dofs[a] < m_shared_dofs[b];
607 });
608 // apply sorting permutation
609 std::vector<index_t> tmp(m_shared_dofs.size());
610 for (std::size_t i = 0; i < idxs.size(); ++i) {
611 tmp[i] = m_shared_dofs[idxs[i]];
612 }
613 std::swap(tmp, m_shared_dofs);
614 for (std::size_t i = 0; i < idxs.size(); ++i) {
615 tmp[i] = m_local_shared_dofs[idxs[i]];
616 }
617 std::swap(tmp, m_local_shared_dofs);
618}

620template <typename MeshType>
621void DofHandler<MeshType>::invertDofToCellMappingAndCompress(
622 auto& dof_to_cell) {
623 constexpr std::size_t static_buffer_size =
624 vmath::pow((index_t)7, MeshType::self_dim);
625 datastructure::CrsMapUncompressed<index_t, static_buffer_size> cell_to_dof;
626 auto n_loc_cell = m_mesh.nLocEntity(MeshType::self_dim);
627 if constexpr (MeshType::parallel()) {
628 n_loc_cell += m_mesh.nGhostEntity(MeshType::self_dim);
629 }
630 cell_to_dof.resize(n_loc_cell);
631 for (std::size_t dof_id = 0; dof_id < dof_to_cell.size(); ++dof_id) {
632 // we exploit the fact that the dofs are sorted later
633 // when we compute the sparsity pattern
634 // TODO: is this already sorted if in serial?
635 std::sort(dof_to_cell[dof_id].begin(), dof_to_cell[dof_id].end());
636 for (const auto global_cell_id : dof_to_cell[dof_id]) {
637 auto loc_cell_id = global_cell_id;
638 if constexpr (MeshType::parallel()) {
639 loc_cell_id =
640 m_mesh.globalToLocal(MeshType::self_dim).at(global_cell_id);
641 }
642 cell_to_dof[loc_cell_id].push_back(dof_id);
643 }
644 }
645
646 m_dof_to_cell = dof_to_cell.template compress<index_t>();
647 m_cell_to_dof = cell_to_dof.template compress<index_t>();
648}

650template <typename MeshType>
651void DofHandler<MeshType>::computeOwnedCellDofsAndBndDofs() {
652 const auto max_dof_idx =
653 m_cell_to_dof.indices()[m_mesh.nLocEntity(MeshType::self_dim)];
654 const auto& data = m_cell_to_dof.data();
655 boost::unordered_flat_set<index_t> owned_cells_dof(
656 data.begin(), data.begin() + max_dof_idx);
657 m_owned_cells_dofs.assign(owned_cells_dof.begin(), owned_cells_dof.end());
658
659 // find which dofs lie on the partition
660 // i.e. they are both shared and belong to owned cells
661 for (auto loc_dof_id : m_owned_cells_dofs) {
662 const auto glob_dof_id = m_local_to_global_dof[loc_dof_id];
663 const auto it = std::lower_bound(
664 m_shared_dofs.begin(), m_shared_dofs.end(), glob_dof_id);
665 if ((it != m_shared_dofs.end()) && (*it == glob_dof_id)) {
666 m_partition_bnd_dofs.push_back(glob_dof_id);
667 }
668 }
669 std::sort(m_partition_bnd_dofs.begin(), m_partition_bnd_dofs.end());
670}

672template <typename MeshType>
673template <
674 std::size_t Dim,
675 typename FloatType,
676 typename DofPolicy,
677 typename FeSpace>
678void DofHandler<MeshType>::distributeDofs(
679 const fe::FeSpaceBase<Dim, FloatType, DofPolicy, FeSpace>& fe_space) {
680 // smallest row size that make sense
681 // namely we fill all the space we have in the `SmallVector`
682 constexpr std::size_t static_row_size =
683 sizeof(std::vector<index_t>) / sizeof(index_t);
684 datastructure::CrsMapUncompressed<index_t, static_row_size> dof_to_cell;
685 static_assert(FeSpace::dim == MeshType::self_dim);
686
687 computeNumDofs(fe_space);
688 dof_to_cell.resize(m_loc_cum_ndof_per_entity[FeSpace::dim + 1]);
689 PROFILE(distributeAllSubentityDofs(fe_space, dof_to_cell));
690
691 if (MeshType::parallel()) {
692 // we sort them because it is easier for rank 0 to use them
693 // when computing the global renumbering
694 PROFILE(sortSharedDofs());
695 }
696
697 PROFILE(invertDofToCellMappingAndCompress(dof_to_cell));
698
699 PROFILE(buildCellToCellDof(fe_space));
700
701 if constexpr (MeshType::parallel()) {
702 PROFILE(computeOwnedCellDofsAndBndDofs());
703 }
704
705 PROFILE(computeSparsityPattern<FeSpace>());
706}

708template <typename MeshType>
709template <bool DofInteractionDistance>
710std::pair<typename MeshType::index_t, typename MeshType::index_t>
711DofHandler<MeshType>::computeSparsityForDof(
712 typename MeshType::index_t dof_id,
713 typename MeshType::index_t* buffer1,
714 typename MeshType::index_t* buffer2) {
715 static std::array<std::size_t, 2> buffers_size = {0, 0};
716 static std::size_t active_buffer = 0;
717 buffers_size = {0, 0};
718 active_buffer = 0;
719 std::array<index_t*, 2> buffers = {buffer1, buffer2};
720
721 // merge all the dofs that belong to this cell (`dofs_view`) and the
722 // others (`buffer[active_buffer]`) into the inactive buffer (`buffer[1 -
723 // active_buffer]`). We assume the dofs are sorted and we remove
724 // duplicates
725 const auto updateBuffer = [&](auto dofs_view) {
726 const index_t* f1 = dofs_view.data();
727 const index_t* l1 = f1 + dofs_view.size();
728 const index_t* f2 = buffers[active_buffer];
729 const index_t* l2 = f2 + buffers_size[active_buffer];
730 index_t* o = buffers[1 - active_buffer];
731 // WARNING: this is not thread-safe
732 buffers_size[1 - active_buffer] =
733 std::distance(o, std::set_union(f1, l1, f2, l2, o));
734 // at each iteration we swap the source buffer with all the dofs and
735 // the destination buffer where we make the union, to avoid an extra
736 // copy
737 buffers_size[active_buffer] = 0;
738 active_buffer = 1 - active_buffer;
739 };
740
741 // for each cell to which this dof belongs to
742 for (index_t global_cell_id : m_dof_to_cell[dof_id]) {
743 auto local_cell_id = global_cell_id;
744 if constexpr (MeshType::parallel()) {
745 local_cell_id =
746 m_mesh.globalToLocal(MeshType::self_dim).at(global_cell_id);
747 }
748 updateBuffer(m_cell_to_dof[local_cell_id]);
749 // if there is interaction with also the neighbours
750 // e.g. if we are in DG each dof belongs to one cell but there
751 // are face couplings
752 if constexpr (DofInteractionDistance) {
753 for (index_t glob_neigh_cell_id : m_mesh.getDualMap()[local_cell_id]) {
754 auto loc_neigh_cell_id = glob_neigh_cell_id;
755 if constexpr (MeshType::parallel()) {
756 loc_neigh_cell_id =
757 m_mesh.globalToLocal(MeshType::self_dim).at(glob_neigh_cell_id);
758 }
759 updateBuffer(m_cell_to_dof[loc_neigh_cell_id]);
760 }
761 }
762 }
763
764 return {active_buffer, buffers_size[active_buffer]};
765}

767template <typename MeshType>
768template <typename FeSpace>
769void DofHandler<MeshType>::computeSparsityPattern() {
770 static constexpr std::size_t n_max_dofs_per_row = 1048576;
771 static std::array<std::array<index_t, n_max_dofs_per_row>, 2> buffers;
772
773 // Probe a good static size for the uncompressed map.
774 // We compute the sparsity pattern for some random dof ids
775 // and check which is the number of nnz at that row.
776 // We then ask for a static size that is greater than the average
777 constexpr index_t probe_size = 1000;
778 index_t nnz = 16;
779 if (m_dof_to_cell.size() > probe_size) {
780 nnz = 0;
781 std::mt19937 gen(0);
782 std::uniform_int_distribution<uint64_t> dis(0, m_dof_to_cell.size() - 1);
783 for (index_t i = 0; i < probe_size; ++i) {
784 const auto random_dof_id = dis(gen);
785 const auto [active_buffer, buffer_size] =
786 computeSparsityForDof<FeSpace::dofsInteractionDistance()>(
787 random_dof_id, buffers[0].data(), buffers[1].data());
788 nnz += buffer_size;
789 }
790 nnz /= probe_size;
791 }
792 std::cout << "Sparsity pattern probing suggested size: " << nnz << std::endl;
793 auto uncompressed_sparsity_pattern =
794 datastructure::getUncompressedCrsMapWithStaticStorageGreaterThan<index_t>(
795 nnz);
796 std::visit(
797 [&](auto& map) { map.resize(m_dof_to_cell.size()); },
798 uncompressed_sparsity_pattern);
799
800 // compute the sparsity pattern for all the dofs
801 for (index_t dof_id = 0; dof_id < m_dof_to_cell.size(); ++dof_id) {
802 const auto [active_buffer, buffer_size] =
803 computeSparsityForDof<FeSpace::dofsInteractionDistance()>(
804 dof_id, buffers[0].data(), buffers[1].data());
805 const auto nnz_dofid = buffer_size;
806 // copy the sparsity pattern from the static buffer to the uncompressed map
807 std::visit(
808 [dof_id, nnz_dofid, active_buffer](auto& map) {
809 map[dof_id].assign(
810 buffers[active_buffer].data(),
811 buffers[active_buffer].data() + nnz_dofid);
812 },
813 uncompressed_sparsity_pattern);
814 }
815
816 // compress the sparsity pattern
817 m_sparsity_pattern = std::visit(
818 [](const auto& map) { return map.template compress<index_t>(); },
819 uncompressed_sparsity_pattern);
820}

822template <typename MeshType>
823void DofHandler<MeshType>::renumberMpiSubdomainWise() {
824 const auto size = mpi::size();
825 // collect the size of owned dofs and the list of shared dofs
826 std::vector<std::size_t> cum_dof_offset(size + 1, 0);
827 m_all_shared_dof.resize(size);
828 m_all_partition_bnd_dofs.resize(size);
829
830 if (mpi::root()) {
831 cum_dof_offset[1] = m_owned_dofs.size();
832 m_all_shared_dof[0] = m_shared_dofs;
833 m_all_partition_bnd_dofs[0] = m_partition_bnd_dofs;
834 for (int p = 1; p < size; ++p) {
835 const auto owned_dof_p = mpi::recv<std::size_t>(p);
836 cum_dof_offset[p + 1] = cum_dof_offset[p] + owned_dof_p;
837 mpi::recv(m_all_shared_dof[p], p);
838 mpi::recv(m_all_partition_bnd_dofs[p], p);
839 }
840 } else {
841 mpi::send(m_owned_dofs.size(), 0);
842 mpi::send(m_shared_dofs, 0);
843 mpi::send(m_partition_bnd_dofs, 0);
844 }
845 // root renumbers the shared dofs and sends them
846 std::vector<index_t> shared_renumbering;
847 m_dof_offset = 0;
848 if (mpi::root()) {
849 // all the shared dofs without duplicates
850 std::vector<index_t> unq_shared_dof;
851 // list of boundary shared dofs "belonging" to proc p
852 // that is a bnd dof is owned by the proc with smallest rank that shares it
853 std::vector<index_t> bnd_dof_belonging_to_p;
854 // global-global old to new numbering
855 m_all_old_to_new_numbering.resize(m_glob_cum_ndof_per_entity.back(), -1);
856 std::vector<index_t> tmp_buffer;
857
858 for (int p = 0; p < size - 1; ++p) {
859 // to be numbered, they belong to p
860 std::set_difference(
861 m_all_partition_bnd_dofs[p].begin(),
862 m_all_partition_bnd_dofs[p].end(), unq_shared_dof.begin(),
863 unq_shared_dof.end(), std::back_inserter(bnd_dof_belonging_to_p));
864 // compute the dof offset taking into account the shared between 0 and p
865 for (int i = p; i < size; ++i) {
866 cum_dof_offset[i + 1] += bnd_dof_belonging_to_p.size();
867 }
868 mpi::send(cum_dof_offset[p + 1], p + 1);
869 const auto offset_without_shared =
870 cum_dof_offset[p + 1] - bnd_dof_belonging_to_p.size();
871 for (std::size_t i = 0; i < bnd_dof_belonging_to_p.size(); ++i) {
872 m_all_old_to_new_numbering[bnd_dof_belonging_to_p[i]] =
873 offset_without_shared + i;
874 }
875 // update unq_shared_dof
876 std::set_union(
877 unq_shared_dof.begin(), unq_shared_dof.end(),
878 m_all_partition_bnd_dofs[p].begin(),
879 m_all_partition_bnd_dofs[p].end(), std::back_inserter(tmp_buffer));
880 std::swap(unq_shared_dof, tmp_buffer);
881 bnd_dof_belonging_to_p.resize(0);
882 tmp_buffer.resize(0);
883 }
884
885 // renumber owning according to offset in sequence for self
886 m_old_to_new_numbering.resize(m_local_to_global_dof.size());
887 for (std::size_t i = 0; i < m_owned_dofs.size(); ++i) {
888 m_old_to_new_numbering[m_global_to_local_dof.at(m_owned_dofs[i])] = i;
889 m_all_old_to_new_numbering[m_owned_dofs[i]] = i;
890 }
891
892 // recive enumeration from all
893 std::vector<index_t> tmp_own_global_old;
894 std::vector<index_t> tmp_own_global_new;
895 for (int p = 1; p < size; ++p) {
896 mpi::recv(tmp_own_global_old, p);
897 mpi::recv(tmp_own_global_new, p);
898 for (std::size_t i = 0; i < tmp_own_global_old.size(); ++i) {
899 m_all_old_to_new_numbering[tmp_own_global_old[i]] =
900 tmp_own_global_new[i];
901 }
902 }
903
904 // find the enumeration for each proc
905 for (int p = 1; p < size; ++p) {
906 shared_renumbering.resize(m_all_shared_dof[p].size());
907 for (std::size_t i = 0; i < m_all_shared_dof[p].size(); ++i) {
908 shared_renumbering[i] =
909 m_all_old_to_new_numbering[m_all_shared_dof[p][i]];
910 }
911 mpi::send(shared_renumbering, p);
912 }
913 // find the enumeration for self
914 shared_renumbering.resize(m_all_shared_dof[0].size());
915 for (std::size_t i = 0; i < m_all_shared_dof[0].size(); ++i) {
916 shared_renumbering[i] =
917 m_all_old_to_new_numbering[m_all_shared_dof[0][i]];
918 }
919 } else {
920 m_dof_offset = mpi::recv<std::size_t>(0);
921 // renumber according to offset in sequence
922 std::vector<index_t> own_global_old;
923 std::vector<index_t> own_global_new;
924 m_old_to_new_numbering.resize(m_local_to_global_dof.size());
925 for (std::size_t i = 0; i < m_owned_dofs.size(); ++i) {
926 const auto glob_own_id = m_owned_dofs[i];
927 const auto loc_own_id = m_global_to_local_dof.at(glob_own_id);
928 const index_t new_glob_id = i + m_dof_offset;
929 m_old_to_new_numbering[loc_own_id] = new_glob_id;
930 own_global_old.push_back(glob_own_id);
931 own_global_new.push_back(new_glob_id);
932 }
933 mpi::send(own_global_old, 0);
934 mpi::send(own_global_new, 0);
935 mpi::recv(shared_renumbering, 0);
936 }
937 // apply the shared dof renumbering
938 assert(shared_renumbering.size() == m_shared_dofs.size());
939 for (std::size_t i = 0; i < m_shared_dofs.size(); ++i) {
940 m_old_to_new_numbering[m_local_shared_dofs[i]] = shared_renumbering[i];
941 }
942 // invert mapping
943 m_new_to_old_numbering.clear();
944 for (std::size_t i = 0; i < m_old_to_new_numbering.size(); ++i) {
945 m_new_to_old_numbering[m_old_to_new_numbering[i]] = i;
946 }
947 // apply renumbering to sparsity pattern
948 for (auto& col : m_sparsity_pattern.data()) {
949 col = m_old_to_new_numbering[col];
950 }
951}

953template <typename MeshType>
954void DofHandler<MeshType>::applyRenumbering(
955 const RenumberingFunctor<typename MeshType::index_t>& renumbering_func,
956 bool reversed) {
957 assert(m_old_to_new_numbering.size() == m_new_to_old_numbering.size());
958 using SparsityPatternType = decltype(m_sparsity_pattern);
959 SparsityPatternType* sparsity_pattern_ptr = &m_sparsity_pattern;
960 std::shared_ptr<SparsityPatternType> owned_sparsity_pattern;
961 if (MeshType::parallel()) {
962 owned_sparsity_pattern =
963 std::make_shared<SparsityPatternType>(computeOwnedSparsityPattern());
964 sparsity_pattern_ptr = owned_sparsity_pattern.get();
965 }
966 const auto [perm, perm_inv] =
967 renumbering_func(*sparsity_pattern_ptr, reversed);
968 assert(perm.size() == perm_inv.size());
969 // apply perm and perm_inv to mappings
970 if constexpr (MeshType::parallel()) {
971 const auto size = mpi::size();
972 // update the old to new for owned dofs
973 std::vector<index_t> new_owned_dof_id(m_owned_dofs.size());
974 {
975 std::vector<index_t> tmp = m_old_to_new_numbering;
976 for (std::size_t i = 0; i < perm.size(); ++i) {
977 tmp[m_local_owned_dofs[i]] =
978 m_old_to_new_numbering[m_local_owned_dofs[perm[i]]];
979 new_owned_dof_id[i] =
980 m_old_to_new_numbering[m_local_owned_dofs[perm[i]]];
981 }
982 m_old_to_new_numbering = std::move(tmp);
983 }
984
985 std::vector<index_t> new_shared_numbering;
986 if (mpi::root()) {
987 // get the new numbering of the dofs owned by each proc
988 std::vector<index_t> owned_dofs_p = m_owned_dofs;
989 for (int p = 0; p < size; ++p) {
990 if (p > 0) {
991 mpi::recv(owned_dofs_p, p);
992 mpi::recv(new_owned_dof_id, p);
993 }
994 assert(new_owned_dof_id.size() == owned_dofs_p.size());
995 for (std::size_t i = 0; i < new_owned_dof_id.size(); ++i) {
996 m_all_old_to_new_numbering[owned_dofs_p[i]] = new_owned_dof_id[i];
997 }
998 }
999
1000 // send new numbering of shared
1001 for (int p = 1; p < size; ++p) {
1002 new_shared_numbering.resize(m_all_shared_dof[p].size());
1003 for (std::size_t i = 0; i < m_all_shared_dof[p].size(); ++i) {
1004 new_shared_numbering[i] =
1005 m_all_old_to_new_numbering[m_all_shared_dof[p][i]];
1006 }
1007 mpi::send(new_shared_numbering, p);
1008 }
1009 // do the same for self
1010 new_shared_numbering.resize(m_shared_dofs.size());
1011 for (std::size_t i = 0; i < m_shared_dofs.size(); ++i) {
1012 new_shared_numbering[i] = m_all_old_to_new_numbering[m_shared_dofs[i]];
1013 }
1014 } else {
1015 mpi::send(m_owned_dofs, 0);
1016 // new glob ids for m_owned_dofs
1017 mpi::send(new_owned_dof_id, 0);
1018 // new enumeration for shared
1019 mpi::recv(new_shared_numbering, 0);
1020 }
1021 // apply new enumeration for shared
1022 assert(m_local_shared_dofs.size() == new_shared_numbering.size());
1023 for (std::size_t i = 0; i < m_local_shared_dofs.size(); ++i) {
1024 m_old_to_new_numbering[m_local_shared_dofs[i]] = new_shared_numbering[i];
1025 }
1026 // update sparsity pattern
1027 for (auto& col : m_sparsity_pattern.data()) {
1028 col = m_old_to_new_numbering[m_new_to_old_numbering[col]];
1029 }
1030 // invert the old_to_new
1031 m_new_to_old_numbering.clear();
1032 for (std::size_t i = 0; i < m_old_to_new_numbering.size(); ++i) {
1033 m_new_to_old_numbering[m_old_to_new_numbering[i]] = i;
1034 }
1035 } else {
1036 if (m_old_to_new_numbering.size() != perm.size()) {
1037 m_old_to_new_numbering = std::move(perm);
1038 m_new_to_old_numbering = std::move(perm_inv);
1039 } else {
1040 for (std::size_t i = 0; i < m_old_to_new_numbering.size(); ++i) {
1041 m_old_to_new_numbering[i] = perm[m_old_to_new_numbering[i]];
1042 m_new_to_old_numbering[i] = perm_inv[m_new_to_old_numbering[i]];
1043 }
1044 }
1045 m_sparsity_pattern = m_sparsity_pattern.permute(perm, perm_inv);
1046 }
1047}

1049template <typename MeshType>
1050auto DofHandler<MeshType>::computeOwnedSparsityPattern() {
1051 if (MeshType::parallel()) {
1052 // build inverse map (from the local id of an owned dof to the owned id)
1053 const auto n_local_dofs = m_owned_dofs.size() + m_shared_dofs.size();
1054 std::vector<index_t> loc_owned_dofs_inv(n_local_dofs, -1);
1055 for (std::size_t i = 0; i < m_local_owned_dofs.size(); ++i) {
1056 loc_owned_dofs_inv[m_local_owned_dofs[i]] = i;
1057 }
1058 //
1059 std::decay_t<decltype(m_sparsity_pattern)> owned_map({}, {});
1060 const auto& indices = m_sparsity_pattern.indices();
1061 const auto& data = m_sparsity_pattern.data();
1062 owned_map.indices().push_back(0);
1063 for (const auto own_loc_dof_id : m_local_owned_dofs) {
1064 const auto row_start = indices[own_loc_dof_id];
1065 const index_t row_size = indices[own_loc_dof_id + 1] - row_start;
1066 index_t real_size = 0;
1067 for (index_t j = 0; j < row_size; ++j) {
1068 const index_t new_glob_id = data[row_start + j];
1069 const index_t loc_id = m_new_to_old_numbering[new_glob_id];
1070 const index_t old_glob_id = m_local_to_global_dof[loc_id];
1071 const auto it = std::lower_bound(
1072 m_shared_dofs.begin(), m_shared_dofs.end(), old_glob_id);
1073 // if it is not shared, than it is owned
1074 // shared are many less, thus it is quicker than to search in owned
1075 if ((it == m_shared_dofs.end()) || (*it != old_glob_id)) {
1076 owned_map.data().push_back(loc_owned_dofs_inv[loc_id]);
1077 real_size++;
1078 }
1079 }
1080 owned_map.indices().push_back(real_size + owned_map.indices().back());
1081 }
1082 return owned_map;
1083 } else {
1084 return m_sparsity_pattern;
1085 }
1086}
