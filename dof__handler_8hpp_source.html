47template <typename FloatType>
48struct Statistics {
49 FloatType mean;
50 FloatType median;
51 FloatType min;
52 FloatType max;
53 FloatType std_dev;
54};
55
56template <typename FloatType>
57std::ostream& operator<<(std::ostream& os, const Statistics<FloatType>& stats) {
58 os << "Min: " << stats.min << "\n"
59 << "Median: " << stats.median << "\n"
60 << "Max: " << stats.max << "\n"
61 << "Mean: " << stats.mean << "\n"
62 << "StdDev: " << stats.std_dev << "\n";
63 return os;
64}

66template <typename FloatType>
67Statistics<FloatType> calculateStatistics(std::vector<FloatType> data) {
68 constexpr FloatType nan = std::numeric_limits<FloatType>::quiet_NaN();
69 Statistics<FloatType> stats = {
70 .mean = nan, .median = nan, .min = nan, .max = nan, .std_dev = nan};
71
72 if (data.empty()) {
73 return stats;
74 }
75
76 FloatType sum = 0.0;
77 stats.min = data[0];
78 stats.max = data[0];
79
80 // Calculate mean, min, and max in one pass
81 for (const FloatType& value : data) {
82 sum += value;
83 if (value < stats.min) {
84 stats.min = value;
85 }
86 if (value > stats.max) {
87 stats.max = value;
88 }
89 }
90
91 stats.mean = sum / data.size();
92
93 // Find the median (need copy to change data order)
94 size_t size = data.size();
95 std::nth_element(data.begin(), data.begin() + size / 2, data.end());
96
97 if (size % 2 == 0) {
98 std::nth_element(data.begin(), data.begin() + size / 2 - 1, data.end());
99 stats.median = (data[size / 2] + data[size / 2 - 1]) / 2.0;
100 } else {
101 stats.median = data[size / 2];
102 }
103
104 // Calculate standard deviation
105 FloatType variance_sum = 0.0;
106 for (const FloatType& value : data) {
107 variance_sum += (value - stats.mean) * (value - stats.mean);
108 }
109 stats.std_dev = std::sqrt(variance_sum / data.size());
110
111 return stats;
112}

456template <typename MeshType>
457template <
458 std::size_t Dim,
459 typename FloatType,
460 typename DofPolicy,
461 typename FeSpace>
462void DofHandler<MeshType>::buildCellToCellDof(
463 const fe::FeSpaceBase<Dim, FloatType, DofPolicy, FeSpace>& fe_space) {
464 m_cell_to_cell_dof = m_cell_to_dof;
465 // cell_id is the local id in MPI numeration
466 for (index_t cell_id = 0; cell_id < m_cell_to_cell_dof.size(); ++cell_id) {
467 for (auto& dof_id : m_cell_to_cell_dof[cell_id]) {
468 // expand information of dof_id by expoiting global (geometric)
469 // enumeration
470 index_t global_dof_id = dof_id;
471 if constexpr (MeshType::parallel()) {
472 global_dof_id = m_local_to_global_dof[dof_id];
473 }
474 index_t enty_dim = 0;
475 while (!(global_dof_id < m_glob_cum_ndof_per_entity[enty_dim + 1])) {
476 enty_dim++;
477 }
478 const index_t entity_dof_id =
479 global_dof_id - m_glob_cum_ndof_per_entity[enty_dim];
480 const index_t n_dofs_per_entity =
481 fe_space.getNumDofsForSubdimEntity(enty_dim);
482 const index_t owning_enty_id = entity_dof_id / n_dofs_per_entity;
483 const index_t enty_multiplicity = entity_dof_id % n_dofs_per_entity;
484 // pass the expanded information to the fe space and update dof_id
485 // from global to local numbering (cell based, not MPI partition based)
486 dof_id = fe_space.dofToCellDof(
487 m_mesh, cell_id, owning_enty_id, enty_dim, enty_multiplicity);
488 }
489 }

492template <typename MeshType>
493template <
494 std::size_t Dim,
495 typename FloatType,
496 typename DofPolicy,
497 typename FeSpace>
498void DofHandler<MeshType>::computeNumDofs(
499 const fe::FeSpaceBase<Dim, FloatType, DofPolicy, FeSpace>& fe_space) {
500 for (std::size_t d = 0; d < FeSpace::dim + 1; ++d) {
501 const auto n_dofs_per_entity = fe_space.getNumDofsForSubdimEntity(d);
502 const auto n_glob_entities = m_mesh.nEntity(d);
503 auto n_loc_entities = m_mesh.nLocEntity(d);
504 if constexpr (MeshType::parallel()) {
505 n_loc_entities += m_mesh.nGhostEntity(d);
506 }
507 m_glob_cum_ndof_per_entity[d + 1] =
508 m_glob_cum_ndof_per_entity[d] + n_dofs_per_entity * n_glob_entities;
509 m_loc_cum_ndof_per_entity[d + 1] =
510 m_loc_cum_ndof_per_entity[d] + n_dofs_per_entity * n_loc_entities;
511 }
512}

514template <typename MeshType>
515template <
516 std::size_t SubentityDim,
517 std::size_t Dim,
518 typename FloatType,
519 typename DofPolicy,
520 typename FeSpace>
521void DofHandler<MeshType>::distributeSubentityDofs(
522 const fe::FeSpaceBase<Dim, FloatType, DofPolicy, FeSpace>& fe_space,
523 auto& dof_to_cell) {
524 const index_t n_dofs_per_entity =
525 fe_space.getNumDofsForSubdimEntity(SubentityDim);
526 if (n_dofs_per_entity > 0) {
527 index_t n_loc_entity = m_mesh.nLocEntity(SubentityDim);
528 if constexpr (MeshType::parallel()) {
529 n_loc_entity += m_mesh.nGhostEntity(SubentityDim);
530 }
531 // for each entity in the mesh, may be vertex, edge, facet, cell
532 for (index_t enty_id = 0; enty_id < n_loc_entity; ++enty_id) {
533 // for each dof that belongs to this entity
534 for (index_t j = 0; j < n_dofs_per_entity; ++j) {
535 // compute global dof id
536 index_t global_enty_id = enty_id;
537 if constexpr (MeshType::parallel()) {
538 global_enty_id = m_mesh.localToGlobal(SubentityDim)[enty_id];
539 }
540 const auto glob_dof_id = m_glob_cum_ndof_per_entity[SubentityDim] +
541 global_enty_id * n_dofs_per_entity + j;
542 auto loc_dof_id = glob_dof_id;
543 if constexpr (MeshType::parallel()) {
544 loc_dof_id = m_local_to_global_dof.size();
545 m_local_to_global_dof.push_back(glob_dof_id);
546 m_global_to_local_dof[glob_dof_id] = loc_dof_id;
547 }
548 // put the `dof_id` in the `dof_to_cell map`
549 // different behavior if we are in a cell or not
550 if constexpr (SubentityDim == Dim) {
551 // we are in a cell, we know to which cell this dof belongs to
552 dof_to_cell[loc_dof_id].push_back(global_enty_id);
553 // in parallel we must keep track of owned and shared dofs
554 if constexpr (MeshType::parallel()) {
555 // we own the cell dof if we own the cell
556 if (enty_id < m_mesh.nLocEntity(SubentityDim)) {
557 m_owned_dofs.push_back(glob_dof_id);
558 m_local_owned_dofs.push_back(loc_dof_id);
559 } else {
560 m_shared_dofs.push_back(glob_dof_id);
561 m_local_shared_dofs.push_back(loc_dof_id);
562 }
563 }
564 } else {
565 const auto& enty_to_cell_map =
566 m_mesh.template getEntityToCellMap<SubentityDim>();
567 // loop over all the cells to which this entity belongs to
568 for (const auto cell_id : enty_to_cell_map[enty_id]) {
569 // facet mappings always have two elements, one of which may be -1
570 // in this case we skip it
571 if constexpr (SubentityDim == Dim - 1) {
572 if (cell_id == static_cast<decltype(cell_id)>(-1)) {
573 continue;
574 }
575 }
576 dof_to_cell[loc_dof_id].push_back(cell_id);
577 // in parallel we must keep track of owned and shared dofs
578 if constexpr (MeshType::parallel()) {
579 // if the dof belongs to an entity that is on a ghost cell
580 // than it is a shared dof
581 const auto local_cell_id = m_mesh.globalToLocal(Dim).at(cell_id);
582 if ((local_cell_id >= m_mesh.nLocEntity(Dim)) &&
583 (m_shared_dofs.empty() ||
584 m_shared_dofs.back() != glob_dof_id)) {
585 m_shared_dofs.push_back(glob_dof_id);
586 m_local_shared_dofs.push_back(loc_dof_id);
587 }
588 }
589 }
590 // if the dof is not shared, than it is owned
591 if constexpr (MeshType::parallel()) {
592 if (!(m_shared_dofs.size() &&
593 (m_shared_dofs.back() == glob_dof_id))) {
594 m_owned_dofs.push_back(glob_dof_id);
595 m_local_owned_dofs.push_back(loc_dof_id);
596 }
597 }
598 }
599 }
600 }
601 }
602}

625template <typename MeshType>
626void DofHandler<MeshType>::sortSharedDofs() {
627 std::vector<index_t> idxs(m_shared_dofs.size());
628 std::iota(idxs.begin(), idxs.end(), 0);
629 std::sort(idxs.begin(), idxs.end(), [this](auto a, auto b) {
630 return m_shared_dofs[a] < m_shared_dofs[b];
631 });
632 // apply sorting permutation
633 std::vector<index_t> tmp(m_shared_dofs.size());
634 for (std::size_t i = 0; i < idxs.size(); ++i) {
635 tmp[i] = m_shared_dofs[idxs[i]];
636 }
637 std::swap(tmp, m_shared_dofs);
638 for (std::size_t i = 0; i < idxs.size(); ++i) {
639 tmp[i] = m_local_shared_dofs[idxs[i]];
640 }
641 std::swap(tmp, m_local_shared_dofs);
642}

644template <typename MeshType>
645void DofHandler<MeshType>::invertDofToCellMappingAndCompress(
646 auto& dof_to_cell) {
647 constexpr std::size_t static_buffer_size =
648 vmath::pow((index_t)7, MeshType::self_dim);
649 datastructure::CrsMapUncompressed<index_t, static_buffer_size> cell_to_dof;
650 auto n_loc_cell = m_mesh.nLocEntity(MeshType::self_dim);
651 if constexpr (MeshType::parallel()) {
652 n_loc_cell += m_mesh.nGhostEntity(MeshType::self_dim);
653 }
654 cell_to_dof.resize(n_loc_cell);
655 for (std::size_t dof_id = 0; dof_id < dof_to_cell.size(); ++dof_id) {
656 // we exploit the fact that the dofs are sorted later
657 // when we compute the sparsity pattern
658 // TODO: is this already sorted if in serial?
659 std::sort(dof_to_cell[dof_id].begin(), dof_to_cell[dof_id].end());
660 for (const auto global_cell_id : dof_to_cell[dof_id]) {
661 auto loc_cell_id = global_cell_id;
662 if constexpr (MeshType::parallel()) {
663 loc_cell_id =
664 m_mesh.globalToLocal(MeshType::self_dim).at(global_cell_id);
665 }
666 cell_to_dof[loc_cell_id].push_back(dof_id);
667 }
668 }
669
670 m_dof_to_cell = dof_to_cell.template compress<index_t>();
671 m_cell_to_dof = cell_to_dof.template compress<index_t>();
672}

674template <typename MeshType>
675void DofHandler<MeshType>::computeOwnedCellDofsAndBndDofs() {
676 const auto max_dof_idx =
677 m_cell_to_dof.indices()[m_mesh.nLocEntity(MeshType::self_dim)];
678 const auto& data = m_cell_to_dof.data();
679 boost::unordered_flat_set<index_t> owned_cells_dof(
680 data.begin(), data.begin() + max_dof_idx);
681 m_owned_cells_dofs.assign(owned_cells_dof.begin(), owned_cells_dof.end());
682
683 // find which dofs lie on the partition
684 // i.e. they are both shared and belong to owned cells
685 for (auto loc_dof_id : m_owned_cells_dofs) {
686 const auto glob_dof_id = m_local_to_global_dof[loc_dof_id];
687 const auto it = std::lower_bound(
688 m_shared_dofs.begin(), m_shared_dofs.end(), glob_dof_id);
689 if ((it != m_shared_dofs.end()) && (*it == glob_dof_id)) {
690 m_partition_bnd_dofs.push_back(glob_dof_id);
691 }
692 }
693 std::sort(m_partition_bnd_dofs.begin(), m_partition_bnd_dofs.end());
694}

696template <typename MeshType>
697template <
698 std::size_t Dim,
699 typename FloatType,
700 typename DofPolicy,
701 typename FeSpace>
702void DofHandler<MeshType>::distributeDofs(
703 const fe::FeSpaceBase<Dim, FloatType, DofPolicy, FeSpace>& fe_space) {
704 // smallest row size that make sense
705 // namely we fill all the space we have in the `SmallVector`
706 constexpr std::size_t static_row_size =
707 sizeof(std::vector<index_t>) / sizeof(index_t);
708 datastructure::CrsMapUncompressed<index_t, static_row_size> dof_to_cell;
709 static_assert(FeSpace::dim == MeshType::self_dim);
710
711 computeNumDofs(fe_space);
712 dof_to_cell.resize(m_loc_cum_ndof_per_entity[FeSpace::dim + 1]);
713 PROFILE(distributeAllSubentityDofs(fe_space, dof_to_cell));
714
715 if (MeshType::parallel()) {
716 // we sort them because it is easier for rank 0 to use them
717 // when computing the global renumbering
718 PROFILE(sortSharedDofs());
719 }
720
721 PROFILE(invertDofToCellMappingAndCompress(dof_to_cell));
722
723 PROFILE(buildCellToCellDof(fe_space));
724
725 if constexpr (MeshType::parallel()) {
726 PROFILE(computeOwnedCellDofsAndBndDofs());
727 }
728
729 PROFILE(computeSparsityPattern<FeSpace>());
730
731 if constexpr (MeshType::parallel()) {
732 renumberMpiSubdomainWise();
733 }
734}

736template <typename MeshType>
737template <bool DofInteractionDistance>
738std::pair<typename MeshType::index_t, typename MeshType::index_t>
739DofHandler<MeshType>::computeSparsityForDof(
740 typename MeshType::index_t dof_id,
741 typename MeshType::index_t* buffer1,
742 typename MeshType::index_t* buffer2) {
743 static std::array<std::size_t, 2> buffers_size = {0, 0};
744 static std::size_t active_buffer = 0;
745 buffers_size = {0, 0};
746 active_buffer = 0;
747 std::array<index_t*, 2> buffers = {buffer1, buffer2};
748
749 // merge all the dofs that belong to this cell (`dofs_view`) and the
750 // others (`buffer[active_buffer]`) into the inactive buffer (`buffer[1 -
751 // active_buffer]`). We assume the dofs are sorted and we remove
752 // duplicates
753 const auto updateBuffer = [&](auto dofs_view) {
754 const index_t* f1 = dofs_view.data();
755 const index_t* l1 = f1 + dofs_view.size();
756 const index_t* f2 = buffers[active_buffer];
757 const index_t* l2 = f2 + buffers_size[active_buffer];
758 index_t* o = buffers[1 - active_buffer];
759 // WARNING: this is not thread-safe
760 buffers_size[1 - active_buffer] =
761 std::distance(o, std::set_union(f1, l1, f2, l2, o));
762 // at each iteration we swap the source buffer with all the dofs and
763 // the destination buffer where we make the union, to avoid an extra
764 // copy
765 buffers_size[active_buffer] = 0;
766 active_buffer = 1 - active_buffer;
767 };
768
769 // for each cell to which this dof belongs to
770 for (index_t global_cell_id : m_dof_to_cell[dof_id]) {
771 auto local_cell_id = global_cell_id;
772 if constexpr (MeshType::parallel()) {
773 local_cell_id =
774 m_mesh.globalToLocal(MeshType::self_dim).at(global_cell_id);
775 }
776 updateBuffer(m_cell_to_dof[local_cell_id]);
777 // if there is interaction with also the neighbours
778 // e.g. if we are in DG each dof belongs to one cell but there
779 // are face couplings
780 if constexpr (DofInteractionDistance) {
781 for (index_t glob_neigh_cell_id : m_mesh.getDualMap()[local_cell_id]) {
782 auto loc_neigh_cell_id = glob_neigh_cell_id;
783 if constexpr (MeshType::parallel()) {
784 loc_neigh_cell_id =
785 m_mesh.globalToLocal(MeshType::self_dim).at(glob_neigh_cell_id);
786 }
787 updateBuffer(m_cell_to_dof[loc_neigh_cell_id]);
788 }
789 }
790 }
791
792 return {active_buffer, buffers_size[active_buffer]};
793}

795template <typename MeshType>
796template <typename FeSpace>
797void DofHandler<MeshType>::computeSparsityPattern() {
798 static constexpr std::size_t n_max_dofs_per_row = 1048576;
799 static std::array<std::array<index_t, n_max_dofs_per_row>, 2> buffers;
800
801 // Probe a good static size for the uncompressed map.
802 // We compute the sparsity pattern for some random dof ids
803 // and check which is the number of nnz at that row.
804 // We then ask for a static size that is greater than the average
805 constexpr index_t probe_size = 1000;
806 index_t nnz = 16;
807 if (m_dof_to_cell.size() > probe_size) {
808 nnz = 0;
809 std::mt19937 gen(0);
810 std::uniform_int_distribution<uint64_t> dis(0, m_dof_to_cell.size() - 1);
811 for (index_t i = 0; i < probe_size; ++i) {
812 const auto random_dof_id = dis(gen);
813 const auto [active_buffer, buffer_size] =
814 computeSparsityForDof<FeSpace::dofsInteractionDistance()>(
815 random_dof_id, buffers[0].data(), buffers[1].data());
816 nnz += buffer_size;
817 }
818 nnz /= probe_size;
819 }
820 std::cout << "Sparsity pattern probing suggested size: " << nnz << std::endl;
821 auto uncompressed_sparsity_pattern =
822 datastructure::getUncompressedCrsMapWithStaticStorageGreaterThan<index_t>(
823 nnz);
824 std::visit(
825 [&](auto& map) { map.resize(m_dof_to_cell.size()); },
826 uncompressed_sparsity_pattern);
827
828 // compute the sparsity pattern for all the dofs
829 for (index_t dof_id = 0; dof_id < m_dof_to_cell.size(); ++dof_id) {
830 const auto [active_buffer, buffer_size] =
831 computeSparsityForDof<FeSpace::dofsInteractionDistance()>(
832 dof_id, buffers[0].data(), buffers[1].data());
833 const auto nnz_dofid = buffer_size;
834 // copy the sparsity pattern from the static buffer to the uncompressed map
835 std::visit(
836 [dof_id, nnz_dofid, active_buffer](auto& map) {
837 map[dof_id].assign(
838 buffers[active_buffer].data(),
839 buffers[active_buffer].data() + nnz_dofid);
840 },
841 uncompressed_sparsity_pattern);
842 }
843
844 // compress the sparsity pattern
845 m_sparsity_pattern = std::visit(
846 [](const auto& map) { return map.template compress<index_t>(); },
847 uncompressed_sparsity_pattern);
848}

850template <typename MeshType>
851void DofHandler<MeshType>::renumberMpiSubdomainWise() {
852 m_has_renumbering = true;
853 const auto size = mpi::size();
854 // collect the size of owned dofs and the list of shared dofs
855 std::vector<std::size_t> cum_dof_offset(size + 1, 0);
856 m_all_shared_dof.resize(size);
857 m_all_partition_bnd_dofs.resize(size);
858
859 if (mpi::root()) {
860 cum_dof_offset[1] = m_owned_dofs.size();
861 m_all_shared_dof[0] = m_shared_dofs;
862 m_all_partition_bnd_dofs[0] = m_partition_bnd_dofs;
863 for (int p = 1; p < size; ++p) {
864 const auto owned_dof_p = mpi::recv<std::size_t>(p);
865 cum_dof_offset[p + 1] = cum_dof_offset[p] + owned_dof_p;
866 mpi::recv(m_all_shared_dof[p], p);
867 mpi::recv(m_all_partition_bnd_dofs[p], p);
868 }
869 } else {
870 mpi::send(m_owned_dofs.size(), 0);
871 mpi::send(m_shared_dofs, 0);
872 mpi::send(m_partition_bnd_dofs, 0);
873 }
874 // root renumbers the shared dofs and sends them
875 std::vector<index_t> shared_renumbering;
876 m_dof_offset = 0;
877 if (mpi::root()) {
878 // all the shared dofs without duplicates
879 std::vector<index_t> unq_shared_dof;
880 // list of boundary shared dofs "belonging" to proc p
881 // that is a bnd dof is owned by the proc with smallest rank that shares it
882 std::vector<index_t> bnd_dof_belonging_to_p;
883 // global-global old to new numbering
884 m_all_old_to_new_numbering.resize(m_glob_cum_ndof_per_entity.back(), -1);
885 std::vector<index_t> tmp_buffer;
886
887 for (int p = 0; p < size - 1; ++p) {
888 // to be numbered, they belong to p
889 std::set_difference(
890 m_all_partition_bnd_dofs[p].begin(),
891 m_all_partition_bnd_dofs[p].end(), unq_shared_dof.begin(),
892 unq_shared_dof.end(), std::back_inserter(bnd_dof_belonging_to_p));
893 // compute the dof offset taking into account the shared between 0 and p
894 for (int i = p; i < size; ++i) {
895 cum_dof_offset[i + 1] += bnd_dof_belonging_to_p.size();
896 }
897 mpi::send(cum_dof_offset[p + 1], p + 1);
898 const auto offset_without_shared =
899 cum_dof_offset[p + 1] - bnd_dof_belonging_to_p.size();
900 for (std::size_t i = 0; i < bnd_dof_belonging_to_p.size(); ++i) {
901 m_all_old_to_new_numbering[bnd_dof_belonging_to_p[i]] =
902 offset_without_shared + i;
903 }
904 // update unq_shared_dof
905 std::set_union(
906 unq_shared_dof.begin(), unq_shared_dof.end(),
907 m_all_partition_bnd_dofs[p].begin(),
908 m_all_partition_bnd_dofs[p].end(), std::back_inserter(tmp_buffer));
909 std::swap(unq_shared_dof, tmp_buffer);
910 bnd_dof_belonging_to_p.resize(0);
911 tmp_buffer.resize(0);
912 }
913
914 // renumber owning according to offset in sequence for self
915 m_old_to_new_numbering.resize(m_local_to_global_dof.size());
916 for (std::size_t i = 0; i < m_owned_dofs.size(); ++i) {
917 m_old_to_new_numbering[m_global_to_local_dof.at(m_owned_dofs[i])] = i;
918 m_all_old_to_new_numbering[m_owned_dofs[i]] = i;
919 }
920
921 // recive enumeration from all
922 std::vector<index_t> tmp_own_global_old;
923 std::vector<index_t> tmp_own_global_new;
924 for (int p = 1; p < size; ++p) {
925 mpi::recv(tmp_own_global_old, p);
926 mpi::recv(tmp_own_global_new, p);
927 for (std::size_t i = 0; i < tmp_own_global_old.size(); ++i) {
928 m_all_old_to_new_numbering[tmp_own_global_old[i]] =
929 tmp_own_global_new[i];
930 }
931 }
932
933 // find the enumeration for each proc
934 for (int p = 1; p < size; ++p) {
935 shared_renumbering.resize(m_all_shared_dof[p].size());
936 for (std::size_t i = 0; i < m_all_shared_dof[p].size(); ++i) {
937 shared_renumbering[i] =
938 m_all_old_to_new_numbering[m_all_shared_dof[p][i]];
939 }
940 mpi::send(shared_renumbering, p);
941 }
942 // find the enumeration for self
943 shared_renumbering.resize(m_all_shared_dof[0].size());
944 for (std::size_t i = 0; i < m_all_shared_dof[0].size(); ++i) {
945 shared_renumbering[i] =
946 m_all_old_to_new_numbering[m_all_shared_dof[0][i]];
947 }
948 } else {
949 m_dof_offset = mpi::recv<std::size_t>(0);
950 // renumber according to offset in sequence
951 std::vector<index_t> own_global_old;
952 std::vector<index_t> own_global_new;
953 m_old_to_new_numbering.resize(m_local_to_global_dof.size());
954 for (std::size_t i = 0; i < m_owned_dofs.size(); ++i) {
955 const auto glob_own_id = m_owned_dofs[i];
956 const auto loc_own_id = m_global_to_local_dof.at(glob_own_id);
957 const index_t new_glob_id = i + m_dof_offset;
958 m_old_to_new_numbering[loc_own_id] = new_glob_id;
959 own_global_old.push_back(glob_own_id);
960 own_global_new.push_back(new_glob_id);
961 }
962 mpi::send(own_global_old, 0);
963 mpi::send(own_global_new, 0);
964 mpi::recv(shared_renumbering, 0);
965 }
966 // broadcast the offset
967 m_all_dof_offset.resize(mpi::size() + 1);
968 m_all_dof_offset.back() = m_glob_cum_ndof_per_entity.back();
969 for (int p = 0; p < mpi::size(); ++p) {
970 m_all_dof_offset[p] = m_dof_offset;
971 mpi::bcast(m_all_dof_offset[p], p);
972 }
973 // apply the shared dof renumbering
974 assert(shared_renumbering.size() == m_shared_dofs.size());
975 for (std::size_t i = 0; i < m_shared_dofs.size(); ++i) {
976 m_old_to_new_numbering[m_local_shared_dofs[i]] = shared_renumbering[i];
977 }
978 // invert mapping
979 m_new_to_old_numbering.clear();
980 for (std::size_t i = 0; i < m_old_to_new_numbering.size(); ++i) {
981 m_new_to_old_numbering[m_old_to_new_numbering[i]] = i;
982 }
983 // apply renumbering to sparsity pattern
984 for (auto& col : m_sparsity_pattern.data()) {
985 col = m_old_to_new_numbering[col];
986 }
987}

989template <typename MeshType>
990void DofHandler<MeshType>::applyRenumbering(
991 const RenumberingFunctor<typename MeshType::index_t>& renumbering_func,
992 bool reversed) {
993 m_has_renumbering = true;
994 assert(m_old_to_new_numbering.size() == m_new_to_old_numbering.size());
995 using SparsityPatternType = decltype(m_sparsity_pattern);
996 SparsityPatternType* sparsity_pattern_ptr = &m_sparsity_pattern;
997 std::shared_ptr<SparsityPatternType> owned_sparsity_pattern;
998 if (MeshType::parallel()) {
999 owned_sparsity_pattern =
1000 std::make_shared<SparsityPatternType>(computeOwnedSparsityPattern());
1001 sparsity_pattern_ptr = owned_sparsity_pattern.get();
1002 }
1003 const auto [perm, perm_inv] =
1004 renumbering_func(*sparsity_pattern_ptr, reversed);
1005 assert(perm.size() == perm_inv.size());
1006 // apply perm and perm_inv to mappings
1007 if constexpr (MeshType::parallel()) {
1008 const auto size = mpi::size();
1009 // update the old to new for owned dofs
1010 std::vector<index_t> new_owned_dof_id(m_owned_dofs.size());
1011 {
1012 std::vector<index_t> tmp = m_old_to_new_numbering;
1013 for (std::size_t i = 0; i < perm.size(); ++i) {
1014 tmp[m_local_owned_dofs[i]] =
1015 m_old_to_new_numbering[m_local_owned_dofs[perm[i]]];
1016 new_owned_dof_id[i] =
1017 m_old_to_new_numbering[m_local_owned_dofs[perm[i]]];
1018 }
1019 m_old_to_new_numbering = std::move(tmp);
1020 }
1021
1022 std::vector<index_t> new_shared_numbering;
1023 if (mpi::root()) {
1024 // get the new numbering of the dofs owned by each proc
1025 std::vector<index_t> owned_dofs_p = m_owned_dofs;
1026 for (int p = 0; p < size; ++p) {
1027 if (p > 0) {
1028 mpi::recv(owned_dofs_p, p);
1029 mpi::recv(new_owned_dof_id, p);
1030 }
1031 assert(new_owned_dof_id.size() == owned_dofs_p.size());
1032 for (std::size_t i = 0; i < new_owned_dof_id.size(); ++i) {
1033 m_all_old_to_new_numbering[owned_dofs_p[i]] = new_owned_dof_id[i];
1034 }
1035 }
1036
1037 // send new numbering of shared
1038 for (int p = 1; p < size; ++p) {
1039 new_shared_numbering.resize(m_all_shared_dof[p].size());
1040 for (std::size_t i = 0; i < m_all_shared_dof[p].size(); ++i) {
1041 new_shared_numbering[i] =
1042 m_all_old_to_new_numbering[m_all_shared_dof[p][i]];
1043 }
1044 mpi::send(new_shared_numbering, p);
1045 }
1046 // do the same for self
1047 new_shared_numbering.resize(m_shared_dofs.size());
1048 for (std::size_t i = 0; i < m_shared_dofs.size(); ++i) {
1049 new_shared_numbering[i] = m_all_old_to_new_numbering[m_shared_dofs[i]];
1050 }
1051 } else {
1052 mpi::send(m_owned_dofs, 0);
1053 // new glob ids for m_owned_dofs
1054 mpi::send(new_owned_dof_id, 0);
1055 // new enumeration for shared
1056 mpi::recv(new_shared_numbering, 0);
1057 }
1058 // apply new enumeration for shared
1059 assert(m_local_shared_dofs.size() == new_shared_numbering.size());
1060 for (std::size_t i = 0; i < m_local_shared_dofs.size(); ++i) {
1061 m_old_to_new_numbering[m_local_shared_dofs[i]] = new_shared_numbering[i];
1062 }
1063 // update sparsity pattern
1064 for (auto& col : m_sparsity_pattern.data()) {
1065 col = m_old_to_new_numbering[m_new_to_old_numbering[col]];
1066 }
1067 // invert the old_to_new
1068 m_new_to_old_numbering.clear();
1069 for (std::size_t i = 0; i < m_old_to_new_numbering.size(); ++i) {
1070 m_new_to_old_numbering[m_old_to_new_numbering[i]] = i;
1071 }
1072 } else {
1073 if (m_old_to_new_numbering.size() != perm.size()) {
1074 m_old_to_new_numbering = std::move(perm);
1075 m_new_to_old_numbering = std::move(perm_inv);
1076 } else {
1077 for (std::size_t i = 0; i < m_old_to_new_numbering.size(); ++i) {
1078 m_old_to_new_numbering[i] = perm[m_old_to_new_numbering[i]];
1079 m_new_to_old_numbering[i] = perm_inv[m_new_to_old_numbering[i]];
1080 }
1081 }
1082 m_sparsity_pattern = m_sparsity_pattern.permute(perm, perm_inv);
1083 }
1084}

1086template <typename MeshType>
1087auto DofHandler<MeshType>::computeOwnedSparsityPattern() {
1088 if (MeshType::parallel()) {
1089 // build inverse map (from the local id of an owned dof to the owned id)
1090 const auto n_local_dofs = m_owned_dofs.size() + m_shared_dofs.size();
1091 std::vector<index_t> loc_owned_dofs_inv(n_local_dofs, -1);
1092 for (std::size_t i = 0; i < m_local_owned_dofs.size(); ++i) {
1093 loc_owned_dofs_inv[m_local_owned_dofs[i]] = i;
1094 }
1095 //
1096 std::decay_t<decltype(m_sparsity_pattern)> owned_map({}, {});
1097 const auto& indices = m_sparsity_pattern.indices();
1098 const auto& data = m_sparsity_pattern.data();
1099 owned_map.indices().push_back(0);
1100 for (const auto own_loc_dof_id : m_local_owned_dofs) {
1101 const auto row_start = indices[own_loc_dof_id];
1102 const index_t row_size = indices[own_loc_dof_id + 1] - row_start;
1103 index_t real_size = 0;
1104 for (index_t j = 0; j < row_size; ++j) {
1105 const index_t new_glob_id = data[row_start + j];
1106 const index_t loc_id = m_new_to_old_numbering[new_glob_id];
1107 const index_t old_glob_id = m_local_to_global_dof[loc_id];
1108 const auto it = std::lower_bound(
1109 m_shared_dofs.begin(), m_shared_dofs.end(), old_glob_id);
1110 // if it is not shared, than it is owned
1111 // shared are many less, thus it is quicker than to search in owned
1112 if ((it == m_shared_dofs.end()) || (*it != old_glob_id)) {
1113 owned_map.data().push_back(loc_owned_dofs_inv[loc_id]);
1114 real_size++;
1115 }
1116 }
1117 owned_map.indices().push_back(real_size + owned_map.indices().back());
1118 }
1119 return owned_map;
1120 } else {
1121 return m_sparsity_pattern;
1122 }
1123}
