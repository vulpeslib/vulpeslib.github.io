84 template <typename U>
85 explicit IndexTracker(const std::vector<U>& vec) {
86 static_assert(
87 std::is_integral<U>::value, "Constructor requires integral vector");
88 m_N = vec.size();
89 m_V = m_N;
90 m_data.reserve(m_N);
91 for (U v : vec) {
92 assert(v >= 0);
93 m_data.push_back(static_cast<IndexType>(v));
94 }
95
96 m_head.assign(m_V, npos);
97 m_next.assign(m_N, npos);
98 m_prev.assign(m_N, npos);
99
100 for (IndexType i = 0; i < m_N; ++i) {
101 IndexType v = m_data[i];
102 assert(v < m_V);
103 m_next[i] = m_head[v];
104 m_prev[i] = npos;
105 if (m_head[v] != npos) {
106 m_prev[m_head[v]] = i;
107 }
108 m_head[v] = i;
109 }
110 }

112 ~IndexTracker() = default;

127 std::vector<IndexType> indicesOf(IndexType v, bool sorted = true) const;

135 template <typename F>
136 void forEachIndex(IndexType v, F f) const {
137 if (v >= m_V) {
138 return;
139 }
140 for (IndexType i = m_head[v]; i != npos; i = m_next[i]) {
141 f(i);
142 }
143 }
144
146 IndexType size() const { return m_N; }

148 const std::vector<IndexType>& getValues() const { return m_data; };
