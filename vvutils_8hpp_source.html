52template <typename Container>
53std::ostream& print(
54 std::ostream& os,
55 const Container& container,
56 const std::string& separator = ",",
57 const std::string& postpend = "\n",
58 const std::string& prepend = "") {
59 auto it = container.begin();
60 if (!prepend.empty()) {
61 os << prepend;
62 }
63 if (it != container.end()) {
64 os << *it;
65 ++it;
66 for (; it != container.end(); ++it) {
67 os << separator << *it;
68 }
69 }
70 if (!postpend.empty()) {
71 os << postpend;
72 }
73 return os;
74}

87 const std::vector<T>& source, const std::vector<IndexType>& indices) {
88 std::vector<T> result;
89 result.reserve(indices.size());
90 for (size_t index : indices) {
