51template <typename Container>
52std::ostream& print(
53 std::ostream& os,
54 const Container& container,
55 const std::string& separator = ",",
56 const std::string& postpend = "\n",
57 const std::string& prepend = "") {
58 auto it = container.begin();
59 if (!prepend.empty()) {
60 os << prepend;
61 }
62 if (it != container.end()) {
63 os << *it;
64 ++it;
65 for (; it != container.end(); ++it) {
66 os << separator << *it;
67 }
68 }
69 if (!postpend.empty()) {
70 os << postpend;
71 }
72 return os;
73}

84template <typename T, typename IndexType>
85std::vector<T> slice(
86 const std::vector<T>& source, const std::vector<IndexType>& indices) {
87 std::vector<T> result;
88 result.reserve(indices.size());
89 for (size_t index : indices) {
90 result.push_back(source[index]);
91 }
