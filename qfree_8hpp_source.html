103template <bool PPoly, typename FloatType, typename IndexType>
104std::pair<
105 AreaIntegral2D<PPoly, FloatType, IndexType>,
106 EdgeIntegral2D<PPoly, FloatType, IndexType>>
107integrateMonomialsQuadratureFree2D(
108 const geometry::Cell2D<FloatType>& cell, IndexType deg) {
109 constexpr FloatType Dim = 2.0;
110 using AreaIntegral = AreaIntegral2D<PPoly, FloatType, IndexType>;
111 using EdgeIntegral = EdgeIntegral2D<PPoly, FloatType, IndexType>;
112
113 const IndexType n_vtxs = cell.vtxs().cols();
114 const IndexType deg1 = deg + 1;
115 AreaIntegral area_ints(deg1);
116 EdgeIntegral edge_ints(deg1, n_vtxs);
117
118 // avoid reallocation of caching buffer for monomial powers
119 // at each loop it is fully over-written
120 std::vector<std::array<FloatType, 2>> vxy(deg1);
121 for (IndexType vi = 0; vi < n_vtxs; ++vi) {
122 const FloatType x1 = cell.vtxs()(0, vi);
123 const FloatType y1 = cell.vtxs()(1, vi);
124 const FloatType x2 = cell.vtxs()(0, (vi + 1) % n_vtxs);
125 const FloatType y2 = cell.vtxs()(1, (vi + 1) % n_vtxs);
126
127 // components of unnormalized normal to the edge
128 const FloatType y2y1 = y2 - y1;
129 const FloatType x1x2 = x1 - x2;
130 // length of the edge
131 const FloatType dij = std::sqrt(y2y1 * y2y1 + x1x2 * x1x2);
132 // bi = signed distance as scalar product with outer normal
133 // we pick the first extrema (x1, y1)
134 const FloatType bi = (y2y1 * x1 + x1x2 * y1) / dij;
135
136 edge_ints(0, 0, vi) = dij;
137 area_ints(0, 0) += bi * dij / Dim;
138
139 // handle the case where i or j is equal to zero (in x^i * y^j)
140 // and cache the pair values (x^k, y^k) to be used later
141 vxy[0] = {{x2, y2}};
142 for (IndexType k = 1; k < deg1; ++k) {
143 vxy[k] = {{vxy[k - 1][0] * x2, vxy[k - 1][1] * y2}};
144 const FloatType den_edge = k + 1;
145 edge_ints(k, 0, vi) =
146 (dij * vxy[k - 1][0] + x1 * k * edge_ints(k - 1, 0, vi)) / den_edge;
147 edge_ints(0, k, vi) =
148 (dij * vxy[k - 1][1] + y1 * k * edge_ints(0, k - 1, vi)) / den_edge;
149 const FloatType den = k + 2;
150 area_ints(k, 0) += bi * edge_ints(k, 0, vi) / den;
151 area_ints(0, k) += bi * edge_ints(0, k, vi) / den;
152 }
153
154 // full recursive formula for edge/area integral
155 for (IndexType i = 1; i < deg1; ++i) {
156 // handle the monomial of the space P^k or Q^k
157 const IndexType j_max = PPoly ? deg1 - i : deg1;
158 for (IndexType j = 1; j < j_max; ++j) {
159 const IndexType i_j = i + j;
160 const auto edge_int_ij = (dij * vxy[i - 1][0] * vxy[j - 1][1] +
161 x1 * i * edge_ints(i - 1, j, vi) +
162 y1 * j * edge_ints(i, j - 1, vi)) /
163 (1 + i_j);
164 edge_ints(i, j, vi) = edge_int_ij;
165 area_ints(i, j) += bi * edge_int_ij / (2 + i_j);
166 }
167 }
168 }
169
170 return {area_ints, edge_ints};
171}

256template <bool PPoly, bool Pruning, typename FloatType, typename IndexType>
257std::pair<
258 VolumeIntegral3D<PPoly, FloatType, IndexType>,
259 LowerIntegral3D<PPoly, FloatType, IndexType>>
260integrateMonomialsQuadratureFree3D(
261 const geometry::Cell3D<FloatType, IndexType>& elem, IndexType deg) {
262 const IndexType n_edges = elem.edges().size();
263 const IndexType n_faces = elem.nfaces();
264 const IndexType n_vtxs = elem.vtxs().cols();
265 const IndexType deg1 = deg + 1;
266
267 LowerIntegral3D<PPoly, FloatType, IndexType> I_edges(deg1, n_edges);
268 LowerIntegral3D<PPoly, FloatType, IndexType> I_faces(deg1, n_faces);
269 VolumeIntegral3D<PPoly, FloatType, IndexType> I_vol(deg1);
270
271 std::vector<FloatType> ds_face(n_faces);
272 std::vector<Point<FloatType, 3>> norm_faces(n_faces);
273 for (IndexType fi = 0; fi < n_faces; ++fi) {
274 const auto p1 = elem.vtxs().col(elem.faces()[fi][0]);
275 const auto p2 = elem.vtxs().col(elem.faces()[fi][1]);
276 const auto p3 = elem.vtxs().col(elem.faces()[fi][2]);
277
278 const Point<FloatType, 3> v1 = p2 - p1;
279 const Point<FloatType, 3> v2 = p3 - p1;
280
281 Point<FloatType, 3> normal_vector = v1.cross(v2);
282 normal_vector /= normal_vector.norm();
283
284 ds_face[fi] = normal_vector.dot(p1);
285 norm_faces[fi] = normal_vector;
286 }
287
288 // length of the edge (distance between two points)
289 std::vector<FloatType> dij(n_edges);
290 // signed distance between the current edge and the origin for
291 // the first/second face to which this edge belongs to
292 std::vector<FloatType> bi1(n_edges);
293 std::vector<FloatType> bi2(n_edges);
294 for (IndexType ei = 0; ei < n_edges; ++ei) {
295 const IndexType e1 = elem.edges()[ei][0];
296 const IndexType e2 = elem.edges()[ei][1];
297 const Point<FloatType, 3> x12 = elem.vtxs().col(e1) - elem.vtxs().col(e2);
298 dij[ei] = x12.norm();
299
300 const IndexType f1 = elem.getEdgeToFaces()[ei][0];
301 const IndexType f2 = elem.getEdgeToFaces()[ei][1];
302
303 // faces orings
304 const Point<FloatType, 3> xF1 = elem.vtxs().col(
305 Pruning ? elem.getFaceCenterIdx()[f1] : elem.faces()[f1][0]);
306 const Point<FloatType, 3> xF2 = elem.vtxs().col(
307 Pruning ? elem.getFaceCenterIdx()[f2] : elem.faces()[f2][0]);
308
309 // normal to the current edge w.r.t. face 1 or 2
310 // each edge `x12` is shared by exactly two faces `f1` and `f2` and it is
311 // walked in opposites directions (when pointing towards the outer normal)
312 const Point<FloatType, 3> normal_edge_1 = norm_faces[f1].cross(x12);
313 const Point<FloatType, 3> normal_edge_2 = x12.cross(norm_faces[f2]);
314 // signed distance w.r.t. face origin
315 bi1[ei] =
316 normal_edge_1.dot(elem.vtxs().col(e1) - xF1) / normal_edge_1.norm();
317 bi2[ei] =
318 normal_edge_2.dot(elem.vtxs().col(e1) - xF2) / normal_edge_2.norm();
319 }
320
321 // Now we compute the integrals by first iterating on all monomials
322 // $x^i y^j z^k$
323 // and for each monomial we start by computing the vertex integral, then
324 // the edge, the face and finally the volume integral
325 //
326 // It follows a manual corner case handling for
327 // (i == 0) || (j == 0) || (k == 0)
328 // into four steps.
329 // In this way we avoid putting an `if` clause in the tripled nested
330 // loop, which is very costly (we reduce the computational cost of ~20%)
331
332 const IndexType eii_max =
333 Pruning ? elem.getEdgesToCompute().size() : elem.edges().size();
334
335 // 0. Case i == 0 && j == 0 && k == 0
336 for (IndexType eii = 0; eii < eii_max; ++eii) {
337 const IndexType ei = Pruning ? elem.getEdgesToCompute()[eii] : eii;
338 I_edges(0, 0, 0, ei) = dij[ei];
339 const IndexType f1 = elem.getEdgeToFaces()[ei][0];
340 I_faces(0, 0, 0, f1) += dij[ei] * bi1[ei] / 2.0;
341 const IndexType f2 = elem.getEdgeToFaces()[ei][1];
342 I_faces(0, 0, 0, f2) += dij[ei] * bi2[ei] / 2.0;
343 }
344
345 for (IndexType fi = 0; fi < n_faces; ++fi) {
346 I_vol(0, 0, 0) += ds_face[fi] * I_faces(0, 0, 0, fi);
347 }
348 I_vol(0, 0, 0) /= 3.0;
349
350 // 1. Case:
351 // ((j == 0) && (k == 0)) || ((i == 0) && (k == 0)) || ((i == 0) && (j == 0))
352 // A single for loop for handling the case where two of the three
353 // indices are zero.
354 for (IndexType l = 1; l < deg1; ++l) {
355 const FloatType ijk1 = l + 1;
356 const FloatType ijk2 = l + 2;
357 const FloatType ijk3 = l + 3;
358 for (IndexType eii = 0; eii < eii_max; ++eii) {
359 const IndexType ei = Pruning ? elem.getEdgesToCompute()[eii] : eii;
360 const Eigen::Vector<FloatType, 3> xF_edge =
361 elem.vtxs().col(elem.edges()[ei][0]);
362 const Eigen::Array<FloatType, 3, 1> x_other =
363 elem.vtxs().col(elem.edges()[ei][1]);
364 const Eigen::Array<FloatType, 3, 1> one = Point<FloatType, 3>::Ones();
365 const Point<FloatType, 3> x_other_l = vmath::pow(x_other, l, one);
366
367 const FloatType Ix_edge =
368 (dij[ei] * x_other_l[0] + l * xF_edge[0] * I_edges(l - 1, 0, 0, ei)) /
369 ijk1;
370 const FloatType Iy_edge =
371 (dij[ei] * x_other_l[1] + l * xF_edge[1] * I_edges(0, l - 1, 0, ei)) /
372 ijk1;
373 const FloatType Iz_edge =
374 (dij[ei] * x_other_l[2] + l * xF_edge[2] * I_edges(0, 0, l - 1, ei)) /
375 ijk1;
376 I_edges(l, 0, 0, ei) += Ix_edge;
377 I_edges(0, l, 0, ei) += Iy_edge;
378 I_edges(0, 0, l, ei) += Iz_edge;
379
380 //
381 const IndexType f1 = elem.getEdgeToFaces()[ei][0];
382 const IndexType f2 = elem.getEdgeToFaces()[ei][1];
383 I_faces(l, 0, 0, f1) += Ix_edge * bi1[ei] / ijk2;
384 I_faces(l, 0, 0, f2) += Ix_edge * bi2[ei] / ijk2;
385 I_faces(0, l, 0, f1) += Iy_edge * bi1[ei] / ijk2;
386 I_faces(0, l, 0, f2) += Iy_edge * bi2[ei] / ijk2;
387 I_faces(0, 0, l, f1) += Iz_edge * bi1[ei] / ijk2;
388 I_faces(0, 0, l, f2) += Iz_edge * bi2[ei] / ijk2;
389 }
390 for (IndexType fi = 0; fi < n_faces; ++fi) {
391 const Eigen::Vector<FloatType, 3> xF_face = elem.vtxs().col(
392 Pruning ? elem.getFaceCenterIdx()[fi] : elem.faces()[fi][0]);
393 const FloatType Ix_face =
394 l * xF_face[0] * I_faces(l - 1, 0, 0, fi) / ijk2;
395 const FloatType Iy_face =
396 l * xF_face[1] * I_faces(0, l - 1, 0, fi) / ijk2;
397 const FloatType Iz_face =
398 l * xF_face[2] * I_faces(0, 0, l - 1, fi) / ijk2;
399 I_faces(l, 0, 0, fi) += Ix_face;
400 I_faces(0, l, 0, fi) += Iy_face;
401 I_faces(0, 0, l, fi) += Iz_face;
402 I_vol(l, 0, 0) += ds_face[fi] * I_faces(l, 0, 0, fi) / ijk3;
403 I_vol(0, l, 0) += ds_face[fi] * I_faces(0, l, 0, fi) / ijk3;
404 I_vol(0, 0, l) += ds_face[fi] * I_faces(0, 0, l, fi) / ijk3;
405 }
406 }
407
408 // 2. Case:
409 // (i == 0) || (j == 0) || (k == 0)
410 // A double for loop for handling the case where one of the three
411 // indices is zero.
412 std::vector<FloatType> monomial_eval_ij0(elem.vtxs().cols());
413 std::vector<FloatType> monomial_eval_i0k(elem.vtxs().cols());
414 std::vector<FloatType> monomial_eval_0jk(elem.vtxs().cols());
415 for (IndexType l1 = 1; l1 < deg1; ++l1) {
416 const IndexType l2_max = PPoly ? deg1 - l1 : deg1;
417 for (IndexType l2 = 1; l2 < l2_max; ++l2) {
418 const IndexType ijk = l1 + l2;
419 const FloatType ijk1 = ijk + 1;
420 const FloatType ijk2 = ijk + 2;
421 const FloatType ijk3 = ijk + 3;
422
423 for (IndexType vi = 0; vi < n_vtxs; ++vi) {
424 monomial_eval_ij0[vi] = vmath::pow(elem.vtxs()(0, vi), l1) *
425 vmath::pow(elem.vtxs()(1, vi), l2);
426 monomial_eval_i0k[vi] = vmath::pow(elem.vtxs()(0, vi), l1) *
427 vmath::pow(elem.vtxs()(2, vi), l2);
428 monomial_eval_0jk[vi] = vmath::pow(elem.vtxs()(1, vi), l1) *
429 vmath::pow(elem.vtxs()(2, vi), l2);
430 }
431 for (IndexType eii = 0; eii < eii_max; ++eii) {
432 const IndexType ei = Pruning ? elem.getEdgesToCompute()[eii] : eii;
433 const Eigen::Vector<FloatType, 3> xF_edge =
434 elem.vtxs().col(elem.edges()[ei][0]);
435 const IndexType x_other_idx = elem.edges()[ei][1];
436
437 const FloatType Ixy_edge =
438 (dij[ei] * monomial_eval_ij0[x_other_idx] +
439 l1 * xF_edge[0] * I_edges(l1 - 1, l2, 0, ei) +
440 l2 * xF_edge[1] * I_edges(l1, l2 - 1, 0, ei)) /
441 ijk1;
442 const FloatType Ixz_edge =
443 (dij[ei] * monomial_eval_i0k[x_other_idx] +
444 l1 * xF_edge[0] * I_edges(l1 - 1, 0, l2, ei) +
445 l2 * xF_edge[2] * I_edges(l1, 0, l2 - 1, ei)) /
446 ijk1;
447 const FloatType Iyz_edge =
448 (dij[ei] * monomial_eval_0jk[x_other_idx] +
449 l1 * xF_edge[1] * I_edges(0, l1 - 1, l2, ei) +
450 l2 * xF_edge[2] * I_edges(0, l1, l2 - 1, ei)) /
451 ijk1;
452 I_edges(l1, l2, 0, ei) += Ixy_edge;
453 I_edges(l1, 0, l2, ei) += Ixz_edge;
454 I_edges(0, l1, l2, ei) += Iyz_edge;
455
456 //
457 const IndexType f1 = elem.getEdgeToFaces()[ei][0];
458 const IndexType f2 = elem.getEdgeToFaces()[ei][1];
459 I_faces(l1, l2, 0, f1) += Ixy_edge * bi1[ei] / ijk2;
460 I_faces(l1, l2, 0, f2) += Ixy_edge * bi2[ei] / ijk2;
461 I_faces(l1, 0, l2, f1) += Ixz_edge * bi1[ei] / ijk2;
462 I_faces(l1, 0, l2, f2) += Ixz_edge * bi2[ei] / ijk2;
463 I_faces(0, l1, l2, f1) += Iyz_edge * bi1[ei] / ijk2;
464 I_faces(0, l1, l2, f2) += Iyz_edge * bi2[ei] / ijk2;
465 }
466 for (IndexType fi = 0; fi < n_faces; ++fi) {
467 const Eigen::Vector<FloatType, 3> xF_face = elem.vtxs().col(
468 Pruning ? elem.getFaceCenterIdx()[fi] : elem.faces()[fi][0]);
469 const FloatType Ixy_face =
470 (l1 * xF_face[0] * I_faces(l1 - 1, l2, 0, fi) +
471 l2 * xF_face[1] * I_faces(l1, l2 - 1, 0, fi)) /
472 ijk2;
473 const FloatType Ixz_face =
474 (l1 * xF_face[0] * I_faces(l1 - 1, 0, l2, fi) +
475 l2 * xF_face[2] * I_faces(l1, 0, l2 - 1, fi)) /
476 ijk2;
477 const FloatType Iyz_face =
478 (l1 * xF_face[1] * I_faces(0, l1 - 1, l2, fi) +
479 l2 * xF_face[2] * I_faces(0, l1, l2 - 1, fi)) /
480 ijk2;
481 I_faces(l1, l2, 0, fi) += Ixy_face;
482 I_faces(l1, 0, l2, fi) += Ixz_face;
483 I_faces(0, l1, l2, fi) += Iyz_face;
484 I_vol(l1, l2, 0) += ds_face[fi] * I_faces(l1, l2, 0, fi) / ijk3;
485 I_vol(l1, 0, l2) += ds_face[fi] * I_faces(l1, 0, l2, fi) / ijk3;
486 I_vol(0, l1, l2) += ds_face[fi] * I_faces(0, l1, l2, fi) / ijk3;
487 }
488 }
489 }
490
491 // 3. Case (i > 0) && (j > 0) && (k > 0)
492 // A triple for loop for handling the case where all indices are different
493 // from zero
494 //
495 // `monomial_eval` is completely over-written at each iteration, we allocate
496 // it here to avoid reallocation (even if it breaks the minimum scope rule)
497 std::vector<FloatType> monomial_eval(elem.vtxs().cols());
498 for (IndexType i = 1; i < deg1; ++i) {
499 const IndexType j_max = PPoly ? deg1 - i : deg1;
500 for (IndexType j = 1; j < j_max; ++j) {
501 const IndexType k_max = PPoly ? deg1 - i - j : deg1;
502 for (IndexType k = 1; k < k_max; ++k) {
503 const IndexType ijk = i + j + k;
504 const FloatType ijk2 = ijk + 2;
505
506 for (IndexType vi = 0; vi < n_vtxs; ++vi) {
507 monomial_eval[vi] = vmath::pow(elem.vtxs()(0, vi), i) *
508 vmath::pow(elem.vtxs()(1, vi), j) *
509 vmath::pow(elem.vtxs()(2, vi), k);
510 }
511
512 for (IndexType eii = 0; eii < eii_max; ++eii) {
513 const IndexType ei = Pruning ? elem.getEdgesToCompute()[eii] : eii;
514 const Eigen::Vector<FloatType, 3> xF_edge =
515 elem.vtxs().col(elem.edges()[ei][0]);
516 FloatType I_edge_tmp = dij[ei] * monomial_eval[elem.edges()[ei][1]];
517 I_edge_tmp += i * xF_edge[0] * I_edges(i - 1, j, k, ei);
518 I_edge_tmp += j * xF_edge[1] * I_edges(i, j - 1, k, ei);
519 I_edge_tmp += k * xF_edge[2] * I_edges(i, j, k - 1, ei);
520 I_edge_tmp /= (ijk + 1);
521 I_edges(i, j, k, ei) = I_edge_tmp;
522
523 //
524 const IndexType f1 = elem.getEdgeToFaces()[ei][0];
525 I_faces(i, j, k, f1) += I_edge_tmp * bi1[ei] / ijk2;
526 const IndexType f2 = elem.getEdgeToFaces()[ei][1];
527 I_faces(i, j, k, f2) += I_edge_tmp * bi2[ei] / ijk2;
528 }
529 for (IndexType fi = 0; fi < n_faces; ++fi) {
530 const Eigen::Vector<FloatType, 3> xF_face = elem.vtxs().col(
531 Pruning ? elem.getFaceCenterIdx()[fi] : elem.faces()[fi][0]);
532 FloatType I_face_tmp = 0.0;
533 I_face_tmp += i * xF_face[0] * I_faces(i - 1, j, k, fi);
534 I_face_tmp += j * xF_face[1] * I_faces(i, j - 1, k, fi);
535 I_face_tmp += k * xF_face[2] * I_faces(i, j, k - 1, fi);
536 I_face_tmp /= ijk2;
537 I_faces(i, j, k, fi) += I_face_tmp;
538 I_vol(i, j, k) += ds_face[fi] * I_faces(i, j, k, fi);
539 }
540 I_vol(i, j, k) /= (ijk + 3);
541 }
542 }
543 }
544 return {I_vol, I_faces};
545}
