257template <
258 bool PPoly,
259 bool Pruning,
260 std::size_t Dim,
261 typename FloatType,
262 typename IndexType>
263auto integrateMonomialsQFree(
264 const geometry::Cell<Dim, FloatType, IndexType>& elem, IndexType deg) {
265 if constexpr (Dim == 2) {
266 return integrateMonomialsQFree2D<PPoly, FloatType, IndexType>(elem, deg);
267 } else if constexpr (Dim == 3) {
268 return integrateMonomialsQFree3D<PPoly, Pruning, FloatType, IndexType>(
269 elem, deg);
270 } else {
271 static_assert(Dim * 0 && "NotImplementedError");
272 }
273}
