121template <bool PPoly, typename FloatType, typename IndexType>
122std::pair<
123 AreaIntegral2D<PPoly, FloatType, IndexType>,
124 EdgeIntegral2D<PPoly, FloatType, IndexType>>
125integrateMonomialsQFree2D(
126 const geometry::Cell2D<FloatType, IndexType>& cell, IndexType deg) {
127 constexpr FloatType Dim = 2.0;
128 using AreaIntegral = AreaIntegral2D<PPoly, FloatType, IndexType>;
129 using EdgeIntegral = EdgeIntegral2D<PPoly, FloatType, IndexType>;
130
131 const IndexType n_vtxs = cell.vtxs().cols();
132 const IndexType deg1 = deg + 1;
133 AreaIntegral area_ints(deg1);
134 EdgeIntegral edge_ints(deg1, n_vtxs);
135
136 // avoid reallocation of caching buffer for monomial powers
137 // at each loop it is fully over-written
138 std::vector<std::array<FloatType, 2>> vxy(deg1);
139 for (IndexType vi = 0; vi < n_vtxs; ++vi) {
140 const FloatType x1 = cell.vtxs()(0, vi);
141 const FloatType y1 = cell.vtxs()(1, vi);
142 const FloatType x2 = cell.vtxs()(0, (vi + 1) % n_vtxs);
143 const FloatType y2 = cell.vtxs()(1, (vi + 1) % n_vtxs);
144
145 // components of unnormalized normal to the edge
146 const FloatType y2y1 = y2 - y1;
147 const FloatType x1x2 = x1 - x2;
148 // length of the edge
149 const FloatType dij = std::sqrt(y2y1 * y2y1 + x1x2 * x1x2);
150 // bi = signed distance as scalar product with outer normal
151 // we pick the first extrema (x1, y1)
152 const FloatType bi = (y2y1 * x1 + x1x2 * y1) / dij;
153
154 edge_ints(0, 0, vi) = dij;
155 area_ints(0, 0) += bi * dij / Dim;
156
157 // handle the case where i or j is equal to zero (in x^i * y^j)
158 // and cache the pair values (x^k, y^k) to be used later
159 vxy[0] = {{x2, y2}};
160 for (IndexType k = 1; k < deg1; ++k) {
161 vxy[k] = {{vxy[k - 1][0] * x2, vxy[k - 1][1] * y2}};
162 const FloatType den_edge = k + 1;
163 edge_ints(k, 0, vi) =
164 (dij * vxy[k - 1][0] + x1 * k * edge_ints(k - 1, 0, vi)) / den_edge;
165 edge_ints(0, k, vi) =
166 (dij * vxy[k - 1][1] + y1 * k * edge_ints(0, k - 1, vi)) / den_edge;
167 const FloatType den = k + 2;
168 area_ints(k, 0) += bi * edge_ints(k, 0, vi) / den;
169 area_ints(0, k) += bi * edge_ints(0, k, vi) / den;
170 }
171
172 // full recursive formula for edge/area integral
173 for (IndexType i = 1; i < deg1; ++i) {
174 // handle the monomial of the space P^k or Q^k
175 const IndexType j_max = PPoly ? deg1 - i : deg1;
176 for (IndexType j = 1; j < j_max; ++j) {
177 const IndexType i_j = i + j;
178 const auto edge_int_ij = (dij * vxy[i - 1][0] * vxy[j - 1][1] +
179 x1 * i * edge_ints(i - 1, j, vi) +
180 y1 * j * edge_ints(i, j - 1, vi)) /
181 (1 + i_j);
182 edge_ints(i, j, vi) = edge_int_ij;
183 area_ints(i, j) += bi * edge_int_ij / (2 + i_j);
184 }
185 }
186 }
187
188 return {area_ints, edge_ints};
189}

274template <bool PPoly, bool Pruning, typename FloatType, typename IndexType>
275std::pair<
276 VolumeIntegral3D<PPoly, FloatType, IndexType>,
277 LowerIntegral3D<PPoly, FloatType, IndexType>>
278integrateMonomialsQFree3D(
279 const geometry::Cell3D<FloatType, IndexType>& elem, IndexType deg) {
280 const IndexType n_edges = elem.edges().size();
281 const IndexType n_faces = elem.nfaces();
282 const IndexType n_vtxs = elem.vtxs().cols();
283 const IndexType deg1 = deg + 1;
284
285 LowerIntegral3D<PPoly, FloatType, IndexType> I_edges(deg1, n_edges);
286 LowerIntegral3D<PPoly, FloatType, IndexType> I_faces(deg1, n_faces);
287 VolumeIntegral3D<PPoly, FloatType, IndexType> I_vol(deg1);
288
289 std::vector<FloatType> ds_face(n_faces);
290 std::vector<Point<FloatType, 3>> norm_faces(n_faces);
291
292 // compute the unit outer normal and the signed face distance
293 for (IndexType fi = 0; fi < n_faces; ++fi) {
294 const auto p1 = elem.vtxs().col(elem.faces()[fi][0]);
295 const auto p2 = elem.vtxs().col(elem.faces()[fi][1]);
296 const auto p3 = elem.vtxs().col(elem.faces()[fi][2]);
297
298 const Point<FloatType, 3> v1 = p2 - p1;
299 const Point<FloatType, 3> v2 = p3 - p1;
300
301 norm_faces[fi] = v1.cross(v2);
302 norm_faces[fi] /= norm_faces[fi].norm();
303
304 ds_face[fi] = norm_faces[fi].dot(p1);
305 }
306
307 // length of the edge (distance between two points)
308 std::vector<FloatType> dij(n_edges);
309 // signed distance between the current edge and the origin for
310 // the first/second face to which this edge belongs to
311 std::vector<FloatType> bi1(n_edges);
312 std::vector<FloatType> bi2(n_edges);
313 for (IndexType ei = 0; ei < n_edges; ++ei) {
314 const IndexType e1 = elem.edges()[ei][0];
315 const IndexType e2 = elem.edges()[ei][1];
316 const Point<FloatType, 3> x12 = elem.vtxs().col(e1) - elem.vtxs().col(e2);
317 dij[ei] = x12.norm();
318
319 const IndexType f1 = elem.getEdgeToFaces()[ei][0];
320 const IndexType f2 = elem.getEdgeToFaces()[ei][1];
321
322 // faces orings
323 const Point<FloatType, 3> xF1 = elem.vtxs().col(
324 Pruning ? elem.getFaceCenterIdx()[f1] : elem.faces()[f1][0]);
325 const Point<FloatType, 3> xF2 = elem.vtxs().col(
326 Pruning ? elem.getFaceCenterIdx()[f2] : elem.faces()[f2][0]);
327
328 // normal to the current edge w.r.t. face 1 or 2
329 // each edge `x12` is shared by exactly two faces `f1` and `f2` and it is
330 // walked in opposites directions (when pointing towards the outer normal)
331 const Point<FloatType, 3> normal_edge_1 = norm_faces[f1].cross(x12);
332 const Point<FloatType, 3> normal_edge_2 = x12.cross(norm_faces[f2]);
333 // signed distance w.r.t. face origin
334 bi1[ei] =
335 normal_edge_1.dot(elem.vtxs().col(e1) - xF1) / normal_edge_1.norm();
336 bi2[ei] =
337 normal_edge_2.dot(elem.vtxs().col(e1) - xF2) / normal_edge_2.norm();
338 }
339
340 // Now we compute the integrals by first iterating on all monomials
341 // $x^i y^j z^k$
342 // and for each monomial we start by computing the vertex integral, then
343 // the edge, the face and finally the volume integral
344 //
345 // It follows a manual corner case handling for
346 // (i == 0) || (j == 0) || (k == 0)
347 // into four steps.
348 // In this way we avoid putting an `if` clause in the tripled nested
349 // loop, which is very costly (we reduce the computational cost of ~20%)
350
351 const IndexType eii_max =
352 Pruning ? elem.getEdgesToCompute().size() : elem.edges().size();
353
354 // 0. Case i == 0 && j == 0 && k == 0
355 for (IndexType eii = 0; eii < eii_max; ++eii) {
356 const IndexType ei = Pruning ? elem.getEdgesToCompute()[eii] : eii;
357 I_edges(0, 0, 0, ei) = dij[ei];
358 const IndexType f1 = elem.getEdgeToFaces()[ei][0];
359 I_faces(0, 0, 0, f1) += dij[ei] * bi1[ei] / 2.0;
360 const IndexType f2 = elem.getEdgeToFaces()[ei][1];
361 I_faces(0, 0, 0, f2) += dij[ei] * bi2[ei] / 2.0;
362 }
363
364 for (IndexType fi = 0; fi < n_faces; ++fi) {
365 I_vol(0, 0, 0) += ds_face[fi] * I_faces(0, 0, 0, fi);
366 }
367 I_vol(0, 0, 0) /= 3.0;
368
369 // 1. Case:
370 // ((j == 0) && (k == 0)) || ((i == 0) && (k == 0)) || ((i == 0) && (j == 0))
371 // A single for loop for handling the case where two of the three
372 // indices are zero.
373 for (IndexType l = 1; l < deg1; ++l) {
374 const FloatType ijk1 = l + 1;
375 const FloatType ijk2 = l + 2;
376 const FloatType ijk3 = l + 3;
377 for (IndexType eii = 0; eii < eii_max; ++eii) {
378 const IndexType ei = Pruning ? elem.getEdgesToCompute()[eii] : eii;
379 const Eigen::Vector<FloatType, 3> xF_edge =
380 elem.vtxs().col(elem.edges()[ei][0]);
381 const Eigen::Array<FloatType, 3, 1> x_other =
382 elem.vtxs().col(elem.edges()[ei][1]);
383 const Eigen::Array<FloatType, 3, 1> one = Point<FloatType, 3>::Ones();
384 const Point<FloatType, 3> x_other_l = vmath::pow(x_other, l, one);
385
386 const FloatType Ix_edge =
387 (dij[ei] * x_other_l[0] + l * xF_edge[0] * I_edges(l - 1, 0, 0, ei)) /
388 ijk1;
389 const FloatType Iy_edge =
390 (dij[ei] * x_other_l[1] + l * xF_edge[1] * I_edges(0, l - 1, 0, ei)) /
391 ijk1;
392 const FloatType Iz_edge =
393 (dij[ei] * x_other_l[2] + l * xF_edge[2] * I_edges(0, 0, l - 1, ei)) /
394 ijk1;
395 I_edges(l, 0, 0, ei) += Ix_edge;
396 I_edges(0, l, 0, ei) += Iy_edge;
397 I_edges(0, 0, l, ei) += Iz_edge;
398
399 //
400 const IndexType f1 = elem.getEdgeToFaces()[ei][0];
401 const IndexType f2 = elem.getEdgeToFaces()[ei][1];
402 I_faces(l, 0, 0, f1) += Ix_edge * bi1[ei] / ijk2;
403 I_faces(l, 0, 0, f2) += Ix_edge * bi2[ei] / ijk2;
404 I_faces(0, l, 0, f1) += Iy_edge * bi1[ei] / ijk2;
405 I_faces(0, l, 0, f2) += Iy_edge * bi2[ei] / ijk2;
406 I_faces(0, 0, l, f1) += Iz_edge * bi1[ei] / ijk2;
407 I_faces(0, 0, l, f2) += Iz_edge * bi2[ei] / ijk2;
408 }
409 for (IndexType fi = 0; fi < n_faces; ++fi) {
410 const Eigen::Vector<FloatType, 3> xF_face = elem.vtxs().col(
411 Pruning ? elem.getFaceCenterIdx()[fi] : elem.faces()[fi][0]);
412 const FloatType Ix_face =
413 l * xF_face[0] * I_faces(l - 1, 0, 0, fi) / ijk2;
414 const FloatType Iy_face =
415 l * xF_face[1] * I_faces(0, l - 1, 0, fi) / ijk2;
416 const FloatType Iz_face =
417 l * xF_face[2] * I_faces(0, 0, l - 1, fi) / ijk2;
418 I_faces(l, 0, 0, fi) += Ix_face;
419 I_faces(0, l, 0, fi) += Iy_face;
420 I_faces(0, 0, l, fi) += Iz_face;
421 I_vol(l, 0, 0) += ds_face[fi] * I_faces(l, 0, 0, fi) / ijk3;
422 I_vol(0, l, 0) += ds_face[fi] * I_faces(0, l, 0, fi) / ijk3;
423 I_vol(0, 0, l) += ds_face[fi] * I_faces(0, 0, l, fi) / ijk3;
424 }
425 }
426
427 // 2. Case:
428 // (i == 0) || (j == 0) || (k == 0)
429 // A double for loop for handling the case where one of the three
430 // indices is zero.
431 std::vector<FloatType> monomial_eval_ij0(elem.vtxs().cols());
432 std::vector<FloatType> monomial_eval_i0k(elem.vtxs().cols());
433 std::vector<FloatType> monomial_eval_0jk(elem.vtxs().cols());
434 for (IndexType l1 = 1; l1 < deg1; ++l1) {
435 const IndexType l2_max = PPoly ? deg1 - l1 : deg1;
436 for (IndexType l2 = 1; l2 < l2_max; ++l2) {
437 const IndexType ijk = l1 + l2;
438 const FloatType ijk1 = ijk + 1;
439 const FloatType ijk2 = ijk + 2;
440 const FloatType ijk3 = ijk + 3;
441
442 for (IndexType vi = 0; vi < n_vtxs; ++vi) {
443 monomial_eval_ij0[vi] = vmath::pow(elem.vtxs()(0, vi), l1) *
444 vmath::pow(elem.vtxs()(1, vi), l2);
445 monomial_eval_i0k[vi] = vmath::pow(elem.vtxs()(0, vi), l1) *
446 vmath::pow(elem.vtxs()(2, vi), l2);
447 monomial_eval_0jk[vi] = vmath::pow(elem.vtxs()(1, vi), l1) *
448 vmath::pow(elem.vtxs()(2, vi), l2);
449 }
450 for (IndexType eii = 0; eii < eii_max; ++eii) {
451 const IndexType ei = Pruning ? elem.getEdgesToCompute()[eii] : eii;
452 const Eigen::Vector<FloatType, 3> xF_edge =
453 elem.vtxs().col(elem.edges()[ei][0]);
454 const IndexType x_other_idx = elem.edges()[ei][1];
455
456 const FloatType Ixy_edge =
457 (dij[ei] * monomial_eval_ij0[x_other_idx] +
458 l1 * xF_edge[0] * I_edges(l1 - 1, l2, 0, ei) +
459 l2 * xF_edge[1] * I_edges(l1, l2 - 1, 0, ei)) /
460 ijk1;
461 const FloatType Ixz_edge =
462 (dij[ei] * monomial_eval_i0k[x_other_idx] +
463 l1 * xF_edge[0] * I_edges(l1 - 1, 0, l2, ei) +
464 l2 * xF_edge[2] * I_edges(l1, 0, l2 - 1, ei)) /
465 ijk1;
466 const FloatType Iyz_edge =
467 (dij[ei] * monomial_eval_0jk[x_other_idx] +
468 l1 * xF_edge[1] * I_edges(0, l1 - 1, l2, ei) +
469 l2 * xF_edge[2] * I_edges(0, l1, l2 - 1, ei)) /
470 ijk1;
471 I_edges(l1, l2, 0, ei) += Ixy_edge;
472 I_edges(l1, 0, l2, ei) += Ixz_edge;
473 I_edges(0, l1, l2, ei) += Iyz_edge;
474
475 //
476 const IndexType f1 = elem.getEdgeToFaces()[ei][0];
477 const IndexType f2 = elem.getEdgeToFaces()[ei][1];
478 I_faces(l1, l2, 0, f1) += Ixy_edge * bi1[ei] / ijk2;
479 I_faces(l1, l2, 0, f2) += Ixy_edge * bi2[ei] / ijk2;
480 I_faces(l1, 0, l2, f1) += Ixz_edge * bi1[ei] / ijk2;
481 I_faces(l1, 0, l2, f2) += Ixz_edge * bi2[ei] / ijk2;
482 I_faces(0, l1, l2, f1) += Iyz_edge * bi1[ei] / ijk2;
483 I_faces(0, l1, l2, f2) += Iyz_edge * bi2[ei] / ijk2;
484 }
485 for (IndexType fi = 0; fi < n_faces; ++fi) {
486 const Eigen::Vector<FloatType, 3> xF_face = elem.vtxs().col(
487 Pruning ? elem.getFaceCenterIdx()[fi] : elem.faces()[fi][0]);
488 const FloatType Ixy_face =
489 (l1 * xF_face[0] * I_faces(l1 - 1, l2, 0, fi) +
490 l2 * xF_face[1] * I_faces(l1, l2 - 1, 0, fi)) /
491 ijk2;
492 const FloatType Ixz_face =
493 (l1 * xF_face[0] * I_faces(l1 - 1, 0, l2, fi) +
494 l2 * xF_face[2] * I_faces(l1, 0, l2 - 1, fi)) /
495 ijk2;
496 const FloatType Iyz_face =
497 (l1 * xF_face[1] * I_faces(0, l1 - 1, l2, fi) +
498 l2 * xF_face[2] * I_faces(0, l1, l2 - 1, fi)) /
499 ijk2;
500 I_faces(l1, l2, 0, fi) += Ixy_face;
501 I_faces(l1, 0, l2, fi) += Ixz_face;
502 I_faces(0, l1, l2, fi) += Iyz_face;
503 I_vol(l1, l2, 0) += ds_face[fi] * I_faces(l1, l2, 0, fi) / ijk3;
504 I_vol(l1, 0, l2) += ds_face[fi] * I_faces(l1, 0, l2, fi) / ijk3;
505 I_vol(0, l1, l2) += ds_face[fi] * I_faces(0, l1, l2, fi) / ijk3;
506 }
507 }
508 }
509
510 // 3. Case (i > 0) && (j > 0) && (k > 0)
511 // A triple for loop for handling the case where all indices are different
512 // from zero
513 //
514 // `monomial_eval` is completely over-written at each iteration, we allocate
515 // it here to avoid reallocation (even if it breaks the minimum scope rule)
516 std::vector<FloatType> monomial_eval(elem.vtxs().cols());
517 for (IndexType i = 1; i < deg1; ++i) {
518 const IndexType j_max = PPoly ? deg1 - i : deg1;
519 for (IndexType j = 1; j < j_max; ++j) {
520 const IndexType k_max = PPoly ? deg1 - i - j : deg1;
521 for (IndexType k = 1; k < k_max; ++k) {
522 const IndexType ijk = i + j + k;
523 const FloatType ijk2 = ijk + 2;
524
525 for (IndexType vi = 0; vi < n_vtxs; ++vi) {
526 monomial_eval[vi] = vmath::pow(elem.vtxs()(0, vi), i) *
527 vmath::pow(elem.vtxs()(1, vi), j) *
528 vmath::pow(elem.vtxs()(2, vi), k);
529 }
530
531 for (IndexType eii = 0; eii < eii_max; ++eii) {
532 const IndexType ei = Pruning ? elem.getEdgesToCompute()[eii] : eii;
533 const Eigen::Vector<FloatType, 3> xF_edge =
534 elem.vtxs().col(elem.edges()[ei][0]);
535 FloatType I_edge_tmp = dij[ei] * monomial_eval[elem.edges()[ei][1]];
536 I_edge_tmp += i * xF_edge[0] * I_edges(i - 1, j, k, ei);
537 I_edge_tmp += j * xF_edge[1] * I_edges(i, j - 1, k, ei);
538 I_edge_tmp += k * xF_edge[2] * I_edges(i, j, k - 1, ei);
539 I_edge_tmp /= (ijk + 1);
540 I_edges(i, j, k, ei) = I_edge_tmp;
541
542 //
543 const IndexType f1 = elem.getEdgeToFaces()[ei][0];
544 I_faces(i, j, k, f1) += I_edge_tmp * bi1[ei] / ijk2;
545 const IndexType f2 = elem.getEdgeToFaces()[ei][1];
546 I_faces(i, j, k, f2) += I_edge_tmp * bi2[ei] / ijk2;
547 }
548 for (IndexType fi = 0; fi < n_faces; ++fi) {
549 const Eigen::Vector<FloatType, 3> xF_face = elem.vtxs().col(
550 Pruning ? elem.getFaceCenterIdx()[fi] : elem.faces()[fi][0]);
551 FloatType I_face_tmp = 0.0;
552 I_face_tmp += i * xF_face[0] * I_faces(i - 1, j, k, fi);
553 I_face_tmp += j * xF_face[1] * I_faces(i, j - 1, k, fi);
554 I_face_tmp += k * xF_face[2] * I_faces(i, j, k - 1, fi);
555 I_face_tmp /= ijk2;
556 I_faces(i, j, k, fi) += I_face_tmp;
557 I_vol(i, j, k) += ds_face[fi] * I_faces(i, j, k, fi);
558 }
559 I_vol(i, j, k) /= (ijk + 3);
560 }
561 }
562 }
563 return {I_vol, I_faces};
564}

566template <
567 bool PPoly,
568 bool Pruning,
569 std::size_t Dim,
570 typename FloatType,
571 typename IndexType>
572auto integrateMonomialsQFree(
573 const geometry::Cell<Dim, FloatType, IndexType>& elem, IndexType deg) {
574 if constexpr (Dim == 2) {
575 return integrateMonomialsQFree2D<PPoly, FloatType, IndexType>(elem, deg);
576 } else if constexpr (Dim == 3) {
577 return integrateMonomialsQFree3D<PPoly, Pruning, FloatType, IndexType>(
578 elem, deg);
579 } else {
580 static_assert(Dim * 0 && "NotImplementedError");
581 }
582}
