256template <
257 bool PPoly,
258 bool Pruning,
259 std::size_t Dim,
260 typename FloatType,
261 typename IndexType>
262auto integrateMonomialsQFree(
263 const geometry::Cell<Dim, FloatType, IndexType>& elem, IndexType deg) {
264 if constexpr (Dim == 2) {
265 return integrateMonomialsQFree2D<PPoly, FloatType, IndexType>(elem, deg);
266 } else if constexpr (Dim == 3) {
267 return integrateMonomialsQFree3D<PPoly, Pruning, FloatType, IndexType>(
268 elem, deg);
269 } else {
270 static_assert(Dim * 0 && "NotImplementedError");
271 }
