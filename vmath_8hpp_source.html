102template <
103 typename U,
104 typename V,
105 typename = std::enable_if_t<std::is_unsigned_v<V> && std::is_integral_v<V>>>
106inline constexpr U upow(U base, V exp, U result = U(1)) {
107 while (exp > 0) {
108 if (exp & 1) {
109 result *= base;
110 }
111 base *= base;
112 exp >>= 1;
113 }
114 return result;
115}

134template <
135 typename U,
136 typename V,
137 typename = std::enable_if_t<std::is_signed_v<V> && std::is_integral_v<V>>>
138inline constexpr U ipow(U base, V exp, U result = U(1)) {
139 if (exp < 0) {
140 exp = -exp;
141 base = result / base;
142 }
143 static_assert(sizeof(V) <= sizeof(std::size_t));
144 // TODO: one extra copy of `base`, can compiler optimize it?
145 // if `sizeof(U)` is less than a word of memory should not matter
146 return upow(base, static_cast<std::size_t>(exp));
147}

269constexpr T epsilon(const T& instance = T(0)) {
270 if constexpr (std::is_floating_point_v<T>) {

315template <
316 typename V,
317 typename U,
318 typename = std::enable_if_t<
319 is_pseudoaritmetic<U>::value && is_pseudoaritmetic<V>::value>>
320inline V to(const U& src) {
321 if constexpr (std::is_same_v<U, V>) {
322 return src;
323 } else if constexpr (std::is_arithmetic_v<U> && std::is_arithmetic_v<V>) {
324 return static_cast<V>(src);
325 } else if constexpr (is_gnump<V>::value) {
326 return V(src);
327 } else if constexpr (
328 is_gnump<U>::value && std::is_floating_point_v<V> &&
329 sizeof(V) <= sizeof(double)) {
330 return static_cast<V>(src.get_d());
331 } else {
332 // dependence on typename is needed to have a well-formed expression
333 static_assert(!sizeof(U), "Not implemented 'to' for this type");
334 }
335}

346template <typename T, typename = std::enable_if_t<is_pseudoaritmetic<T>::value>>
347T factorial(T n) {
348 T result = T(1);
349 while (n > 1) {
350 result *= n;
351 n -= T(1);
352 }
353 return result;
354}

384template <typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
385inline T binomial(T n, T k) {
386 if (k > n - k) { // Use symmetry property (minimize number of loops)
387 k = n - k;
388 }
389
390 T result = 1;
391 for (T i = 0; i < k; ++i) {
392 result *= (n - i);
393 // at 1st iter divide by 1. at 2nd iter for sure there is a factor 2
394 // and so on
395 result /= (i + 1);
396 }
397 return result;
398}

409template <typename T, int Options>
410inline T sarrus_det(const Eigen::Matrix<T, 3, 3, Options>& mat) {
411 return mat(0, 0) * mat(1, 1) * mat(2, 2) + mat(1, 0) * mat(2, 1) * mat(0, 2) +
412 mat(2, 0) * mat(0, 1) * mat(1, 2) - mat(2, 0) * mat(1, 1) * mat(0, 2) -
413 mat(1, 0) * mat(0, 1) * mat(2, 2) - mat(0, 0) * mat(2, 1) * mat(1, 2);
414}

426template <typename T, int Dim, int Options>
427inline T determinant(const Eigen::Matrix<T, Dim, Dim, Options>& mat) {
428 if constexpr (Dim == 2) {
429 return mat(0, 0) * mat(1, 1) - mat(0, 1) * mat(1, 0);
430 } else if constexpr (Dim == 3) {
431 return sarrus_det(mat);
432 } else {
433 static_assert(Dim * 0, "Inline determinant only for Dim 2 or 3");
434 }
435}

437template <typename T, int Dim, int Options>
438inline T area(const Eigen::Matrix<T, Dim, Dim - 1, Options>& mat) {
439 if constexpr (Dim == 2) {
440 return mat.norm();
441 } else if constexpr (Dim == 3) {
442 return mat.col(0).cross(mat.col(1)).norm();
443 } else {
444 static_assert(Dim * 0, "Inline determinant only for Dim 2 or 3");
445 }
446}
