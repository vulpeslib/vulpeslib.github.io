102template <
103 typename U,
104 typename V,
105 typename = std::enable_if_t<std::is_unsigned_v<V> && std::is_integral_v<V>>>
106inline constexpr U upow(U base, V exp, U result = U(1)) {
107 while (exp > 0) {
108 if (exp & 1) {
109 result *= base;
110 }
111 base *= base;
112 exp >>= 1;
113 }
114 return result;
115}

134template <
135 typename U,
136 typename V,
137 typename = std::enable_if_t<std::is_signed_v<V> && std::is_integral_v<V>>>
138inline constexpr U ipow(U base, V exp, U result = U(1)) {
139 if (exp < 0) {
140 exp = -exp;
141 base = result / base;
142 }
143 static_assert(sizeof(V) <= sizeof(std::size_t));
144 return upow(base, static_cast<std::size_t>(exp));
145}

313template <
314 typename V,
315 typename U,
316 typename = std::enable_if_t<
317 is_pseudoaritmetic<U>::value && is_pseudoaritmetic<V>::value>>
318inline V to(const U& src) {
319 if constexpr (std::is_same_v<U, V>) {
320 return src;
321 } else if constexpr (std::is_arithmetic_v<U> && std::is_arithmetic_v<V>) {
322 return static_cast<V>(src);
323 } else if constexpr (is_gnump<V>::value) {
324 return V(src);
325 } else if constexpr (
326 is_gnump<U>::value && std::is_floating_point_v<V> &&
327 sizeof(V) <= sizeof(double)) {
328 return static_cast<V>(src.get_d());
329 } else {
330 // dependence on typename is needed to have a well-formed expression
331 static_assert(!sizeof(U), "Not implemented 'to' for this type");
332 }
333}

344template <typename T, typename = std::enable_if_t<is_pseudoaritmetic<T>::value>>
345inline constexpr T factorial(T n) {
346 T result = T(1);
347 while (n > 1) {
348 result *= n;
349 n -= T(1);
350 }
351 return result;
352}

382template <typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
383constexpr inline T binomial(T n, T k) {
384 if (k > n - k) { // Use symmetry property (minimize number of loops)
385 k = n - k;
386 }
387
388 T result = 1;
389 for (T i = 0; i < k; ++i) {
390 result *= (n - i);
391 // at 1st iter divide by 1. at 2nd iter for sure there is a factor 2
392 // and so on
393 result /= (i + 1);
394 }
395 return result;
396}

407template <typename T, int Options>
408inline T sarrus_det(const Eigen::Matrix<T, 3, 3, Options>& mat) {
409 return mat(0, 0) * mat(1, 1) * mat(2, 2) + mat(1, 0) * mat(2, 1) * mat(0, 2) +
410 mat(2, 0) * mat(0, 1) * mat(1, 2) - mat(2, 0) * mat(1, 1) * mat(0, 2) -
411 mat(1, 0) * mat(0, 1) * mat(2, 2) - mat(0, 0) * mat(2, 1) * mat(1, 2);
412}

424template <typename T, int Dim, int Options>
425inline T determinant(const Eigen::Matrix<T, Dim, Dim, Options>& mat) {
426 if constexpr (Dim == 2) {
427 return mat(0, 0) * mat(1, 1) - mat(0, 1) * mat(1, 0);
428 } else if constexpr (Dim == 3) {
429 return sarrus_det(mat);
430 } else {
431 static_assert(Dim * 0, "Inline determinant only for Dim 2 or 3");

435template <typename T, int Dim, int Options>
436inline T area(const Eigen::Matrix<T, Dim, Dim - 1, Options>& mat) {
437 if constexpr (Dim == 2) {
438 return mat.norm();
439 } else if constexpr (Dim == 3) {
440 return mat.col(0).cross(mat.col(1)).norm() / 2.0;
441 } else {
442 static_assert(Dim * 0, "Inline determinant only for Dim 2 or 3");

473template <typename FloatType, typename IndexType>
474inline void newellNormal(
475 const FloatType* vertices_data,
476 const IndexType* face_indices_data,
477 IndexType num_face_vertices,
478 FloatType* out_normal) {
479 for (IndexType i = 0; i < num_face_vertices; ++i) {
480 // Get vertex indices for the current edge.
481 const IndexType idx_current = face_indices_data[i];
482 const IndexType idx_next = face_indices_data[(i + 1) % num_face_vertices];
483
484 // Create pointers to the start of each vertex's data (x, y, z).
485 const FloatType* p_current = vertices_data + idx_current * 3;
486 const FloatType* p_next = vertices_data + idx_next * 3;
487
488 // Apply Newell's formula using array indexing from the pointers.
489 out_normal[0] += (p_current[1] - p_next[1]) * (p_current[2] + p_next[2]);
490 out_normal[1] += (p_current[2] - p_next[2]) * (p_current[0] + p_next[0]);
491 out_normal[2] += (p_current[0] - p_next[0]) * (p_current[1] + p_next[1]);
492 }
493
494 const FloatType norm = std::sqrt(
495 out_normal[0] * out_normal[0] + out_normal[1] * out_normal[1] +
496 out_normal[2] * out_normal[2]);
497
498 assert(norm > vmath::epsilon<FloatType>());
499 out_normal[0] /= norm;
500 out_normal[1] /= norm;
501 out_normal[2] /= norm;
502}
