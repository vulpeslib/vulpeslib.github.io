106// split a tuple in fe::Expression that are linear-bilinear
108
109// Helper: if the I-th element's type has is_bilinear==true, wrap it in a tuple;
110// otherwise, return an empty tuple.
111template <typename Tuple, std::size_t I>
112constexpr auto extract_true(Tuple&& t) {
113 if constexpr (
114 std::remove_pointer_t<
115 std::tuple_element_t<I, std::decay_t<Tuple>>>::is_bilinear ==
116 fe::IsBilinear) {
117 return std::make_tuple(std::get<I>(std::forward<Tuple>(t)));
118 } else {
119 return std::tuple<>{};
120 }
121}
122
123// Similarly, for false:
124template <typename Tuple, std::size_t I>
125constexpr auto extract_false(Tuple&& t) {
126 if constexpr (!(std::remove_pointer_t<std::tuple_element_t<
127 I, std::decay_t<Tuple>>>::is_bilinear ==
128 fe::IsBilinear)) {
129 return std::make_tuple(std::get<I>(std::forward<Tuple>(t)));
130 } else {
131 return std::tuple<>{};
132 }
133}
134
135// Main implementation: iterate over all indices of the tuple.
136template <typename Tuple, std::size_t... Is>
137constexpr auto split_tuple_impl(Tuple&& t, std::index_sequence<Is...>) {
138 auto true_tuple = std::tuple_cat(extract_true<Tuple, Is>(t)...);
139 auto false_tuple = std::tuple_cat(extract_false<Tuple, Is>(t)...);
140 return std::make_pair(true_tuple, false_tuple);
141}
142
143// Convenience wrapper that deduces the tuple size.
144template <typename... Ts>
145constexpr auto split_expression_tuple(std::tuple<Ts...> t) {
146 return split_tuple_impl(t, std::index_sequence_for<Ts...>{});
147}
