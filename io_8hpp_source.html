47Eigen::MatrixXd readCSVToEigenMatrix(
48 const std::string& filename, char delimiter = ',') {
49 std::ifstream file(filename);
50 if (!file.is_open()) {
51 throw std::runtime_error("Could not open file: " + filename);
52 }
53
54 std::string line;
55 std::vector<double> data;
56 size_t numRows = 0;
57 size_t numCols = 0;
58 while (std::getline(file, line)) {
59 std::stringstream ss(line);
60 std::string cell;
61 size_t colsInCurrentRow = 0;
62
63 while (std::getline(ss, cell, delimiter)) {
64 // Trim whitespace
65 cell.erase(0, cell.find_first_not_of(" \t\r\n"));
66 cell.erase(cell.find_last_not_of(" \t\r\n") + 1);
67
68 // Convert to double and add to data
69 try {
70 data.push_back(std::stod(cell));
71 } catch (const std::exception& e) {
72 throw std::runtime_error(
73 "Failed to convert value to double at row " +
74 std::to_string(numRows + 1) + ", column " +
75 std::to_string(colsInCurrentRow + 1) + ": " + cell);
76 }
77
78 colsInCurrentRow++;
79 }
80
81 // For the first row, establish the column count
82 if (numRows == 0) {
83 numCols = colsInCurrentRow;
84 } else if (colsInCurrentRow != numCols) {
85 // For subsequent rows, check column consistency
86 throw std::runtime_error(
87 "Inconsistent number of columns at row " +
88 std::to_string(numRows + 1) + ": expected " +
89 std::to_string(numCols) + " but got " +
90 std::to_string(colsInCurrentRow));
91 }
92
93 numRows++;
94 }
95
96 return Eigen::Map<Eigen::Matrix<double, -1, -1, Eigen::RowMajor>>(
97 data.data(), numRows, numCols);
98}
