57template <Eigen::Index CrossId, typename ArgType>
58struct evaluator<EigenVulpes::SymMat<CrossId, ArgType>>
59 : evaluator_base<EigenVulpes::SymMat<CrossId, ArgType>> {
60 typedef EigenVulpes::SymMat<CrossId, ArgType> XprType;
61 typedef typename nested_eval<ArgType, XprType::ColsAtCompileTime>::type
62 ArgTypeNested;
63 typedef remove_all_t<ArgTypeNested> ArgTypeNestedCleaned;
64 typedef typename XprType::CoeffReturnType CoeffReturnType;
65
66 enum {
67 CoeffReadCost = evaluator<ArgTypeNestedCleaned>::CoeffReadCost,
68 Flags = Eigen::ColMajor
69 };
70
71 explicit evaluator(const XprType& xpr)
72 : m_argImpl(xpr.m_arg), m_rows(xpr.rows()) {}
73
74 CoeffReturnType coeff(Index row, Index col) const {
75 if (row == CrossId) {
76 if (col == CrossId) {
77 return m_argImpl.coeff(CrossId);
78 }
79 return m_argImpl.coeff(col) / 2.0;
80 } else if (col == CrossId) {
81 return m_argImpl.coeff(row) / 2.0;
82 } else {
83 return 0.0;
84 }
85 }
86
87 evaluator<ArgTypeNestedCleaned> m_argImpl;
88 const Index m_rows;
89};
90} // namespace internal
91} // namespace Eigen
