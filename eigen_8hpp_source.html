59template <Eigen::Index CrossId, typename ArgType>
60struct evaluator<EigenVulpes::SymMat<CrossId, ArgType>>
61 : evaluator_base<EigenVulpes::SymMat<CrossId, ArgType>> {
62 typedef EigenVulpes::SymMat<CrossId, ArgType> XprType;
63 typedef typename nested_eval<ArgType, XprType::ColsAtCompileTime>::type
64 ArgTypeNested;
65 typedef remove_all_t<ArgTypeNested> ArgTypeNestedCleaned;
66 typedef typename XprType::CoeffReturnType CoeffReturnType;
67
68 enum {
69 CoeffReadCost = evaluator<ArgTypeNestedCleaned>::CoeffReadCost,
70 Flags = Eigen::ColMajor
71 };
72
73 explicit evaluator(const XprType& xpr)
74 : m_argImpl(xpr.m_arg), m_rows(xpr.rows()) {}
75
76 CoeffReturnType coeff(Index row, Index col) const {
77 if (row == CrossId) {
78 if (col == CrossId) {
79 return m_argImpl.coeff(CrossId);
80 }
81 return m_argImpl.coeff(col) / 2.0;
82 } else if (col == CrossId) {
83 return m_argImpl.coeff(row) / 2.0;
84 } else {
85 return 0.0;
86 }
87 }
88
89 evaluator<ArgTypeNestedCleaned> m_argImpl;
90 const Index m_rows;
91};
92} // namespace internal
93} // namespace Eigen
