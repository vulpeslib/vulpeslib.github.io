62template <Eigen::Index CrossId, typename ArgType>
63struct evaluator<EigenVulpes::SymMat<CrossId, ArgType>>
64 : evaluator_base<EigenVulpes::SymMat<CrossId, ArgType>> {
65 typedef EigenVulpes::SymMat<CrossId, ArgType> XprType;
66 typedef typename nested_eval<ArgType, XprType::ColsAtCompileTime>::type
67 ArgTypeNested;
68 typedef remove_all_t<ArgTypeNested> ArgTypeNestedCleaned;
69 typedef typename XprType::CoeffReturnType CoeffReturnType;
70
71 enum {
72 CoeffReadCost = evaluator<ArgTypeNestedCleaned>::CoeffReadCost,
73 Flags = Eigen::ColMajor
74 };
75
76 explicit evaluator(const XprType& xpr)
77 : m_argImpl(xpr.m_arg), m_rows(xpr.rows()) {}
78
79 CoeffReturnType coeff(Index row, Index col) const {
80 if (row == CrossId) {
81 if (col == CrossId) {
82 return m_argImpl.coeff(CrossId);
83 }
84 return m_argImpl.coeff(col) / 2.0;
85 } else if (col == CrossId) {
86 return m_argImpl.coeff(row) / 2.0;
87 } else {
88 return 0.0;
89 }
90 }
91
92 evaluator<ArgTypeNestedCleaned> m_argImpl;
93 const Index m_rows;
94};
95} // namespace internal
96} // namespace Eigen

318inline bool compareEigenMatrices(
319 const Eigen::MatrixXd& mat1,
320 const Eigen::MatrixXd& mat2,
321 const double tolerance = 1e-12) {
322 if (mat1.rows() != mat2.rows() || mat1.cols() != mat2.cols()) {
323 std::cout << "Matrix sizes do not match: (" << mat1.rows() << ","
324 << mat1.cols() << ") vs (" << mat2.rows() << "," << mat2.cols()
325 << ")" << std::endl;
326 return false;
327 }
328
329 double norm_diff = (mat1 - mat2).norm();
330 double norm_ref = mat2.norm();
331
332 bool result = norm_diff < tolerance * (norm_ref + 1.0);
333
334 if (!result) {
335 std::cout << "Matrix comparison failed: ||diff||=" << norm_diff
336 << ", ||ref||=" << norm_ref << ", tolerance=" << tolerance
337 << std::endl;
338 }
339
340 return result;
341}
342
343inline bool compareEigenVectors(
344 const Eigen::VectorXd& vec1,
345 const Eigen::VectorXd& vec2,
346 const double tolerance = 1e-12) {
347 if (vec1.size() != vec2.size()) {
348 std::cout << "Vector sizes do not match: " << vec1.size() << " vs "
349 << vec2.size() << std::endl;
350 return false;
351 }
352
353 double norm_diff = (vec1 - vec2).norm();
354 double norm_ref = vec2.norm();
355
356 bool result = norm_diff < tolerance * (norm_ref + 1.0);
357
358 if (!result) {
359 std::cout << "Vector comparison failed: ||diff||=" << norm_diff
360 << ", ||ref||=" << norm_ref << ", tolerance=" << tolerance
361 << std::endl;
362 }
363
364 return result;
365}
