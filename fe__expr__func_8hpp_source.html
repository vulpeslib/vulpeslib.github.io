450template <
451 std::size_t Dim,
452 typename FeSpace,
453 typename DofHandler,
454 std::size_t OutputDim = 1>
455class FeExpressionFeProjection
456 : public FeExpressionBase<
457 Dim,
458 FeExpressionFeProjection<Dim, FeSpace, DofHandler, OutputDim>> {
459 public:
460 static constexpr bool is_leaf = true;
461 static constexpr uint8_t is_bilinear = IsNothing;
462
463 using OutputType = std::
464 conditional_t<OutputDim == 1, double, Eigen::Vector<double, OutputDim>>;
465
466 explicit constexpr FeExpressionFeProjection(
467 const la::Vector& projection,
468 const FeSpace& fe_space,
469 const DofHandler& dof_handler)
470 : m_projection(projection),
471 m_fe_space(fe_space),
472 m_dof_handler(dof_handler) {};
473 FeExpressionFeProjection(const FeExpressionFeProjection&) = delete;
474 FeExpressionFeProjection& operator=(const FeExpressionFeProjection&) = delete;
475 FeExpressionFeProjection(FeExpressionFeProjection&&) = delete;
476 FeExpressionFeProjection& operator=(FeExpressionFeProjection&&) = delete;
477
478 template <std::size_t I = 0, std::size_t J = 0>
479 decltype(auto) evaluate(
480 const FeValues<Dim>&, std::size_t, std::size_t, std::size_t q) const {
481 return m_evals[q];
482 }
483
484 decltype(auto) evaluatePoly(
485 const FeValues<Dim>&,
486 std::size_t,
487 std::size_t,
488 std::size_t,
489 std::size_t) const {
490 const auto& basis = m_fe_space.getBasis();
491 using PolynomialType = std::decay_t<decltype(basis[0])>;
492 const Eigen::Index n_comps = vmath::binomial(m_fe_space.order() + Dim, Dim);
493 assert(static_cast<std::size_t>(n_comps) == basis.size());
494 Eigen::Matrix<PolynomialType, 1, -1> polys(1, n_comps);
495 for (Eigen::Index i = 0; i < n_comps; ++i) {
496 polys(0, i) = basis[i];
497 }
498 return polys;
499 }
500
501 decltype(auto) evaluatePolyCoef(
502 const auto&, const auto& fe_values, std::size_t, std::size_t) const {
503 using ScalarType = typename FeValues<Dim>::real_t;
504 const Eigen::Index n_comps = vmath::binomial(m_fe_space.order() + Dim, Dim);
505 const auto cell_id = fe_values.getCellId();
506 const auto dof_ids = m_dof_handler.getCellToDof()[cell_id];
507 Eigen::Matrix<ScalarType, 1, -1> coefs(1, n_comps);
508 const auto range = m_projection.getOwnershipRange();
509 const auto data = m_projection.toEigenMap();
510 for (Eigen::Index i = 0; i < n_comps; ++i) {
511 coefs(0, i) = data[dof_ids[i] - range[0]];
512 }
513 return coefs;
514 }
515
516 constexpr UpdateFlagsType computeUpdateFlags() const { return UpdateNothing; }
517
518 constexpr std::size_t computeOrder() const { return m_fe_space.order(); }
519
520 constexpr bool precompute(
521 const FeValues<Dim>& fe_values, const CellsType<Dim>&) const {
522 const Eigen::Index nqpts = fe_values.weights().size();
523 assert(fe_values.qpts().cols() == nqpts);
524 m_evals.resize(nqpts);
525 std::fill(m_evals.begin(), m_evals.end(), 0.0);
526 const auto cell_id = fe_values.getCellId();
527 const auto dof_ids = m_dof_handler.getCellToDof()[cell_id];
528 const auto& basis = m_fe_space.getBasisFast();
529 const auto range = m_projection.getOwnershipRange();
530 const auto data = m_projection.toEigenMap();
531 for (Eigen::Index q = 0; q < nqpts; ++q) {
532 const Eigen::Vector<double, Dim> ref_pt = fe_values.qptsRef().col(q);
533 for (std::size_t fn_id = 0; fn_id < basis.size(); ++fn_id) {
534 const double fn_eval = basis[fn_id].evaluate(ref_pt);
535 const auto local_id = dof_ids[fn_id] - range[0];
536 m_evals[q] += data[local_id] * fn_eval;
537 }
538 }
539 return true;
540 }
541
542 protected:
543 const la::Vector& m_projection;
544 const FeSpace& m_fe_space;
545 const DofHandler& m_dof_handler;
546 mutable std::vector<OutputType> m_evals;
547};

610 decltype(auto) evaluatePolyCoef(
611 const auto&, const auto& fe_values, std::size_t, std::size_t) const {
612 using ScalarType = typename FeValues<Dim>::real_t;
613 const Eigen::Index n_comps = vmath::binomial(m_fe_space.order() + Dim, Dim);
614 const auto cell_id = fe_values.getCellId();
615 const auto dof_ids = m_dof_handler.getCellToDof()[cell_id];

617 const auto range = m_projection.getOwnershipRange();
618 const auto data = m_projection.toEigenMap();
619 for (Eigen::Index i = 0; i < n_comps; ++i) {

629 constexpr bool precompute(
630 const FeValues<Dim>& fe_values, const CellsType<Dim>&) const {
631 const Eigen::Index nqpts = fe_values.weights().size();
632 assert(fe_values.qpts().cols() == nqpts);
633 m_evals.resize(nqpts);
634 std::fill(m_evals.begin(), m_evals.end(), 0.0);
635 const auto cell_id = fe_values.getCellId();
636 const auto dof_ids = m_dof_handler.getCellToDof()[cell_id];
637 const auto& basis = m_fe_space.getBasisFast();
638 const auto range = m_projection.getOwnershipRange();
639 const auto data = m_projection.toEigenMap();
640 for (Eigen::Index q = 0; q < nqpts; ++q) {
641 const Eigen::Vector<double, Dim> ref_pt = fe_values.qptsRef().col(q);
642 for (std::size_t fn_id = 0; fn_id < basis.size(); ++fn_id) {
643 const double fn_eval = basis[fn_id].evaluate(ref_pt);
644 const auto local_id = dof_ids[fn_id] - range[0];
645 m_evals[q] += data[local_id] * fn_eval;
646 }
647
648 if constexpr (m_fe_space.needBasisScaling()) {
649 // `FeValues` does not store or expose the mapping object, so we
650 // recover the necessary scaling factor from the quadrature weights.
651 // For affine mappings the ratio weights/weightsRef is constant and
652 // equal to the Jacobian determinant. Using the weights has the
653 // advantage of working even when the jacobian varies with the
654 // quadrature point.
655 const double weight_ref = fe_values.weightsRef()[q];
656 const double weight_phys = fe_values.weights()[q];
657 assert(weight_ref > 0.0);
658 const double scaling = std::sqrt(weight_phys / weight_ref);
659 m_evals[q] /= scaling;
660 }
661 }
662 return true;
663 }

712 constexpr std::size_t computeOrder() const { return m_order; }
713 constexpr bool precompute(
714 const FeValues<Dim>& fe_values, const CellsType<Dim>&) const {
715 const Eigen::Index nqpts = fe_values.weights().size();
716 assert(fe_values.qpts().cols() == nqpts);
717 m_evals.resize(nqpts);
718 for (Eigen::Index q = 0; q < nqpts; ++q) {
719 m_evals[q] = m_func(fe_values.qpts().col(q));
720 }
721 return true;
722 }
723
724 decltype(auto) evaluatePoly(
725 const FeValues<Dim>&,
726 std::size_t,
727 std::size_t,
728 std::size_t,
729 std::size_t) const {
730 return Eigen::Vector<double, Dim * Dim>::Ones().eval();
731 }
732
733 decltype(auto) evaluatePolyCoef(
734 const auto&, const auto&, std::size_t, std::size_t) const {
735 return Eigen::Vector<double, Dim * Dim>{m_evals[0].reshaped()};
736 }

738 protected:

740 mutable std::vector<OutputType> m_evals;
741 const std::size_t m_order = 0;
742};
