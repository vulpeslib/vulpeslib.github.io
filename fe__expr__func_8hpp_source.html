242template <std::size_t Dim, typename E>
243class FeExpressionSkipper
244 : public FeExpressionBase<Dim, FeExpressionSkipper<Dim, E>> {
245 public:
246 static constexpr bool is_leaf = false;
247 static constexpr uint8_t is_bilinear = E::is_bilinear;
248
249 using FuncType =
250 std::function<double(const FeValues<Dim>&, const CellsType<Dim>&)>;
251
252 explicit constexpr FeExpressionSkipper(const FuncType& func, const E& expr)
253 : m_func(func), m_expr(expr) {};
254 FeExpressionSkipper(const FeExpressionSkipper&) = delete;
255 FeExpressionSkipper& operator=(const FeExpressionSkipper&) = delete;
256 FeExpressionSkipper(FeExpressionSkipper&&) = delete;
257 FeExpressionSkipper& operator=(FeExpressionSkipper&&) = delete;
258
259 template <std::size_t I = 0, std::size_t J = 0>
260 decltype(auto) evaluate(
261 const FeValues<Dim>& fe_values,
262 std::size_t i,
263 std::size_t j,
264 std::size_t q) const {
265 return m_expr.template evaluate<I, J>(fe_values, i, j, q);
266 }
267
268 constexpr UpdateFlagsType computeUpdateFlags() const {
269 return m_expr.computeUpdateFlags();
270 }
271 constexpr std::size_t computeOrder() const { return m_expr.computeOrder(); }
272
273 decltype(auto) evaluatePoly(
274 const FeValues<Dim>& fe_values,
275 std::size_t i,
276 std::size_t j,
277 std::size_t,
278 std::size_t) const {
279 return m_expr.evaluatePoly(fe_values, i, j);
280 }
281 decltype(auto) evaluatePolyCoef(
282 const auto& mapping, std::size_t, std::size_t) const {
283 return m_expr.evaluatePolyCoef(mapping);
284 }
285 constexpr bool precompute(
286 const FeValues<Dim>& fe_values, const CellsType<Dim>& cells) const {
287 if (!m_func(fe_values, cells)) {
288 return false;
289 }
290 return m_expr.precompute(fe_values, cells);
291 }
292
293 protected:
294 const FuncType m_func;
295 typename std::conditional_t<E::is_leaf, const E&, const E> m_expr;
296};

306template <std::size_t Dim>
307class FeExpressionFunctionWrapper
308 : public FeExpressionBase<Dim, FeExpressionFunctionWrapper<Dim>> {
309 public:
310 static constexpr bool is_leaf = true;
311 static constexpr uint8_t is_bilinear = IsNothing;
312
313 explicit constexpr FeExpressionFunctionWrapper(
314 const std::function<double(const Eigen::Vector<double, Dim>&)>& func,
315 std::size_t order = 0)
316 : m_func(func), m_order(order) {};
317 FeExpressionFunctionWrapper(const FeExpressionFunctionWrapper&) = delete;
318 FeExpressionFunctionWrapper& operator=(const FeExpressionFunctionWrapper&) =
319 delete;
320 FeExpressionFunctionWrapper(FeExpressionFunctionWrapper&&) = delete;
321 FeExpressionFunctionWrapper& operator=(FeExpressionFunctionWrapper&&) =
322 delete;
323
324 template <std::size_t I = 0, std::size_t J = 0>
325 decltype(auto) evaluate(
326 const FeValues<Dim>&, std::size_t, std::size_t, std::size_t q) const {
327 return m_evals[q];
328 }
329
330 constexpr UpdateFlagsType computeUpdateFlags() const {
331 return UpdateQuadraturePts;
332 }
333 constexpr std::size_t computeOrder() const { return m_order; }
334 constexpr bool precompute(
335 const FeValues<Dim>& fe_values, const CellsType<Dim>&) const {
336 const Eigen::Index nqpts = fe_values.weights().size();
337 assert(fe_values.qpts().cols() == nqpts);
338 m_evals.resize(nqpts);
339 for (Eigen::Index q = 0; q < nqpts; ++q) {
340 m_evals[q] = m_func(fe_values.qpts().col(q));
341 }
342 return true;
343 }
344
345 protected:
346 const std::function<double(const Eigen::Vector<double, Dim>&)> m_func;
347 mutable std::vector<double> m_evals;
348 const std::size_t m_order = 0;
349};
