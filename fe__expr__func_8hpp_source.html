326 return m_expr.template evaluate<I, J>(fe_values, i, j, q);
327 }
328
329 constexpr UpdateFlagsType computeUpdateFlags() const {

331 }

422template <
423 std::size_t Dim,
424 typename FeSpace,
425 typename DofHandler,
426 std::size_t OutputDim = 1>
427class FeExpressionFeProjection
428 : public FeExpressionBase<
429 Dim,
430 FeExpressionFeProjection<Dim, FeSpace, DofHandler, OutputDim>> {
431 public:
432 static constexpr bool is_leaf = true;
433 static constexpr uint8_t is_bilinear = IsNothing;
434
435 using OutputType = std::
436 conditional_t<OutputDim == 1, double, Eigen::Vector<double, OutputDim>>;
437
438 explicit constexpr FeExpressionFeProjection(
439 const la::Vector& projection,
440 const FeSpace& fe_space,
441 const DofHandler& dof_handler)
442 : m_projection(projection),
443 m_fe_space(fe_space),
444 m_dof_handler(dof_handler) {};
445 FeExpressionFeProjection(const FeExpressionFeProjection&) = delete;
446 FeExpressionFeProjection& operator=(const FeExpressionFeProjection&) = delete;
447 FeExpressionFeProjection(FeExpressionFeProjection&&) = delete;
448 FeExpressionFeProjection& operator=(FeExpressionFeProjection&&) = delete;
449
450 template <std::size_t I = 0, std::size_t J = 0>
451 decltype(auto) evaluate(
452 const FeValues<Dim>&, std::size_t, std::size_t, std::size_t q) const {
453 return m_evals[q];
454 }
455
456 decltype(auto) evaluatePoly(
457 const FeValues<Dim>&,
458 std::size_t,
459 std::size_t,
460 std::size_t,
461 std::size_t) const {
462 const auto& basis = m_fe_space.getBasis();
463 using PolynomialType = std::decay_t<decltype(basis[0])>;
464 const Eigen::Index n_comps = vmath::binomial(m_fe_space.order() + Dim, Dim);
465 assert(static_cast<std::size_t>(n_comps) == basis.size());
466 Eigen::Matrix<PolynomialType, 1, -1> polys(1, n_comps);
467 for (Eigen::Index i = 0; i < n_comps; ++i) {
468 polys(0, i) = basis[i];
469 }
470 return polys;
471 }
472
473 decltype(auto) evaluatePolyCoef(
474 const auto&, const auto& fe_values, std::size_t, std::size_t) const {
475 using ScalarType = typename FeValues<Dim>::real_t;
476 const Eigen::Index n_comps = vmath::binomial(m_fe_space.order() + Dim, Dim);
477 const auto cell_id = fe_values.getCellId();
478 const auto dof_ids = m_dof_handler.getCellToDof()[cell_id];
479 Eigen::Matrix<ScalarType, 1, -1> coefs(1, n_comps);
480 const auto range = m_projection.getOwnershipRange();
481 const auto data = m_projection.toEigenMap();
482 for (Eigen::Index i = 0; i < n_comps; ++i) {
483 coefs(0, i) = data[dof_ids[i] - range[0]];
484 }
485 return coefs;
486 }
487
488 constexpr UpdateFlagsType computeUpdateFlags() const { return UpdateNothing; }
489
490 constexpr std::size_t computeOrder() const { return m_fe_space.order(); }
491
492 constexpr bool precompute(
493 const FeValues<Dim>& fe_values, const CellsType<Dim>&) const {
494 const Eigen::Index nqpts = fe_values.weights().size();
495 assert(fe_values.qpts().cols() == nqpts);
496 m_evals.resize(nqpts);
497 std::fill(m_evals.begin(), m_evals.end(), 0.0);
498 const auto cell_id = fe_values.getCellId();
499 const auto dof_ids = m_dof_handler.getCellToDof()[cell_id];
500 const auto& basis = m_fe_space.getBasisFast();
501 const auto range = m_projection.getOwnershipRange();
502 const auto data = m_projection.toEigenMap();
503 for (Eigen::Index q = 0; q < nqpts; ++q) {
504 const Eigen::Vector<double, Dim> ref_pt = fe_values.qptsRef().col(q);
505 for (std::size_t fn_id = 0; fn_id < basis.size(); ++fn_id) {
506 const double fn_eval = basis[fn_id].evaluate(ref_pt);
507 const auto local_id = dof_ids[fn_id] - range[0];
508 m_evals[q] += data[local_id] * fn_eval;
509 }
510 }
511 return true;
512 }
513
514 protected:
515 const la::Vector& m_projection;
516 const FeSpace& m_fe_space;
517 const DofHandler& m_dof_handler;
518 mutable std::vector<OutputType> m_evals;
519};
