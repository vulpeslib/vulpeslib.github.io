44template <typename T>
45inline T jacobiP(
46 const T& x, std::size_t alpha, std::size_t beta, std::size_t N) {
47 const std::size_t alpha_1 = alpha + 1;
48 const std::size_t beta_1 = beta + 1;
49 const std::size_t alpha_beta_1 = alpha_1 + beta;
50 const T alpha_beta_1_T = T(alpha_beta_1);
51 //
52 const T gamma0 = T(vmath::pow(2, alpha_beta_1)) / alpha_beta_1_T *
53 T(vmath::gamma(alpha_1) * vmath::gamma(beta_1)) /
54 vmath::gamma(alpha_beta_1);
55 T jacobi_old_old = 1.0 / vmath::sqrt(gamma0);
56 if (N == 0) {
57 return jacobi_old_old;
58 }
59
60 const T alpha_1_beta_1_T = T(alpha_1 * beta_1);
61 const T alpha_beta_2_T = T(alpha_beta_1 + 1);
62 const T alpha_beta_3_T = T(alpha_beta_1 + 2);
63 const T _2 = T(2);
64 //
65 const T gamma1 = alpha_1_beta_1_T / alpha_beta_3_T * gamma0;
66 T jacobi_old =
67 (alpha_beta_2_T * x / 2 + (T(alpha) - T(beta)) / 2) / vmath::sqrt(gamma1);
68 if (N == 1) {
69 return jacobi_old;
70 }
71
72 T aold =
73 _2 / alpha_beta_2_T * vmath::sqrt(T(alpha_1_beta_1_T / alpha_beta_3_T));
74 T jacobi;
75 for (std::size_t i = 0; i < N - 1; ++i) {
76 const std::size_t h1 = 2 * (i + 1) + alpha + beta;
77 const T h1_2_T = T(h1 + 2);
78 const T anew = _2 / h1_2_T *
79 vmath::sqrt(
80 T(T((i + 2) * (i + 2 + alpha + beta) * (i + 2 + alpha) *
81 (i + 2 + beta)) /
82 T(h1 + 1) / T(h1 + 3)));
83 const T bnew = -(T(alpha * alpha) - T(beta * beta)) / T(h1) / h1_2_T;
84
85 jacobi = (1 / anew) * (-aold * jacobi_old_old + (x - bnew) * jacobi_old);
86 aold = anew;
87
88 jacobi_old_old = jacobi_old;
89 jacobi_old = jacobi;
90 }
91
92 return jacobi;
93}

99template <typename T>
100T simplex2DP(
101 const Eigen::Vector<T, 2>& ab, const std::array<std::size_t, 2>& ij) {
102 const T h1 = jacobiP(ab[0], 0, 0, ij[0]);
103 const T h2 = jacobiP(ab[1], 2 * ij[0] + 1, 0, ij[1]);
104 return vmath::sqrt(T(2.0)) * h1 * h2 * T(vmath::pow(T(1 - ab[1]), ij[0]));
105}

111template <typename T>
112T simplex3DP(
113 const Eigen::Vector<T, 3>& abc, const std::array<std::size_t, 3>& ijk) {
114 const T h1 = jacobiP(abc[0], 0, 0, ijk[0]);
115 const T h2 = jacobiP(abc[1], 2 * ijk[0] + 1, 0, ijk[1]);
116 const T h3 = jacobiP(abc[2], 2 * (ijk[0] + ijk[1]) + 2, 0, ijk[2]);
117 return T(2) * vmath::sqrt(T(2.0)) * h1 * h2 *
118 T(vmath::pow(T(1 - abc[1]), ijk[0])) * h3 *
119 vmath::pow(T(1 - abc[2]), ijk[0] + ijk[1]);
120}

125template <typename T>
126inline Eigen::Vector<T, 2> rs2ab(const Eigen::Vector<T, 2>& rs) {
127 if (rs[1] == T(1)) {
128 return {T(-1), rs[1]};
129 }
130 return {2 * (1 + rs[0]) / (T(1) - rs[1]) - 1, rs[1]};
131}

136template <typename T>
137inline Eigen::Vector<T, 3> rst2abc(const Eigen::Vector<T, 3>& rst) {
138 const T a = (rst[1] + rst[2] != T(0))
139 ? 2 * (1 + rst[0]) / (-rst[1] - rst[2]) - 1
140 : T(-1);
141 const T b = (rst[2] != T(1)) ? 2 * (1 + rst[1]) / (1 - rst[2]) - 1 : T(-1);
142 return {a, b, rst[2]};
143}

149template <typename T, std::size_t Dim>
150T evalSimplex(const Eigen::Vector<T, Dim>& x, std::size_t n) {
151 if constexpr (Dim == 1) {
152 return jacobiP(x[0], 0, 0, n) * vmath::sqrt(T(2.0 / T(2 * n + 1)));
153 } else if constexpr (Dim == 2) {
154 return simplex2DP(rs2ab(x), monomial::getMonomial2D<std::size_t>(n));
155 } else if constexpr (Dim == 3) {
156 return simplex3DP(rst2abc(x), monomial::getMonomial3D<std::size_t>(n));
157 } else {
158 static_assert(Dim * 0 && "evalSimplex not implemented for this Dim");
159 }
160}

166template <typename T, std::size_t Dim>
167Eigen::Vector<T, -1> interpolateFunctionAtNodes(
168 const std::vector<Eigen::Vector<T, Dim>>& nodes,
169 const std::function<T(const Eigen::Vector<T, Dim>&)>& f,
170 std::size_t deg) {
171 using MatrixType = Eigen::Matrix<T, -1, -1>;
172 const std::size_t n_monomials = vmath::binomial(deg + Dim, Dim);
173 MatrixType vandermonde = MatrixType::Ones(nodes.size(), n_monomials);
174 assert(vandermonde.cols() <= vandermonde.rows());
175 for (std::size_t j = 0; j < n_monomials; ++j) {
176 const auto exps = monomial::getMonomial<Dim, std::size_t>(j);
177 for (std::size_t i = 0; i < nodes.size(); ++i) {
178 const auto& pt = nodes[i];
179 for (std::size_t d = 0; d < Dim; ++d) {
180 vandermonde(i, j) *= vmath::pow(pt[d], exps[d]);
181 }
182 }
183 }
184 Eigen::Vector<T, -1> evals(nodes.size());
185 for (std::size_t i = 0; i < nodes.size(); ++i) {
186 evals(i) = f(nodes[i]);
187 }
188
189 Eigen::ColPivHouseholderQR<MatrixType> qr_decomp(vandermonde);
190 const Eigen::Vector<T, -1> coef = qr_decomp.solve(evals);
191#ifndef DNDEBUG
192 const Eigen::Vector<T, -1> res = vandermonde * coef - evals;
193 assert(res.norm() < 1e-30);
194#endif
195 return coef;
196}

203template <std::size_t Dim, typename DofPolicy>
204class FeSpaceModalP : public FeSpaceBase<
205 Dim,
206 internal::real_t,
207 DofPolicy,
208 FeSpaceModalP<Dim, DofPolicy>> {
209 public:
210 using Polynomial = polynomial::NdPolynomial<internal::real_t, Dim>;
211 explicit FeSpaceModalP(std::size_t order)
212 : FeSpaceBase<
213 Dim,
214 internal::real_t,
215 DofPolicy,
216 FeSpaceModalP<Dim, DofPolicy>>(order) {
217 const std::size_t basis_size = vmath::binomial(order + Dim, Dim);
218 for (std::size_t i = 0; i < basis_size; ++i) {
219 const auto exp = monomial::getMonomial<Dim, std::size_t>(i);
220 const auto sum = std::accumulate(exp.begin(), exp.end(), 0);
221 const auto nodes_mpq = internal::LagrangianNodesPolicy<Dim>::computeNodes(
222 std::max(sum + 1, 2));
223 std::vector<Eigen::Vector<internal::real_t, Dim>> nodes;
224 constexpr double delta = 0.1;
225 for (const auto& pt : nodes_mpq) {
226 nodes.push_back(
227 pt.template cast<mpf_class>().array() /
228 internal::real_t(1 - 2 * delta) +
229 internal::real_t(delta));
230 }
231 const auto coef =
232 internal::interpolateFunctionAtNodes<internal::real_t, Dim>(
233 nodes,
234 [&](const Eigen::Vector<internal::real_t, Dim>& x) {
235 return internal::evalSimplex<internal::real_t, Dim>(x, i);
236 },
237 sum);
238
239 m_polys.emplace_back(coef.size());
240 for (int64_t mono_id = 0; mono_id < coef.size(); ++mono_id) {
241 // get the exponents of the current monomial
242 const std::array<std::size_t, Dim> exps =
243 monomial::getMonomial<Dim, std::size_t>(mono_id);
244 // set the coefficient of the polynomial (dimension indipendent)
245 std::apply(
246 [&](auto... args) {
247 const auto c = vmath::to<internal::real_t>(coef(mono_id));
248 const auto c_clip = vmath::abs(c) < 1e-30 ? 0.0 : c;
249 m_polys.back().coef(args...) = c_clip;
250 },
251 exps);
252 }
253 }
254 }
255
256 const auto& getBasis() const { return m_polys; }

265 std::vector<Polynomial> m_polys;
266};
