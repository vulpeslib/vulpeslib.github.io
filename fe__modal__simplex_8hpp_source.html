40template <typename T>
41inline T jacobiP(
42 const T& x, std::size_t alpha, std::size_t beta, std::size_t N) {
43 const std::size_t alpha_1 = alpha + 1;
44 const std::size_t beta_1 = beta + 1;
45 const std::size_t alpha_beta_1 = alpha_1 + beta;
46 const T alpha_beta_1_T = T(alpha_beta_1);
47 //
48 const T gamma0 = T(vmath::pow(2, alpha_beta_1)) / alpha_beta_1_T *
49 T(vmath::gamma(alpha_1) * vmath::gamma(beta_1)) /
50 vmath::gamma(alpha_beta_1);
51 T jacobi_old_old = 1.0 / vmath::sqrt(gamma0);
52 if (N == 0) {
53 return jacobi_old_old;
54 }
55
56 const T alpha_1_beta_1_T = T(alpha_1 * beta_1);
57 const T alpha_beta_2_T = T(alpha_beta_1 + 1);
58 const T alpha_beta_3_T = T(alpha_beta_1 + 2);
59 const T _2 = T(2);
60 //
61 const T gamma1 = alpha_1_beta_1_T / alpha_beta_3_T * gamma0;
62 T jacobi_old =
63 (alpha_beta_2_T * x / 2 + (T(alpha) - T(beta)) / 2) / vmath::sqrt(gamma1);
64 if (N == 1) {
65 return jacobi_old;
66 }
67
68 T aold =
69 _2 / alpha_beta_2_T * vmath::sqrt(T(alpha_1_beta_1_T / alpha_beta_3_T));
70 T jacobi;
71 for (std::size_t i = 0; i < N - 1; ++i) {
72 const std::size_t h1 = 2 * (i + 1) + alpha + beta;
73 const T h1_2_T = T(h1 + 2);
74 const T anew = _2 / h1_2_T *
75 vmath::sqrt(
76 T(T((i + 2) * (i + 2 + alpha + beta) * (i + 2 + alpha) *
77 (i + 2 + beta)) /
78 T(h1 + 1) / T(h1 + 3)));
79 const T bnew = -(T(alpha * alpha) - T(beta * beta)) / T(h1) / h1_2_T;
80
81 jacobi = (1 / anew) * (-aold * jacobi_old_old + (x - bnew) * jacobi_old);
82 aold = anew;
83
84 jacobi_old_old = jacobi_old;
85 jacobi_old = jacobi;
86 }
87
88 return jacobi;
89}

91template <typename T>
92T simplex2DP(
93 const Eigen::Vector<T, 2>& ab, const std::array<std::size_t, 2>& ij) {
94 const T h1 = jacobiP(ab[0], 0, 0, ij[0]);
95 const T h2 = jacobiP(ab[1], 2 * ij[0] + 1, 0, ij[1]);
96 return vmath::sqrt(T(2.0)) * h1 * h2 * T(vmath::pow(T(1 - ab[1]), ij[0]));
97}
98
99template <typename T>
100T simplex3DP(
101 const Eigen::Vector<T, 3>& abc, const std::array<std::size_t, 3>& ijk) {
102 const T h1 = jacobiP(abc[0], 0, 0, ijk[0]);
103 const T h2 = jacobiP(abc[1], 2 * ijk[0] + 1, 0, ijk[1]);
104 const T h3 = jacobiP(abc[2], 2 * (ijk[0] + ijk[1]) + 2, 0, ijk[2]);
105 return T(2) * vmath::sqrt(T(2.0)) * h1 * h2 *
106 T(vmath::pow(T(1 - abc[1]), ijk[0])) * h3 *
107 vmath::pow(T(1 - abc[2]), ijk[0] + ijk[1]);
108}

110template <typename T>
111inline Eigen::Vector<T, 2> rs2ab(const Eigen::Vector<T, 2>& rs) {
112 if (rs[1] == T(1)) {
113 return {T(-1), rs[1]};
114 }
115 return {2 * (1 + rs[0]) / (T(1) - rs[1]) - 1, rs[1]};
116}
117
118template <typename T>
119inline Eigen::Vector<T, 3> rst2abc(const Eigen::Vector<T, 3>& rst) {
120 const T a = (rst[1] + rst[2] != T(0))
121 ? 2 * (1 + rst[0]) / (-rst[1] - rst[2]) - 1
122 : T(-1);
123 const T b = (rst[2] != T(1)) ? 2 * (1 + rst[1]) / (1 - rst[2]) - 1 : T(-1);
124 return {a, b, rst[2]};
125}

127template <typename T, std::size_t Dim>
128T evalSimplex(const Eigen::Vector<T, Dim>& x, std::size_t n) {
129 if constexpr (Dim == 1) {
130 return jacobiP(x[0], 0, 0, n) * vmath::sqrt(T(2.0 / T(2 * n + 1)));
131 } else if constexpr (Dim == 2) {
132 return simplex2DP(rs2ab(x), monomial::getMonomial2D<std::size_t>(n));
133 } else if constexpr (Dim == 3) {
134 return simplex3DP(rst2abc(x), monomial::getMonomial3D<std::size_t>(n));
135 } else {
136 static_assert(Dim * 0 && "evalSimplex not implemented for this Dim");
137 }
138}

140template <typename T, std::size_t Dim>
141Eigen::Vector<T, -1> interpolateFunctionAtNodes(
142 const std::vector<Eigen::Vector<T, Dim>>& nodes,
143 const std::function<T(const Eigen::Vector<T, Dim>&)>& f,
144 std::size_t deg) {
145 using MatrixType = Eigen::Matrix<T, -1, -1>;
146 const std::size_t n_monomials = vmath::binomial(deg + Dim, Dim);
147 MatrixType vandermonde = MatrixType::Ones(nodes.size(), n_monomials);
148 assert(vandermonde.cols() <= vandermonde.rows());
149 for (std::size_t j = 0; j < n_monomials; ++j) {
150 const auto exps = monomial::getMonomial<Dim, std::size_t>(j);
151 for (std::size_t i = 0; i < nodes.size(); ++i) {
152 const auto& pt = nodes[i];
153 for (std::size_t d = 0; d < Dim; ++d) {
154 vandermonde(i, j) *= vmath::pow(pt[d], exps[d]);
155 }
156 }
157 }
158 Eigen::Vector<T, -1> evals(nodes.size());
159 for (std::size_t i = 0; i < nodes.size(); ++i) {
160 evals(i) = f(nodes[i]);
161 }
162
163 Eigen::ColPivHouseholderQR<MatrixType> qr_decomp(vandermonde);
164 const Eigen::Vector<T, -1> coef = qr_decomp.solve(evals);
165#ifndef DNDEBUG
166 const Eigen::Vector<T, -1> res = vandermonde * coef - evals;
167 assert(res.norm() < 1e-30);
168#endif
169 return coef;
170}

174template <std::size_t Dim, typename DofPolicy>
175class FeSpaceModalP : public FeSpaceBase<
176 Dim,
177 internal::real_t,
178 DofPolicy,
179 FeSpaceModalP<Dim, DofPolicy>> {
180 public:
181 using Polynomial = polynomial::NdPolynomial<internal::real_t, Dim>;
182 explicit FeSpaceModalP(std::size_t order)
183 : FeSpaceBase<
184 Dim,
185 internal::real_t,
186 DofPolicy,
187 FeSpaceModalP<Dim, DofPolicy>>(order) {
188 const std::size_t basis_size = vmath::binomial(order + Dim, Dim);
189 for (std::size_t i = 0; i < basis_size; ++i) {
190 const auto exp = monomial::getMonomial<Dim, std::size_t>(i);
191 const auto sum = std::accumulate(exp.begin(), exp.end(), 0);
192 const auto nodes_mpq = internal::LagrangianNodesPolicy<Dim>::computeNodes(
193 std::max(sum + 1, 2));
194 std::vector<Eigen::Vector<internal::real_t, Dim>> nodes;
195 constexpr double delta = 0.1;
196 for (const auto& pt : nodes_mpq) {
197 nodes.push_back(
198 pt.template cast<mpf_class>().array() /
199 internal::real_t(1 - 2 * delta) +
200 internal::real_t(delta));
201 }
202 const auto coef =
203 internal::interpolateFunctionAtNodes<internal::real_t, Dim>(
204 nodes,
205 [&](const Eigen::Vector<internal::real_t, Dim>& x) {
206 return internal::evalSimplex<internal::real_t, Dim>(x, i);
207 },
208 sum);
209
210 m_polys.emplace_back(coef.size());
211 for (int64_t mono_id = 0; mono_id < coef.size(); ++mono_id) {
212 // get the exponents of the current monomial
213 const std::array<std::size_t, Dim> exps =
214 monomial::getMonomial<Dim, std::size_t>(mono_id);
215 // set the coefficient of the polynomial (dimension indipendent)
216 std::apply(
217 [&](auto... args) {
218 const auto c = vmath::to<internal::real_t>(coef(mono_id));
219 const auto c_clip = vmath::abs(c) < 1e-30 ? 0.0 : c;
220 m_polys.back().coef(args...) = c_clip;
221 },
222 exps);
223 }
224 }
225 }
226
227 const auto& getBasis() const { return m_polys; }
228
229 template <typename CellType>
230 inline static constexpr auto getMapping(const CellType& cell) {
231 return 0;
232 };
233
234 protected:
235 std::vector<Polynomial> m_polys;
236};
