71template <typename T>
72constexpr torch::ScalarType toTorchType() {
73 if constexpr (std::is_same_v<T, float>) {
74 return torch::kFloat32;
75 } else if constexpr (std::is_same_v<T, double>) {
76 return torch::kFloat64;
77 } else if constexpr (std::is_same_v<T, int32_t>) {
78 return torch::kInt32;
79 } else if constexpr (std::is_same_v<T, int64_t>) {
80 return torch::kInt64;
81 } else {
82 static_assert(
83 std::is_same_v<T, void>,
84 "Unsupported type: must be float, double, int32_t, or int64_t");
85 }
86}
