32template <
33 std::size_t IndexBegin,
34 std::size_t IndexEnd,
35 std::size_t Dim,
36 typename FeSpace,
37 bool IsTest = false>
38class FeExpressionFunc
39 : public FeExpressionBase<
40 Dim,
41 FeExpressionFunc<IndexBegin, IndexEnd, Dim, FeSpace, IsTest>> {
42 public:
43 static constexpr bool is_leaf = true;
44 static constexpr uint8_t is_bilinear = IsTest ? IsLinearTest : IsLinearTrial;
45 using FeSpaceType = FeSpace;
46 using PolynomialType = std::decay_t<decltype(FeSpace{0}.getBasis()[0])>;
47
48 static constexpr std::size_t k_index_begin = IndexBegin;
49 static constexpr std::size_t k_index_end = IndexEnd;
50
51 explicit constexpr FeExpressionFunc(const FeSpace& fe_space)
52 : m_fe_space(fe_space) {};
53 FeExpressionFunc(const FeExpressionFunc&) = delete;
54 FeExpressionFunc& operator=(const FeExpressionFunc&) = delete;
55 FeExpressionFunc(FeExpressionFunc&&) = delete;
56 FeExpressionFunc& operator=(FeExpressionFunc&&) = delete;
57
58 constexpr const FeSpace& feSpace() const { return m_fe_space; }
59
60 template <std::size_t I = 0, std::size_t J = 0>
61 decltype(auto) evaluate(
62 const FeValues<Dim>& fe_values,
63 std::size_t i,
64 std::size_t j,
65 std::size_t q) const {
66 using ScalarType = typename FeValues<Dim>::real_t;
67 if constexpr (IsTest) {
68 if constexpr (k_index_begin <= J && J < k_index_end) {
69 if constexpr (k_index_end - k_index_begin == 1) {
70 return fe_values.getShapeEval()(j, q);
71 } else {
72 static_assert(k_index_end - k_index_begin == Dim);
73 static constexpr auto DJ = J - IndexBegin;
74 return EigenVulpes::OneComponentVector<DJ, ScalarType, Dim>(
75 fe_values.getShapeEval()(j, q));
76 }
77 } else {
78 return void();
79 }
80 } else {
81 if constexpr (k_index_begin <= I && I < k_index_end) {
82 if constexpr (k_index_end - k_index_begin == 1) {
83 return fe_values.getShapeEval()(i, q);
84 } else {
85 static_assert(k_index_end - k_index_begin == Dim);
86 static constexpr auto DI = I - IndexBegin;
87 return EigenVulpes::OneComponentVector<DI, ScalarType, Dim>(
88 fe_values.getShapeEval()(i, q));
89 }
90 } else {
91 return void();
92 }
93 }
94 }
95
96 constexpr UpdateFlagsType computeUpdateFlags() const { return UpdateValues; }
97 constexpr std::size_t computeOrder() const { return m_fe_space.order(); }

99 decltype(auto) evaluatePoly(
100 const FeValues<Dim>&,
101 std::size_t i,
102 std::size_t j,
103 std::size_t bi,
104 std::size_t bj) const {
105 // TODO: in vectorial case return vector
106 if constexpr (IsTest) {
107 if (k_index_begin <= bj && bj < k_index_end) {
108 if constexpr (k_index_end - k_index_begin == 1) {
109 return Eigen::Vector<PolynomialType, 1>{m_fe_space.getBasis()[j]};
110 } else {
111 static_assert(k_index_end - k_index_begin == Dim);
112 Eigen::Vector<PolynomialType, Dim> result;
113 result(bj - IndexBegin) = m_fe_space.getBasis()[j];
114 return result;
115 }
116 } else {
117 if constexpr (k_index_end - k_index_begin == 1) {
118 return Eigen::Vector<PolynomialType, 1>{};
119 } else {
120 return Eigen::Vector<PolynomialType, Dim>{};
121 }
122 }
123 } else {
124 if (k_index_begin <= bi && bi < k_index_end) {
125 if constexpr (k_index_end - k_index_begin == 1) {
126 return Eigen::Vector<PolynomialType, 1>{m_fe_space.getBasis()[i]};
127 } else {
128 static_assert(k_index_end - k_index_begin == Dim);
129 Eigen::Vector<PolynomialType, Dim> result;
130 result(bi - IndexBegin) = m_fe_space.getBasis()[i];
131 return result;
132 }
133 } else {
134 if constexpr (k_index_end - k_index_begin == 1) {
135 return Eigen::Vector<PolynomialType, 1>{};
136 } else {
137 return Eigen::Vector<PolynomialType, Dim>{};
138 }
139 }
140 }
141 }
142
143 decltype(auto) evaluatePolyCoef(
144 const auto&, const auto&, std::size_t, std::size_t) const {
145 using ScalarType = typename FeValues<Dim>::real_t;
146 if constexpr (k_index_end - k_index_begin == 1) {
147 return Eigen::Vector<ScalarType, 1>{1.0};
148 } else {
149 static_assert(k_index_end - k_index_begin == Dim);
150 return Eigen::Vector<ScalarType, Dim>{
151 Eigen::Vector<ScalarType, Dim>::Ones()};
152 }
153 }
154 constexpr bool precompute(const FeValues<Dim>&, const CellsType<Dim>&) const {
155 return true;
156 }

158 protected:
159 const FeSpace& m_fe_space;

279 if constexpr (k_index_end - k_index_begin == 1) {

288 decltype(auto) evaluatePolyCoef(
289 const auto& mapping, const auto&, std::size_t, std::size_t) const {
290 using CoefType = std::decay_t<decltype(mapping.jacobianInv({}))>;
291 static_assert(CoefType::RowsAtCompileTime > 0);
292 static_assert(CoefType::ColsAtCompileTime == 1);
293 if constexpr (k_index_end - k_index_begin == 1) {
294 return mapping.jacobianInv({});
295 } else {
296 const auto row_jac = mapping.jacobianInv({});
297 using ScalarType = typename FeValues<Dim>::real_t;
298 Eigen::Vector<ScalarType, Dim * Dim> result;
299 for (std::size_t i = 0; i < Dim * Dim; ++i) {
300 result(i) = row_jac(i / Dim);
301 }
302 return result;
303 }
304 }

433 decltype(auto) evaluatePolyCoef(
434 const auto& mapping, const auto&, std::size_t bi, std::size_t bj) const {
435 static_assert(
436 k_index_end - k_index_begin == Dim &&
437 "The SymGrad is defined only for vector spaces.");
438 const auto jac_row = mapping.jacobianInv({});
439 using ScalarType = typename FeValues<Dim>::real_t;
440 Eigen::Matrix<ScalarType, Dim, Dim> jac_mat =
441 Eigen::Matrix<ScalarType, Dim, Dim>::Zero();

443 jac_mat.row(bj - IndexBegin) = jac_row;
444 jac_mat.col(bj - IndexBegin) = jac_row;
445 } else {
446 jac_mat.row(bi - IndexBegin) = jac_row;
447 jac_mat.col(bi - IndexBegin) = jac_row;
448 }
449 return Eigen::Vector<ScalarType, Dim * Dim>{jac_mat.reshaped()};
450 }

611 using FeSpaceType = FeSpace;
612
613 static constexpr std::size_t k_index_begin = IndexBegin;

616 explicit FeExpressionNeighborFunc(
617 const FeExpressionFunc<IndexBegin, IndexEnd, Dim, FeSpace, IsTest>& w)
618 : m_w(w) {}

620 constexpr const auto& feSpace() const { return m_w.feSpace(); }
621
622 template <std::size_t I = 0, std::size_t J = 0>
623 decltype(auto) evaluate(
624 const FeValues<Dim>& fe_values,
625 std::size_t i,
626 std::size_t j,
627 std::size_t q) const {
628 assert(&fe_values.neighValues());
629 return m_w.template evaluate<I, J>(fe_values.neighValues(), i, j, q);

638 const FeValues<Dim>& fe_values,
639 std::size_t i,
640 std::size_t j,
641 std::size_t bi,
642 std::size_t bj) const {
643 using TypeW =
644 std::decay_t<decltype(m_w.evaluatePoly(fe_values, i, j, 0, 0))>;
645 constexpr int RowsW = TypeW::RowsAtCompileTime;
646 constexpr int ColsW = TypeW::ColsAtCompileTime;
647 // NOLINTBEGIN
648 using PolyTypeLifted =
649 std::decay_t<decltype(TypeW{}(0, 0).applyCartesianMapping())>;
650 // NOLINTEND
651 return Eigen::Matrix<PolyTypeLifted, RowsW, ColsW>{
652 m_w.evaluatePoly(fe_values, i, j, bi, bj)
653 .unaryExpr(
654 [](const auto& poly) { return poly.applyCartesianMapping(); })};
655 }

658 const auto& mapping,
659 const auto& fe_values,
