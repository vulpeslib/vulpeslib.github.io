50template <typename FloatType, std::size_t Dim>
51std::tuple<
52 Eigen::Matrix<FloatType, -1, -1>,
53 Eigen::Matrix<FloatType, -1, -1>,
54 Eigen::Matrix<FloatType, Dim, -1>>
55vandermondeDecomposition(std::size_t order) {
56 const auto gl_rule = qrule::gmpGaussLegendreQuadrature(order + 1, 256);
57
58 const std::size_t n_pts = vmath::pow(order + 1, Dim);
59 const std::size_t n_monomials = vmath::binomial(order + Dim, Dim);
60
61 Eigen::Matrix<FloatType, -1, -1> vandermondeT(n_monomials, n_pts);
62 Eigen::Matrix<FloatType, Dim, -1> pts(Dim, n_pts);
63
64 for (std::size_t j = 0; j < n_pts; ++j) {
65 // write `j` in radix `order + 1` to find the cartesian index of the point
66 std::size_t idx = j;
67 for (std::size_t d = 0; d < Dim; ++d) {
68 pts(d, j) = vmath::to<FloatType>(
69 gl_rule.nodes[order - (idx % (order + 1))].point[0]);
70 idx /= (order + 1);
71 }
72 for (std::size_t i = 0; i < n_monomials; ++i) {
73 const auto exp = monomial::getMonomial<Dim, std::size_t>(i);
74 FloatType monomial = 1;
75 for (std::size_t d = 0; d < Dim; ++d) {
76 monomial *= vmath::pow(pts(d, j), exp[d]);
77 }
78 vandermondeT(i, j) = monomial;
79 }
80 }
81 Eigen::ColPivHouseholderQR<decltype(vandermondeT)> qr(vandermondeT);
82 Eigen::Matrix<FloatType, -1, -1> Q = qr.householderQ();
83 Eigen::Matrix<FloatType, -1, -1> R =
84 qr.matrixQR().template triangularView<Eigen::Upper>();
85 Eigen::VectorXi permutationIndices = qr.colsPermutation().indices();
86 Eigen::Matrix<FloatType, -1, -1> R_truncated =
87 R.block(0, 0, n_monomials, n_monomials);
88 Eigen::Matrix<FloatType, Dim, -1> pts_truncated =
89 pts(Eigen::placeholders::all, permutationIndices.head(n_monomials));
90
91 return std::tuple(Q, R_truncated, pts_truncated);
92}
93} // namespace internal

95template <typename FloatType, typename IndexType, std::size_t Dim>
96qrule::QuadratureRule<FloatType, Dim> computeFrugalQuadrature(
97 datastructure::SimplexDataView<Dim, true, FloatType, IndexType> integrals) {
98 using MatrixType = Eigen::Matrix<FloatType, -1, -1>;
99 using QRType = std::tuple<MatrixType, MatrixType, MatrixType>;
100 static std::vector<QRType> s_vandermonde_collection;
101 static std::vector<bool> s_vandermonde_collection_built;
102 assert(integrals.dim() > 0);
103 const std::size_t order = integrals.dim() - 1;
104 if (s_vandermonde_collection.size() <= order) {
105 s_vandermonde_collection.resize(order + 1);
106 s_vandermonde_collection_built.resize(order + 1, false);
107 }
108 if (!s_vandermonde_collection_built[order]) {
109 s_vandermonde_collection_built[order] = true;
110 s_vandermonde_collection[order] =
111 internal::vandermondeDecomposition<FloatType, Dim>(order);
112 }
113 const auto& [Q, R, pts] = s_vandermonde_collection[order];
114 const Eigen::Map<const Eigen::Vector<FloatType, -1>> integral_map(
115 integrals.begin(), integrals.numel());
116 assert(integrals.numel() == pts.cols());
117 const Eigen::Vector<FloatType, -1> y =
118 R.template triangularView<Eigen::Upper>().solve(
119 Q.transpose() * integral_map);
120
121 qrule::QuadratureRule<FloatType, Dim> rule;
122 rule.degree = order;
123 for (int64_t i = 0; i < pts.cols(); ++i) {
124 std::array<FloatType, Dim> pt;
125 for (std::size_t d = 0; d < Dim; ++d) {
126 pt[d] = pts(d, i);
127 }
128 rule.nodes.emplace_back(pt, y(i));
129 }
130 return rule;
131}
