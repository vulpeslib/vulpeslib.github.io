50template <typename DofHandlerType>
51inline Vector extractComponent(
52 const Vector& blocked_vec,
53 const DofHandlerType& dof_handler,
54 std::size_t f) {
55 assert(
56 static_cast<PetscInt>(f) < blocked_vec.getBlockSize() &&
57 "Component index out of range");
58
59 Vector comp(dof_handler);
60 PetscCallAbortWorld(VecStrideGather(
61 blocked_vec.getPetscVector(), static_cast<PetscInt>(f),
62 comp.getPetscVector(), INSERT_VALUES));
63 return comp;
64}

74template <std::size_t NFields, typename DofHandlerType>
75inline std::vector<Vector> extractAllComponents(
76 const Vector& blocked_vec, const DofHandlerType& dof_handler) {
77 assert(
78 blocked_vec.getBlockSize() == static_cast<PetscInt>(NFields) &&
79 "blocked_vec block size must equal NFields");
80
81 std::vector<Vector> components;
82 components.reserve(NFields);
83 for (std::size_t f = 0; f < NFields; ++f) {
84 components.push_back(extractComponent(blocked_vec, dof_handler, f));
85 }
86 return components;
87}

97template <std::size_t NFields, typename DofHandlerType>
98inline Vector mergeBlockedVector(
99 const std::vector<Vector>& scalars, const DofHandlerType& dof_handler) {
100 assert(scalars.size() == NFields && "scalars.size() must equal NFields");
101
102 Vector blocked(dof_handler, static_cast<PetscInt>(NFields));
103
104 for (std::size_t f = 0; f < NFields; ++f) {
105 assert(
106 scalars[f].getBlockSize() == 1 &&
107 "scalar components must be plain vectors");
108 PetscCallAbortWorld(VecStrideScatter(
109 scalars[f].getPetscVector(), static_cast<PetscInt>(f),
110 blocked.getPetscVector(), INSERT_VALUES));
111 }
112 return blocked;
113}

122// Maximum scalar dimension for which toEigenDense() is permitted.
123// Beyond this limit the O(N^2) dense allocation would exhaust memory.
124static constexpr PetscInt kMaxDenseExtractSize = 2000;
125
126inline void extractDiagonalBlockComponent(
127 const SparseMatrix& blocked_mat,
128 std::size_t component,
129 SparseMatrix& scalar_mat,
130 double zero_tol = 0.0) {
131 const PetscInt bs = blocked_mat.getBlockSize();
132 assert(bs > 0);
133 assert(static_cast<PetscInt>(component) < bs);
134
135 const auto size = blocked_mat.getSize();
136 assert(size[0] == size[1]);
137 assert(size[0] % bs == 0);
138
139 const PetscInt n_scalar = size[0] / bs;
140 assert(
141 scalar_mat.getSize()[0] == n_scalar &&
142 scalar_mat.getSize()[1] == n_scalar);
143
144 if (n_scalar > kMaxDenseExtractSize) {
145 std::cerr << "[warning] extractDiagonalBlockComponent: matrix too large ("
146 << n_scalar << " > " << kMaxDenseExtractSize
147 << ") for dense extraction -- skipping, scalar_mat left empty.\n";
148 return;
149 }
150
151 const Eigen::MatrixXd dense = blocked_mat.toEigenDense();
152 for (PetscInt i = 0; i < n_scalar; ++i) {
153 for (PetscInt j = 0; j < n_scalar; ++j) {
154 const double v = dense(
155 i * bs + static_cast<PetscInt>(component),
156 j * bs + static_cast<PetscInt>(component));
157 if (std::abs(v) <= zero_tol) {
158 continue;
159 }
160 scalar_mat.insertValue(v, i, j, INSERT_VALUES);
161 }
162 }
163 scalar_mat.compress();
164}

173template <std::size_t NFields>
174inline void mergeDiagonalBlockedMatrix(
175 const std::array<std::reference_wrapper<const SparseMatrix>, NFields>&
176 scalar_components,
177 SparseMatrix& blocked_mat,
178 double zero_tol = 0.0) {
179 const PetscInt bs = blocked_mat.getBlockSize();
180 assert(bs == static_cast<PetscInt>(NFields));
181
182 const auto blocked_size = blocked_mat.getSize();
183 assert(blocked_size[0] == blocked_size[1]);
184 assert(blocked_size[0] % bs == 0);
185 const PetscInt n_scalar = blocked_size[0] / bs;
186
187 if (n_scalar > kMaxDenseExtractSize) {
188 std::cerr << "[warning] mergeDiagonalBlockedMatrix: matrix too large ("
189 << n_scalar << " > " << kMaxDenseExtractSize
190 << ") for dense merge -- skipping, blocked_mat left empty.\n";
191 return;
192 }
193
194 for (std::size_t f = 0; f < NFields; ++f) {
195 const Eigen::MatrixXd dense = scalar_components[f].get().toEigenDense();
196 assert(dense.rows() == n_scalar && dense.cols() == n_scalar);
197 for (PetscInt i = 0; i < n_scalar; ++i) {
198 for (PetscInt j = 0; j < n_scalar; ++j) {
199 const double v = dense(i, j);
200 if (std::abs(v) <= zero_tol) {
201 continue;
202 }
203 blocked_mat.insertValue(
204 v, i * bs + static_cast<PetscInt>(f),
205 j * bs + static_cast<PetscInt>(f), INSERT_VALUES);
206 }
207 }
208 }
209 blocked_mat.compress();
210}
