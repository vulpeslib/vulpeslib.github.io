71 explicit Tensor(const std::string& filename, const std::string& dataset_name)
72 : m_vec_data(nullptr) {
73 VecCreate(PETSC_COMM_WORLD, &m_vec_data);
74 PetscObjectSetName((PetscObject)m_vec_data, dataset_name.c_str());
75
76 PetscViewer viewer;
77 PetscCallAbortWorld(PetscViewerHDF5Open(
78 PETSC_COMM_WORLD, filename.c_str(), FILE_MODE_READ, &viewer));
79 PetscCallAbortWorld(VecLoad(m_vec_data, viewer));
80 PetscCallAbortWorld(PetscViewerDestroy(&viewer));
81
82 VecSetFromOptions(m_vec_data);
83 }
84
85 // Delete copy operations
86 Tensor(const Tensor&) = delete;
87 Tensor& operator=(const Tensor&) = delete;
88
96 template <int Dim>
97 LocalMap<Dim> getEigenMap(PetscInt local_index) const {
98 const PetscInt components_per_cell = Dim * Dim;
99 if (!m_raw_ptr) {
100 PetscCallAbortWorld(VecGetArrayRead(m_vec_data, &m_raw_ptr));
101 }
102 // Calculate the memory offset for this specific local cell
103 return LocalMap<Dim>(m_raw_ptr + (local_index * components_per_cell));
104 }
105
113 template <int Dim, typename IndexT>
114 auto getEvaluation(IndexT local_index) const {
115 return getEigenMap<Dim>(static_cast<PetscInt>(local_index)).eval();
116 }
117
118 ~Tensor() {
119 if (m_vec_data) {
120 VecDestroy(&m_vec_data);
121 }
122 }
123
124 private:
126 Vec m_vec_data;
128 mutable const PetscScalar* m_raw_ptr = nullptr;
129};
