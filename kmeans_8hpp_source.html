57template <typename T, int Size>
58std::pair<std::vector<size_t>, Eigen::Matrix<T, Size, -1>> kmeans(
59 Eigen::Map<const Eigen::Matrix<T, Size, Eigen::Dynamic>> data,
60 size_t k,
61 T toll = 10 * std::numeric_limits<T>::epsilon(),
62 size_t max_iters = 100,
63 size_t seed = 0) {
64 size_t num_points = data.cols();
65 std::mt19937 gen(seed);
66 std::uniform_int_distribution<size_t> dist(0, num_points - 1);
67
68 Eigen::Matrix<T, Size, -1> centroids(k);
69 Eigen::Matrix<T, Size, -1> new_centroids(k);
70 std::vector<size_t> cluster_sizes(k, 0);
71 std::vector<size_t> cluster_assignment(num_points);
72 for (size_t i = 0; i < k; ++i) {
73 centroids.col(i) = data.col(dist(gen));
74 }
75
76 T err = std::numeric_limits<T>::max();
77 size_t iter = 0;
78 while ((err > toll) && (iter < max_iters)) {
79 for (size_t i = 0; i < num_points; ++i) {
80 T best_distance = std::numeric_limits<T>::max();
81 for (size_t j = 0; j < k; ++j) {
82 const T distance = (data.col(i) - centroids.col(j)).squaredNorm();
83 if (distance < best_distance) {
84 best_distance = distance;
85 cluster_assignment[i] = j;
86 }
87 }
88 }
89
90 std::fill(cluster_sizes.begin(), cluster_sizes.end(), 0);
91 new_centroids.setZero();
92 for (size_t i = 0; i < num_points; ++i) {
93 size_t cluster = cluster_assignment[i];
94 new_centroids.col(cluster) += data.col(i);
95 ++cluster_sizes[cluster];
96 }
97
98 for (size_t j = 0; j < k; ++j) {
99 if (cluster_sizes[j] > 0) {
100 new_centroids.col(j) /= cluster_sizes[j];
101 } else {
102 new_centroids.col(j) = data.col(dist(gen));
103 }
104 }
105
106 err = (centroids - new_centroids).norm();
107 std::swap(centroids, new_centroids);
108 ++iter;
109 }
110
111 return {cluster_assignment, centroids};
112}
