114template <typename T, size_t Dim>
115class KMeansNaive : public KMeansBase<T, Dim> {
116 public:
117 using LabelType = KMeansBase<T, Dim>::LabelType;
118 using DataType = KMeansBase<T, Dim>::DataType;
119 using CentersType = KMeansBase<T, Dim>::CentersType;
120
121 KMeansNaive(
122 DataType x,
123 size_t k,
124 size_t max_iter = 1e6,
125 T tol = 1e-6,
126 std::size_t seed = 0)
127 : KMeansBase<T, Dim>(k, max_iter, tol, seed), m_data_points(x) {
128 assert(x.cols() >= k);
129 this->m_n_samples = x.cols();
130 this->m_centroids.resize(Dim, k);
131 this->m_labels.resize(this->m_n_samples);
132 };
133
134 void fit() override;
135
136 private:
137 DataType m_data_points;
138};
