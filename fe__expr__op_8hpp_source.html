47// While I would like to avoid macros as much as possible this is a lot of
48// boilerplate to copy and paste on each binary expression.
49// Inheritance is out of the possibilities because it is needed to inherit
50// directly from the base class to solve at compile time the `decltype(auto)`
51// with the CRTP
52#define FE_EXPRESSION_BINARY_IMPL(Dim, E1, E2) \
53 public: \
54 static constexpr bool is_leaf = false; \
55 static constexpr uint8_t is_bilinear = E1::is_bilinear | E2::is_bilinear; \
56 static_assert( \
57 std::is_same_v<typename E1::FeSpaceType, void> || \
58 std::is_same_v<typename E2::FeSpaceType, void> || \
59 std::is_same_v<typename E1::FeSpaceType, typename E2::FeSpaceType>); \
60 using FeSpaceType = std::conditional_t< \
61 std::is_same_v<typename E1::FeSpaceType, void>, \
62 typename E2::FeSpaceType, typename E1::FeSpaceType>; \
63 constexpr const auto& feSpace() const { \
64 if constexpr (std::is_same_v<FeSpaceType, void>) { \
65 /* this is always false, see static assert */ \
66 return nullptr; \
67 } else if constexpr (std::is_same_v<typename E1::FeSpaceType, void>) { \
68 return m_v.feSpace(); \
69 } else { \
70 return m_u.feSpace(); \
71 } \
72 } \
73 using PolynomialType = internal::PolynomialTypeHelper<FeSpaceType>::type; \
74 static constexpr std::size_t k_index_begin = \
75 std::min(E1::k_index_begin, E2::k_index_begin); \
76 static constexpr std::size_t k_index_end = \
77 std::max(E1::k_index_end, E2::k_index_end); \
78 constexpr UpdateFlagsType computeUpdateFlags() const { \
79 return m_u.computeUpdateFlags() | m_v.computeUpdateFlags(); \
80 } \
81 constexpr bool precompute( \
82 const FeValues<Dim>& fe_values, const CellsType<Dim>& cells) const { \
83 return m_u.precompute(fe_values, cells) && \
84 m_v.precompute(fe_values, cells); \
85 } \
86 \
87 protected: \
88 typename std::conditional_t<E1::is_leaf, const E1&, const E1> m_u; \
89 typename std::conditional_t<E2::is_leaf, const E2&, const E2> m_v;

95 : public FeExpressionBase<Dim, FeExpressionInner<Dim, E1, E2>> {

100 template <std::size_t I = 0, std::size_t J = 0>
101 decltype(auto) evaluate(
102 const FeValues<Dim>& fe_values,
103 std::size_t i,
104 std::size_t j,
105 std::size_t q) const {
106 using TypeEvalU = decltype(m_u.template evaluate<I, J>(fe_values, i, j, q));
107 using TypeEvalV = decltype(m_v.template evaluate<I, J>(fe_values, i, j, q));
108 if constexpr (
109 std::is_same_v<TypeEvalU, void> || std::is_same_v<TypeEvalV, void>) {
110 return void();
111 } else {
112 return m_u.template evaluate<I, J>(fe_values, i, j, q)
113 .cwiseProduct(m_v.template evaluate<I, J>(fe_values, i, j, q))
114 .sum();
115 }
116 }
117
118 constexpr std::size_t computeOrder() const {
119 return m_u.computeOrder() + m_v.computeOrder();
120 }

122 decltype(auto) evaluatePoly(
123 const FeValues<Dim>& fe_values,
124 std::size_t i,
125 std::size_t j,
126 std::size_t bi,
127 std::size_t bj) const {
128 // TODO: implement general case with different number of cols
129 using TypeU =
130 std::decay_t<decltype(m_u.evaluatePoly(fe_values, i, j, bi, bj))>;
131 using TypeV =
132 std::decay_t<decltype(m_v.evaluatePoly(fe_values, i, j, bi, bj))>;
133 if constexpr (std::is_same_v<void, TypeU>) {
134 return m_v.evaluatePoly(fe_values, i, j, bi, bj).transpose().eval();
135 } else if constexpr (std::is_same_v<void, TypeV>) {
136 return m_u.evaluatePoly(fe_values, i, j, bi, bj).transpose().eval();
137 } else {
138 const auto [pu, pv] = internal::lift_to_same_space(
139 m_u.evaluatePoly(fe_values, i, j, bi, bj),
140 m_v.evaluatePoly(fe_values, i, j, bi, bj));
141 using PolyLiftedType = decltype(pu)::Scalar;
142 // on different rows we have different spatial components
143 // on different cols we have different sum components of the final result
144 // e.g. if we have u * v + inner(grad(u), grad(v))
145 // the u * v and inner(grad(u), grad(v)) have different transformation
146 // when mapping the integral from the reference element to the physical
147 // space, thus, we keep each different addend of the sum on different cols
148 assert(pu.rows() == pv.rows());
149 if ((pu.cols() == pv.cols()) && (pu.cols() == 1)) {
150 return Eigen::Matrix<PolyLiftedType, 1, -1>{
151 pu.cwiseProduct(pv).reshaped()};
152 } else {
153 // for ex. this is the case of inner(sigma * grad(u), grad(v))
154 // where sigma is a dim x dim tensor
155 return Eigen::Matrix<PolyLiftedType, 1, -1>{
156 internal::cartesian_hcat(pu, pv).reshaped()};
157 }
158 }
159 }
160
161 decltype(auto) evaluatePolyCoef(
162 const auto& mapping,
163 const auto& fe_values,
164 std::size_t bi,
165 std::size_t bj) const {
166 const auto cu = m_u.evaluatePolyCoef(mapping, fe_values, bi, bj);
167 const auto cv = m_v.evaluatePolyCoef(mapping, fe_values, bi, bj);
168 assert(cu.rows() == cv.rows());
169 if ((cu.cols() == cv.cols()) && (cu.cols() == 1)) {
170 return Eigen::Matrix<double, 1, -1>{(cu.cwiseProduct(cv)).transpose()};
171 } else {
172 return Eigen::Matrix<double, 1, -1>{
173 internal::cartesian_hcat(cu, cv).reshaped()};
174 }
175 }
176};

196 template <std::size_t I = 0, std::size_t J = 0>
197 decltype(auto) evaluate(
198 const FeValues<Dim>& fe_values,
199 std::size_t i,
200 std::size_t j,
201 std::size_t q) const {
202 using TypeEvalU = decltype(m_u.template evaluate<I, J>(fe_values, i, j, q));
203 using TypeEvalV = decltype(m_v.template evaluate<I, J>(fe_values, i, j, q));
204 if constexpr (
205 std::is_same_v<TypeEvalU, void> || std::is_same_v<TypeEvalV, void>) {
206 return void();
207 } else {
208 return m_u.template evaluate<I, J>(fe_values, i, j, q) *
209 m_v.template evaluate<I, J>(fe_values, i, j, q).transpose();

217 decltype(auto) evaluatePoly(
218 const FeValues<Dim>& fe_values,
219 std::size_t i,
220 std::size_t j,
221 std::size_t bi,
222 std::size_t bj) const {
223 using TypeU =
224 std::decay_t<decltype(m_u.evaluatePoly(fe_values, i, j, bi, bj))>;
225 using TypeV =
226 std::decay_t<decltype(m_v.evaluatePoly(fe_values, i, j, bi, bj))>;
227 if constexpr (std::is_same_v<void, TypeU>) {
228 const auto pv = m_v.evaluatePoly(fe_values, i, j, bi, bj);
229 assert(pv.rows() == Dim);
230 using PolyType = typename TypeV::Scalar;
231 Eigen::Matrix<PolyType, Dim * Dim, -1> result(Dim * Dim, pv.cols());
232 for (Eigen::Index k = 0; k < pv.cols(); ++k) {
233 result.col(k) = pv.col(k).transpose().replicate(Dim, 1).reshaped();
234 }
235 return result;
236 } else if constexpr (std::is_same_v<void, TypeV>) {
237 const auto pu = m_u.evaluatePoly(fe_values, i, j, bi, bj);
238 assert(pu.rows() == Dim);
239 using PolyType = typename TypeU::Scalar;
240 Eigen::Matrix<PolyType, Dim * Dim, -1> result(Dim * Dim, pu.cols());
241 for (Eigen::Index k = 0; k < pu.cols(); ++k) {
242 result.col(k) = pu.col(k).replicate(1, Dim).reshaped();
243 }
244 return result;
245 } else {
246 const auto [pu, pv] = internal::lift_to_same_space(
247 m_u.evaluatePoly(fe_values, i, j, bi, bj),
248 m_v.evaluatePoly(fe_values, i, j, bi, bj));
249 assert(pu.rows() == pv.rows());
250 assert(pu.rows() == Dim);
251 constexpr auto Dim2 = Dim * Dim;
252 using PolyLiftType = decltype(pu)::Scalar;
253 Eigen::Matrix<PolyLiftType, Dim2, -1> result(Dim2, pu.cols() * pv.cols());
254 Eigen::Index idx = 0;
255 for (Eigen::Index ki = 0; ki < pu.cols(); ++ki) {
256 for (Eigen::Index kj = 0; kj < pv.cols(); ++kj) {
257 result.col(idx++) = (pu.col(ki) * pv.col(kj).transpose()).reshaped();
258 }
259 }
260 return result;
261 }
262 }
263
264 decltype(auto) evaluatePolyCoef(
265 const auto& mapping,
266 const auto& fe_values,
267 std::size_t bi,
268 std::size_t bj) const {
269 const auto cu = m_u.evaluatePolyCoef(mapping, fe_values, bi, bj);
270 const auto cv = m_v.evaluatePolyCoef(mapping, fe_values, bi, bj);
271 assert(cu.rows() == Dim);
272 assert(cv.rows() == Dim);
273 constexpr auto Dim2 = Dim * Dim;
274 Eigen::Matrix<double, Dim2, -1> result(Dim2, cu.cols() * cv.cols());
275 Eigen::Index idx = 0;
276 for (Eigen::Index i = 0; i < cu.cols(); ++i) {
277 for (Eigen::Index j = 0; j < cv.cols(); ++j) {
278 result.col(idx++) = (cu.col(i) * cv.col(j).transpose()).reshaped();
279 }
280 }
281 return result;
282 }

312 std::is_same_v<TypeEvalU, void> || std::is_same_v<TypeEvalV, void>) {
313 return void();

324 decltype(auto) evaluatePoly(
325 const FeValues<Dim>& fe_values,
326 std::size_t i,
327 std::size_t j,
328 std::size_t bi,
329 std::size_t bj) const {
330 using TypeU =
331 std::decay_t<decltype(m_u.evaluatePoly(fe_values, i, j, bi, bj))>;
332 using TypeV =
333 std::decay_t<decltype(m_v.evaluatePoly(fe_values, i, j, bi, bj))>;
334 if constexpr (std::is_same_v<void, TypeU>) {
335 // constant * v
336 return m_v.evaluatePoly(fe_values, i, j, bi, bj);
337 } else if constexpr (std::is_same_v<void, TypeV>) {
338 // u * constant
339 return m_u.evaluatePoly(fe_values, i, j, bi, bj);
340 } else {
341 auto [pu, pv] = internal::lift_to_same_space(
342 m_u.evaluatePoly(fe_values, i, j, bi, bj),
343 m_v.evaluatePoly(fe_values, i, j, bi, bj));
344 using LiftedPolyType = decltype(pu)::Scalar;
345 using ResultType = Eigen::Matrix<LiftedPolyType, -1, -1>;
346 if ((pu.rows() == pv.rows()) && (pu.rows() == 1)) {
347 // product between scalars
348 return ResultType{internal::cartesian_hcat(pu, pv)};
349 } else if ((pu.rows() == Dim * Dim) && (pv.rows() == Dim)) {
350 // matvec product
351 Eigen::Map<ResultType> rpu(pu.data(), Dim, Dim * pu.cols());
352 assert(rpu.size() == pu.size());
353 // transpose Dim x Dim blocks to simulate mat vec product
354 // indeed the cartesian hcat is a col-by-col prod
355 // while the mat vec is row-by-col prod
356 for (Eigen::Index k = 0; k < pu.cols(); ++k) {
357 rpu.template block<Dim, Dim>(0, k * Dim).transposeInPlace();
358 }
359 auto tmp = internal::cartesian_hcat(rpu, pv);
360 // transpose Dim x Dim blocks to simulate mat vec product
361 for (Eigen::Index k = 0; k < pu.cols(); ++k) {
362 tmp.template block<Dim, Dim>(0, k * Dim).transposeInPlace();
363 }
364 return ResultType{tmp};
365 } else {
366 // TODO: implement
367 // - mat mat product
368 // - vec (row) mat product
369 assert(false);
370 return ResultType{};
371 }
372 }
373 }

375 decltype(auto) evaluatePolyCoef(
376 const auto& mapping,
377 const auto& fe_values,
378 std::size_t bi,
379 std::size_t bj) const {
380 auto cu = m_u.evaluatePolyCoef(mapping, fe_values, bi, bj);
381 auto cv = m_v.evaluatePolyCoef(mapping, fe_values, bi, bj);
382 if ((cu.rows() == cv.rows()) && (cu.rows() == 1)) {
383 // product between scalars
384 return Eigen::MatrixXd{internal::cartesian_hcat(cu, cv)};
385 } else if ((cu.rows() == Dim * Dim) && (cv.rows() == Dim)) {
386 // matvec product
387 using ReshapedTypeU = Eigen::Matrix<double, Dim, -1>;
388 Eigen::Map<ReshapedTypeU> rcu(cu.data(), Dim, Dim * cu.cols());
389 assert(rcu.size() == cu.size());
390 // transpose Dim x Dim blocks to simulate mat vec product
391 // indeed the cartesian hcat is a col-by-col prod
392 // while the mat vec is row-by-col prod
393 for (Eigen::Index k = 0; k < cu.cols(); ++k) {
394 rcu.template block<Dim, Dim>(0, k * Dim).transposeInPlace();
395 }
396 auto tmp = internal::cartesian_hcat(rcu, cv);
397 // transpose Dim x Dim blocks to simulate mat vec product
398 for (Eigen::Index k = 0; k < cu.cols(); ++k) {
399 tmp.template block<Dim, Dim>(0, k * Dim).transposeInPlace();
400 }
401 return Eigen::MatrixXd{tmp};
402 } else {
403 // TODO: implement
404 // - mat mat product
405 // - vec (row) mat product
406 assert(false);
407 return Eigen::MatrixXd{};
408 }
409 }
