33// While I would like to avoid macros as much as possible this is a lot of
34// boilerplate to copy and paste on each binary expression.
35// Inheritance is out of the possibilities because it is needed to inherit
36// directly from the base class to solve at compile time the `decltype(auto)`
37// with the CRTP
38#define FE_EXPRESSION_BINARY_IMPL(Dim, E1, E2) \
39 public: \
40 static constexpr bool is_leaf = false; \
41 static constexpr uint8_t is_bilinear = E1::is_bilinear | E2::is_bilinear; \
42 static_assert( \
43 std::is_same_v<typename E1::FeSpaceType, void> || \
44 std::is_same_v<typename E2::FeSpaceType, void> || \
45 std::is_same_v<typename E1::FeSpaceType, typename E2::FeSpaceType>); \
46 using FeSpaceType = std::conditional_t< \
47 std::is_same_v<typename E1::FeSpaceType, void>, \
48 typename E2::FeSpaceType, typename E1::FeSpaceType>; \
49 static_assert(!std::is_same_v<FeSpaceType, void>); \
50 constexpr const auto& feSpace() const { \
51 if constexpr (std::is_same_v<FeSpaceType, void>) { \
52 /* this is always false, see static assert */ \
53 return nullptr; \
54 } else if constexpr (std::is_same_v<typename E1::FeSpaceType, void>) { \
55 return m_v.feSpace(); \
56 } else { \
57 return m_u.feSpace(); \
58 } \
59 } \
60 using PolynomialType = std::decay_t<decltype(FeSpaceType{0}.getBasis()[0])>; \
61 static constexpr std::size_t k_index_begin = \
62 std::min(E1::k_index_begin, E2::k_index_begin); \
63 static constexpr std::size_t k_index_end = \
64 std::max(E1::k_index_end, E2::k_index_end); \
65 constexpr UpdateFlagsType computeUpdateFlags() const { \
66 return m_u.computeUpdateFlags() | m_v.computeUpdateFlags(); \
67 } \
68 constexpr bool precompute( \
69 const FeValues<Dim>& fe_values, const CellsType<Dim>& cells) const { \
70 return m_u.precompute(fe_values, cells) && \
71 m_v.precompute(fe_values, cells); \
72 } \
73 \
74 protected: \
75 typename std::conditional_t<E1::is_leaf, const E1&, const E1> m_u; \
76 typename std::conditional_t<E2::is_leaf, const E2&, const E2> m_v;

80template <std::size_t Dim, typename E1, typename E2>
81class FeExpressionInner
82 : public FeExpressionBase<Dim, FeExpressionInner<Dim, E1, E2>> {
83 FE_EXPRESSION_BINARY_IMPL(Dim, E1, E2)
84 public:
85 FeExpressionInner(E1 const& u, E2 const& v) : m_u(u), m_v(v) {}
86
87 template <std::size_t I = 0, std::size_t J = 0>
88 decltype(auto) evaluate(
89 const FeValues<Dim>& fe_values,
90 std::size_t i,
91 std::size_t j,
92 std::size_t q) const {
93 using TypeEvalU = decltype(m_u.template evaluate<I, J>(fe_values, i, j, q));
94 using TypeEvalV = decltype(m_v.template evaluate<I, J>(fe_values, i, j, q));
95 if constexpr (
96 std::is_same_v<TypeEvalU, void> || std::is_same_v<TypeEvalV, void>) {
97 return void();
98 } else {
99 return m_u.template evaluate<I, J>(fe_values, i, j, q)
100 .cwiseProduct(m_v.template evaluate<I, J>(fe_values, i, j, q))
101 .sum();
102 }
103 }
104
105 constexpr std::size_t computeOrder() const {
106 return m_u.computeOrder() + m_v.computeOrder();
107 }
108
109 decltype(auto) evaluatePoly(
110 const FeValues<Dim>& fe_values,
111 std::size_t i,
112 std::size_t j,
113 std::size_t bi,
114 std::size_t bj) const {
115 // TODO: implement general case with different number of cols
116 using TypeU =
117 std::decay_t<decltype(m_u.evaluatePoly(fe_values, i, j, bi, bj))>;
118 using TypeV =
119 std::decay_t<decltype(m_v.evaluatePoly(fe_values, i, j, bi, bj))>;
120 if constexpr (std::is_same_v<void, TypeU>) {
121 return Eigen::Matrix<PolynomialType, 1, Dim>{
122 m_v.evaluatePoly(fe_values, i, j, bi, bj).transpose()};
123 } else if constexpr (std::is_same_v<void, TypeV>) {
124 return Eigen::Matrix<PolynomialType, 1, Dim>{
125 m_u.evaluatePoly(fe_values, i, j, bi, bj).transpose()};
126 } else {
127 static_assert(TypeU::ColsAtCompileTime == TypeV::ColsAtCompileTime);
128 static_assert(TypeU::RowsAtCompileTime == TypeV::RowsAtCompileTime);
129 constexpr auto Rows = TypeU::RowsAtCompileTime;
130 constexpr auto Cols = TypeU::ColsAtCompileTime;
131 // on different rows we have different spatial components
132 // on different cols we have different sum components of the final result
133 // e.g. if we have u * v + inner(grad(u), grad(v))
134 // the u * v and inner(grad(u), grad(v)) have different transformation
135 // when mapping the integral from the reference element to the physical
136 // space, thus, we keep each different addend of the sum on different cols
137 Eigen::Matrix<PolynomialType, Rows, Cols> tmp =
138 m_u.evaluatePoly(fe_values, i, j, bi, bj)
139 .cwiseProduct(m_v.evaluatePoly(fe_values, i, j, bi, bj));
140 using ResultType = Eigen::Matrix<PolynomialType, 1, Rows * Cols>;
141 return ResultType{Eigen::Map<ResultType>(tmp.data())};
142 }
143 }
144
145 decltype(auto) evaluatePolyCoef(
146 const auto& mapping, std::size_t bi, std::size_t bj) const {
147 constexpr auto N = std::decay_t<decltype(m_u.evaluatePolyCoef(
148 mapping, bi, bj))>::RowsAtCompileTime;
149 return Eigen::Matrix<double, 1, N>{
150 (m_u.evaluatePolyCoef(mapping, bi, bj).array() *
151 m_v.evaluatePolyCoef(mapping, bi, bj).array())
152 .transpose()};
153 }
154};

174 template <std::size_t I = 0, std::size_t J = 0>
175 decltype(auto) evaluate(
176 const FeValues<Dim>& fe_values,
177 std::size_t i,
178 std::size_t j,
179 std::size_t q) const {
180 using TypeEvalU = decltype(m_u.template evaluate<I, J>(fe_values, i, j, q));
181 using TypeEvalV = decltype(m_v.template evaluate<I, J>(fe_values, i, j, q));

184 return void();
185 } else {

187 m_v.template evaluate<I, J>(fe_values, i, j, q);

195 decltype(auto) evaluatePoly(
196 const FeValues<Dim>& fe_values,
197 std::size_t i,
198 std::size_t j,
199 std::size_t bi,
200 std::size_t bj) const {
201 using TypeU =
202 std::decay_t<decltype(m_u.evaluatePoly(fe_values, i, j, bi, bj))>;
203 using TypeV =
204 std::decay_t<decltype(m_v.evaluatePoly(fe_values, i, j, bi, bj))>;
205 if constexpr (std::is_same_v<void, TypeU>) {
206 return m_v.evaluatePoly(fe_values, i, j, bi, bj);
207 } else if constexpr (std::is_same_v<void, TypeV>) {
208 return m_u.evaluatePoly(fe_values, i, j, bi, bj);
209 } else {
210 return internal::cartesian_hcat(
211 m_u.evaluatePoly(fe_values, i, j, bi, bj),
212 m_v.evaluatePoly(fe_values, i, j, bi, bj));
213 }
214 }

216 decltype(auto) evaluatePolyCoef(
217 const auto& mapping, std::size_t bi, std::size_t bj) const {
218 return internal::cartesian_hcat(
219 m_u.evaluatePolyCoef(mapping, bi, bj),
220 m_v.evaluatePolyCoef(mapping, bi, bj));
221 }
222};

245 std::size_t i,
246 std::size_t j,

297template <std::size_t Dim, typename E1, typename E2>
298class FeExpressionDifference
299 : public FeExpressionBase<Dim, FeExpressionDifference<Dim, E1, E2>> {
300 FE_EXPRESSION_BINARY_IMPL(Dim, E1, E2)
301 public:
302 FeExpressionDifference(E1 const& u, E2 const& v) : m_u(u), m_v(v) {}
303
304 template <std::size_t I = 0, std::size_t J = 0>
305 decltype(auto) evaluate(
306 const FeValues<Dim>& fe_values,
307 std::size_t i,
308 std::size_t j,
309 std::size_t q) const {
310 using TypeEvalU = decltype(m_u.template evaluate<I, J>(fe_values, i, j, q));
311 using TypeEvalV = decltype(m_v.template evaluate<I, J>(fe_values, i, j, q));
312 if constexpr (
313 std::is_same_v<TypeEvalU, void> || std::is_same_v<TypeEvalV, void>) {
314 return void();

316 return -m_v.template evaluate<I, J>(fe_values, i, j, q);
317 } else if constexpr (std::is_same_v<TypeEvalV, void>) {
318 return m_u.template evaluate<I, J>(fe_values, i, j, q);
319 } else {
320 return m_u.template evaluate<I, J>(fe_values, i, j, q) -
321 m_v.template evaluate<I, J>(fe_values, i, j, q);
322 }
323 }
324
325 constexpr std::size_t computeOrder() const {
326 return std::max(m_u.computeOrder(), m_v.computeOrder());
327 }

333 std::size_t bi,
334 std::size_t bj) const {
