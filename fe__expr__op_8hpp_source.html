52// While I would like to avoid macros as much as possible this is a lot of
53// boilerplate to copy and paste on each binary expression.
54// Inheritance is out of the possibilities because it is needed to inherit
55// directly from the base class to solve at compile time the `decltype(auto)`
56// with the CRTP
57#define FE_EXPRESSION_BINARY_IMPL(Dim, E1, E2) \
58 public: \
59 static constexpr bool is_leaf = false; \
60 static constexpr uint8_t is_bilinear = E1::is_bilinear | E2::is_bilinear; \
61 static_assert( \
62 std::is_same_v<typename E1::FeSpaceType, void> || \
63 std::is_same_v<typename E2::FeSpaceType, void> || \
64 std::is_same_v<typename E1::FeSpaceType, typename E2::FeSpaceType>); \
65 using FeSpaceType = std::conditional_t< \
66 std::is_same_v<typename E1::FeSpaceType, void>, \
67 typename E2::FeSpaceType, typename E1::FeSpaceType>; \
68 constexpr const auto& feSpace() const { \
69 if constexpr (std::is_same_v<FeSpaceType, void>) { \
70 /* this is always false, see static assert */ \
71 return nullptr; \
72 } else if constexpr (std::is_same_v<typename E1::FeSpaceType, void>) { \
73 return m_v.feSpace(); \
74 } else { \
75 return m_u.feSpace(); \
76 } \
77 } \
78 using PolynomialType = internal::PolynomialTypeHelper<FeSpaceType>::type; \
79 static constexpr std::size_t k_index_begin = \
80 std::min(E1::k_index_begin, E2::k_index_begin); \
81 static constexpr std::size_t k_index_end = \
82 std::max(E1::k_index_end, E2::k_index_end); \
83 constexpr UpdateFlagsType computeUpdateFlags() const { \
84 return m_u.computeUpdateFlags() | m_v.computeUpdateFlags(); \
85 } \
86 constexpr bool precompute( \
87 const FeValues<Dim>& fe_values, const CellsType<Dim>& cells) const { \
88 return m_u.precompute(fe_values, cells) && \
89 m_v.precompute(fe_values, cells); \
90 } \
91 \
92 protected: \
93 typename std::conditional_t<E1::is_leaf, const E1&, const E1> m_u; \
94 typename std::conditional_t<E2::is_leaf, const E2&, const E2> m_v;

105 template <std::size_t J = 0, std::size_t I = 0>
106 decltype(auto) evaluate(
107 const FeValues<Dim>& fe_values,
108 std::size_t j,
109 std::size_t i,
110 std::size_t q) const {
111 using TypeEvalU = decltype(m_u.template evaluate<J, I>(fe_values, j, i, q));
112 using TypeEvalV = decltype(m_v.template evaluate<J, I>(fe_values, j, i, q));
113 if constexpr (
114 std::is_same_v<TypeEvalU, void> || std::is_same_v<TypeEvalV, void>) {
115 return void();
116 } else if constexpr (
117 std::is_arithmetic_v<std::decay_t<TypeEvalU>> &&
118 std::is_arithmetic_v<std::decay_t<TypeEvalV>>) {
119 return m_u.template evaluate<J, I>(fe_values, j, i, q) *
120 m_v.template evaluate<J, I>(fe_values, j, i, q);
121 } else {
122 return m_u.template evaluate<J, I>(fe_values, j, i, q)
123 .cwiseProduct(m_v.template evaluate<J, I>(fe_values, j, i, q))
124 .sum();
125 }
126 }
127
128 constexpr std::size_t computeOrder() const {
129 return m_u.computeOrder() + m_v.computeOrder();
130 }

132 decltype(auto) evaluatePoly(
133 const FeValues<Dim>& fe_values,
134 std::size_t j,
135 std::size_t i,
136 std::size_t bi,
137 std::size_t bj) const {
138 // TODO: implement general case with different number of cols
139 using TypeU =
140 std::decay_t<decltype(m_u.evaluatePoly(fe_values, j, i, bi, bj))>;
141 using TypeV =
142 std::decay_t<decltype(m_v.evaluatePoly(fe_values, j, i, bi, bj))>;
143 if constexpr (std::is_same_v<void, TypeU>) {
144 return m_v.evaluatePoly(fe_values, j, i, bi, bj).transpose().eval();
145 } else if constexpr (std::is_same_v<void, TypeV>) {
146 return m_u.evaluatePoly(fe_values, j, i, bi, bj).transpose().eval();
147 } else {
148 const auto [pu, pv] = internal::lift_to_same_space(
149 m_u.evaluatePoly(fe_values, j, i, bi, bj),
150 m_v.evaluatePoly(fe_values, j, i, bi, bj));
151 using PolyLiftedType = decltype(pu)::Scalar;
152 // on different rows we have different spatial components
153 // on different cols we have different sum components of the final result
154 // e.g. if we have u * v + inner(grad(u), grad(v))
155 // the u * v and inner(grad(u), grad(v)) have different transformation
156 // when mapping the integral from the reference element to the physical
157 // space, thus, we keep each different addend of the sum on different cols
158 assert(pu.rows() == pv.rows());
159 if ((pu.cols() == pv.cols()) && (pu.cols() == 1)) {
160 return Eigen::Matrix<PolyLiftedType, 1, -1>{
161 pu.cwiseProduct(pv).reshaped()};
162 } else {
163 // for ex. this is the case of inner(sigma * grad(u), grad(v))
164 // where sigma is a dim x dim tensor
165 return Eigen::Matrix<PolyLiftedType, 1, -1>{
166 internal::cartesian_hcat(pu, pv).reshaped()};
167 }
168 }
169 }
170
171 decltype(auto) evaluatePolyCoef(
172 const auto& mapping,
173 const auto& fe_values,
174 std::size_t bi,
175 std::size_t bj) const {
176 const auto cu = m_u.evaluatePolyCoef(mapping, fe_values, bi, bj);
177 const auto cv = m_v.evaluatePolyCoef(mapping, fe_values, bi, bj);
178 assert(cu.rows() == cv.rows());
179 if ((cu.cols() == cv.cols()) && (cu.cols() == 1)) {
180 return Eigen::Matrix<double, 1, -1>{(cu.cwiseProduct(cv)).transpose()};
181 } else {
182 return Eigen::Matrix<double, 1, -1>{
183 internal::cartesian_hcat(cu, cv).reshaped()};
184 }
185 }
186};

206 template <std::size_t J = 0, std::size_t I = 0>
207 decltype(auto) evaluate(
208 const FeValues<Dim>& fe_values,
209 std::size_t j,
210 std::size_t i,
211 std::size_t q) const {
212 using TypeEvalU = decltype(m_u.template evaluate<J, I>(fe_values, j, i, q));
213 using TypeEvalV = decltype(m_v.template evaluate<J, I>(fe_values, j, i, q));
214 if constexpr (
215 std::is_same_v<TypeEvalU, void> || std::is_same_v<TypeEvalV, void>) {
216 return void();
217 } else {
218 return m_u.template evaluate<J, I>(fe_values, j, i, q) *
219 m_v.template evaluate<J, I>(fe_values, j, i, q).transpose();
220 }
221 }

223 constexpr std::size_t computeOrder() const {
224 return m_u.computeOrder() + m_v.computeOrder();

227 decltype(auto) evaluatePoly(
228 const FeValues<Dim>& fe_values,
229 std::size_t j,
230 std::size_t i,
231 std::size_t bi,
232 std::size_t bj) const {
233 using TypeU =
234 std::decay_t<decltype(m_u.evaluatePoly(fe_values, j, i, bi, bj))>;
235 using TypeV =
236 std::decay_t<decltype(m_v.evaluatePoly(fe_values, j, i, bi, bj))>;
237 if constexpr (std::is_same_v<void, TypeU>) {
238 const auto pv = m_v.evaluatePoly(fe_values, j, i, bi, bj);
239 const Eigen::Index nrows = pv.rows();
240 using PolyType = typename TypeV::Scalar;
241 Eigen::Matrix<PolyType, -1, -1> result(nrows * nrows, pv.cols());
242 for (Eigen::Index k = 0; k < pv.cols(); ++k) {
243 result.col(k) = pv.col(k).transpose().replicate(nrows, 1).reshaped();
244 }
245 return result;
246 } else if constexpr (std::is_same_v<void, TypeV>) {
247 const auto pu = m_u.evaluatePoly(fe_values, j, i, bi, bj);
248 const Eigen::Index nrows = pu.rows();
249 using PolyType = typename TypeU::Scalar;
250 Eigen::Matrix<PolyType, -1, -1> result(nrows * nrows, pu.cols());
251 for (Eigen::Index k = 0; k < pu.cols(); ++k) {
252 result.col(k) = pu.col(k).replicate(1, nrows).reshaped();
253 }
254 return result;
255 } else {
256 const auto [pu, pv] = internal::lift_to_same_space(
257 m_u.evaluatePoly(fe_values, j, i, bi, bj),
258 m_v.evaluatePoly(fe_values, j, i, bi, bj));
259 assert(pu.rows() == pv.rows());
260 const Eigen::Index nrows = pu.rows();
261 using PolyLiftType = decltype(pu)::Scalar;
262 Eigen::Matrix<PolyLiftType, -1, -1> result(
263 nrows * nrows, pu.cols() * pv.cols());
264 Eigen::Index idx = 0;
265 for (Eigen::Index ki = 0; ki < pu.cols(); ++ki) {
266 for (Eigen::Index kj = 0; kj < pv.cols(); ++kj) {
267 result.col(idx++) = (pu.col(ki) * pv.col(kj).transpose()).reshaped();
268 }
269 }
270 return result;
271 }
272 }

274 decltype(auto) evaluatePolyCoef(
275 const auto& mapping,
276 const auto& fe_values,
277 std::size_t bi,
278 std::size_t bj) const {
279 const auto cu = m_u.evaluatePolyCoef(mapping, fe_values, bi, bj);
280 const auto cv = m_v.evaluatePolyCoef(mapping, fe_values, bi, bj);
281 assert(cu.rows() == cv.rows());
282 const Eigen::Index nrows = cu.rows();
283 Eigen::Matrix<double, -1, -1> result(nrows * nrows, cu.cols() * cv.cols());
284 Eigen::Index idx = 0;
285 for (Eigen::Index i = 0; i < cu.cols(); ++i) {
286 for (Eigen::Index j = 0; j < cv.cols(); ++j) {
287 result.col(idx++) = (cu.col(i) * cv.col(j).transpose()).reshaped();
288 }
289 }
290 return result;
291 }

315 std::size_t j,

319 using TypeEvalV = decltype(m_v.template evaluate<J, I>(fe_values, j, i, q));

321 std::is_same_v<TypeEvalU, void> || std::is_same_v<TypeEvalV, void>) {
322 return void();
323 } else {
324 return m_u.template evaluate<J, I>(fe_values, j, i, q) *
325 m_v.template evaluate<J, I>(fe_values, j, i, q);

333 decltype(auto) evaluatePoly(
334 const FeValues<Dim>& fe_values,
335 std::size_t j,
336 std::size_t i,
337 std::size_t bi,
338 std::size_t bj) const {
339 using TypeU =
340 std::decay_t<decltype(m_u.evaluatePoly(fe_values, j, i, bi, bj))>;
341 using TypeV =
342 std::decay_t<decltype(m_v.evaluatePoly(fe_values, j, i, bi, bj))>;
343 if constexpr (std::is_same_v<void, TypeU>) {
344 // constant * v
345 return m_v.evaluatePoly(fe_values, j, i, bi, bj);
346 } else if constexpr (std::is_same_v<void, TypeV>) {
347 // u * constant
348 return m_u.evaluatePoly(fe_values, j, i, bi, bj);
349 } else {
350 auto [pu, pv] = internal::lift_to_same_space(
351 m_u.evaluatePoly(fe_values, j, i, bi, bj),
352 m_v.evaluatePoly(fe_values, j, i, bi, bj));
353 using LiftedPolyType = decltype(pu)::Scalar;
354 using ResultType = Eigen::Matrix<LiftedPolyType, -1, -1>;
355 if ((pu.rows() == pv.rows()) && (pu.rows() == 1)) {
356 // product between scalars
357 return ResultType{internal::cartesian_hcat(pu, pv)};
358 } else if ((pu.rows() == pv.rows() * pv.rows()) && (pv.rows() > 1)) {
359 // matvec product
360 const Eigen::Index nrows = pv.rows();
361 Eigen::Map<ResultType> rpu(pu.data(), nrows, nrows * pu.cols());
362 assert(rpu.size() == pu.size());
363 // transpose nrows x nrows blocks to simulate mat vec product
364 // indeed the cartesian hcat is a col-by-col prod
365 // while the mat vec is row-by-col prod
366 for (Eigen::Index k = 0; k < pu.cols(); ++k) {
367 rpu.block(0, k * nrows, nrows, nrows).transposeInPlace();
368 }
369 auto tmp = internal::cartesian_hcat(rpu, pv);
370 // transpose nrows x nrows blocks to simulate mat vec product
371 for (Eigen::Index k = 0; k < pu.cols(); ++k) {
372 tmp.block(0, k * nrows, nrows, nrows).transposeInPlace();
373 }
374 return ResultType{tmp};
375 } else {
376 // TODO: implement
377 // - mat mat product
378 // - vec (row) mat product
379 assert(false);
380 return ResultType{};
381 }
382 }
383 }
384
385 decltype(auto) evaluatePolyCoef(
386 const auto& mapping,
387 const auto& fe_values,
388 std::size_t bi,
389 std::size_t bj) const {
390 auto cu = m_u.evaluatePolyCoef(mapping, fe_values, bi, bj);
391 auto cv = m_v.evaluatePolyCoef(mapping, fe_values, bi, bj);
392 if ((cu.rows() == cv.rows()) && (cu.rows() == 1)) {
393 // product between scalars
394 return Eigen::MatrixXd{internal::cartesian_hcat(cu, cv)};
395 } else if ((cu.rows() == cv.rows() * cv.rows()) && (cv.rows() > 1)) {
396 // matvec product
397 const Eigen::Index nrows = cv.rows();
398 using ReshapedTypeU = Eigen::Matrix<double, -1, -1>;
399 Eigen::Map<ReshapedTypeU> rcu(cu.data(), nrows, nrows * cu.cols());
400 assert(rcu.size() == cu.size());
401 // transpose nrows x nrows blocks to simulate mat vec product
402 // indeed the cartesian hcat is a col-by-col prod
403 // while the mat vec is row-by-col prod
404 for (Eigen::Index k = 0; k < cu.cols(); ++k) {
405 rcu.block(0, k * nrows, nrows, nrows).transposeInPlace();
406 }
407 auto tmp = internal::cartesian_hcat(rcu, cv);
408 // transpose nrows x nrows blocks to simulate mat vec product
409 for (Eigen::Index k = 0; k < cu.cols(); ++k) {
410 tmp.block(0, k * nrows, nrows, nrows).transposeInPlace();
411 }
412 return Eigen::MatrixXd{tmp};
413 } else {
414 // TODO: implement
415 // - mat mat product
416 // - vec (row) mat product
417 assert(false);
418 return Eigen::MatrixXd{};
419 }
420 }
421};

444 std::size_t j,

446 std::size_t q) const {

457 return m_u.template evaluate<J, I>(fe_values, j, i, q) +
458 m_v.template evaluate<J, I>(fe_values, j, i, q);

564// make_vec(e_0, ..., e_{N-1}): assemble N scalar expressions into an
565// N-dimensional Eigen vector at each quadrature point. The number of arguments
566// fixes the output dimension (2 args -> 2D vector, 3 args -> 3D vector). This
567// is used to build vector coefficient expressions from scalar components. It is
568// a pure coefficient expression (FeSpaceType = void inherited from the base):
569// when combined with a test/trial function via inner(...), the surrounding
570// binary expression supplies the FE space and k-index range.
571template <std::size_t Dim, typename... ScalarExprs>
572class FeExpressionMakeVec
573 : public FeExpressionBase<Dim, FeExpressionMakeVec<Dim, ScalarExprs...>> {
574 public:
575 static constexpr std::size_t N = sizeof...(ScalarExprs);
576 static_assert(N >= 1, "make_vec requires at least one component");
577
578 static constexpr bool is_leaf = false;
579 static constexpr uint8_t is_bilinear = (ScalarExprs::is_bilinear | ...);
580
581 using OutputType = Eigen::Vector<double, static_cast<int>(N)>;
582
583 explicit FeExpressionMakeVec(const ScalarExprs&... exprs)
584 : m_exprs(exprs...) {}
585
586 template <std::size_t J = 0, std::size_t I = 0>
587 OutputType evaluate(
588 const FeValues<Dim>& fe_values,
589 std::size_t j,
590 std::size_t i,
591 std::size_t q) const {
592 OutputType result;
593 evalAssign<J, I>(fe_values, j, i, q, std::make_index_sequence<N>{}, result);
594 return result;
595 }
596
597 constexpr std::size_t computeOrder() const {
598 return computeOrderImpl(std::make_index_sequence<N>{});
599 }
600
601 constexpr UpdateFlagsType computeUpdateFlags() const {
602 return computeUpdateFlagsImpl(std::make_index_sequence<N>{});
603 }
604
605 constexpr bool precompute(
606 const FeValues<Dim>& fe_values, const CellsType<Dim>& cells) const {
607 return precomputeImpl(fe_values, cells, std::make_index_sequence<N>{});
608 }
609
610 protected:
611 std::tuple<typename std::conditional_t<
612 ScalarExprs::is_leaf,
613 const ScalarExprs&,
614 const ScalarExprs>...>
615 m_exprs;
616
617 private:
618 template <std::size_t J, std::size_t I, std::size_t... Ks>
619 void evalAssign(
620 const FeValues<Dim>& fe_values,
621 std::size_t j,
622 std::size_t i,
623 std::size_t q,
624 std::index_sequence<Ks...>,
625 OutputType& result) const {
626 ((result[static_cast<Eigen::Index>(Ks)] =
627 std::get<Ks>(m_exprs).template evaluate<J, I>(fe_values, j, i, q)),
628 ...);
629 }
630
631 template <std::size_t... Ks>
632 constexpr std::size_t computeOrderImpl(std::index_sequence<Ks...>) const {
633 return std::max({std::get<Ks>(m_exprs).computeOrder()...});
634 }
635
636 template <std::size_t... Ks>
637 constexpr UpdateFlagsType computeUpdateFlagsImpl(
638 std::index_sequence<Ks...>) const {
639 return (std::get<Ks>(m_exprs).computeUpdateFlags() | ...);
640 }
641
642 template <std::size_t... Ks>
643 constexpr bool precomputeImpl(
644 const FeValues<Dim>& fe_values,
645 const CellsType<Dim>& cells,
646 std::index_sequence<Ks...>) const {
647 return (std::get<Ks>(m_exprs).precompute(fe_values, cells) && ...);
648 }
649};
650
651template <std::size_t Dim, typename... ScalarExprs>
652FeExpressionMakeVec<Dim, ScalarExprs...> make_vec(
653 const FeExpressionBase<Dim, ScalarExprs>&... exprs) {
654 return FeExpressionMakeVec<Dim, ScalarExprs...>(
655 *static_cast<const ScalarExprs*>(&exprs)...);
656}
