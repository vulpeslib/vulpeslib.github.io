116 const CellType& getCell(index_t cell_id) {
117 Profiler::start("getCell");
118 constexpr auto cell_options = FeType::cellInitOptions();
119
120 // create cell
121 if (!m_cells[cell_id]) {
122 m_cells[cell_id] = std::make_unique<geometry::Cell<dim, double, index_t>>(
123 m_mesh, cell_id, cell_options);
124 }
125
126 // map cell to reference frame
127 auto& cell = *m_cells[cell_id];
128 std::unique_ptr<MappingType> mapping;
129 if (!m_cell_is_normalized[cell_id]) {
130 mapping = std::make_unique<MappingType>(
131 m_fe_space.getMapping(*m_cells[cell_id]));
132 mapping->transformInv_(cell.vtxs());
133 m_cell_is_normalized[cell_id] = true;
134 }
135
136 // build mapping
137 if (!m_mappings[cell_id]) {
138 m_mappings[cell_id] = std::make_unique<MappingType>(
139 m_fe_space.getMapping(*m_cells[cell_id]));
140 }
141 Profiler::stop("getCell");
142 return cell;
143 }

168 NeighInfo findCellNeigh(index_t cell_id, index_t loc_face_id) {
169 auto& neigh_info = m_cell_to_facet_neigh[cell_id][loc_face_id];
170 if (neigh_info.built) {
171 return neigh_info;
172 }
173 // global means not of the cell, it is not global in MPI sense
174 const auto glob_face_id = m_mesh.getCellToFacetMap()[cell_id][loc_face_id];
175 // find the two cells to which this face belongs to
176 const auto cell_ids = m_mesh.getFacetToCellMap()[glob_face_id];
177 // find the cell that is not cell_id
178 neigh_info.neigh_cell_id = cell_ids[(cell_ids[0] == cell_id)];
179 neigh_info.have_neigh =
180 (neigh_info.neigh_cell_id != static_cast<index_t>(-1));
181 if (neigh_info.have_neigh) {
182 // find the local_cell_id in dual map from global id
183 const auto& dual_map = m_mesh.getDualMap();
184 const auto neighs_begin = dual_map[cell_id].begin();
185 const auto neighs_end = dual_map[cell_id].end();
186 neigh_info.loc_neigh_cell_id = std::distance(
187 neighs_begin,
188 std::find(neighs_begin, neighs_end, neigh_info.neigh_cell_id));
189 assert(neigh_info.loc_neigh_cell_id < dual_map[cell_id].size());
190 const auto neigh_cell_to_facet =
191 m_mesh.getCellToFacetMap()[neigh_info.neigh_cell_id];
192 const auto loc_neigh_face_it = std::find(
193 neigh_cell_to_facet.begin(), neigh_cell_to_facet.end(), glob_face_id);
194 neigh_info.loc_neigh_face_id =
195 std::distance(neigh_cell_to_facet.begin(), loc_neigh_face_it);
196 assert(neigh_info.loc_neigh_face_id < neigh_cell_to_facet.size());
197 }
198 neigh_info.built = true;
199 return neigh_info;
200 }

214 Eigen::Array<double, MeshType::self_dim, 1> getPeriodicShift(
215 index_t neigh_cell_id, index_t current_cell_id) const {
216 const auto& curr_cell = m_cells[current_cell_id];
217 const auto& neigh_cell = m_cells[neigh_cell_id];
218
219 if (!curr_cell || !neigh_cell) {
220 return Eigen::Array<double, dim, 1>::Zero();
221 }
222
223 // Extract period values from metadata
224 const auto& md = m_mesh.getMetadata();
225 Eigen::Array<double, dim, 1> period = Eigen::Array<double, dim, 1>::Zero();
226
227 if (md.doubles.find("XPeriod") != md.doubles.end()) {
228 period[0] = md.doubles.at("XPeriod")[0];
229 }
230 if constexpr (dim > 1) {
231 if (md.doubles.find("YPeriod") != md.doubles.end()) {
232 period[1] = md.doubles.at("YPeriod")[0];
233 }
234 }
235 if constexpr (dim > 2) {
236 if (md.doubles.find("ZPeriod") != md.doubles.end()) {
237 throw std::logic_error("3D periodic conditions not yet implemented.");
238 period[2] = md.doubles.at("ZPeriod")[0];
239 }
240 }
241
242 return geometry::computePeriodicShift<dim>(
243 curr_cell->bbox(), neigh_cell->bbox(), period);
244 }
