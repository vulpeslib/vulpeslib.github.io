69 CellManager(const MeshType& mesh, const FeSpaceType& fe_space)
70 : m_mesh(mesh), m_fe_space(fe_space) {

85 const CellType& getCell(index_t cell_id) {
86 Profiler::start("getCell");
87 constexpr auto cell_options = FeType::cellInitOptions();
88
89 // create cell
90 if (!m_cells[cell_id]) {
91 m_cells[cell_id] = std::make_unique<geometry::Cell<dim, double, index_t>>(
92 m_mesh, cell_id, cell_options);
93 }
94
95 // map cell to reference frame
96 auto& cell = *m_cells[cell_id];
97 std::unique_ptr<MappingType> mapping;
98 if (!m_cell_is_normalized[cell_id]) {
99 mapping = std::make_unique<MappingType>(
100 m_fe_space.getMapping(*m_cells[cell_id]));
101 mapping->transformInv_(cell.vtxs());
102 m_cell_is_normalized[cell_id] = true;
103 }
104
105 // build mapping
106 if (!m_mappings[cell_id]) {
107 m_mappings[cell_id] = std::make_unique<MappingType>(
108 m_fe_space.getMapping(*m_cells[cell_id]));
109 }
110 Profiler::stop("getCell");
111 return cell;
112 }

124 NeighInfo findCellNeigh(index_t cell_id, index_t loc_face_id) {
125 auto& neigh_info = m_cell_to_facet_neigh[cell_id][loc_face_id];
126 if (neigh_info.built) {
127 return neigh_info;
128 }
129 // find global face id
130 const auto glob_face_id = m_mesh.getCellToFacetMap()[cell_id][loc_face_id];
131 // find the two cells to which this face belongs to
132 const auto cell_ids = m_mesh.getFacetToCellMap()[glob_face_id];
133 // find the cell that is not cell_id
134 neigh_info.neigh_cell_id = cell_ids[(cell_ids[0] == cell_id)];
135 neigh_info.have_neigh =
136 (neigh_info.neigh_cell_id != static_cast<index_t>(-1));
137 if (neigh_info.have_neigh) {
138 // find the local_cell_id in dual map from global id
139 const auto& dual_map = m_mesh.getDualMap();
140 const auto neighs_begin = dual_map[cell_id].begin();
141 const auto neighs_end = dual_map[cell_id].end();
142 neigh_info.loc_neigh_cell_id = std::distance(
143 neighs_begin,
144 std::find(neighs_begin, neighs_end, neigh_info.neigh_cell_id));
145 assert(neigh_info.loc_neigh_cell_id < dual_map[cell_id].size());
146 const auto neigh_cell_to_facet =
147 m_mesh.getCellToFacetMap()[neigh_info.neigh_cell_id];
148 const auto loc_neigh_face_it = std::find(
149 neigh_cell_to_facet.begin(), neigh_cell_to_facet.end(), glob_face_id);
150 neigh_info.loc_neigh_face_id =
151 std::distance(neigh_cell_to_facet.begin(), loc_neigh_face_it);
152 assert(neigh_info.loc_neigh_face_id < neigh_cell_to_facet.size());
153 }
154 neigh_info.built = true;
155 return neigh_info;
156 }
