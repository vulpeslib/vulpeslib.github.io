113 const CellType& getCell(index_t cell_id) {
114 Profiler::start("getCell");
115 constexpr auto cell_options = FeType::cellInitOptions();
116
117 // create cell
118 if (!m_cells[cell_id]) {
119 m_cells[cell_id] = std::make_unique<geometry::Cell<dim, double, index_t>>(
120 m_mesh, cell_id, cell_options);
121 }
122
123 // map cell to reference frame
124 auto& cell = *m_cells[cell_id];
125 std::unique_ptr<MappingType> mapping;
126 if (!m_cell_is_normalized[cell_id]) {
127 mapping = std::make_unique<MappingType>(
128 m_fe_space.getMapping(*m_cells[cell_id]));
129 mapping->transformInv_(cell.vtxs());
130 m_cell_is_normalized[cell_id] = true;
131 }
132
133 // build mapping
134 if (!m_mappings[cell_id]) {
135 m_mappings[cell_id] = std::make_unique<MappingType>(
136 m_fe_space.getMapping(*m_cells[cell_id]));
137 }
138 Profiler::stop("getCell");
139 return cell;
140 }

165 NeighInfo findCellNeigh(index_t cell_id, index_t loc_face_id) {
166 auto& neigh_info = m_cell_to_facet_neigh[cell_id][loc_face_id];
167 if (neigh_info.built) {
168 return neigh_info;
169 }
170 // global means not of the cell, it is not global in MPI sense
171 const auto glob_face_id = m_mesh.getCellToFacetMap()[cell_id][loc_face_id];
172 // find the two cells to which this face belongs to
173 const auto cell_ids = m_mesh.getFacetToCellMap()[glob_face_id];
174 // find the cell that is not cell_id
175 neigh_info.neigh_cell_id = cell_ids[(cell_ids[0] == cell_id)];
176 neigh_info.have_neigh =
177 (neigh_info.neigh_cell_id != static_cast<index_t>(-1));
178 if (neigh_info.have_neigh) {
179 // find the local_cell_id in dual map from global id
180 const auto& dual_map = m_mesh.getDualMap();
181 const auto neighs_begin = dual_map[cell_id].begin();
182 const auto neighs_end = dual_map[cell_id].end();
183 neigh_info.loc_neigh_cell_id = std::distance(
184 neighs_begin,
185 std::find(neighs_begin, neighs_end, neigh_info.neigh_cell_id));
186 assert(neigh_info.loc_neigh_cell_id < dual_map[cell_id].size());
187 const auto neigh_cell_to_facet =
188 m_mesh.getCellToFacetMap()[neigh_info.neigh_cell_id];
189 const auto loc_neigh_face_it = std::find(
190 neigh_cell_to_facet.begin(), neigh_cell_to_facet.end(), glob_face_id);
191 neigh_info.loc_neigh_face_id =
192 std::distance(neigh_cell_to_facet.begin(), loc_neigh_face_it);
193 assert(neigh_info.loc_neigh_face_id < neigh_cell_to_facet.size());
194 }
195 neigh_info.built = true;
196 return neigh_info;
197 }
