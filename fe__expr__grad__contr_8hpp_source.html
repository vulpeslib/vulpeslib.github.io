119 template <std::size_t J = 0, std::size_t I = 0>
120 decltype(auto) evaluate(
121 const FeValues<Dim>& fe_values,
122 std::size_t j,
123 std::size_t i,
124 std::size_t q) const {
125 if constexpr (J != I) {
126 // Off-diagonal block: test component I != trial component J.
127 // No coupling between distinct components in this expression.
128 return void();
129 } else if constexpr (IndexBegin <= I && I < IndexEnd) {
130 // Diagonal block: F(q) . grad(phi_i) * phi_j.
131 using TypeEvalFunc =
132 decltype(m_func.template evaluate<J, I>(fe_values, j, i, q));
133 if constexpr (std::is_same_v<TypeEvalFunc, void>) {
134 return void();
135 } else {
136 return m_func.template evaluate<J, I>(fe_values, j, i, q)
137 .dot(fe_values.getGradEval()(i, q)) *
138 fe_values.getShapeEval()(j, q);
139 }
140 } else {
141 // Component I outside [IndexBegin, IndexEnd): not active.
142 return void();
143 }
144 }
145
146 constexpr UpdateFlagsType computeUpdateFlags() const {
147 // Requires shape values (for u), gradients (for grad v), and whatever
148 // flags F needs (typically quadrature point coordinates).
149 return UpdateGradients | UpdateValues | m_func.computeUpdateFlags();
150 }
151
152 constexpr std::size_t computeOrder() const {
153 // The integrand contains F(x) . grad(v_I) * u_I, so the quadrature-order
154 // estimate must include the function expression, the trial basis, and the
155 // test-gradient contribution. For a p-th order basis, grad(v_I) has order
156 // p - 1; clamp at 0 to avoid unsigned underflow for constant spaces.
157 const std::size_t u_order = m_u.computeOrder();
158 const std::size_t grad_v_order = (u_order > 0) ? (u_order - 1) : 0;
159 return m_func.computeOrder() + u_order + grad_v_order;
160 }

334 template <std::size_t J = 0, std::size_t I = 0>
335 decltype(auto) evaluate(
336 const FeValues<Dim>& fe_values,
337 std::size_t j,
338 std::size_t i,

340 if constexpr (J == TrialComp && I == TestComp) {
341 using TypeEvalFunc =

343 if constexpr (std::is_same_v<TypeEvalFunc, void>) {
