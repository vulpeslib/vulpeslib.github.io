58inline double computePhiM(std::size_t m) {
59 double phi = 1.0;
60 constexpr std::size_t iter_max = 30;
61 for (std::size_t it = 0; it < iter_max; ++it) {
62 const double phim = vmath::upow(phi, m);
63 const double dphi =
64 (phi * (phim - 1) - 1) / (static_cast<double>(m + 1) * phim - 1);
65 phi -= dphi;
66 if (std::abs(dphi) < 2 * std::numeric_limits<double>::epsilon()) {
67 break;
68 }
69 }
70 return phi;
71}

101template <std::size_t M>
102std::vector<std::array<double, M>> additiveRecurrence(
103 std::size_t n0, std::size_t n1) {
104 std::array<double, M> phi_inv_Md;
105 // if M is large this is not accurate since phi approaches 1
106 phi_inv_Md[0] = 1.0 / internal::computePhiM(M);
107 for (std::size_t m = 1; m < M; ++m) {
108 phi_inv_Md[m] = phi_inv_Md[m - 1] * phi_inv_Md[0];
109 }
110 std::vector<std::array<double, M>> pts;
111 for (std::size_t i = n0; i < n1; ++i) {
112 std::array<double, M> pt;
113 for (std::size_t m = 0; m < M; ++m) {
114 pt[m] = std::fmod((0.5 + phi_inv_Md[m] * i), 1.0);
115 }
116 pts.push_back(pt);
117 }
118 return pts;
119}
