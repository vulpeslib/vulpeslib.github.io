55 template <typename DofHandler>
56 void init(const DofHandler& dof_handler, Type type = Plain) {
57 createMatrix(type);
58
59 const PetscInt n_glob = dof_handler.ndofsGlobal();
60 PetscInt dof_begin = 0;
61 PetscInt dof_end = n_glob;
62 if constexpr (DofHandler::parallel) {
63 dof_begin = dof_handler.dofOffset()[mpi::rank()];
64 dof_end = dof_handler.dofOffset()[mpi::rank() + 1];
65 }
66 const PetscInt n_loc = dof_end - dof_begin;
67 // TODO: check not too imbalanced
68 std::cout << "nloc" << n_loc << " rank" << mpi::rank() << std::endl;
69 PetscCallAbortWorld(
70 MatSetSizes(m_petsc_matrix, n_loc, n_loc, n_glob, n_glob));
71
72 std::vector<PetscInt> dnn(n_loc, 0), onn(n_loc, 0);
73 computePreallocationData(dof_handler, dof_begin, dof_end, dnn, onn);
74
75 allocateMatrix(type, dnn, onn);
76 }

154 template <typename DofHandler>
155 void computePreallocationData(
156 const DofHandler& dof_handler,
157 const PetscInt dof_begin,
158 const PetscInt dof_end,
159 std::vector<PetscInt>& dnn,
160 std::vector<PetscInt>& onn) {
161 for (auto dof_id = dof_begin; dof_id < dof_end; ++dof_id) {
162 // assert that the key is in the map
163 const auto loc_dof_id =
164 dof_handler.hasRenumbering()
165 ? dof_handler.getCurrToRefNumbering().at(dof_id)
166 : dof_id;
167 const auto sp_row = dof_handler.getSparsityPattern()[loc_dof_id];
168 for (const auto glob_dof_nz : sp_row) {
169 if (dof_begin <= glob_dof_nz && glob_dof_nz < dof_end) {
170 dnn[dof_id - dof_begin]++;
171 } else {
172 onn[dof_id - dof_begin]++;
173 }
174 }
175 }
176 }
