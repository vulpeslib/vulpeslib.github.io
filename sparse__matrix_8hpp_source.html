90 template <typename DofHandler>
91 void init(
92 const DofHandler& dof_handler,
93 Type type = Plain,
94 PetscInt block_size = 1) {
95 if ((block_size != 1) && (type != Block)) {
96 std::cout
97 << "ERROR: block size != 1 available only with block type matrices"
98 << std::endl;
99 std::exit(-1);
100 }
101 createMatrix(type);
102
103 const PetscInt n_glob = dof_handler.ndofsGlobal();
104 PetscInt dof_begin = 0;
105 PetscInt dof_end = n_glob;
106 if constexpr (DofHandler::parallel) {
107 dof_begin = dof_handler.dofOffset()[mpi::rank()];
108 dof_end = dof_handler.dofOffset()[mpi::rank() + 1];
109 }
110 const PetscInt n_loc = dof_end - dof_begin;
111 // TODO: check not too imbalanced
112 std::cout << "[SparseMatrix::init] own: " << n_loc << "/" << n_glob
113 << "@rank:" << mpi::rank() << std::endl;
114 if (type == Block) {
115 PetscCallAbortWorld(MatSetBlockSize(m_petsc_matrix, block_size));
116 setMatrixSizes(PETSC_DECIDE, n_glob);
117 }
118 setMatrixSizes(n_loc * block_size, n_glob * block_size);
119
120 if (DofHandler::parallel && (!dof_handler.hasRenumbering())) {
121 // if we do not have MPI renumbering, we do not know the sparsity pattern
122 return;
123 }
124 std::vector<PetscInt> dnn(n_loc, 0), onn(n_loc, 0);
125 computePreallocationData(dof_handler, dof_begin, dof_end, dnn, onn);
126 allocateMatrix(type, dnn, onn, block_size);
127 }

314 template <typename DofHandler>
315 void computePreallocationData(
316 const DofHandler& dof_handler,
317 const PetscInt dof_begin,
318 const PetscInt dof_end,
319 std::vector<PetscInt>& dnn,
320 std::vector<PetscInt>& onn) {
321 for (auto dof_id = dof_begin; dof_id < dof_end; ++dof_id) {
322 // assert that the key is in the map
323 auto loc_dof_id = dof_id;
324 if (DofHandler::parallel) {
325 loc_dof_id = dof_handler.getCurrToRefNumbering().at(dof_id);
326 }
327 const auto sp_row = dof_handler.getSparsityPattern()[loc_dof_id];
328 for (const auto glob_dof_nz : sp_row) {
329 if (dof_begin <= glob_dof_nz && glob_dof_nz < dof_end) {
330 dnn[dof_id - dof_begin]++;
331 } else {
332 onn[dof_id - dof_begin]++;
333 }
334 }
335 }
336 }
