42template <typename T>
43MPI_Datatype getMpiType() {
44 if constexpr (std::is_same_v<T, unsigned char>) {
45 return MPI_UNSIGNED_CHAR;
46 } else if constexpr (std::is_same_v<T, char>) {
47 return MPI_CHAR;
48 } else if constexpr (std::is_same_v<T, short>) { // NOLINT(runtime/int)
49 return MPI_SHORT;
50 } else if constexpr (std::is_same_v<T, int>) {
51 return MPI_INT;
52 } else if constexpr (std::is_same_v<T, long>) { // NOLINT(runtime/int)
53 return MPI_LONG;
54 } else if constexpr (std::is_same_v<T, long long>) { // NOLINT(runtime/int)
55 return MPI_LONG_LONG;
56 } else if constexpr (std::is_same_v<
57 T, unsigned short>) { // NOLINT(runtime/int)
58 return MPI_UNSIGNED_SHORT;
59 } else if constexpr (std::is_same_v<T, unsigned int>) {
60 return MPI_UNSIGNED;
61 } else if constexpr (std::is_same_v<
62 T, unsigned long>) { // NOLINT(runtime/int)
63 return MPI_UNSIGNED_LONG;
64 } else if constexpr (std::is_same_v<
65 T, unsigned long long>) { // NOLINT(runtime/int)
66 return MPI_UNSIGNED_LONG_LONG;
67 } else if constexpr (std::is_same_v<T, float>) {
68 return MPI_FLOAT;
69 } else if constexpr (std::is_same_v<T, double>) {
70 return MPI_DOUBLE;
71 } else if constexpr (std::is_same_v<T, long double>) {
72 return MPI_LONG_DOUBLE;
73 } else {
74 static_assert(sizeof(T) * 0 && "Unknown conversion to MPI type.");
75 }
76}

416class MpiHandle {
417 public:
419 static int init(
420 int& argc,
421 char**& argv,
422 int required = s_default_thread_required,
423 int* provided = nullptr) {
424 return init(&argc, &argv, required, provided);
425 }
427 static int init(
428 int* argc = nullptr,
429 char*** argv = nullptr,
430 int required = s_default_thread_required,
431 int* provided = nullptr) {
432 assert(!isInitialized() && "MPI already initialized!");
433 int mpi_res;
434 if (required == MPI_THREAD_SINGLE) {
435 mpi_res = MPI_Init(argc, argv);
436 } else {
437 mpi_res = MPI_Init_thread(argc, argv, required, provided);
438 }
439 if (mpi_res != MPI_SUCCESS) {
440 std::cout << "Error during MPI_Init()!" << std::endl;
441 std::exit(-1);
442 }
443 PetscCall(PetscInitialize(argc, argv, nullptr, nullptr));
444 singleton();
445 return 0;
446 }
447
448 private:
450 inline constexpr static int s_default_thread_required = MPI_THREAD_SINGLE;
452 static MpiHandle& singleton() {
453 static MpiHandle mpi;
454 return mpi;
455 }
457 static int finalize() {
458 if (isPetscInitialized() && !isPetscFinalized()) {
459 PetscCall(PetscFinalize());
460 }
461 if (isInitialized() && !isFinalized()) {
462 MPI_Finalize();
463 }
464 return 0;
465 }
467 ~MpiHandle() { finalize(); }
469 MpiHandle() = default;
470};
