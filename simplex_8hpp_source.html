54template <typename T, std::size_t Dim>
55QuadratureRule<T, Dim - 1> baricentrycToCartesian(
56 const QuadratureRule<T, Dim>& rule_bary) {
57 static_assert(Dim > 2);
58 QuadratureRule<T, Dim - 1> rule;
59 rule.degree = rule_bary.degree;
60 rule.nodes.reserve(rule_bary.nodes.size());
61 // volume of n-simplex = |det(A)| / n!
62 // where the columns of A are the difference of the vertexes with respect to
63 // the first vertex. |det(A)| = 1 for the unit ortogonal simplex.
64 // We multiply by 2^n for rescaling in [-1, 1]^n
65 mpf_class weight0 = mpq_class(
66 (1U << (Dim - 1)), vmath::factorial(static_cast<uint32_t>(Dim - 1)));
67 for (const auto& node : rule_bary.nodes) {
68 rule.nodes.emplace_back();
69 rule.nodes.back().weight = node.weight * weight0;
70 for (std::size_t d = 0; d < (Dim - 1); ++d) {
71 // matvec with matrix of vertices
72 // we assume that the vertexes are zero and the versors
73 // then we apply a homotety and a translation
74 rule.nodes.back().point[d] = node.point[d + 1] * T(2) - T(1);
75 }
76 }
77 return rule;
78}

101template <std::size_t Dim>
102QuadratureRule<mpf_class, Dim + 1> mpfGrundammMoellerQuadrature(
103 std::size_t deg, std::size_t precision) {
104 mpf_set_default_prec(precision);
105 // works only for odd degree
106 if (deg % 2 == 0) {
107 deg++;
108 }
109 std::size_t order = (deg - 1) / 2;
110 // pre-compute exponents of monomial of order up to `order` (grouped by order)
111 // this implementation is rather wastfull in terms of computations but it
112 // is not recursive, moreover, it should be easier to understand
113 using MonomialGroup = std::vector<std::array<std::size_t, Dim + 1>>;
114 std::vector<MonomialGroup> ordered_exponents(order + 1);
115 for (std::size_t i = 0; i < vmath::upow(order + 1, Dim + 1); ++i) {
116 std::array<std::size_t, Dim + 1> exponent;
117 std::size_t idx = i;
118 std::size_t sum = 0;
119 // write `i` in radix `order + 1`
120 for (std::size_t d = 0; d < Dim + 1; ++d) {
121 std::size_t digit = idx % (order + 1);
122 idx /= (order + 1);
123 exponent[d] = digit;
124 sum += digit;
125 }
126 // we are generating the cartesian product, skip if order too high
127 if (sum > order) {
128 continue;
129 }
130 ordered_exponents[sum].push_back(exponent);
131 }
132 // compute rule
133 QuadratureRule<mpf_class, Dim + 1> rule_barycentric;
134 rule_barycentric.degree = deg;
135 for (std::size_t i = 0; i < order + 1; ++i) {
136 const mpq_class num =
137 vmath::pow1(i) * vmath::upow(mpz_class(deg + Dim - 2 * i), deg);
138 const mpq_class den =
139 (vmath::upow(mpz_class(2), 2 * order) * vmath::factorial(mpz_class(i)) *
140 vmath::factorial(mpz_class(deg + Dim - i)));
141 mpq_class weight = num / den;
142 weight.canonicalize();
143
144 std::array<mpf_class, Dim + 1> node;
145 for (const auto& exponent : ordered_exponents[order - i]) {
146 for (std::size_t d = 0; d < Dim + 1; ++d) {
147 node[d] = mpq_class(2 * exponent[d] + 1, deg + Dim - 2 * i);
148 }
149 rule_barycentric.nodes.emplace_back(node, weight);
150 }
151 }
152
153 // normalize weights
154 mpf_class weights_sum = 0.0;
155 for (const auto& node : rule_barycentric.nodes) {
156 weights_sum += node.weight;
157 }
158 for (auto& node : rule_barycentric.nodes) {
159 node.weight /= weights_sum;
160 }
161
162 return rule_barycentric;
163}
