175 template <typename MeshType>
176 constexpr MeshType::index_t dofToCellDof(
177 const MeshType& mesh,
178 MeshType::index_t cell_id,
179 MeshType::index_t owning_enty_id,
180 MeshType::index_t enty_dim,
181 MeshType::index_t enty_multiplicity) const {
182 using index_t = MeshType::index_t;
183 index_t owning_enty_id_local = owning_enty_id;
184 if constexpr (MeshType::parallel()) {
185 owning_enty_id_local = mesh.globalToLocal(enty_dim).at(owning_enty_id);
186 }
187 if (enty_dim == 0) {
188 const auto& cell_vtxs = mesh.getCellToVtxMap();
189 index_t loc_vtx_id = 0;
190 for (const auto& vtx_id : cell_vtxs[cell_id]) {
191 if (vtx_id == owning_enty_id) {
192 break;
193 }
194 loc_vtx_id++;
195 }
196 std::array<CartDofIndexType, Dim + 1> cart_dof = {0};
197 assert(loc_vtx_id < index_t(cart_dof.size()));
198 cart_dof[loc_vtx_id] = this->m_order;
199 const auto dof_it =
200 std::lower_bound(m_dofs.begin(), m_dofs.end(), cart_dof);
201 assert(*dof_it == cart_dof);
202 const auto dist = std::distance(m_dofs.begin(), dof_it);
203 return m_dofs_perm[dist];
204 } else if (enty_dim == 1) {
205 if constexpr (Dim == 1) {
206 assert(cell_id == owning_enty_id_local);
207 return enty_multiplicity + 1;
208 }
209 static_assert(Dim > 0 && Dim <= 3);
210 using EdgeToVtxMapType = std::conditional_t<
211 Dim == 2, typename MeshType::FacetToVtxMap,
212 typename MeshType::EdgeToVtxMap>;
213 const EdgeToVtxMapType* edge_to_vtx_map;
214 if constexpr (Dim == 2) {
215 edge_to_vtx_map = &mesh.getFacetToVtxMap();
216 }
217 if constexpr (Dim == 3) {
218 edge_to_vtx_map = &mesh.getEdgeToVtxMap();
219 }
220 const auto cell_vtxs = mesh.getCellToVtxMap()[cell_id];
221 std::array<index_t, 2> loc_edge_vtx_id = {0};
222 for (std::size_t i = 0; i < Dim + 1; ++i) {
223 const auto edge_vtxs = (*edge_to_vtx_map)[owning_enty_id_local];
224 loc_edge_vtx_id[0] += i * (edge_vtxs[0] == cell_vtxs[i]);
225 loc_edge_vtx_id[1] += i * (edge_vtxs[1] == cell_vtxs[i]);
226 }
227 std::array<CartDofIndexType, Dim + 1> cart_dof = {0};
228 assert(loc_edge_vtx_id[0] < index_t(cart_dof.size()));
229 assert(loc_edge_vtx_id[1] < index_t(cart_dof.size()));
230 cart_dof[loc_edge_vtx_id[0]] = this->m_order - enty_multiplicity;
231 cart_dof[loc_edge_vtx_id[1]] = enty_multiplicity;
232 const auto dof_it =
233 std::lower_bound(m_dofs.begin(), m_dofs.end(), cart_dof);
234 assert(*dof_it == cart_dof);
235 const auto dist = std::distance(m_dofs.begin(), dof_it);
236 return m_dofs_perm[dist];
237 } else if (enty_dim == 2) {
238 std::array<index_t, 3> loc_face_vtx_id = {0};
239 if constexpr (Dim == 2) {
240 loc_face_vtx_id = {0, 1, 2};
241 }
242 if constexpr (Dim == 3) {
243 const auto& face_to_vtx_map = mesh.getFacetToVtxMap();
244 const auto cell_vtxs = mesh.getCellToVtxMap()[cell_id];
245 for (std::size_t i = 0; i < Dim + 1; ++i) {
246 for (std::size_t j = 0; j < 3; ++j) {
247 loc_face_vtx_id[j] +=
248 i * (face_to_vtx_map[owning_enty_id_local][j] == cell_vtxs[i]);
249 }
250 }
251 }
252 const auto cart_idx_xy =
253 monomial::getMonomial2D<CartDofIndexType>(enty_multiplicity);
254 std::array<CartDofIndexType, Dim + 1> cart_dof = {0};
255 cart_dof[loc_face_vtx_id[1]] = cart_idx_xy[0] + 1;
256 cart_dof[loc_face_vtx_id[2]] = cart_idx_xy[1] + 1;
257 cart_dof[loc_face_vtx_id[0]] =
258 this->m_order - 2 - cart_idx_xy[0] - cart_idx_xy[1];
259 const auto dof_it =
260 std::lower_bound(m_dofs.begin(), m_dofs.end(), cart_dof);
261 assert(*dof_it == cart_dof);
262 const auto dist = std::distance(m_dofs.begin(), dof_it);
263 return m_dofs_perm[dist];
264 } else if (enty_dim == 3) {
265 if constexpr (Dim > 2) {
266 const auto cart_idx_xyz =
267 monomial::getMonomial3D<CartDofIndexType>(enty_multiplicity);
268 std::array<CartDofIndexType, Dim + 1> cart_dof = {0};
269 cart_dof[0] = this->m_order;
270 for (std::size_t d = 0; d < 3; ++d) {
271 cart_dof[d + 1] = cart_idx_xyz[d] + 1;
272 cart_dof[0] -= cart_dof[d + 1];
273 }
274 const auto dof_it =
275 std::lower_bound(m_dofs.begin(), m_dofs.end(), cart_dof);
276 assert(*dof_it == cart_dof);
277 const auto dist = std::distance(m_dofs.begin(), dof_it);
278 return m_dofs_perm[dist];
279 }
280 }
281 assert(false);
282 return -1;
283 }
284
285 void buildDofTable() {
286 const auto n_pts = this->m_order + 1;
287 for (std::size_t j = 0; j < vmath::pow(n_pts, Dim); ++j) {
288 std::array<CartDofIndexType, Dim + 1> cart_dof_id;
289 std::size_t idx = j;
290 std::size_t idx_sum = 0;
291 for (std::size_t d = 0; d < Dim; ++d) {
292 const auto idx_d = idx % n_pts;
293 idx_sum += idx_d;
294 cart_dof_id[d + 1] = idx_d;
295 idx /= n_pts;
296 }
297 if (idx_sum > this->m_order) {
298 continue;
299 }
300 cart_dof_id[0] = this->m_order - idx_sum;
301 m_dofs.push_back(cart_dof_id);
302 }
303 m_dofs_perm.resize(m_dofs.size());
304 std::iota(m_dofs_perm.begin(), m_dofs_perm.end(), 0);
305 std::sort(m_dofs_perm.begin(), m_dofs_perm.end(), [this](auto a, auto b) {
306 return this->m_dofs[a] < this->m_dofs[b];
307 });
308 const auto dofs_perm = m_dofs_perm;
309 const auto dofs = m_dofs;
310 for (std::size_t i = 0; i < m_dofs_perm.size(); ++i) {
311 m_dofs_perm[dofs_perm[i]] = i;
312 m_dofs[i] = dofs[dofs_perm[i]];
313 }
314 }
315
316 std::vector<std::array<CartDofIndexType, Dim + 1>> m_dofs;
317 std::vector<std::size_t> m_dofs_perm;
318};

411template <std::size_t Dim>
412struct DofPolicyFeSpaceVEM
413 : public DofPolicyBase<Dim, DofPolicyFeSpaceVEM<Dim>> {
414 explicit DofPolicyFeSpaceVEM(std::size_t order)
415 : DofPolicyBase<Dim, DofPolicyFeSpaceVEM<Dim>>(order) {}
416 static std::array<std::size_t, Dim + 1> buildDofSubdimEntity(
417 std::size_t order) {
418 assert(order > 0);
419 std::array<std::size_t, Dim + 1> ndofs;
420 using IndexType = decltype(order);
421 if constexpr (Dim == 1) {
422 ndofs[0] = 1;
423 ndofs[1] = order > 1 ? (order - 2) : 0;
424 } else if constexpr (Dim == 2) {
425 ndofs[0] = 1;
426 ndofs[1] = order > 1 ? (order - 2) : 0;
427 ndofs[2] =
428 order > 1 ? vmath::binomial(order, static_cast<IndexType>(2)) : 0;
429 } else if constexpr (Dim == 3) {
430 ndofs[0] = 1;
431 ndofs[1] = order > 1 ? (order - 2) : 0;
432 ndofs[2] =
433 order > 1 ? vmath::binomial(order, static_cast<IndexType>(2)) : 0;
434 ndofs[3] =
435 order > 1 ? vmath::binomial(order + 1, static_cast<IndexType>(3)) : 0;
436 } else {
437 static_assert(Dim * 0 && "Not implemented");
438 }
439 return ndofs;
440 }
441 inline static constexpr bool dofsInteractionDistance() { //
442 return false;
443 };

445 template <typename MeshType>
446 constexpr MeshType::index_t dofToCellDof(
447 const MeshType& mesh,
448 MeshType::index_t cell_id,
449 MeshType::index_t owning_enty_id,
450 MeshType::index_t enty_dim,
451 MeshType::index_t enty_multiplicity) const {
452 using index_t = MeshType::index_t;
453 static_assert(Dim > 1); // if dim == 1, use standard elements
454 if (enty_dim == 0) {
455 const auto& cell_vtxs = mesh.getCellToVtxMap();
456 const auto vtx_it = std::find(
457 cell_vtxs[cell_id].begin(), cell_vtxs[cell_id].end(), owning_enty_id);
458 assert(vtx_it != cell_vtxs[cell_id].end());
459 // local vertex id
460 return std::distance(cell_vtxs[cell_id].begin(), vtx_it);
461 } else if (enty_dim == 1) {
462 const auto n_verts = mesh.getCellToVtxMap()[cell_id].size();
463 const auto edge_mult = this->m_order - 1;
464 index_t loc_edge_id = -1;
465 if constexpr (Dim == 2) {
466 const auto edge_it = std::find(
467 mesh.getCellToFacetMap()[cell_id].begin(),
468 mesh.getCellToFacetMap()[cell_id].end(), owning_enty_id);
469 assert(edge_it != mesh.getCellToFacetMap()[cell_id].end());
470 loc_edge_id =
471 std::distance(mesh.getCellToFacetMap()[cell_id].begin(), edge_it);
472 }
473 if constexpr (Dim == 3) {
474 const auto edge_it = std::find(
475 mesh.getCellToEdgeMap()[cell_id].begin(),
476 mesh.getCellToEdgeMap()[cell_id].end(), owning_enty_id);
477 assert(edge_it != mesh.getCellToEdgeMap()[cell_id].end());
478 loc_edge_id =
479 std::distance(mesh.getCellToEdgeMap()[cell_id].begin(), edge_it);
480 }
481 return n_verts + loc_edge_id * edge_mult + enty_multiplicity;
482 } else if (enty_dim == 2) {
483 const index_t n_verts = mesh.getCellToVtxMap()[cell_id].size();
484 const index_t edge_mult = this->m_order - 1;
485 const index_t n_edges = mesh.getCellToEdgeMap()[cell_id].size();
486 if constexpr (Dim == 2) {
487 assert(cell_id == owning_enty_id);
488 return n_verts + edge_mult * n_edges + enty_multiplicity;
489 }
490 if constexpr (Dim == 3) {
491 const auto face_it = std::find(
492 mesh.getCellToFacetMap()[cell_id].begin(),
493 mesh.getCellToFacetMap()[cell_id].end(), owning_enty_id);
494 assert(face_it != mesh.getCellToFacetMap()[cell_id].end());
495 const index_t loc_face_id =
496 std::distance(mesh.getCellToFacetMap()[cell_id].begin(), face_it);
497 const index_t face_mult = this->m_order * (this->m_order - 1) / 2;
498 return n_verts + edge_mult * n_edges + enty_multiplicity +
499 loc_face_id * face_mult + enty_multiplicity;
500 }
501 } else if (enty_dim == 3) {
502 assert(cell_id == owning_enty_id);
503 const index_t n_verts = mesh.getCellToVtxMap()[cell_id].size();
504 const index_t edge_mult = this->m_order - 1;
505 const index_t n_edges = mesh.getCellToEdgeMap()[cell_id].size();
506 const index_t face_mult = this->m_order * (this->m_order - 1) / 2;
507 const index_t n_faces = mesh.getCellToFacetMap()[cell_id].size();
508 return n_verts + edge_mult * n_edges + face_mult * n_faces +
509 enty_multiplicity;
510 }
511 assert(false);
512 return -1;
513 }
