43template <
44 typename MeshType,
45 typename FeExpressionTuple,
46 typename FeExpressionFaceTuple = std::tuple<>>
47class WeakFormIntegratorGauss : public WeakFormIntegratorBase<
48 MeshType,
49 FeExpressionTuple,
50 FeExpressionFaceTuple> {
51 public:
52 using index_t = typename MeshType::index_t;
53 using WeakFormIntegratorBaseType = WeakFormIntegratorBase<
54 MeshType,
55 FeExpressionTuple,
56 FeExpressionFaceTuple>;
57 using FeType = std::remove_pointer_t<
58 std::tuple_element_t<0, FeExpressionTuple>>::FeSpaceType;
59 static inline constexpr std::size_t dim = MeshType::self_dim;
60 static inline constexpr uint32_t s_n_expr =
61 WeakFormIntegratorBaseType::s_n_expr;
62 static inline constexpr uint32_t s_n_face_expr =
63 WeakFormIntegratorBaseType::s_n_face_expr;
64
65 WeakFormIntegratorGauss(
66 const MeshType& mesh,
67 const FeExpressionTuple& fe_expression,
68 const FeExpressionFaceTuple& fe_expression_face = FeExpressionFaceTuple{})
69 : WeakFormIntegratorBaseType(mesh, fe_expression, fe_expression_face) {
70 // cache quadrature rule depending on element type
71 constexpr auto elem_type = FeType::elemType();
72
73 m_qrule_d = qrule::castRule<double>(m_qrule_mpf);
74 const auto& fe_space = std::get<0>(this->m_fe_expression)->feSpace();
75 m_fe_values.evalOnReferenceFrame(fe_space.getBasisFast(), m_qrule_d);
76 }
77
78 void integrate(MeshType::index_t cell_id) {
79 const Eigen::Index n_q_nodes = m_qrule_d.nodes.size();
80 const auto& fe_space = std::get<0>(this->m_fe_expression)->feSpace();
81 const Eigen::Index n_dof_loc = fe_space.getBasis().size();
82 constexpr auto cell_options = FeType::cellInitOptions();
83 this->m_cells[cell_id] =
84 std::make_unique<geometry::Cell<dim, double, index_t>>(
85 this->m_mesh, cell_id, cell_options);
86 const auto& cell = *this->m_cells[cell_id];
87 const auto mapping = fe_space.getMapping(cell);
88 m_fe_values.evalAndApplyMapping(
89 mapping, this->m_update_flags | fe::UpdateWeights);
90 this->m_integrals.setZero();
91 for (Eigen::Index q = 0; q < n_q_nodes; ++q) {
92 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
93 for (Eigen::Index j = 0; j < n_dof_loc; ++j) {
94 uint32_t expr_id = 0;
95 std::apply(
96 [this, &expr_id, i, j, q](auto&&... exprs) {
97 ((this->m_integrals(i, j, expr_id++) +=
98 exprs->evaluate(m_fe_values, i, j, q) *
99 m_fe_values.weights()[q]),
100 ...);
101 },
102 this->m_fe_expression);
103 }
104 }
105 }
106 }
107
108 protected:
109 fe::FeValues<dim> m_fe_values;
110
111 qrule::QuadratureRule<mpf_class, dim> m_qrule_mpf;
112 qrule::QuadratureRule<double, dim> m_qrule_d;
113 qrule::QuadratureRule<mpf_class, dim - 1> m_qrule_face_mpf;
114 qrule::QuadratureRule<double, dim - 1> m_qrule_face_d;
115};
