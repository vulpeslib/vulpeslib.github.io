100 void integrate(MeshType::index_t cell_id) {
101 // TODO: assert cell is compatible with fe space
102 const Eigen::Index n_q_nodes = m_qrule_d.nodes.size();
103 const auto& fe_space = std::get<0>(this->m_fe_expression)->feSpace();
104 const Eigen::Index n_dof_loc = fe_space.getBasis().size();
105 // TODO: cache cell
106 constexpr auto cell_options = FeType::cellInitOptions();
107 this->m_cells[cell_id] =
108 std::make_unique<geometry::Cell<dim, double, index_t>>(
109 this->m_mesh, cell_id, cell_options);
110 const auto& cell = *this->m_cells[cell_id];
111 const auto mapping = fe_space.getMapping(cell);
112 m_fe_values.evalAndApplyMapping(
113 mapping, this->m_update_flags | fe::UpdateWeights);
114 this->m_integrals.setZero();
115 for (Eigen::Index q = 0; q < n_q_nodes; ++q) {
116 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
117 // TODO: it is worth it to make further local copies
118 // of data used in the expression to increase cache locality
119 // this should have ~10% speed-up
120 for (Eigen::Index j = 0; j < n_dof_loc; ++j) {
121 uint32_t expr_id = 0;
122 // TODO: evaluate if the std::apply is better as the most inner
123 // loop (like now) or in the most outer loop
124 std::apply(
125 [this, &expr_id, i, j, q](auto&&... exprs) {
126 ((this->m_integrals(i, j, expr_id++) +=
127 exprs->evaluate(m_fe_values, i, j, q) *
128 m_fe_values.weights()[q]),
129 ...);
130 },
131 this->m_fe_expression);
132 }
133 }
134 }
