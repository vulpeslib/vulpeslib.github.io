44class Profiler {
45 public:
46#ifndef NPROFILE
47 // Static method to start profiling a block of code
48 static void start(const std::string& label) {
49 Profiler::start_times[label] = std::chrono::high_resolution_clock::now();
50 }
51
52 // Static method to stop profiling and record the elapsed time
53 static void stop(const std::string& label) {
54 const auto end_time = std::chrono::high_resolution_clock::now();
55 const auto start_time = Profiler::start_times[label];
56 const std::size_t duration =
57 std::chrono::duration_cast<std::chrono::nanoseconds>(
58 end_time - start_time)
59 .count();
60 Profiler::time_map[label] +=
61 duration; // Accumulate the duration for the label
62 }
63
64 // Static method to print the accumulated times for all labels
65 static void print() {
66 std::cout << "Profiler results (in milliseconds):\n";
67
68 // Find the longest label for proper alignment
69 size_t maxLabelLength = 0;
70 for (const auto& entry : Profiler::time_map) {
71 maxLabelLength = std::max(maxLabelLength, entry.first.length());
72 }
73
74 // Create a vector of entries for sorting
75 std::vector<std::pair<std::string, std::size_t>> sorted_entries(
76 Profiler::time_map.begin(), Profiler::time_map.end());
77
78 // Sort by time in decreasing order
79 std::sort(
80 sorted_entries.begin(), sorted_entries.end(),
81 [](const auto& a, const auto& b) { return a.second > b.second; });
82
83 // Find the maximum width needed for time values with fixed precision
84 size_t maxTimeWidth = 0;
85 std::ostringstream oss;
86 for (const auto& entry : sorted_entries) {
87 oss.str("");
88 oss << std::fixed << std::setprecision(3)
89 << (static_cast<double>(entry.second) / 1e6);
90 maxTimeWidth = std::max(maxTimeWidth, oss.str().length());
91 }
92
93 // Print table header with proper alignment
94 std::cout << std::left << std::setw(maxLabelLength + 2) << "Label"
95 << "| Time (ms)" << std::endl;
96
97 // Print separator line
98 std::cout << std::string(maxLabelLength + 2, '-') << "+"
99 << std::string(maxTimeWidth + 2, '-') << std::endl;
100
101 // Print each entry with aligned times
102 for (const auto& entry : sorted_entries) {
103 double time_ms = static_cast<double>(entry.second) / 1e6;
104
105 std::cout << std::left << std::setw(maxLabelLength + 2) << entry.first
106 << "| " << std::right << std::setw(maxTimeWidth) << std::fixed
107 << std::setprecision(3) << time_ms << std::endl;
108 }
109 }
110
111 static void clear() { Profiler::time_map.clear(); }
112
113#else
114 // If DISABLE_PROFILER is defined, these functions do nothing
115 static void start(const std::string&) {}
116 static void stop(const std::string&) {}
117 static void print() {}
118 static void clear() {}
119#endif
120
121 using time_point = std::chrono::high_resolution_clock::time_point;
122 static std::unordered_map<std::string, std::size_t> time_map;
123 static std::unordered_map<std::string, time_point> start_times;
124};
