57template <typename FloatType, int32_t Dim, typename Distance>
58class DistanceExpression {
59 public:
64 static constexpr bool is_leaf = false;
65
75 static constexpr size_t size() { return Distance::size(); }
76
88 void operator()(
89 const Point<FloatType, Dim>& point, FloatType* distance_buffer) const {
90 static_cast<const Distance&>(*this)(point, distance_buffer);
91 }
92

102 return static_cast<const Distance*>(this)->bbox();

116template <typename FloatType>
117class RectangleDistance
118 : public DistanceExpression<FloatType, 2, RectangleDistance<FloatType>> {
119 public:
124 static constexpr bool is_leaf = true;
130 static constexpr size_t size() { return 5; }
131
138 RectangleDistance(
139 const Point<FloatType, 2>& corner_a, const Point<FloatType, 2>& corner_b)
140 : m_corner_a(corner_a), m_corner_b(corner_b) {}
141
149 void operator()(
150 const Point<FloatType, 2>& point, FloatType* distance_buffer) const {
151 distance_buffer[0] = m_corner_a[0] - point[0];
152 distance_buffer[1] = point[0] - m_corner_b[0];
153 distance_buffer[2] = m_corner_a[1] - point[1];
154 distance_buffer[3] = point[1] - m_corner_b[1];
155 distance_buffer[4] =
156 *std::max_element(distance_buffer, distance_buffer + 4);
157 }
158
164 std::array<Point<FloatType, 2>, 2> bbox() const {
165 return {m_corner_a, m_corner_b};
166 }
167
168 private:
169 const Point<FloatType, 2> m_corner_a;
170 const Point<FloatType, 2> m_corner_b;
171};

181template <typename FloatType>
182class CircleDistance
183 : public DistanceExpression<FloatType, 2, CircleDistance<FloatType>> {
184 public:
189 static constexpr bool is_leaf = true;
190
196 static constexpr size_t size() { return 2; }
197
204 CircleDistance(const Point<FloatType, 2>& center, FloatType radius)
205 : m_center(center), m_radius(radius) {}
206
214 void operator()(
215 const Point<FloatType, 2>& point, FloatType* distance_buffer) const {
216 const FloatType dx = point[0] - m_center[0];
217 const FloatType dy = point[1] - m_center[1];
218 distance_buffer[0] = std::sqrt(dx * dx + dy * dy) - m_radius;
219 distance_buffer[1] = distance_buffer[0];
220 }
221
227 std::array<Point<FloatType, 2>, 2> bbox() const {
228 return {m_center.array() - m_radius, m_center.array() + m_radius};
229 }
230
231 private:
232 const Point<FloatType, 2> m_center;
233 const FloatType m_radius;
234};

382template <typename FloatType, int32_t Dim, typename D1, typename D2>
383class UnionDistance : public DistanceExpression<
384 FloatType,
385 Dim,
386 UnionDistance<FloatType, Dim, D1, D2>> {
389 typename std::conditional_t<D1::is_leaf, const D1&, const D1> m_u;
392 typename std::conditional_t<D2::is_leaf, const D2&, const D2> m_v;
393
394 public:
395 static constexpr bool is_leaf = false;
396
403 UnionDistance(D1 const& u, D2 const& v) : m_u(u), m_v(v) {}
404
413 static constexpr size_t size() {
414 return std::decay_t<decltype(m_v)>::size() +
415 std::decay_t<decltype(m_u)>::size() - 1;
416 }
417
428 void operator()(
429 const Point<FloatType, Dim>& point, FloatType* distance_buffer) const {
430 constexpr auto u_distance_size = std::decay_t<decltype(m_u)>::size();
431 constexpr auto this_size = std::decay_t<decltype(*this)>::size();
432 m_u(point, distance_buffer);
433 FloatType tmp = distance_buffer[u_distance_size - 1];
434 m_v(point, distance_buffer + u_distance_size - 1);
435 distance_buffer[this_size - 1] =
436 std::min(tmp, distance_buffer[this_size - 1]);
437 }
438
446 std::array<Point<FloatType, Dim>, 2> bbox() const {
447 const auto bbox_u = m_u.bbox();
448 const auto bbox_v = m_v.bbox();
449 Point<FloatType, Dim> corner_low;
450 Point<FloatType, Dim> corner_high;
451 for (size_t d = 0; d < Dim; ++d) {
452 corner_low[d] = std::min(bbox_u[0][d], bbox_v[0][d]);
453 corner_high[d] = std::max(bbox_u[1][d], bbox_v[1][d]);
454 }
455 return {corner_low, corner_high};
456 }
457};
