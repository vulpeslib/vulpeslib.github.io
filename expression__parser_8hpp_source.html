34template <size_t N>
35class ExpressionParser {
36 public:
37 template <typename... Args>
38 explicit ExpressionParser(const char* expression, Args&&... args) {
39 init(expression, args...);
40 }
41 template <typename... Args>
42 explicit ExpressionParser(const std::string& expression, Args&&... args) {
43 init(expression.c_str(), args...);
44 }
45
46 double evaluate(const double* data) const {
47 return m_expr.evaluate(const_cast<double*>(data));
48 }
49
50 private:
51 template <typename... Args>
52 void init(const char* expression, Args&&... args) {
53 static_assert(
54 sizeof...(Args) == N, "Incorrect number of variable names provided");
55 m_context.addBuiltIns();
56 std::array<std::string_view, N> variable_names{std::forward<Args>(args)...};
57 for (size_t i = 0; i < N; ++i) {
58 m_context.addVariable(variable_names[i].data(), i * sizeof(double));
59 }
60
61 const auto options = mathpresso::kNoOptions;
62 auto err = m_expr.compile(m_context, expression, options);
63 if (err != mathpresso::kErrorOk) {
64 throw std::runtime_error(
65 "Failed to compile expression: '" + std::string(expression) +
66 "'. Error: " + std::to_string(err));
67 }
68 }
69
70 mathpresso::Context m_context;
71 mathpresso::Expression m_expr;
72};
