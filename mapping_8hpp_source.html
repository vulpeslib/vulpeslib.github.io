57template <typename FloatType, std::size_t Dim>
58class CartesianMapping
59 : MappingBase<FloatType, Dim, CartesianMapping<FloatType, Dim>> {
60 public:
61 constexpr CartesianMapping(
62 const Point<FloatType, Dim>& stretches,
63 const Point<FloatType, Dim>& translations)
64 : m_jacobian_diag(stretches),
65 m_translations(translations),
66 m_jacobian_det(m_jacobian_diag.prod()),
67 m_jacobian_inv_diag(m_jacobian_diag.array().inverse()),
68 m_translations_inv(
69 -m_jacobian_inv_diag.array() * m_translations.array()) {}
70
71 explicit CartesianMapping(const Eigen::Matrix<double, Dim, 2>& bbox)
72 : CartesianMapping(
73 (bbox.col(1) - bbox.col(0)) * 0.5,
74 (bbox.col(1) + bbox.col(0)) * 0.5) {}
75
76 template <typename Derived>
77 constexpr void transform_(Eigen::MatrixBase<Derived>& pts) const {
78 assert(pts.rows() == Dim);
79 for (Eigen::Index i = 0; i < pts.cols(); ++i) {
80 for (std::size_t d = 0; d < Dim; ++d) {
81 pts(d, i) *= m_jacobian_diag[d];
82 pts(d, i) += m_translations[d];
83 }
84 }
85 }
86 template <typename Derived>
87 constexpr void transformInv_(Eigen::MatrixBase<Derived>& pts) const {
88 assert(pts.rows() == Dim);
89 for (Eigen::Index i = 0; i < pts.cols(); ++i) {
90 for (std::size_t d = 0; d < Dim; ++d) {
91 pts(d, i) *= m_jacobian_inv_diag[d];
92 pts(d, i) += m_translations_inv[d];
93 }
94 }
95 }
96 constexpr Point<FloatType, Dim> transform(
97 const Point<FloatType, Dim>& pt) const {
98 return (pt.array() * m_jacobian_diag.array()).matrix() + m_translations;
99 }
100 constexpr Point<FloatType, Dim> transformInv(
101 const Point<FloatType, Dim>& pt) const {
102 return pt.array() * m_jacobian_inv_diag.array() + m_translations_inv;
103 }
104
105 constexpr void transform(
106 const Eigen::Matrix<FloatType, Dim, -1>& pts_in,
107 Eigen::Matrix<FloatType, Dim, -1>& pts_out) const {
108 for (Eigen::Index i = 0; i < pts_in.cols(); ++i) {
109 for (std::size_t d = 0; d < Dim; ++d) {
110 pts_out.coeffRef(d, i) =
111 m_jacobian_diag[d] * pts_in(d, i) + m_translations[d];
112 }
113 }
114 }
115 constexpr void transformInv(
116 const Eigen::Matrix<FloatType, Dim, -1>& pts_in,
117 Eigen::Matrix<FloatType, Dim, -1>& pts_out) const {
118 for (Eigen::Index i = 0; i < pts_in.cols(); ++i) {
119 for (std::size_t d = 0; d < Dim; ++d) {
120 pts_out.coeffRef(d, i) =
121 m_jacobian_inv_diag[d] * pts_in(d, i) + m_translations_inv[d];
122 }
123 }
124 }
125
126 constexpr FloatType jacobianDet(const Point<FloatType, Dim>&) const {
127 return m_jacobian_det;
128 }
129 constexpr Point<FloatType, Dim> jacobianInv(
130 const Point<FloatType, Dim>&) const {
131 return m_jacobian_inv_diag;
132 }
133
134 void setJacobianDet(FloatType det) { m_jacobian_det = det; }
135
136 const auto& getJacobian() const { return m_jacobian_diag; }
137 const auto& getTranslation() const { return m_translations; }
138 const auto& getJacobianInv() const { return m_jacobian_inv_diag; }
139 const auto& getTranslationInv() const { return m_translations_inv; }
140
141 protected:
142 const Point<FloatType, Dim> m_jacobian_diag;
143 const Point<FloatType, Dim> m_translations;
144 FloatType m_jacobian_det;
145 const Point<FloatType, Dim> m_jacobian_inv_diag;
146 const Point<FloatType, Dim> m_translations_inv;
147};

149template <typename FloatType, std::size_t Dim>
150class SimplexMapping
151 : MappingBase<FloatType, Dim, SimplexMapping<FloatType, Dim>> {
152 public:
153 explicit constexpr SimplexMapping(
154 const Eigen::Matrix<FloatType, Dim, Dim + 1>& vtxs)
155 : m_translations(vtxs.col(0)),
156 m_homogeneous(vtxs.rightCols(Dim).colwise() - m_translations) {
157 Eigen::FullPivLU<Eigen::Matrix<FloatType, Dim, Dim>> lu_decomp(
158 m_homogeneous);
159 m_homogeneous_inv = lu_decomp.inverse();
160 m_jacobian_det = lu_decomp.determinant();
161 }
162
163 explicit constexpr SimplexMapping(
164 const Eigen::Matrix<FloatType, Dim, -1>& vtxs)
165 : m_translations(vtxs.col(0)),
166 m_homogeneous(vtxs.rightCols(Dim).colwise() - m_translations) {
167 assert(vtxs.cols() == Dim + 1);
168 Eigen::FullPivLU<Eigen::Matrix<FloatType, Dim, Dim>> lu_decomp(
169 m_homogeneous);
170 m_homogeneous_inv = lu_decomp.inverse();
171 m_jacobian_det = lu_decomp.determinant();
172 }
173
174 template <typename Derived>
175 explicit constexpr SimplexMapping(
176 const Eigen::MatrixBase<Derived>& vtxs,
177 const Eigen::Vector<int32_t, Dim + 1>& idxs,
178 bool compute_inverse = true)
179 : m_translations(vtxs.col(idxs(0))) {
180 for (std::size_t d = 0; d < Dim; ++d) {
181 m_homogeneous.col(d) = vtxs.col(idxs(d + 1)) - m_translations;
182 }
183 assert(vtxs.cols() == Dim + 1);
184 if (compute_inverse) {
185 Eigen::FullPivLU<Eigen::Matrix<FloatType, Dim, Dim>> lu_decomp(
186 m_homogeneous);
187 m_homogeneous_inv = lu_decomp.inverse();
188 m_jacobian_det = lu_decomp.determinant();
189 }
190 }
191
192 constexpr Point<FloatType, Dim> transform(
193 const Point<FloatType, Dim>& pt) const {
194 return m_homogeneous * pt + m_translations;
195 }
196 constexpr Point<FloatType, Dim> transformInv(
197 const Point<FloatType, Dim>& pt) const {
198 return m_homogeneous_inv * (pt - m_translations);
199 }
200 constexpr FloatType jacobianDet(const Point<FloatType, Dim>&) const {
201 return m_jacobian_det;
202 }
203
204 protected:
205 const Point<FloatType, Dim> m_translations;
206 Eigen::Matrix<FloatType, Dim, Dim> m_homogeneous;
207 Eigen::Matrix<FloatType, Dim, Dim> m_homogeneous_inv;
208 FloatType m_jacobian_det;
209};
