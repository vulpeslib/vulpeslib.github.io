31template <typename FloatType, std::size_t Dim, typename Mapping>
32class MappingBase {
33 public:
34 constexpr Point<FloatType, Dim> transform(
35 const Point<FloatType, Dim>& pt) const {
36 return self().transform(pt);
37 }
38 constexpr void transform_(Eigen::Map<Point<FloatType, Dim>> pt) const {
39 self().transform(pt);
40 }
41 constexpr Eigen::Matrix<FloatType, Dim, Dim> jacobian(
42 const Point<FloatType, Dim>& pt) const {
43 return self().jacobian(pt);
44 }
45 constexpr FloatType jacobianDet(const Point<FloatType, Dim>& pt) const {
46 return self().jacobianDet(pt);
47 }
48
49 protected:
50 auto& self() { return static_cast<Mapping const&>(*this); }

54template <typename FloatType, std::size_t Dim>
55class CartesianMapping
56 : MappingBase<FloatType, Dim, CartesianMapping<FloatType, Dim>> {
57 public:
58 constexpr CartesianMapping(
59 const Point<FloatType, Dim>& stretches,
60 const Point<FloatType, Dim>& translations)
61 : m_jacobian_diag(stretches),
62 m_translations(translations),
63 m_jacobian_det(m_jacobian_diag.prod()),
64 m_jacobian_inv_diag(m_jacobian_diag.array().inverse()),
65 m_translations_inv(
66 -m_jacobian_inv_diag.array() * m_translations.array()) {}
67
68 explicit CartesianMapping(const Eigen::Matrix<double, Dim, 2>& bbox)
69 : CartesianMapping(
70 (bbox.col(1) - bbox.col(0)) * 0.5,
71 (bbox.col(1) + bbox.col(0)) * 0.5) {}
72
73 constexpr void transform_(Eigen::Map<Point<FloatType, Dim>> pt) const {
74 pt.array() *= m_jacobian_diag.array();
75 pt += m_translations;
76 }
77 constexpr void transformInv_(Eigen::Map<Point<FloatType, Dim>> pt) const {
78 pt.array() *= m_jacobian_inv_diag.array();
79 pt += m_translations_inv;
80 }
81 constexpr Point<FloatType, Dim> transform(
82 const Point<FloatType, Dim>& pt) const {
83 return (pt.array() * m_jacobian_diag.array()).matrix() + m_translations;
84 }
85 constexpr Point<FloatType, Dim> transformInv(
86 const Point<FloatType, Dim>& pt) const {
87 return pt.array() * m_jacobian_inv_diag.array() + m_translations_inv;
88 }
89 constexpr FloatType jacobianDet(const Point<FloatType, Dim>&) const {
90 return m_jacobian_det;
91 }
92 constexpr Point<FloatType, Dim> jacobianInv(
93 const Point<FloatType, Dim>&) const {
94 return m_jacobian_inv_diag;
95 }
96
97 void setJacobianDet(FloatType det) { m_jacobian_det = det; }
98
99 const auto& getJacobian() const { return m_jacobian_diag; }
100 const auto& getTranslation() const { return m_translations; }
101 const auto& getJacobianInv() const { return m_jacobian_inv_diag; }
102 const auto& getTranslationInv() const { return m_translations_inv; }
103
104 protected:
105 const Point<FloatType, Dim> m_jacobian_diag;
106 const Point<FloatType, Dim> m_translations;
107 FloatType m_jacobian_det;
108 const Point<FloatType, Dim> m_jacobian_inv_diag;
109 const Point<FloatType, Dim> m_translations_inv;
110};

112template <typename FloatType, std::size_t Dim>
113class SimplexMapping
114 : MappingBase<FloatType, Dim, SimplexMapping<FloatType, Dim>> {
115 public:
116 explicit constexpr SimplexMapping(
117 const Eigen::Matrix<FloatType, Dim, Dim + 1>& vtxs)
118 : m_translations(vtxs.col(0)),
119 m_homogeneous(vtxs.rightCols(Dim).colwise() - m_translations) {
120 Eigen::FullPivLU<Eigen::Matrix<FloatType, Dim, Dim>> lu_decomp(
121 m_homogeneous);
122 m_homogeneous_inv = lu_decomp.inverse();
123 m_jacobian_det = lu_decomp.determinant();
124 }
125
126 constexpr Point<FloatType, Dim> transform(
127 const Point<FloatType, Dim>& pt) const {
128 return m_homogeneous * pt + m_translations;
129 }
130 constexpr Point<FloatType, Dim> transformInv(
131 const Point<FloatType, Dim>& pt) const {
132 return m_homogeneous_inv * (pt - m_translations);
133 }
134 constexpr FloatType jacobianDet(const Point<FloatType, Dim>&) const {
135 return m_jacobian_det;
136 }
137
138 protected:
139 const Point<FloatType, Dim> m_translations;
140 Eigen::Matrix<FloatType, Dim, Dim> m_homogeneous;
141 Eigen::Matrix<FloatType, Dim, Dim> m_homogeneous_inv;
142 FloatType m_jacobian_det;
143};
