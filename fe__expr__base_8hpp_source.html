55// horizontally concatenate two matrices (that have the same number of rows)
56static inline VULPES_ALWAYS_INLINE auto hcat(const auto& u, const auto& v) {
57 using TypeU = std::decay_t<decltype(u)>;
58 using PolyTypeU = typename TypeU::Scalar;
59 using TypeV = std::decay_t<decltype(v)>;
60 using PolyTypeV = typename TypeV::Scalar;
61 static_assert(std::is_same_v<PolyTypeU, PolyTypeV>);
62 assert(u.rows() == v.rows());
63 return Eigen::Matrix<PolyTypeU, -1, -1>{
64 (Eigen::Matrix<PolyTypeU, -1, -1>(u.rows(), u.cols() + v.cols()) << u, v)
65 .finished()};
66}

68// lift a matrix of plain scalars (double) into polynomials
69template <typename FromMat, typename ToPoly>
70auto liftConstant(const FromMat& m) {
71 constexpr Eigen::Index Rows = FromMat::RowsAtCompileTime;
72 constexpr Eigen::Index Cols = FromMat::ColsAtCompileTime;
73 using Lifted = Eigen::Matrix<ToPoly, Rows, Cols>;
74
75 Lifted out;
76 // build the "1" polynomial
77 const typename ToPoly::ScalarType one_scalar(1);
78 const ToPoly one_poly(one_scalar);
79
80 for (Eigen::Index i = 0; i < m.rows(); ++i) {
81 for (Eigen::Index j = 0; j < m.cols(); ++j) {
82 assert(m(i, j) == 1.0);
83 out(i, j) = one_poly;
84 }
85 }
86 return out;
87}

89// lift a lower-dim polynomial into a higher-dim one
90template <typename FromMat, typename ToPoly>
91auto liftDimension(const FromMat& m) {
92 constexpr Eigen::Index Rows = FromMat::RowsAtCompileTime;
93 constexpr Eigen::Index Cols = FromMat::ColsAtCompileTime;
94 using Lifted = Eigen::Matrix<ToPoly, Rows, Cols>;
95
96 Lifted out;
97 for (Eigen::Index i = 0; i < m.rows(); ++i) {
98 for (Eigen::Index j = 0; j < m.cols(); ++j) {
99 out(i, j) = m(i, j).template liftDimension<0, 2 * FromMat::Scalar::dim>();
100 }
101 }
102 return out;
103}

105// lift the polynomial of lower dimension (between u and v) to the larger one
106static inline VULPES_ALWAYS_INLINE auto lift_to_same_space(
107 const auto& u, const auto& v) {
108 using TypeU = std::decay_t<decltype(u)>;
109 using PolyU = typename TypeU::Scalar;
110 using TypeV = std::decay_t<decltype(v)>;
111 using PolyV = typename TypeV::Scalar;
112
113 if constexpr (std::is_same_v<PolyU, PolyV>) {
114 return std::pair<TypeU, TypeV>{u, v};
115 } else if constexpr (std::is_same_v<PolyU, double>) {
116 // lift constant to polynomial, the actual coef should be
117 // handled by the evaluatePolyCoef method
118 return std::pair{liftConstant<TypeU, PolyV>(u), v};
119 } else if constexpr (std::is_same_v<PolyV, double>) {
120 // lift constant to polynomial, the actual coef should be
121 // handled by the evaluatePolyCoef method
122 return std::pair{u, liftConstant<TypeV, PolyU>(v)};
123 } else if constexpr (PolyU::dim < PolyV::dim) {
124 return std::pair{liftDimension<TypeU, PolyV>(u), v};
125 } else if constexpr (PolyU::dim > PolyV::dim) {
126 return std::pair{u, liftDimension<TypeV, PolyU>(v)};
127 } else {
128 // this shouldn't happen if dims are non-equal & caught above
129 assert(false);
130 return std::pair<TypeU, TypeV>{};
131 }
132}

134// horizontally concatenate the columns obtained as the cartesian product of the
135// set of columns of u and the set of columns of v (that have the same number of
136// rows)
137static inline VULPES_ALWAYS_INLINE auto cartesian_hcat(
138 const auto& u, const auto& v) {
139 using TypeU = std::decay_t<decltype(u)>;
140 using PolyTypeU = typename TypeU::Scalar;
141 using TypeV = std::decay_t<decltype(v)>;
142 using PolyTypeV = typename TypeV::Scalar;
143 // assume that polynomials are already lifted to the correct space
144 static_assert(std::is_same_v<PolyTypeU, PolyTypeV>);
145 assert(u.rows() == v.rows());
146 Eigen::Matrix<PolyTypeU, -1, -1> result(u.rows(), u.cols() * v.cols());
147 Eigen::Index idx = 0;
148 for (Eigen::Index i = 0; i < u.cols(); ++i) {
149 for (Eigen::Index j = 0; j < v.cols(); ++j) {
150 for (Eigen::Index k = 0; k < u.rows(); ++k) {
151 result(k, idx) = u(k, i) * v(k, j);
152 }
153 idx++;
154 }
155 }
156 return result;
157}
