41template <std::size_t Dim>
42struct GatheredMonomial {
43 uint16_t integr_id;
44 std::array<uint8_t, Dim * 2> evalut_id;
45 std::vector<std::pair<double, std::array<uint8_t, Dim * 2>>> gather;
46};
47
48struct CompressedMonomial {
49 double coef;
50 std::size_t integr_id;
51
52 bool operator<(const CompressedMonomial& other) const {
53 return std::tie(coef, integr_id) < std::tie(other.coef, other.integr_id);
54 }
55};
56
57struct CompressedGatheredMonomial {
58 uint16_t integr_id;
59 uint16_t evalut_id;
60 uint32_t gather_id;
61
62 bool operator<(const CompressedGatheredMonomial& other) const {
63 return std::tie(integr_id, evalut_id, gather_id) <
64 std::tie(other.integr_id, other.evalut_id, other.gather_id);
65 }
66};

68template <
69 typename MeshType,
70 typename FeExpressionTuple,
71 typename FeExpressionFaceTuple = std::tuple<>>
72class WeakFormIntegratorQFree : public WeakFormIntegratorBase<
73 MeshType,
74 FeExpressionTuple,
75 FeExpressionFaceTuple> {
76 public:
77 using WeakFormIntegratorBaseType = WeakFormIntegratorBase<
78 MeshType,
79 FeExpressionTuple,
80 FeExpressionFaceTuple>;
81 static inline constexpr std::size_t dim = MeshType::self_dim;
82 static inline constexpr uint32_t s_n_expr =
83 WeakFormIntegratorBaseType::s_n_expr;
84 static inline constexpr uint32_t s_n_face_expr =
85 WeakFormIntegratorBaseType::s_n_face_expr;
86
87 using FeType = std::remove_pointer_t<
88 std::tuple_element_t<0, FeExpressionTuple>>::FeSpaceType;
89 using PolyType = FeType::FeSpaceType::PolynomialFast;
90 using MonoType = PolyType::MonomialType;
91 // when considering the neighbour polynomial, the basis polynomial
92 // is lifted in a higher dimensional space to account for the mapping
93 // into the neigh bounding box (2*dim extra dimensions: diag(J) + transl)
94 using PolyTypeLifted =
95 std::decay_t<decltype(PolyType{}.template liftDimension<0, 2 * dim>())>;
96 using MonoTypeLifted = PolyTypeLifted::MonomialType;
97 using index_t = typename MeshType::index_t;
98
99 static inline constexpr index_t s_n_blocks =
100 WeakFormIntegratorBaseType::nblocks;
101
102 // uncompressed gathered polynomial
103 using GatheredPolynomial = std::vector<GatheredMonomial<dim>>;
104 // gathered coefficient for a compressed gathered polynomial
105 using GatheredPolynomialCoef = std::vector<std::pair<double, uint16_t>>;
106 using FlatPolyTypeLifted = datastructure::CrsMap<MonoTypeLifted, index_t>;
107 using FlatPolyType = datastructure::CrsMap<CompressedMonomial, index_t>;
108 using CompressedListGatheredPoly =
109 datastructure::CrsMap<CompressedGatheredMonomial, index_t>;
110
111 // transform an uncompressed table of uncompressed monomials
112 // in a compressed table of compressed monomials
113 FlatPolyType compress(const auto& uncompressed_table) {
114 // compress the table
115 const auto compr_table_uncompr_monos =
116 uncompressed_table.template compress<index_t>();
117 // compress the monomials
118 FlatPolyType compr_table_compr_monos;
119 compr_table_compr_monos.indices() = compr_table_uncompr_monos.indices();
120 std::array<std::size_t, dim> exp;
121 for (const auto& mono : compr_table_uncompr_monos.data()) {
122 for (std::size_t d = 0; d < dim; ++d) {
123 exp[d] = mono.exp()[d];
124 }
125 compr_table_compr_monos.data().emplace_back(
126 mono.coef(), monomial::getLinearIndex<dim, std::size_t>(exp));
127 }
128 return compr_table_compr_monos;
129 }

131 WeakFormIntegratorQFree(
132 const MeshType& mesh,
133 CellManager<MeshType, FeType>& cell_manager,
134 const FeExpressionTuple& fe_expression,
135 const FeExpressionFaceTuple& fe_expression_face = FeExpressionFaceTuple{})
136 : WeakFormIntegratorBaseType(
137 mesh, cell_manager, fe_expression, fe_expression_face) {
138 const Eigen::Index n_dof_loc = this->m_fe_space.getBasis().size();
139
140 // cache the polynomials of the expressions
141 // they are computed in reference configuration, thus they do not change
142 // to take into account the transformation we split the polynomials
143 // into components that need different transformations
144 // Finally, the coef of the transformation is computed on the fly during
145 // the actual integration.
146 vutils::for_each_idxed(
147 fe_expression, [this, n_dof_loc](const auto& expr, std::size_t idx) {
148 datastructure::CrsMapUncompressed<MonoType> poly_table_tmp;
149 std::size_t poly_idx = 0;
150 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
151 for (Eigen::Index bj = 0; bj < s_n_blocks; ++bj) {
152 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
153 for (Eigen::Index j = 0; j < n_dof_loc; ++j) {
154 const Eigen::Matrix<PolyType, 1, -1> poly_comps =
155 expr->evaluatePoly(this->m_fe_values, i, j, bi, bj)
156 .unaryExpr([](const auto& p) {
157 return p.template to<double>();
158 });
159 m_npoly_comps_volume[idx] = poly_comps.size();
160 for (Eigen::Index k = 0; k < poly_comps.size(); ++k) {
161 auto& row = poly_table_tmp.getRow(poly_idx);
162 for (const auto& monomial : poly_comps[k].data()) {
163 row.push_back(monomial);
164 }
165 poly_idx++;
166 }
167 }
168 }
169 }
170 }
171 m_poly_table_flat[idx] = compress(poly_table_tmp);
172 });
173
174 // Expressions on the boundary may need the mapping of the neighbour
175 // In these cases the polynomials are lifted in higher dimension,
176 // that is, we add dummy variables representing the coef of the map
177 vutils::for_each_idxed(
178 fe_expression_face,
179 [this, n_dof_loc](const auto& expr, std::size_t idx) {
180 using PolyMatType = std::decay_t<decltype(expr->evaluatePoly(
181 this->m_fe_values, 0, 0, 0, 0))>;
182 constexpr auto expr_poly_dim = PolyMatType::Scalar::dim;
183
184 using MonoTypeCurr = std::conditional_t<
185 expr_poly_dim == dim, MonoType, MonoTypeLifted>;
186 datastructure::CrsMapUncompressed<MonoTypeCurr> poly_table_tmp;
187 std::size_t poly_idx = 0;
188 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
189 for (Eigen::Index bj = 0; bj < s_n_blocks; ++bj) {
190 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
191 for (Eigen::Index j = 0; j < n_dof_loc; ++j) {
192 using ExprPolyType = std::conditional_t<
193 expr_poly_dim == dim, PolyType, PolyTypeLifted>;
194 const Eigen::Matrix<ExprPolyType, 1, -1> poly_comps =
195 expr->evaluatePoly(this->m_fe_values, i, j, bi, bj)
196 .unaryExpr([](const auto& p) {
197 return p.template to<double>();
198 });
199 m_npoly_comps_face[idx] = poly_comps.size();
200 for (Eigen::Index k = 0; k < poly_comps.size(); ++k) {
201 auto& row = poly_table_tmp.getRow(poly_idx);
202 for (const auto& monomial : poly_comps[k].data()) {
203 row.push_back(monomial);
204 }
205 poly_idx++;
206 }
207 }
208 }
209 }
210 }
211
212 if constexpr (expr_poly_dim == dim) {
213 m_poly_table_face_flat[idx] = compress(poly_table_tmp);
214 } else {
215 // compressing lifted monomials is more complex, we first have
216 // to save them all, than apply gathering
217 m_poly_table_face_lifted_flat[idx] =
218 poly_table_tmp.template compress<index_t>();
219 }
220 });
221 // speed-up polynomial evaluation by gathering common monomials
222 gatherMonomials();
223 // often polynomials are repeated in a local matrix
224 // we find out where are the repetions and build a table
225 // with one of each kind of polynomial and a table with the
226 // list of entriens where it should be put
227 buildLookupTables();
228 }

230 void buildLookupTables() {
231 const Eigen::Index n_dof_loc = this->m_fe_space.getBasis().size();
232
233 // since we have to apply the same procedure to the integrals
234 // in the volume, on the boundary and with the neighbour
235 // we wrap together the datastructures to avoid code duplication
236 const auto tables = std::tuple(
237 std::tuple(
238 &m_poly_table_flat, //
239 &m_poly_table_flat_nodupes, //
240 &m_poly_table_flat_ijs_table),
241 std::tuple(
242 &m_poly_table_face_flat, //
243 &m_poly_table_face_flat_nodupes, //
244 &m_poly_table_face_flat_ijs_table),
245 std::tuple(
246 &m_poly_table_face_lifted_gathered,
247 &m_poly_table_face_lifted_gathered_nodupes,
248 &m_poly_table_face_lifted_gathered_ijs_table));
249
250 vutils::for_each(tables, [n_dof_loc, this](auto tpts) {
251 auto tpt = std::get<0>(tpts);
252 auto tpt_nodup = std::get<1>(tpts);
253 auto tpt_ijk = std::get<2>(tpts);
254 for (std::size_t exi = 0; exi < tpt->size(); ++exi) {
255 using MonomialType = std::decay_t<decltype((*tpt)[0][0][0])>;
256 using PolynomialType = std::vector<MonomialType>;
257 const auto npolys = (*tpt)[exi].size();
258 const auto slice_size = n_dof_loc * n_dof_loc * s_n_blocks * s_n_blocks;
259 const auto n_k_comps = npolys / slice_size;
260
261 index_t poly_idx = 0;
262 std::vector<index_t> keep_idxs;
263 datastructure::CrsMapUncompressed<TypeIJ> table;
264 std::map<PolynomialType, std::size_t> unq_polys;
265 // dummy just to compute automatically the `kbibj` offset
266 Eigen::Tensor<double, 3> dummy(n_k_comps, s_n_blocks, s_n_blocks);
267 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
268 for (Eigen::Index bj = 0; bj < s_n_blocks; ++bj) {
269 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
270 for (Eigen::Index j = 0; j < n_dof_loc; ++j) {
271 for (Eigen::Index k = 0; k < n_k_comps; ++k) {
272 const auto kbibj = &dummy(k, bi, bj) - dummy.data();
273 const TypeIJ ij = {
274 i * s_n_blocks + bi, j * s_n_blocks + bj, kbibj};
275 PolynomialType poly;
276 for (auto mono : (*tpt)[exi][poly_idx]) {
277 poly.push_back(mono);
278 }
279 std::sort(poly.begin(), poly.end());
280 if (poly.size()) {
281 const auto [it, inserted] =
282 unq_polys.emplace(poly, unq_polys.size());
283 table.getRow(it->second).push_back(ij);
284 if (inserted) {
285 keep_idxs.push_back(poly_idx);
286 }
287 }
288 poly_idx++;
289 }
290 }
291 }
292 }
293 }
294
295 (*tpt_nodup)[exi] = (*tpt)[exi].slice(keep_idxs);
296 (*tpt_ijk)[exi] = table.template compress<index_t>();
297 }
298 });
299 }

305 std::array<std::vector<GatheredPolynomial>, s_n_face_expr>
306 gatherUpcompressedMonomials() {
307 // gather the lifted monomials
308 std::array<std::vector<GatheredPolynomial>, s_n_face_expr>
309 gathered_monos_table;
310 for (std::size_t i = 0; i < m_poly_table_face_lifted_flat.size(); ++i) {
311 const auto& old_table = m_poly_table_face_lifted_flat[i];
312 for (index_t poly_id = 0; poly_id < old_table.size(); ++poly_id) {
313 GatheredPolynomial gathered_poly;
314 for (const auto& mono : old_table[poly_id]) {
315 // in the monomial exponent, split the first dim components (the ones
316 // that must be integrated and with respect to which we gather), and
317 // the evaluation components (the last 2 * dim ones, which come from
318 // the cartesian mapping)
319 std::array<uint32_t, dim> int_ids;
320 for (std::size_t d = 0; d < dim; ++d) {
321 int_ids[d] = mono.exp()[d];
322 }
323 std::array<uint8_t, dim * 2> eva_ids;
324 for (std::size_t d = 0; d < dim * 2; ++d) {
325 eva_ids[d] = mono.exp()[d + dim];
326 }
327 const uint16_t int_id =
328 monomial::getLinearIndex<dim, uint32_t>(int_ids);
329
330 // search among gathered monomials if we already have
331 // a monomial with the common term
332 std::size_t mono_id = -1;
333 for (std::size_t j = 0; j < gathered_poly.size(); ++j) {
334 if (gathered_poly[j].integr_id == int_id) {
335 mono_id = j;
336 break;
337 }
338 }
339 if (mono_id == std::size_t(-1)) {
340 // if the common term is not present, we add a new term
341 gathered_poly.push_back(GatheredMonomial<dim>{
342 .integr_id = int_id,
343 .evalut_id = {0},
344 .gather = {{mono.coef(), eva_ids}}});
345 } else {
346 // the term is already presend, we append the non-common part
347 gathered_poly[mono_id].gather.push_back({mono.coef(), eva_ids});
348 }
349 }
350 // extract for each gathered monomial the common part in the
351 // gathered part. E.g. x(ab + a) -> xa(b + 1)
352 for (auto& mono : gathered_poly) {
353 std::array<uint8_t, dim * 2> min_id;
354 min_id.fill(255);
355 // find the common part
356 for (const auto& c : mono.gather) {
357 for (std::size_t d = 0; d < dim * 2; ++d) {
358 min_id[d] = std::min(min_id[d], c.second[d]);
359 }
360 }
361 // remove the common part from the gathered
362 mono.evalut_id = min_id;
363 for (auto& c : mono.gather) {
364 for (std::size_t d = 0; d < dim * 2; ++d) {
365 c.second[d] -= min_id[d];
366 }
367 }
368 }
369 gathered_monos_table[i].push_back(gathered_poly);
370 }
371 }
372 return gathered_monos_table;
373 }

375 // Builds a unique evaluation coefficient table from all gathered monomials.
376 std::vector<std::array<uint8_t, dim * 2>> buildUniqueEvaTable(
377 const std::array<std::vector<GatheredPolynomial>, s_n_face_expr>&
378 gathered_monos_table) {
379 std::set<std::array<uint8_t, dim * 2>> eva_set;
380 for (const auto& gathered_monos : gathered_monos_table) {
381 for (const auto& poly : gathered_monos) {
382 for (const auto& mono : poly) {
383 eva_set.insert(mono.evalut_id);
384 for (const auto& entry : mono.gather) {
385 eva_set.insert(entry.second);
386 }
387 }
388 }
389 }
390 return std::vector<std::array<uint8_t, dim * 2>>(
391 eva_set.begin(), eva_set.end());
392 }

402 void compressGatheredPolynomials(
403 const std::array<std::vector<GatheredPolynomial>, s_n_face_expr>&
404 gathered_monos_table) {
405 // compress the monomials, consists in two steps
406 // 1. the GatheredMonomial becomes a CompressedGatheredMonomial
407 // 2. find the unique gather coefficients needed for the compressed monos
408 using GatheredPolynomialCoef = std::vector<std::pair<double, uint16_t>>;
409 std::map<GatheredPolynomialCoef, uint32_t> gather_table;
410 using CompressedGatheredPolynomial =
411 std::vector<CompressedGatheredMonomial>;
412 std::array<std::vector<CompressedGatheredPolynomial>, s_n_face_expr>
413 compr_gathered_monos_table;
414 for (std::size_t expr_id = 0; expr_id < compr_gathered_monos_table.size();
415 ++expr_id) {
416 auto& compr_gathered_monos = compr_gathered_monos_table[expr_id];
417 compr_gathered_monos.resize(gathered_monos_table[expr_id].size());
418 for (std::size_t i = 0; i < gathered_monos_table[expr_id].size(); ++i) {
419 const auto& poly = gathered_monos_table[expr_id][i];
420 compr_gathered_monos[i].resize(poly.size());
421 for (std::size_t j = 0; j < poly.size(); ++j) {
422 const auto& mono = poly[j];
423 auto& compr_mono = compr_gathered_monos[i][j];
424 compr_mono.integr_id = mono.integr_id;
425 // transform the exponent array into a index
426 const auto it = std::lower_bound(
427 m_eva_idx_table.begin(), m_eva_idx_table.end(), mono.evalut_id);
428 assert(it != m_eva_idx_table.end());
429 assert(*it == mono.evalut_id);
430 compr_mono.evalut_id = std::distance(m_eva_idx_table.begin(), it);
431 // transform the gather part into a index
432 // this is done by suitably updating the gather_table
433 std::vector<std::pair<double, uint16_t>> sorted_gather;
434 for (const auto& [coef, eval_id] : mono.gather) {
435 const auto eval_it = std::lower_bound(
436 m_eva_idx_table.begin(), m_eva_idx_table.end(), eval_id);
437 assert(eval_it != m_eva_idx_table.end());
438 assert(*eval_it == eval_id);
439 sorted_gather.push_back(
440 {coef, std::distance(m_eva_idx_table.begin(), eval_it)});
441 }
442 std::sort(sorted_gather.begin(), sorted_gather.end());
443 const auto [gather_it, inserted] = gather_table.emplace(
444 sorted_gather, (uint32_t)gather_table.size());
445 compr_mono.gather_id = gather_it->second;
446 }
447 }
448
449 m_poly_table_face_lifted_gathered[expr_id] =
450 CompressedListGatheredPoly{compr_gathered_monos_table[expr_id]};
451 }
452 assert(m_eva_idx_table.size() < (1 << 16));
453 assert(gather_table.size() < (1UL << 32UL));
454 // compress the gather table
455 std::vector<GatheredPolynomialCoef> gather_table_flat(gather_table.size());
456 for (const auto& [gather, id] : gather_table) {
457 gather_table_flat[id] = gather;
458 }
459 m_gathered_monos = decltype(m_gathered_monos)(gather_table_flat);
460 }

470 void gatherMonomials() {
471 const auto gathered_monos_table = gatherUpcompressedMonomials();
472 // find all the unique coefficients of the evaluate part
473 m_eva_idx_table = buildUniqueEvaTable(gathered_monos_table);
474 // compress the gathered monomials
475 compressGatheredPolynomials(gathered_monos_table);
476 }

478 // composition of ref-to-physic physic-to-ref[neigh]
479 // first column is the jacobian (diagonal), the second the translation
480 auto computeRefToNeighRefMap() const {
481 Eigen::Matrix<double, dim, 2> neigh_flat_mapping;
482 if (this->m_need_neigh_values && this->m_cell_neigh_info.have_neigh) {
483 const auto& neigh_map = *this->m_cell_manager.getMapping(
484 this->m_cell_neigh_info.neigh_cell_id);
485 neigh_flat_mapping.col(0) = neigh_map.getJacobianInv().array() *
486 this->m_mapping->getJacobian().array();
487 neigh_flat_mapping.col(1) =
488 neigh_map.getJacobianInv().array() *
489 this->m_mapping->getTranslation().array() +
490 neigh_map.getTranslationInv().array();
491 }
492 return neigh_flat_mapping;
493 }

495 void integrateVolumeCell(const double* aints_data) {
496 Profiler::start("integrate-volume-qfree");
497 const double jac_det = this->m_mapping->jacobianDet({});
498 std::vector<double> buffer;
499 vutils::for_each_idxed(
500 this->m_fe_expression, [this, jac_det, aints_data, &buffer](
501 const auto& expr, Eigen::Index expr_id) {
502 {
503 const auto& table = m_poly_table_flat_nodupes[expr_id];
504 buffer.resize(table.size());
505 for (index_t i = 0; i < table.size(); ++i) {
506 double integral = 0.0;
507 for (const auto mono : table[i]) {
508 integral += mono.coef * aints_data[mono.integr_id];
509 }
510 buffer[i] = integral;
511 }
512 }
513 expr->precompute(this->m_fe_values, this->m_cell_manager.getCells());
514 const auto& table = m_poly_table_flat_ijs_table[expr_id];
515
516 Profiler::start("evaluatePolyCoef+cacheVolume");
517 const auto K = m_npoly_comps_volume[expr_id];
518 Eigen::Tensor<double, 3> coefs(K, s_n_blocks, s_n_blocks);
519 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
520 for (Eigen::Index bj = 0; bj < s_n_blocks; ++bj) {
521 const auto coef = expr->evaluatePolyCoef(
522 *(this->m_mapping), this->m_fe_values, bi, bj);
523 assert(coef.size() == K);
524 assert(coef.rows() == 1);
525 for (Eigen::Index k = 0; k < K; ++k) {
526 coefs(k, bi, bj) = coef(0, k);
527 }
528 }
529 }
530 Profiler::stop("evaluatePolyCoef+cacheVolume");
531
532 for (index_t pid = 0; pid < table.size(); ++pid) {
533 const auto integral = jac_det * buffer[pid];
534 for (const auto ijk : table[pid]) {
535 this->m_integrals(ijk[0], ijk[1], expr_id) +=
536 coefs.data()[ijk[2]] * integral;
537 }
538 }
539 });
540 Profiler::stop("integrate-volume-qfree");
541 }

543 double updateFaceNormalAndGetNanson() {
544 const double jac_det = this->m_mapping->jacobianDet({});
545 const auto jac = this->m_mapping->getJacobian();
546 const auto& cell = this->getCell(this->m_cell_id);
547 const Eigen::Vector<double, dim> ns = cell.normals().col(this->m_face_id);
548 const Eigen::Vector<double, dim> Fns = jac.array() * ns.array();
549 this->m_fe_values.setFaceNormal(ns);
550 // ||JF^-TN||, where F jacobian of ref-to-physic, J is |det(F)|,
551 // N unit outer in ref. By expoiting linearity and F is diagonal we have
552 // J/||Fn||, where n is the unit normal in physical space. See:
553 // https://en.wikipedia.org/wiki/Finite_strain_theory#Transformation_of_a_surface_and_volume_element
554 const double nanson = jac_det / Fns.norm();
555 return nanson;
556 }

558 void precomputeMonomialCoefsForFace() {
559 Profiler::start("integrate-bnd-qfree-precomp");
560 const Eigen::Matrix<double, dim, 2> neigh_map = computeRefToNeighRefMap();
561 Eigen::Matrix<double, dim * 2, -1> eval_table(dim * 2, 32);
562 eval_table.col(0) = Eigen::VectorXd::Ones(dim * 2);
563 eval_table.col(1) =
564 Eigen::Map<const Eigen::VectorXd>(neigh_map.data(), dim * 2);
565 for (Eigen::Index i = 2; i < eval_table.cols(); ++i) {
566 eval_table.col(i) =
567 eval_table.col(i - 1).array() * eval_table.col(1).array();
568 }
569 m_eval_table_ext.resize(m_eva_idx_table.size());
570 std::fill(m_eval_table_ext.begin(), m_eval_table_ext.end(), 1.0);
571 for (std::size_t i = 0; i < m_eval_table_ext.size(); ++i) {
572 for (std::size_t d = 0; d < 2 * dim; ++d) {
573 m_eval_table_ext[i] *= eval_table(d, m_eva_idx_table[i][d]);
574 }
575 }
576 Profiler::stop("integrate-bnd-qfree-precomp");
577
578 Profiler::start("integrate-bnd-qfree-precomp-v2");
579 m_gather_coef.resize(m_gathered_monos.size());
580 std::fill(m_gather_coef.begin(), m_gather_coef.end(), 0.0);
581 for (std::size_t i = 0; i < m_gather_coef.size(); ++i) {
582 for (const auto& [coef, id] : m_gathered_monos[i]) {
583 m_gather_coef[i] += coef * m_eval_table_ext[id];
584 }

589 void integrateFaceCell(const double* eints_data, double nanson) {
590 Profiler::start("integrate-bnd-qfree");
591 std::vector<double> buffer;
592 vutils::for_each_idxed(
593 this->m_fe_expression_face,
594 [this, nanson, eints_data, &buffer](
595 const auto& expr, Eigen::Index expr_id) {
596 using PolyMatType = std::decay_t<decltype(expr->evaluatePoly(
597 this->m_fe_values, 0, 0, 0, 0))>;
598 constexpr auto expr_poly_dim = PolyMatType::Scalar::dim;
599
600 if constexpr (expr_poly_dim == dim) {
601 const auto& table = m_poly_table_face_flat_nodupes[expr_id];
602 buffer.resize(table.size());
603 for (index_t i = 0; i < table.size(); ++i) {
604 double integral = 0.0;
605 for (const auto mono : table[i]) {
606 integral += mono.coef * eints_data[mono.integr_id];
607 }
608 buffer[i] = integral;
609 }
610 } else {
611 const auto& table =
612 m_poly_table_face_lifted_gathered_nodupes[expr_id];
613 buffer.resize(table.size());
614 for (index_t i = 0; i < table.size(); ++i) {
615 double integral = 0.0;
616 for (const auto mono : table[i]) {
617 integral += m_gather_coef[mono.gather_id] *
618 m_eval_table_ext[mono.evalut_id] *
619 eints_data[mono.integr_id];
620 }
621 buffer[i] = integral;
622 }
623 }
624 auto curr_integral = this->getCurrFaceIntegral(expr, expr_id);
625 if (!curr_integral.size()) {
626 return;
627 }
628
629 Profiler::start("evaluatePolyCoef+cacheFace");
630 const auto K = m_npoly_comps_face[expr_id];
631 Eigen::Tensor<double, 3> coefs(K, s_n_blocks, s_n_blocks);
632 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
633 for (Eigen::Index bj = 0; bj < s_n_blocks; ++bj) {
634 const auto coef = expr->evaluatePolyCoef(
635 *(this->m_mapping), this->m_fe_values, bi, bj);
636 assert(coef.size() == K);
637 assert(coef.rows() == 1);
638 for (Eigen::Index k = 0; k < K; ++k) {
639 coefs(k, bi, bj) = coef(0, k);
640 }
641 }
642 }
643 Profiler::stop("evaluatePolyCoef+cacheFace");
644
645 using TableType = std::decay_t<std::conditional_t<
646 expr_poly_dim == dim,
647 decltype(m_poly_table_face_flat_ijs_table[0]),
648 decltype(m_poly_table_face_lifted_gathered_ijs_table[0])>>;
649 const TableType* table = nullptr;
650 if constexpr (expr_poly_dim == dim) {
651 table = &m_poly_table_face_flat_ijs_table[expr_id];
652 } else {
653 table = &m_poly_table_face_lifted_gathered_ijs_table[expr_id];
654 }
655 for (index_t pid = 0; pid < (*table).size(); ++pid) {
656 const auto integral = nanson * buffer[pid];
657 for (const auto ijk : (*table)[pid]) {
658 curr_integral(ijk[0], ijk[1]) += coefs.data()[ijk[2]] * integral;
659 }
660 }
661 });
662 Profiler::stop("integrate-bnd-qfree");
663 }

665 void integrate(index_t cell_id) {
666 this->preallocate(cell_id);
667 auto& cell = this->getCell(this->m_cell_id, true);
668
669 // TODO: change integration P/Q depending of polynomial type
670 const auto order = std::max(this->m_order, this->m_order_face);
671 Profiler::start("integrateMonomialsQFree");
672 const auto integrateMonomials =
673 qfree::integrateMonomialsQFree<true, false, dim, double, index_t>;
674 const auto [aints, eints] = integrateMonomials(cell, order);
675 Profiler::stop("integrateMonomialsQFree");
676
677 integrateVolumeCell(aints.begin());
678
679 const auto eints_data = eints.begin();
680 const auto slice_dim = aints.numel();
681 for (this->m_face_id = 0; this->m_face_id < cell.nfaces();
682 ++this->m_face_id) {
683 this->findCellNeigh();
684 const double* eints_curr_data = eints_data + this->m_face_id * slice_dim;
685 const double nanson = updateFaceNormalAndGetNanson();
686 precomputeMonomialCoefsForFace();
687 integrateFaceCell(eints_curr_data, nanson);
688 }
689
690 this->applyBasisScaling();
691 }
692
693 protected:
694 using TypeIJ = std::array<Eigen::Index, 3>;
695 using RepeatingTableIJ = datastructure::CrsMap<TypeIJ, index_t>;
696 // poly table for the volume expressions
697 std::array<FlatPolyType, s_n_expr> m_poly_table_flat;
698 std::array<FlatPolyType, s_n_expr> m_poly_table_flat_nodupes;
699 std::array<RepeatingTableIJ, s_n_expr> m_poly_table_flat_ijs_table;
700 // poly table for the face expressions
701 std::array<FlatPolyType, s_n_face_expr> m_poly_table_face_flat;
702 std::array<FlatPolyType, s_n_face_expr> m_poly_table_face_flat_nodupes;
703 std::array<RepeatingTableIJ, s_n_face_expr> m_poly_table_face_flat_ijs_table;
704 // poly table for the face expressions that must be lifted (need neigh map)
705 std::array<FlatPolyTypeLifted, s_n_face_expr> m_poly_table_face_lifted_flat;
706 // compressed and gathered version of `m_poly_table_face_lifted_flat`
707 std::array<CompressedListGatheredPoly, s_n_face_expr>
708 m_poly_table_face_lifted_gathered;
709 std::array<CompressedListGatheredPoly, s_n_face_expr>
710 m_poly_table_face_lifted_gathered_nodupes;
711 std::array<RepeatingTableIJ, s_n_face_expr>
712 m_poly_table_face_lifted_gathered_ijs_table;
713 // table to map a monomial exponent to an integer and vice-versa
714 // the monomials are ordered lexicographycally
715 std::vector<std::array<uint8_t, dim * 2>> m_eva_idx_table;
716 // table that stores the gathered monomials
717 // each row represents a coef of the gathered monomial of the type
718 // x(c_1 a_1 + c_2 a_2 + ... ) the pair is [c_i, a_i] where a_i is a monomial
719 // taken from the eval table built from `m_eva_idx_table`
720 datastructure::CrsMap<std::pair<double, uint16_t>, index_t> m_gathered_monos;
721
722 std::vector<double> m_gather_coef;
723 std::vector<double> m_eval_table_ext;
724
725 std::array<Eigen::Index, s_n_expr> m_npoly_comps_volume;
726 std::array<Eigen::Index, s_n_face_expr> m_npoly_comps_face;
727};
