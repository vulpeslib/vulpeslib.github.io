442 Eigen::Matrix<double, dim, 2> computeRefToNeighRefMap() const {
443 Eigen::Matrix<double, dim, 2> neigh_flat_mapping;
444 if (this->m_need_neigh_values && this->m_cell_neigh_info.have_neigh) {
445 const auto& neigh_map = *this->m_cell_manager.getMapping(
446 this->m_cell_neigh_info.neigh_cell_id);
447
448 // Get periodic shift in physical coordinates (if mesh is periodic)
449 const auto periodic_shift = this->m_cell_manager.getPeriodicShift(
450 this->m_cell_neigh_info.neigh_cell_id, this->m_cell_id);
451
452 // Composed transformation: current->physical->neighbor with periodic
453 // shift
454 neigh_flat_mapping.col(0) = neigh_map.getJacobianInv().array() *
455 this->m_mapping->getJacobian().array();
456 neigh_flat_mapping.col(1) =
457 neigh_map.getJacobianInv().array() *
458 (this->m_mapping->getTranslation().array() +
459 periodic_shift.array()) +
460 neigh_map.getTranslationInv().array();
461 }
462 return neigh_flat_mapping;
463 }

611WeakFormIntegratorQFree<
612 MeshType,
613 FeExpressionTuple,
614 FeExpressionFaceTuple>::FlatPolyType
615WeakFormIntegratorQFree<MeshType, FeExpressionTuple, FeExpressionFaceTuple>::
616 compress(const datastructure::CrsMapUncompressed<WeakFormIntegratorQFree<
617 MeshType,
618 FeExpressionTuple,
619 FeExpressionFaceTuple>::MonoType>& uncompressed_table) {
620 // compress the table
621 const auto compr_table_uncompr_monos =
622 uncompressed_table.template compress<index_t>();
623 // compress the monomials
624 FlatPolyType compr_table_compr_monos;
625 compr_table_compr_monos.indices() = compr_table_uncompr_monos.indices();
626 std::array<std::size_t, dim> exp;
627 for (const auto& mono : compr_table_uncompr_monos.data()) {
628 for (std::size_t d = 0; d < dim; ++d) {
629 exp[d] = mono.exp()[d];
630 }
631 compr_table_compr_monos.data().emplace_back(
632 mono.coef(), monomial::getLinearIndex<dim, std::size_t>(exp));
633 }
634 return compr_table_compr_monos;
635};

637template <
638 typename MeshType,
639 typename FeExpressionTuple,
640 typename FeExpressionFaceTuple>
641WeakFormIntegratorQFree<MeshType, FeExpressionTuple, FeExpressionFaceTuple>::
642 WeakFormIntegratorQFree(
643 const MeshType& mesh,
644 CellManager<MeshType, FeType>& cell_manager,
645 const FeExpressionTuple& fe_expression,
646 const FeExpressionFaceTuple& fe_expression_face)
647 : WeakFormIntegratorBaseType(
648 mesh, cell_manager, fe_expression, fe_expression_face) {
649 const Eigen::Index n_dof_loc = this->m_fe_space.getBasis().size();
650
651 // Cache polynomials for all expressions in reference configuration.
652 // The coefficient from the physical mapping is applied on the fly during
653 // integration via evaluatePolyCoef. Bilinear forms loop over (bj,bi,j,i);
654 // linear (RHS) forms loop over (bi,i) only -- no trial function.
655 const auto [bilinear_exprs, linear_exprs] =
656 internal::split_expression_tuple(fe_expression);
657
658 vutils::for_each_idxed(
659 bilinear_exprs, [this, n_dof_loc](const auto& expr, std::size_t idx) {
660 datastructure::CrsMapUncompressed<MonoType> poly_table_tmp;
661 std::size_t poly_idx = 0;
662 for (Eigen::Index bj = 0; bj < s_n_blocks; ++bj) {
663 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
664 for (Eigen::Index j = 0; j < n_dof_loc; ++j) {
665 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
666 const Eigen::Matrix<PolyType, 1, -1> poly_comps =
667 expr->evaluatePoly(this->m_fe_values, j, i, bj, bi)
668 .unaryExpr([](const auto& p) {
669 return p.template to<double>();
670 });
671 m_npoly_comps_volume[idx] = poly_comps.size();
672 for (Eigen::Index k = 0; k < poly_comps.size(); ++k) {
673 auto& row = poly_table_tmp.getRow(poly_idx);
674 for (const auto& monomial : poly_comps[k].data()) {
675 row.push_back(monomial);
676 }
677 poly_idx++;
678 }
679 }
680 }
681 }
682 }
683 m_poly_table_flat[idx] = compress(poly_table_tmp);
684 });
685
686 vutils::for_each_idxed(
687 linear_exprs, [this, n_dof_loc](const auto& expr, std::size_t idx) {
688 datastructure::CrsMapUncompressed<MonoType> poly_table_tmp;
689 std::size_t poly_idx = 0;
690 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
691 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
692 const Eigen::Matrix<PolyType, 1, -1> poly_comps =
693 expr->evaluatePoly(this->m_fe_values, 0, i, 0, bi)
694 .unaryExpr(
695 [](const auto& p) { return p.template to<double>(); });
696 m_npoly_comps_volume_linear[idx] = poly_comps.size();
697 for (Eigen::Index k = 0; k < poly_comps.size(); ++k) {
698 auto& row = poly_table_tmp.getRow(poly_idx);
699 for (const auto& monomial : poly_comps[k].data()) {
700 row.push_back(monomial);
701 }
702 poly_idx++;
703 }
704 }
705 }
706 m_poly_table_flat_linear[idx] = compress(poly_table_tmp);
707 });
708
709 // Face expressions: bilinear forms may need neighbor mapping (lifted
710 // polynomials); linear forms use local test basis only (non-lifted).
711 const auto [bilinear_face_exprs, linear_face_exprs] =
712 internal::split_expression_tuple(fe_expression_face);
713
714 vutils::for_each_idxed(
715 bilinear_face_exprs,
716 [this, n_dof_loc](const auto& expr, std::size_t idx) {
717 using PolyMatType = std::decay_t<decltype(expr->evaluatePoly(
718 this->m_fe_values, 0, 0, 0, 0))>;
719 constexpr auto expr_poly_dim = PolyMatType::Scalar::dim;
720
721 using MonoTypeCurr =
722 std::conditional_t<expr_poly_dim == dim, MonoType, MonoTypeLifted>;
723 datastructure::CrsMapUncompressed<MonoTypeCurr> poly_table_tmp;
724 std::size_t poly_idx = 0;
725 for (Eigen::Index bj = 0; bj < s_n_blocks; ++bj) {
726 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
727 for (Eigen::Index j = 0; j < n_dof_loc; ++j) {
728 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
729 using ExprPolyType = std::conditional_t<
730 expr_poly_dim == dim, PolyType, PolyTypeLifted>;
731 const Eigen::Matrix<ExprPolyType, 1, -1> poly_comps =
732 expr->evaluatePoly(this->m_fe_values, j, i, bj, bi)
733 .unaryExpr([](const auto& p) {
734 return p.template to<double>();
735 });
736 m_npoly_comps_face[idx] = poly_comps.size();
737 for (Eigen::Index k = 0; k < poly_comps.size(); ++k) {
738 auto& row = poly_table_tmp.getRow(poly_idx);
739 for (const auto& monomial : poly_comps[k].data()) {
740 row.push_back(monomial);
741 }
742 poly_idx++;
743 }
744 }
745 }
746 }
747 }
748
749 if constexpr (expr_poly_dim == dim) {
750 m_poly_table_face_flat[idx] = compress(poly_table_tmp);
751 } else {
752 // compressing lifted monomials is more complex, we first have
753 // to save them all, than apply gathering
754 m_poly_table_face_lifted_flat[idx] =
755 poly_table_tmp.template compress<index_t>();
756 }
757 });
758
759 vutils::for_each_idxed(
760 linear_face_exprs, [this, n_dof_loc](const auto& expr, std::size_t idx) {
761 datastructure::CrsMapUncompressed<MonoType> poly_table_tmp;
762 std::size_t poly_idx = 0;
763 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
764 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
765 const Eigen::Matrix<PolyType, 1, -1> poly_comps =
766 expr->evaluatePoly(this->m_fe_values, 0, i, 0, bi)
767 .unaryExpr(
768 [](const auto& p) { return p.template to<double>(); });
769 m_npoly_comps_face_linear[idx] = poly_comps.size();
770 for (Eigen::Index k = 0; k < poly_comps.size(); ++k) {
771 auto& row = poly_table_tmp.getRow(poly_idx);
772 for (const auto& monomial : poly_comps[k].data()) {
773 row.push_back(monomial);
774 }
775 poly_idx++;
776 }
777 }
778 }
779 m_poly_table_face_flat_linear[idx] = compress(poly_table_tmp);
780 });
781 // speed-up polynomial evaluation by gathering common monomials
782 gatherLiftedMonomials();
783 // often polynomials are repeated in a local matrix
784 // we find out where are the repetions and build a table
785 // with one of each kind of polynomial and a table with the
786 // list of entriens where it should be put
787 buildLookupTables();
788}

790template <
791 typename MeshType,
792 typename FeExpressionTuple,
793 typename FeExpressionFaceTuple>
794void WeakFormIntegratorQFree<
795 MeshType,
796 FeExpressionTuple,
797 FeExpressionFaceTuple>::buildLookupTables() {
798 const Eigen::Index n_dof_loc = this->m_fe_space.getBasis().size();
799
800 // Bilinear (matrix) polynomial tables: slice is (bj,bi,j,i,k).
801 // Apply the same deduplication to volume, face, and lifted-face tables.
802 const auto bilinear_tables = std::tuple(
803 std::tuple(
804 &m_poly_table_flat, //
805 &m_poly_table_flat_nodupes, //
806 &m_poly_table_flat_ijs_table),
807 std::tuple(
808 &m_poly_table_face_flat, //
809 &m_poly_table_face_flat_nodupes, //
810 &m_poly_table_face_flat_ijs_table),
811 std::tuple(
812 &m_poly_table_face_lifted_gathered,
813 &m_poly_table_face_lifted_gathered_nodupes,
814 &m_poly_table_face_lifted_gathered_ijs_table));
815
816 vutils::for_each(bilinear_tables, [n_dof_loc, this](auto tpts) {
817 auto tpt = std::get<0>(tpts);
818 auto tpt_nodup = std::get<1>(tpts);
819 auto tpt_ijk = std::get<2>(tpts);
820 for (std::size_t exi = 0; exi < tpt->size(); ++exi) {
821 using MonomialType = std::decay_t<decltype((*tpt)[0][0][0])>;
822 using PolynomialType = std::vector<MonomialType>;
823 const auto npolys = (*tpt)[exi].size();
824 const auto slice_size = n_dof_loc * n_dof_loc * s_n_blocks * s_n_blocks;
825 const auto n_k_comps = npolys / slice_size;
826
827 index_t poly_idx = 0;
828 std::vector<index_t> keep_idxs;
829 datastructure::CrsMapUncompressed<TypeIJ> table;
830 std::map<PolynomialType, std::size_t> unq_polys;
831 // dummy just to compute automatically the `kbibj` offset
832 Eigen::Tensor<double, 3> dummy(n_k_comps, s_n_blocks, s_n_blocks);
833 for (Eigen::Index bj = 0; bj < s_n_blocks; ++bj) {
834 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
835 for (Eigen::Index j = 0; j < n_dof_loc; ++j) {
836 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
837 for (Eigen::Index k = 0; k < n_k_comps; ++k) {
838 const auto kbibj = &dummy(k, bi, bj) - dummy.data();
839 const TypeIJ ij = {
840 i * s_n_blocks + bi, j * s_n_blocks + bj, kbibj};
841 PolynomialType poly;
842 for (auto mono : (*tpt)[exi][poly_idx]) {
843 poly.push_back(mono);
844 }
845 std::sort(poly.begin(), poly.end());
846 if (poly.size()) {
847 const auto [it, inserted] =
848 unq_polys.emplace(poly, unq_polys.size());
849 table.getRow(it->second).push_back(ij);
850 if (inserted) {
851 keep_idxs.push_back(poly_idx);
852 }
853 }
854 poly_idx++;
855 }
856 }
857 }
858 }
859 }
860
861 (*tpt_nodup)[exi] = (*tpt)[exi].slice(keep_idxs);
862 (*tpt_ijk)[exi] = table.template compress<index_t>();
863 }
864 });
865
866 // Linear (RHS) polynomial tables: slice is (bi,i,k) -- no trial index.
867 const auto linear_tables = std::tuple(
868 std::tuple(
869 &m_poly_table_flat_linear, &m_poly_table_flat_linear_nodupes,
870 &m_poly_table_flat_linear_ijs_table),
871 std::tuple(
872 &m_poly_table_face_flat_linear,
873 &m_poly_table_face_flat_linear_nodupes,
874 &m_poly_table_face_flat_linear_ijs_table));
875
876 vutils::for_each(linear_tables, [n_dof_loc, this](auto tpts) {
877 auto tpt = std::get<0>(tpts);
878 auto tpt_nodup = std::get<1>(tpts);
879 auto tpt_i = std::get<2>(tpts);
880 for (std::size_t exi = 0; exi < tpt->size(); ++exi) {
881 using MonomialType = std::decay_t<decltype((*tpt)[0][0][0])>;
882 using PolynomialType = std::vector<MonomialType>;
883 const auto npolys = (*tpt)[exi].size();
884 const auto slice_size = n_dof_loc * s_n_blocks;
885 const auto n_k_comps = npolys / slice_size;
886
887 index_t poly_idx = 0;
888 std::vector<index_t> keep_idxs;
889 datastructure::CrsMapUncompressed<TypeI> table;
890 std::map<PolynomialType, std::size_t> unq_polys;
891 // dummy to compute kbi offset in (k, bi) layout
892 Eigen::Tensor<double, 2> dummy(n_k_comps, s_n_blocks);
893 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
894 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
895 for (Eigen::Index k = 0; k < n_k_comps; ++k) {
896 const auto kbi = &dummy(k, bi) - dummy.data();
897 const TypeI i_entry = {i * s_n_blocks + bi, kbi};
898 PolynomialType poly;
899 for (auto mono : (*tpt)[exi][poly_idx]) {
900 poly.push_back(mono);
901 }
902 std::sort(poly.begin(), poly.end());
903 if (poly.size()) {
904 const auto [it, inserted] =
905 unq_polys.emplace(poly, unq_polys.size());
906 table.getRow(it->second).push_back(i_entry);
907 if (inserted) {
908 keep_idxs.push_back(poly_idx);
909 }
910 }
911 poly_idx++;
912 }
913 }
914 }
915
916 (*tpt_nodup)[exi] = (*tpt)[exi].slice(keep_idxs);
917 (*tpt_i)[exi] = table.template compress<index_t>();
918 }
919 });

934 FeExpressionFaceTuple>::s_n_face_expr_bilinear>
935WeakFormIntegratorQFree<MeshType, FeExpressionTuple, FeExpressionFaceTuple>::
936 gatherUncompressedLiftedMonomials() {
937 // gather the lifted monomials
938 std::array<std::vector<GatheredPolynomial>, s_n_face_expr_bilinear>
939 gathered_monos_table;
940 for (std::size_t i = 0; i < m_poly_table_face_lifted_flat.size(); ++i) {
941 const auto& old_table = m_poly_table_face_lifted_flat[i];
942 for (index_t poly_id = 0; poly_id < old_table.size(); ++poly_id) {
943 GatheredPolynomial gathered_poly;
944 for (const auto& mono : old_table[poly_id]) {
945 // in the monomial exponent, split the first dim components (the ones
946 // that must be integrated and with respect to which we gather), and
947 // the evaluation components (the last 2 * dim ones, which come from
948 // the cartesian mapping)
949 std::array<uint32_t, dim> int_ids;
950 for (std::size_t d = 0; d < dim; ++d) {
951 int_ids[d] = mono.exp()[d];
952 }
953 std::array<uint8_t, dim * 2> eva_ids;
954 for (std::size_t d = 0; d < dim * 2; ++d) {
955 eva_ids[d] = mono.exp()[d + dim];
956 }
957 const uint16_t int_id =
958 monomial::getLinearIndex<dim, uint32_t>(int_ids);
959
960 // search among gathered monomials if we already have
961 // a monomial with the common term
962 std::size_t mono_id = -1;
963 for (std::size_t j = 0; j < gathered_poly.size(); ++j) {
964 if (gathered_poly[j].integr_id == int_id) {
965 mono_id = j;
966 break;
967 }
968 }
969 if (mono_id == std::size_t(-1)) {
970 // if the common term is not present, we add a new term
971 gathered_poly.push_back(GatheredMonomial<dim>{
972 .integr_id = int_id,
973 .evalut_id = {0},
974 .gather = {{mono.coef(), eva_ids}}});
975 } else {
976 // the term is already presend, we append the non-common part
977 gathered_poly[mono_id].gather.push_back({mono.coef(), eva_ids});
978 }
979 }
980 // extract for each gathered monomial the common part in the
981 // gathered part. E.g. x(ab + a) -> xa(b + 1)
982 for (auto& mono : gathered_poly) {
983 std::array<uint8_t, dim * 2> min_id;
984 min_id.fill(255);
985 // find the common part
986 for (const auto& c : mono.gather) {
987 for (std::size_t d = 0; d < dim * 2; ++d) {
988 min_id[d] = std::min(min_id[d], c.second[d]);
989 }
990 }
991 // remove the common part from the gathered
992 mono.evalut_id = min_id;
993 for (auto& c : mono.gather) {
994 for (std::size_t d = 0; d < dim * 2; ++d) {
995 c.second[d] -= min_id[d];
996 }
997 }
998 }
999 gathered_monos_table[i].push_back(gathered_poly);
1000 }
1001 }
1002 return gathered_monos_table;
1003}

1041void WeakFormIntegratorQFree<
1042 MeshType,
1043 FeExpressionTuple,
1044 FeExpressionFaceTuple>::
1045 compressGatheredLiftedPolynomials(
1046 const std::array<
1047 std::vector<GatheredPolynomial>,
1048 s_n_face_expr_bilinear>& gathered_monos_table) {
1049 // compress the monomials, consists in two steps
1050 // 1. the GatheredMonomial becomes a CompressedGatheredMonomial
1051 // 2. find the unique gather coefficients needed for the compressed monos
1052 using GatheredPolynomialCoef = std::vector<std::pair<double, uint16_t>>;
1053 std::map<GatheredPolynomialCoef, uint32_t> gather_table;
1054 using CompressedGatheredPolynomial = std::vector<CompressedGatheredMonomial>;
1055 std::array<std::vector<CompressedGatheredPolynomial>, s_n_face_expr_bilinear>
1056 compr_gathered_monos_table;
1057 for (std::size_t expr_id = 0; expr_id < compr_gathered_monos_table.size();
1058 ++expr_id) {
1059 auto& compr_gathered_monos = compr_gathered_monos_table[expr_id];
1060 compr_gathered_monos.resize(gathered_monos_table[expr_id].size());
1061 for (std::size_t i = 0; i < gathered_monos_table[expr_id].size(); ++i) {
1062 const auto& poly = gathered_monos_table[expr_id][i];
1063 compr_gathered_monos[i].resize(poly.size());
1064 for (std::size_t j = 0; j < poly.size(); ++j) {
1065 const auto& mono = poly[j];
1066 auto& compr_mono = compr_gathered_monos[i][j];
1067 compr_mono.integr_id = mono.integr_id;
1068 // transform the exponent array into a index
1069 const auto it = std::lower_bound(
1070 m_eva_idx_table.begin(), m_eva_idx_table.end(), mono.evalut_id);
1071 assert(it != m_eva_idx_table.end());
1072 assert(*it == mono.evalut_id);
1073 compr_mono.evalut_id = std::distance(m_eva_idx_table.begin(), it);
1074 // transform the gather part into a index
1075 // this is done by suitably updating the gather_table
1076 std::vector<std::pair<double, uint16_t>> sorted_gather;
1077 for (const auto& [coef, eval_id] : mono.gather) {
1078 const auto eval_it = std::lower_bound(
1079 m_eva_idx_table.begin(), m_eva_idx_table.end(), eval_id);
1080 assert(eval_it != m_eva_idx_table.end());
1081 assert(*eval_it == eval_id);
1082 sorted_gather.push_back(
1083 {coef, std::distance(m_eva_idx_table.begin(), eval_it)});
1084 }
1085 std::sort(sorted_gather.begin(), sorted_gather.end());
1086 const auto [gather_it, inserted] =
1087 gather_table.emplace(sorted_gather, (uint32_t)gather_table.size());
1088 compr_mono.gather_id = gather_it->second;
1089 }
1090 }
1091
1092 m_poly_table_face_lifted_gathered[expr_id] =
1093 CompressedListGatheredPoly{compr_gathered_monos_table[expr_id]};
1094 }
1095 assert(m_eva_idx_table.size() < (1 << 16));
1096 assert(gather_table.size() < (1UL << 32UL));
1097 // compress the gather table
1098 std::vector<GatheredPolynomialCoef> gather_table_flat(gather_table.size());
1099 for (const auto& [gather, id] : gather_table) {
1100 gather_table_flat[id] = gather;
1101 }
1102 m_gathered_monos = decltype(m_gathered_monos)(gather_table_flat);
1103}

1225 updateFaceNormalAndGetNanson() {

1227 const auto jac = this->m_mapping->getJacobian();
1228 const auto& cell = this->getCell(this->m_cell_id);

1233 // N unit outer in ref. By expoiting linearity and F is diagonal we have
1234 // J/||Fn||, where n is the unit normal in physical space. See:
1235 // https://en.wikipedia.org/wiki/Finite_strain_theory#Transformation_of_a_surface_and_volume_element
1236 const double nanson = jac_det / Fns.norm();
1237 return nanson;
