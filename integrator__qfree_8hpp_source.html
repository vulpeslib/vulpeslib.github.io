545template <
546 typename MeshType,
547 typename FeExpressionTuple,
548 typename FeExpressionFaceTuple>
549WeakFormIntegratorQFree<
550 MeshType,
551 FeExpressionTuple,
552 FeExpressionFaceTuple>::FlatPolyType
553WeakFormIntegratorQFree<MeshType, FeExpressionTuple, FeExpressionFaceTuple>::
554 compress(const datastructure::CrsMapUncompressed<WeakFormIntegratorQFree<
555 MeshType,
556 FeExpressionTuple,
557 FeExpressionFaceTuple>::MonoType>& uncompressed_table) {
558 // compress the table
559 const auto compr_table_uncompr_monos =
560 uncompressed_table.template compress<index_t>();
561 // compress the monomials
562 FlatPolyType compr_table_compr_monos;
563 compr_table_compr_monos.indices() = compr_table_uncompr_monos.indices();
564 std::array<std::size_t, dim> exp;
565 for (const auto& mono : compr_table_uncompr_monos.data()) {
566 for (std::size_t d = 0; d < dim; ++d) {
567 exp[d] = mono.exp()[d];
568 }
569 compr_table_compr_monos.data().emplace_back(
570 mono.coef(), monomial::getLinearIndex<dim, std::size_t>(exp));
571 }
572 return compr_table_compr_monos;
573};

575template <
576 typename MeshType,
577 typename FeExpressionTuple,
578 typename FeExpressionFaceTuple>
579WeakFormIntegratorQFree<MeshType, FeExpressionTuple, FeExpressionFaceTuple>::
580 WeakFormIntegratorQFree(
581 const MeshType& mesh,
582 CellManager<MeshType, FeType>& cell_manager,
583 const FeExpressionTuple& fe_expression,
584 const FeExpressionFaceTuple& fe_expression_face)
585 : WeakFormIntegratorBaseType(
586 mesh, cell_manager, fe_expression, fe_expression_face) {
587 const Eigen::Index n_dof_loc = this->m_fe_space.getBasis().size();
588
589 // cache the polynomials of the expressions
590 // they are computed in reference configuration, thus they do not change
591 // to take into account the transformation we split the polynomials
592 // into components that need different transformations
593 // Finally, the coef of the transformation is computed on the fly during
594 // the actual integration.
595 vutils::for_each_idxed(
596 fe_expression, [this, n_dof_loc](const auto& expr, std::size_t idx) {
597 datastructure::CrsMapUncompressed<MonoType> poly_table_tmp;
598 std::size_t poly_idx = 0;
599 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
600 for (Eigen::Index bj = 0; bj < s_n_blocks; ++bj) {
601 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
602 for (Eigen::Index j = 0; j < n_dof_loc; ++j) {
603 const Eigen::Matrix<PolyType, 1, -1> poly_comps =
604 expr->evaluatePoly(this->m_fe_values, i, j, bi, bj)
605 .unaryExpr([](const auto& p) {
606 return p.template to<double>();
607 });
608 m_npoly_comps_volume[idx] = poly_comps.size();
609 for (Eigen::Index k = 0; k < poly_comps.size(); ++k) {
610 auto& row = poly_table_tmp.getRow(poly_idx);
611 for (const auto& monomial : poly_comps[k].data()) {
612 row.push_back(monomial);
613 }
614 poly_idx++;
615 }
616 }
617 }
618 }
619 }
620 m_poly_table_flat[idx] = compress(poly_table_tmp);
621 });
622
623 // Expressions on the boundary may need the mapping of the neighbour
624 // In these cases the polynomials are lifted in higher dimension,
625 // that is, we add dummy variables representing the coef of the map
626 vutils::for_each_idxed(
627 fe_expression_face, [this, n_dof_loc](const auto& expr, std::size_t idx) {
628 using PolyMatType = std::decay_t<decltype(expr->evaluatePoly(
629 this->m_fe_values, 0, 0, 0, 0))>;
630 constexpr auto expr_poly_dim = PolyMatType::Scalar::dim;
631
632 using MonoTypeCurr =
633 std::conditional_t<expr_poly_dim == dim, MonoType, MonoTypeLifted>;
634 datastructure::CrsMapUncompressed<MonoTypeCurr> poly_table_tmp;
635 std::size_t poly_idx = 0;
636 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
637 for (Eigen::Index bj = 0; bj < s_n_blocks; ++bj) {
638 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
639 for (Eigen::Index j = 0; j < n_dof_loc; ++j) {
640 using ExprPolyType = std::conditional_t<
641 expr_poly_dim == dim, PolyType, PolyTypeLifted>;
642 const Eigen::Matrix<ExprPolyType, 1, -1> poly_comps =
643 expr->evaluatePoly(this->m_fe_values, i, j, bi, bj)
644 .unaryExpr([](const auto& p) {
645 return p.template to<double>();
646 });
647 m_npoly_comps_face[idx] = poly_comps.size();
648 for (Eigen::Index k = 0; k < poly_comps.size(); ++k) {
649 auto& row = poly_table_tmp.getRow(poly_idx);
650 for (const auto& monomial : poly_comps[k].data()) {
651 row.push_back(monomial);
652 }
653 poly_idx++;
654 }
655 }
656 }
657 }
658 }
659
660 if constexpr (expr_poly_dim == dim) {
661 m_poly_table_face_flat[idx] = compress(poly_table_tmp);
662 } else {
663 // compressing lifted monomials is more complex, we first have
664 // to save them all, than apply gathering
665 m_poly_table_face_lifted_flat[idx] =
666 poly_table_tmp.template compress<index_t>();
667 }
668 });
669 // speed-up polynomial evaluation by gathering common monomials
670 gatherLiftedMonomials();
671 // often polynomials are repeated in a local matrix
672 // we find out where are the repetions and build a table
673 // with one of each kind of polynomial and a table with the
674 // list of entriens where it should be put
675 buildLookupTables();
676}

678template <
679 typename MeshType,
680 typename FeExpressionTuple,
681 typename FeExpressionFaceTuple>
682void WeakFormIntegratorQFree<
683 MeshType,
684 FeExpressionTuple,
685 FeExpressionFaceTuple>::buildLookupTables() {
686 const Eigen::Index n_dof_loc = this->m_fe_space.getBasis().size();
687
688 // since we have to apply the same procedure to the integrals
689 // in the volume, on the boundary and with the neighbour
690 // we wrap together the datastructures to avoid code duplication
691 const auto tables = std::tuple(
692 std::tuple(
693 &m_poly_table_flat, //
694 &m_poly_table_flat_nodupes, //
695 &m_poly_table_flat_ijs_table),
696 std::tuple(
697 &m_poly_table_face_flat, //
698 &m_poly_table_face_flat_nodupes, //
699 &m_poly_table_face_flat_ijs_table),
700 std::tuple(
701 &m_poly_table_face_lifted_gathered,
702 &m_poly_table_face_lifted_gathered_nodupes,
703 &m_poly_table_face_lifted_gathered_ijs_table));
704
705 vutils::for_each(tables, [n_dof_loc, this](auto tpts) {
706 auto tpt = std::get<0>(tpts);
707 auto tpt_nodup = std::get<1>(tpts);
708 auto tpt_ijk = std::get<2>(tpts);
709 for (std::size_t exi = 0; exi < tpt->size(); ++exi) {
710 using MonomialType = std::decay_t<decltype((*tpt)[0][0][0])>;
711 using PolynomialType = std::vector<MonomialType>;
712 const auto npolys = (*tpt)[exi].size();
713 const auto slice_size = n_dof_loc * n_dof_loc * s_n_blocks * s_n_blocks;
714 const auto n_k_comps = npolys / slice_size;
715
716 index_t poly_idx = 0;
717 std::vector<index_t> keep_idxs;
718 datastructure::CrsMapUncompressed<TypeIJ> table;
719 std::map<PolynomialType, std::size_t> unq_polys;
720 // dummy just to compute automatically the `kbibj` offset
721 Eigen::Tensor<double, 3> dummy(n_k_comps, s_n_blocks, s_n_blocks);
722 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
723 for (Eigen::Index bj = 0; bj < s_n_blocks; ++bj) {
724 for (Eigen::Index i = 0; i < n_dof_loc; ++i) {
725 for (Eigen::Index j = 0; j < n_dof_loc; ++j) {
726 for (Eigen::Index k = 0; k < n_k_comps; ++k) {
727 const auto kbibj = &dummy(k, bi, bj) - dummy.data();
728 const TypeIJ ij = {
729 i * s_n_blocks + bi, j * s_n_blocks + bj, kbibj};
730 PolynomialType poly;
731 for (auto mono : (*tpt)[exi][poly_idx]) {
732 poly.push_back(mono);
733 }
734 std::sort(poly.begin(), poly.end());
735 if (poly.size()) {
736 const auto [it, inserted] =
737 unq_polys.emplace(poly, unq_polys.size());
738 table.getRow(it->second).push_back(ij);
739 if (inserted) {
740 keep_idxs.push_back(poly_idx);
741 }
742 }
743 poly_idx++;
744 }
745 }
746 }
747 }
748 }
749
750 (*tpt_nodup)[exi] = (*tpt)[exi].slice(keep_idxs);
751 (*tpt_ijk)[exi] = table.template compress<index_t>();
752 }
753 });
754}

756template <
757 typename MeshType,
758 typename FeExpressionTuple,
759 typename FeExpressionFaceTuple>
760std::array<
761 std::vector<typename WeakFormIntegratorQFree<
762 MeshType,
763 FeExpressionTuple,
764 FeExpressionFaceTuple>::GatheredPolynomial>,
765 WeakFormIntegratorQFree<
766 MeshType,
767 FeExpressionTuple,
768 FeExpressionFaceTuple>::s_n_face_expr>
769WeakFormIntegratorQFree<MeshType, FeExpressionTuple, FeExpressionFaceTuple>::
770 gatherUncompressedLiftedMonomials() {
771 // gather the lifted monomials
772 std::array<std::vector<GatheredPolynomial>, s_n_face_expr>
773 gathered_monos_table;
774 for (std::size_t i = 0; i < m_poly_table_face_lifted_flat.size(); ++i) {
775 const auto& old_table = m_poly_table_face_lifted_flat[i];
776 for (index_t poly_id = 0; poly_id < old_table.size(); ++poly_id) {
777 GatheredPolynomial gathered_poly;
778 for (const auto& mono : old_table[poly_id]) {
779 // in the monomial exponent, split the first dim components (the ones
780 // that must be integrated and with respect to which we gather), and
781 // the evaluation components (the last 2 * dim ones, which come from
782 // the cartesian mapping)
783 std::array<uint32_t, dim> int_ids;
784 for (std::size_t d = 0; d < dim; ++d) {
785 int_ids[d] = mono.exp()[d];
786 }
787 std::array<uint8_t, dim * 2> eva_ids;
788 for (std::size_t d = 0; d < dim * 2; ++d) {
789 eva_ids[d] = mono.exp()[d + dim];
790 }
791 const uint16_t int_id =
792 monomial::getLinearIndex<dim, uint32_t>(int_ids);
793
794 // search among gathered monomials if we already have
795 // a monomial with the common term
796 std::size_t mono_id = -1;
797 for (std::size_t j = 0; j < gathered_poly.size(); ++j) {
798 if (gathered_poly[j].integr_id == int_id) {
799 mono_id = j;
800 break;
801 }
802 }
803 if (mono_id == std::size_t(-1)) {
804 // if the common term is not present, we add a new term
805 gathered_poly.push_back(GatheredMonomial<dim>{
806 .integr_id = int_id,
807 .evalut_id = {0},
808 .gather = {{mono.coef(), eva_ids}}});
809 } else {
810 // the term is already presend, we append the non-common part
811 gathered_poly[mono_id].gather.push_back({mono.coef(), eva_ids});
812 }
813 }
814 // extract for each gathered monomial the common part in the
815 // gathered part. E.g. x(ab + a) -> xa(b + 1)
816 for (auto& mono : gathered_poly) {
817 std::array<uint8_t, dim * 2> min_id;
818 min_id.fill(255);
819 // find the common part
820 for (const auto& c : mono.gather) {
821 for (std::size_t d = 0; d < dim * 2; ++d) {
822 min_id[d] = std::min(min_id[d], c.second[d]);
823 }
824 }
825 // remove the common part from the gathered
826 mono.evalut_id = min_id;
827 for (auto& c : mono.gather) {
828 for (std::size_t d = 0; d < dim * 2; ++d) {
829 c.second[d] -= min_id[d];
830 }
831 }
832 }
833 gathered_monos_table[i].push_back(gathered_poly);
834 }
835 }
836 return gathered_monos_table;
837}

839// Builds a unique evaluation coefficient table from all gathered monomials.
840template <
841 typename MeshType,
842 typename FeExpressionTuple,
843 typename FeExpressionFaceTuple>
844std::vector<std::array<
845 uint8_t,
846 WeakFormIntegratorQFree<
847 MeshType,
848 FeExpressionTuple,
849 FeExpressionFaceTuple>::dim *
850 2>>
851WeakFormIntegratorQFree<MeshType, FeExpressionTuple, FeExpressionFaceTuple>::
852 buildUniqueEvaTableLiftedMonomials(
853 const std::array<std::vector<GatheredPolynomial>, s_n_face_expr>&
854 gathered_monos_table) {
855 std::set<std::array<uint8_t, dim * 2>> eva_set;
856 for (const auto& gathered_monos : gathered_monos_table) {
857 for (const auto& poly : gathered_monos) {
858 for (const auto& mono : poly) {
859 eva_set.insert(mono.evalut_id);
860 for (const auto& entry : mono.gather) {
861 eva_set.insert(entry.second);
862 }
863 }
864 }
865 }
866 return std::vector<std::array<uint8_t, dim * 2>>(
867 eva_set.begin(), eva_set.end());
868}

870template <
871 typename MeshType,
872 typename FeExpressionTuple,
873 typename FeExpressionFaceTuple>
874void WeakFormIntegratorQFree<
875 MeshType,
876 FeExpressionTuple,
877 FeExpressionFaceTuple>::
878 compressGatheredLiftedPolynomials(
879 const std::array<std::vector<GatheredPolynomial>, s_n_face_expr>&
880 gathered_monos_table) {
881 // compress the monomials, consists in two steps
882 // 1. the GatheredMonomial becomes a CompressedGatheredMonomial
883 // 2. find the unique gather coefficients needed for the compressed monos
884 using GatheredPolynomialCoef = std::vector<std::pair<double, uint16_t>>;
885 std::map<GatheredPolynomialCoef, uint32_t> gather_table;
886 using CompressedGatheredPolynomial = std::vector<CompressedGatheredMonomial>;
887 std::array<std::vector<CompressedGatheredPolynomial>, s_n_face_expr>
888 compr_gathered_monos_table;
889 for (std::size_t expr_id = 0; expr_id < compr_gathered_monos_table.size();
890 ++expr_id) {
891 auto& compr_gathered_monos = compr_gathered_monos_table[expr_id];
892 compr_gathered_monos.resize(gathered_monos_table[expr_id].size());
893 for (std::size_t i = 0; i < gathered_monos_table[expr_id].size(); ++i) {
894 const auto& poly = gathered_monos_table[expr_id][i];
895 compr_gathered_monos[i].resize(poly.size());
896 for (std::size_t j = 0; j < poly.size(); ++j) {
897 const auto& mono = poly[j];
898 auto& compr_mono = compr_gathered_monos[i][j];
899 compr_mono.integr_id = mono.integr_id;
900 // transform the exponent array into a index
901 const auto it = std::lower_bound(
902 m_eva_idx_table.begin(), m_eva_idx_table.end(), mono.evalut_id);
903 assert(it != m_eva_idx_table.end());
904 assert(*it == mono.evalut_id);
905 compr_mono.evalut_id = std::distance(m_eva_idx_table.begin(), it);
906 // transform the gather part into a index
907 // this is done by suitably updating the gather_table
908 std::vector<std::pair<double, uint16_t>> sorted_gather;
909 for (const auto& [coef, eval_id] : mono.gather) {
910 const auto eval_it = std::lower_bound(
911 m_eva_idx_table.begin(), m_eva_idx_table.end(), eval_id);
912 assert(eval_it != m_eva_idx_table.end());
913 assert(*eval_it == eval_id);
914 sorted_gather.push_back(
915 {coef, std::distance(m_eva_idx_table.begin(), eval_it)});
916 }
917 std::sort(sorted_gather.begin(), sorted_gather.end());
918 const auto [gather_it, inserted] =
919 gather_table.emplace(sorted_gather, (uint32_t)gather_table.size());
920 compr_mono.gather_id = gather_it->second;
921 }
922 }
923
924 m_poly_table_face_lifted_gathered[expr_id] =
925 CompressedListGatheredPoly{compr_gathered_monos_table[expr_id]};
926 }
927 assert(m_eva_idx_table.size() < (1 << 16));
928 assert(gather_table.size() < (1UL << 32UL));
929 // compress the gather table

954 typename FeExpressionTuple,
955 typename FeExpressionFaceTuple>
956void WeakFormIntegratorQFree<
957 MeshType,
958 FeExpressionTuple,
959 FeExpressionFaceTuple>::integrateVolumeCell(const double* aints_data) {
960 Profiler::start("integrate-volume-qfree");
961 const double jac_det = this->m_mapping->jacobianDet({});
962 std::vector<double> buffer;
963 vutils::for_each_idxed(
964 this->m_fe_expression, [this, jac_det, aints_data, &buffer](
965 const auto& expr, Eigen::Index expr_id) {
966 {
967 const auto& table = m_poly_table_flat_nodupes[expr_id];
968 buffer.resize(table.size());
969 for (index_t i = 0; i < table.size(); ++i) {
970 double integral = 0.0;
971 for (const auto mono : table[i]) {
972 integral += mono.coef * aints_data[mono.integr_id];
973 }
974 buffer[i] = integral;
975 }
976 }
977 expr->precompute(this->m_fe_values, this->m_cell_manager.getCells());
978 const auto& table = m_poly_table_flat_ijs_table[expr_id];
979
980 Profiler::start("evaluatePolyCoef+cacheVolume");
981 const auto K = m_npoly_comps_volume[expr_id];
982 Eigen::Tensor<double, 3> coefs(K, s_n_blocks, s_n_blocks);
983 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
984 for (Eigen::Index bj = 0; bj < s_n_blocks; ++bj) {
985 const auto coef = expr->evaluatePolyCoef(
986 *(this->m_mapping), this->m_fe_values, bi, bj);
987 assert(coef.size() == K);
988 assert(coef.rows() == 1);
989 for (Eigen::Index k = 0; k < K; ++k) {
990 coefs(k, bi, bj) = coef(0, k);
991 }
992 }
993 }
994 Profiler::stop("evaluatePolyCoef+cacheVolume");
995
996 for (index_t pid = 0; pid < table.size(); ++pid) {
997 const auto integral = jac_det * buffer[pid];
998 for (const auto ijk : table[pid]) {
999 this->m_integrals(ijk[0], ijk[1], expr_id) +=
1000 coefs.data()[ijk[2]] * integral;
1001 }
1002 }
1003 });
1004 Profiler::stop("integrate-volume-qfree");

1007template <
1008 typename MeshType,
1009 typename FeExpressionTuple,
1010 typename FeExpressionFaceTuple>
1011double
1012WeakFormIntegratorQFree<MeshType, FeExpressionTuple, FeExpressionFaceTuple>::
1013 updateFaceNormalAndGetNanson() {
1014 const double jac_det = this->m_mapping->jacobianDet({});
1015 const auto jac = this->m_mapping->getJacobian();
1016 const auto& cell = this->getCell(this->m_cell_id);
1017 const Eigen::Vector<double, dim> ns = cell.normals().col(this->m_face_id);
1018 const Eigen::Vector<double, dim> Fns = jac.array() * ns.array();
1019 this->m_fe_values.setFaceNormal(ns);
1020 // ||JF^-TN||, where F jacobian of ref-to-physic, J is |det(F)|,
1021 // N unit outer in ref. By expoiting linearity and F is diagonal we have
1022 // J/||Fn||, where n is the unit normal in physical space. See:
1023 // https://en.wikipedia.org/wiki/Finite_strain_theory#Transformation_of_a_surface_and_volume_element
1024 const double nanson = jac_det / Fns.norm();
1025 return nanson;
1026}

1032void WeakFormIntegratorQFree<
1033 MeshType,
1034 FeExpressionTuple,
1035 FeExpressionFaceTuple>::precomputeMonomialCoefsForFace() {
1036 Profiler::start("integrate::precomputeMonomialCoefsForFace::pt1");
1037 const Eigen::Matrix<double, dim, 2> neigh_map = computeRefToNeighRefMap();
1038 Eigen::Matrix<double, dim * 2, -1> eval_table(dim * 2, 32);
1039 eval_table.col(0) = Eigen::VectorXd::Ones(dim * 2);
1040 eval_table.col(1) =
1041 Eigen::Map<const Eigen::VectorXd>(neigh_map.data(), dim * 2);
1042 for (Eigen::Index i = 2; i < eval_table.cols(); ++i) {
1043 eval_table.col(i) =
1044 eval_table.col(i - 1).array() * eval_table.col(1).array();

1046 m_eval_table_ext.resize(m_eva_idx_table.size());
1047 std::fill(m_eval_table_ext.begin(), m_eval_table_ext.end(), 1.0);
1048 for (std::size_t i = 0; i < m_eval_table_ext.size(); ++i) {
1049 for (std::size_t d = 0; d < 2 * dim; ++d) {
1050 m_eval_table_ext[i] *= eval_table(d, m_eva_idx_table[i][d]);
1051 }
1052 }
1053 Profiler::stop("integrate::precomputeMonomialCoefsForFace::pt1");
1054
1055 Profiler::start("integrate::precomputeMonomialCoefsForFace::pt2");
1056 m_gather_coef.resize(m_gathered_monos.size());
1057 std::fill(m_gather_coef.begin(), m_gather_coef.end(), 0.0);
1058 for (std::size_t i = 0; i < m_gather_coef.size(); ++i) {
1059 for (const auto& [coef, id] : m_gathered_monos[i]) {
1060 m_gather_coef[i] += coef * m_eval_table_ext[id];
1061 }
1062 }
1063 Profiler::stop("integrate::precomputeMonomialCoefsForFace::pt2");

1066template <
1067 typename MeshType,
1068 typename FeExpressionTuple,
1069 typename FeExpressionFaceTuple>
1070void WeakFormIntegratorQFree<
1071 MeshType,
1072 FeExpressionTuple,
1073 FeExpressionFaceTuple>::
1074 integrateFaceCell(const double* eints_data, double nanson) {
1075 std::vector<double> buffer;
1076 vutils::for_each_idxed(
1077 this->m_fe_expression_face, [this, nanson, eints_data, &buffer](
1078 const auto& expr, Eigen::Index expr_id) {
1079 using PolyMatType = std::decay_t<decltype(expr->evaluatePoly(
1080 this->m_fe_values, 0, 0, 0, 0))>;
1081 constexpr auto expr_poly_dim = PolyMatType::Scalar::dim;
1082
1083 auto curr_integral = this->getCurrFaceIntegral(expr, expr_id);
1084 if (!curr_integral.size()) {
1085 return;
1086 }
1087
1088 // pre-compute integrals of monomials
1089 Profiler::start("integrate::integrateFaceCell::init");
1090 if constexpr (expr_poly_dim == dim) {
1091 const auto& table = m_poly_table_face_flat_nodupes[expr_id];
1092 buffer.resize(table.size());
1093 for (index_t i = 0; i < table.size(); ++i) {
1094 double integral = 0.0;
1095 for (const auto mono : table[i]) {
1096 integral += mono.coef * eints_data[mono.integr_id];
1097 }
1098 buffer[i] = integral;
1099 }
1100 } else {
1101 const auto& table =
1102 m_poly_table_face_lifted_gathered_nodupes[expr_id];
1103 buffer.resize(table.size());
1104 for (index_t i = 0; i < table.size(); ++i) {
1105 double integral = 0.0;
1106 for (const auto mono : table[i]) {
1107 integral += m_gather_coef[mono.gather_id] *
1108 m_eval_table_ext[mono.evalut_id] *
1109 eints_data[mono.integr_id];
1110 }
1111 buffer[i] = integral;
1112 }
1113 }
1114 Profiler::stop("integrate::integrateFaceCell::init");
1115
1116 // prepare buffer of coefficients with suitable duplicates
1117 // (the look-up is more expensive than putting everything in cache)
1118 Profiler::start("integrate::integrateFaceCell::cache");
1119 const auto K = m_npoly_comps_face[expr_id];
1120 Eigen::Tensor<double, 3> coefs(K, s_n_blocks, s_n_blocks);
1121 for (Eigen::Index bi = 0; bi < s_n_blocks; ++bi) {
1122 for (Eigen::Index bj = 0; bj < s_n_blocks; ++bj) {
1123 const auto coef = expr->evaluatePolyCoef(
1124 *(this->m_mapping), this->m_fe_values, bi, bj);
1125 assert(coef.size() == K);
1126 assert(coef.rows() == 1);
1127 for (Eigen::Index k = 0; k < K; ++k) {
1128 coefs(k, bi, bj) = coef(0, k);
1129 }
1130 }
1131 }
1132 Profiler::stop("integrate::integrateFaceCell::cache");
1133
1134 // actually compute the integrals by combining integrals of the
1135 // monomials and their coefficient
1136 Profiler::start("integrate::integrateFaceCell::eval");
1137 using TableType = std::decay_t<std::conditional_t<
1138 expr_poly_dim == dim, decltype(m_poly_table_face_flat_ijs_table[0]),
1139 decltype(m_poly_table_face_lifted_gathered_ijs_table[0])>>;
1140 const TableType* table = nullptr;
1141 if constexpr (expr_poly_dim == dim) {
1142 table = &m_poly_table_face_flat_ijs_table[expr_id];
1143 } else {
1144 table = &m_poly_table_face_lifted_gathered_ijs_table[expr_id];
1145 }
1146 for (index_t pid = 0; pid < (*table).size(); ++pid) {
1147 const auto integral = nanson * buffer[pid];
1148 for (const auto ijk : (*table)[pid]) {
1149 curr_integral(ijk[0], ijk[1]) += coefs.data()[ijk[2]] * integral;
1150 }
1151 }
1152 Profiler::stop("integrate::integrateFaceCell::eval");
1153 });
1154}
