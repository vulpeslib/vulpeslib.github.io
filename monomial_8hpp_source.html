69template <typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
70inline std::array<T, 2> getMonomial2D(std::size_t n) {
71 // float32 max safe int should be 16777215, ok for all reasonable polynomials
72 // degrees
73 using real_t = float;
74 // for ease of changing `real_t`
75 constexpr real_t one = 1;
76 constexpr real_t two = 2;
77 constexpr real_t eight = 8;
78 // solve the quadratic
79 const real_t tria_lvl_f =
80 (std::sqrt(one + eight * static_cast<real_t>(n)) - one) / two;
81 // compiler by default should trunc (we need trunc and nothing else!)
82 // possible packing in avx2: https://stackoverflow.com/q/41144668
83 const uint32_t tria_lvl = static_cast<uint32_t>(tria_lvl_f);
84 const uint32_t before = tria_lvl * (tria_lvl + uint32_t(1)) / uint32_t(2);
85 assert(static_cast<std::size_t>(before) <= n);
86 assert(n < static_cast<std::size_t>((tria_lvl + 1) * (tria_lvl + 2) / 2));
87 const uint32_t remain = n - before;
88 return {{static_cast<T>(tria_lvl - remain), static_cast<T>(remain)}};
89}

118template <typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
119inline std::array<T, 3> getMonomial3D(std::size_t n) {
120 if (n == 0) {
121 return {0, 0, 0};
122 }
123 using real_t = float;
124 // for ease of changing `real_t`
125 constexpr real_t one = 1;
126 constexpr real_t two = 2;
127 constexpr real_t three = 3;
128 constexpr real_t six = 6;
129 constexpr real_t cbrt6 = 1.817120592832139658891211756327;
130 const real_t n_f = static_cast<real_t>(n);
131 // solve the cubic: initial guess
132 const real_t x = cbrt6 * std::cbrt(n_f) - one;
133 // solve the cubic: newton iter
134 const real_t pyramid_lvl_f =
135 (six * n_f + (two * x + three) * x * x) / (three * x * (x + two) + two);
136 // supposing compiler trunc the float
137 const uint32_t pyramid_lvl = static_cast<uint32_t>(pyramid_lvl_f);
138 const uint32_t before =
139 pyramid_lvl * (pyramid_lvl + 1) * (pyramid_lvl + 2) / 6;
140 assert(static_cast<std::size_t>(before) <= n);
141 assert(
142 n < static_cast<std::size_t>(
143 (pyramid_lvl + 1) * (pyramid_lvl + 2) * (pyramid_lvl + 3) / 6));
144 const uint32_t lvl_id = n - before;
145 const auto ij = getMonomial2D<uint32_t>(lvl_id);
146 return {
147 {static_cast<T>(pyramid_lvl - ij[0] - ij[1]), static_cast<T>(ij[0]),
148 static_cast<T>(ij[1])}};
149}

154template <std::size_t Dim, typename T>
155inline std::array<T, Dim> getMonomial(std::size_t n) {
156 if constexpr (Dim == 1) {
157 return {n};
158 } else if constexpr (Dim == 2) {
159 return getMonomial2D<T>(n);
160 } else if constexpr (Dim == 3) {
161 return getMonomial3D<T>(n);
162 } else {
163 static_assert(Dim * 0 && "Not implemented for dimension > 3");
164 }
165}

185template <std::size_t Dim, typename IndexType, typename CoefType, bool IsScaled>
186class Monomial : public std::conditional_t<
187 IsScaled,
188 internal::ScaledMonomial<Dim, IndexType, CoefType>,
189 internal::NonScaledMonomial> {
190 public:
194 constexpr Monomial() : m_coefficient(0) {
195 m_exponent.fill(0);
196 if constexpr (IsScaled) {
197 this->m_scaling = 0;
198 this->m_translation.fill(0);
199 }
200 }
201
205 explicit constexpr Monomial(const CoefType& coeff) : m_coefficient(coeff) {
206 m_exponent.fill(0);
207 if constexpr (IsScaled) {
208 this->m_scaling = 1;
209 this->m_translation.fill(0);
210 }
211 }
212
216 constexpr Monomial(
217 const std::array<IndexType, Dim>& exponent, const CoefType& coefficient)
218 : m_exponent(exponent), m_coefficient(coefficient) {
219 if constexpr (IsScaled) {
220 this->m_scaling = 1; // Default scaling to 1
221 this->m_translation.fill(0); // Default translation to 0
222 }
223 }
224
228 constexpr Monomial(
229 const std::array<IndexType, Dim>& exponent,
230 const CoefType& coefficient,
231 const CoefType& scaling,
232 const std::array<IndexType, Dim>& translation)
233 : m_exponent(exponent), m_coefficient(coefficient) {
234 if constexpr (IsScaled) {
235 this->m_scaling = scaling;
236 this->m_translation = translation;
237 }
238 }
239
243 constexpr CoefType evaluate(const Eigen::Vector<CoefType, Dim>& point) const {
244 return evaluate(point.data());
245 }
246
250 constexpr CoefType evaluate(const CoefType* point) const {
251 CoefType value = m_coefficient;
252 for (std::size_t i = 0; i < Dim; ++i) {
253 CoefType x = point[i];
254 if constexpr (IsScaled) {
255 x = (x - this->m_translation[i]) * this->m_scaling;
256 }
257 value *= vmath::pow(x, m_exponent[i]);
258 }
259 return value;
260 }
261
268 constexpr Monomial derivative(std::size_t component) const {
269 assert(component < Dim);
270
271 if (m_exponent[component] == 0) {
272 return Monomial{};
273 }
274
275 std::array<IndexType, Dim> new_exponent = m_exponent;
276 new_exponent[component] -= 1;
277
278 CoefType new_coefficient = m_coefficient * m_exponent[component];
279
280 if constexpr (IsScaled) {
281 return Monomial(
282 new_exponent, new_coefficient, this->m_scaling, this->m_translation);
283 } else {
284 return Monomial(new_exponent, new_coefficient);
285 }
286 }
287
291 constexpr bool empty() const { return m_coefficient == CoefType(0); }
292
296 constexpr Eigen::Vector<Monomial, Dim> gradient() const {
297 Eigen::Vector<Monomial, Dim> grad;
298 for (std::size_t d = 0; d < Dim; ++d) {
299 grad[d] = this->derivative(d);
300 }
301 return grad;
302 }

307 template <typename U>
308 Monomial<Dim, IndexType, U, IsScaled> to() const {
309 if constexpr (IsScaled) {
310 std::array<U, Dim> trasl;
311 for (std::size_t d = 0; d < Dim; ++d) {
312 trasl[d] = vmath::to<U>(this->m_traslation[d]);
313 }
314 return Monomial<Dim, IndexType, U, IsScaled>(
315 exp, vmath::to<U>(m_coefficient), vmath::to<U>(this->m_scaling),
316 trasl);
317 } else {
318 return Monomial<Dim, IndexType, U, IsScaled>(
319 m_exponent, vmath::to<U>(m_coefficient));
320 }
321 }

357 friend constexpr bool operator==(const Monomial& lhs, const Monomial& rhs) {
358 if constexpr (IsScaled) {
359 return lhs.m_coefficient == rhs.m_coefficient &&
360 lhs.m_exponent == rhs.m_exponent &&
361 lhs.m_scaling == rhs.m_scaling &&
362 lhs.m_translation == rhs.m_translation;
363 } else {
364 return lhs.m_coefficient == rhs.m_coefficient &&
365 lhs.m_exponent == rhs.m_exponent;
366 }
367 }
368
375 friend constexpr bool operator<(const Monomial& lhs, const Monomial& rhs) {
376 // Compare exponents lexicographically
377 if (lhs.m_exponent != rhs.m_exponent) {
378 return lhs.m_exponent < rhs.m_exponent;
379 }
380
381 // Compare coefficients
382 if (lhs.m_coefficient != rhs.m_coefficient) {
383 return lhs.m_coefficient < rhs.m_coefficient;
384 }
385
386 if constexpr (IsScaled) {
387 // Compare scaling factors
388 if (lhs.m_scaling != rhs.m_scaling) {
389 return lhs.m_scaling < rhs.m_scaling;
390 }
391
392 // Compare translation factors lexicographically
393 return lhs.m_translation < rhs.m_translation;
394 }
395
396 return false;
397 }

404};
