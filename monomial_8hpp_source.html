46template <typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
47inline constexpr std::array<T, 2> getMonomial2D_Impl(std::size_t n) {
48 // float32 max safe int should be 16777215, ok for all reasonable polynomials
49 // degrees
50 using real_t = float;
51 // for ease of changing `real_t`
52 constexpr real_t one = 1;
53 constexpr real_t two = 2;
54 constexpr real_t eight = 8;
55 // solve the quadratic
56 const real_t tria_lvl_f =
57 (std::sqrt(one + eight * static_cast<real_t>(n)) - one) / two;
58 // compiler by default should trunc (we need trunc and nothing else!)
59 // possible packing in avx2: https://stackoverflow.com/q/41144668
60 const std::size_t tria_lvl = static_cast<std::size_t>(tria_lvl_f);
61 const std::size_t before =
62 tria_lvl * (tria_lvl + std::size_t(1)) / std::size_t(2);
63 assert(static_cast<std::size_t>(before) <= n);
64 assert(n < static_cast<std::size_t>((tria_lvl + 1) * (tria_lvl + 2) / 2));
65 const std::size_t remain = n - before;
66 return {{static_cast<T>(tria_lvl - remain), static_cast<T>(remain)}};
67}

69template <typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
70inline constexpr std::array<T, 3> getMonomial3D_Impl(std::size_t n) {
71 if (n == 0) {
72 return {0, 0, 0};
73 }
74 using real_t = float;
75 // for ease of changing `real_t`
76 constexpr real_t one = 1;
77 constexpr real_t two = 2;
78 constexpr real_t three = 3;
79 constexpr real_t six = 6;
80 constexpr real_t cbrt6 = 1.817120592832139658891211756327;
81 const real_t n_f = static_cast<real_t>(n);
82 // the cubic: pyramid_lvl(pyramid_lvl+1)(pyramid_lvl+2)/6 = n
83 // solve the cubic: initial guess
84 const real_t x = cbrt6 * std::cbrt(n_f) - one;
85 // solve the cubic: newton iter
86 const real_t pyramid_lvl_f =
87 (six * n_f + (two * x + three) * x * x) / (three * x * (x + two) + two);
88 // supposing compiler trunc the float
89 const std::size_t pyramid_lvl = static_cast<std::size_t>(pyramid_lvl_f);
90 const std::size_t before =
91 pyramid_lvl * (pyramid_lvl + 1) * (pyramid_lvl + 2) / 6;
92 assert(static_cast<std::size_t>(before) <= n);
93 assert(
94 n < static_cast<std::size_t>(
95 (pyramid_lvl + 1) * (pyramid_lvl + 2) * (pyramid_lvl + 3) / 6));
96 const std::size_t lvl_id = n - before;
97 const auto ij = getMonomial2D_Impl<std::size_t>(lvl_id);
98 return {
99 {static_cast<T>(pyramid_lvl - ij[0] - ij[1]), static_cast<T>(ij[0]),
100 static_cast<T>(ij[1])}};
101}

130template <typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
131inline std::array<T, 2> getMonomial2D(std::size_t n) {
132 static constexpr std::size_t max_deg = 10;
133 static constexpr std::size_t max_cache = (max_deg + 1) * (max_deg + 2) / 2;
134 static constexpr std::array<std::array<T, 2>, max_cache> cache = []() {
135 std::array<std::array<T, 2>, max_cache> cache;
136 for (std::size_t i = 0; i < max_cache; ++i) {
137 cache[i] = internal::getMonomial2D_Impl<T>(i);
138 }
139 return cache;
140 }();
141 if (VULPES_LIKELY(n < max_cache)) {
142 return cache[n];
143 } else {
144 return internal::getMonomial2D_Impl<T>(n);
145 }
146}

197template <std::size_t Dim, typename T>
198inline std::array<T, Dim> getMonomial(std::size_t n) {
199 if constexpr (Dim == 1) {
200 return {n};
201 } else if constexpr (Dim == 2) {
202 return getMonomial2D<T>(n);
203 } else if constexpr (Dim == 3) {
204 return getMonomial3D<T>(n);
205 } else {
206 static_assert(Dim * 0 && "Not implemented for dimension > 3");
207 }
208}

216template <std::size_t Dim, typename T>
217inline T getLinearIndex(const std::array<T, Dim>& monomial) {
218 if constexpr (Dim == 1) {
219 return monomial[0];
220 } else if constexpr (Dim == 2) {
221 const T ij = monomial[0] + monomial[1];
222 return ij * (ij + 1) / 2 + monomial[1];
223 } else if constexpr (Dim == 3) {
224 const T j_k = monomial[1] + monomial[2];
225 const T i_j_k = monomial[0] + j_k;
226 return i_j_k * (i_j_k + 1) * (i_j_k + 2) / 6 + j_k * (j_k + 1) / 2 +
227 monomial[2];
228 } else {
229 static_assert(Dim * 0 && "Not implemented for dimension > 3");
230 }
231}

251template <std::size_t Dim, typename IndexType, typename CoefType, bool IsScaled>
252class Monomial : public std::conditional_t<
253 IsScaled,
254 internal::ScaledMonomial<Dim, IndexType, CoefType>,
255 internal::NonScaledMonomial> {
256 public:
260 constexpr Monomial() : m_coefficient(0) {
261 m_exponent.fill(0);
262 if constexpr (IsScaled) {
263 this->m_scaling = 0;
264 this->m_translation.fill(0);
265 }
266 }
267
271 explicit constexpr Monomial(const CoefType& coeff) : m_coefficient(coeff) {
272 m_exponent.fill(0);
273 if constexpr (IsScaled) {
274 this->m_scaling = 1;
275 this->m_translation.fill(0);
276 }
277 }
278
282 constexpr Monomial(
283 const std::array<IndexType, Dim>& exponent, const CoefType& coefficient)
284 : m_exponent(exponent), m_coefficient(coefficient) {
285 if constexpr (IsScaled) {
286 this->m_scaling = 1; // Default scaling to 1
287 this->m_translation.fill(0); // Default translation to 0
288 }
289 }
290
294 constexpr Monomial(
295 const std::array<IndexType, Dim>& exponent,
296 const CoefType& coefficient,
297 const CoefType& scaling,
298 const std::array<IndexType, Dim>& translation)
299 : m_exponent(exponent), m_coefficient(coefficient) {
300 if constexpr (IsScaled) {
301 this->m_scaling = scaling;
302 this->m_translation = translation;
303 }
304 }
305
309 constexpr CoefType evaluate(const Eigen::Vector<CoefType, Dim>& point) const {
310 return evaluate(point.data());
311 }
312
316 constexpr CoefType evaluate(const CoefType* point) const {
317 CoefType value = m_coefficient;
318 for (std::size_t i = 0; i < Dim; ++i) {
319 CoefType x = point[i];
320 if constexpr (IsScaled) {
321 x = (x - this->m_translation[i]) * this->m_scaling;
322 }
323 value *= vmath::pow(x, m_exponent[i]);
324 }
325 return value;
326 }
327
334 constexpr Monomial derivative(std::size_t component) const {
335 assert(component < Dim);
336
337 if (m_exponent[component] == 0) {
338 return Monomial{};
339 }
340
341 std::array<IndexType, Dim> new_exponent = m_exponent;
342 new_exponent[component] -= 1;
343
344 CoefType new_coefficient = m_coefficient * m_exponent[component];
345
346 if constexpr (IsScaled) {
347 return Monomial(
348 new_exponent, new_coefficient, this->m_scaling, this->m_translation);
349 } else {
350 return Monomial(new_exponent, new_coefficient);
351 }
352 }
353
357 constexpr bool empty() const { return m_coefficient == CoefType(0); }
358
362 constexpr Eigen::Vector<Monomial, Dim> gradient() const {
363 Eigen::Vector<Monomial, Dim> grad;
364 for (std::size_t d = 0; d < Dim; ++d) {
365 grad[d] = this->derivative(d);
366 }
367 return grad;
368 }

373 template <typename U>
374 Monomial<Dim, IndexType, U, IsScaled> to() const {
375 if constexpr (IsScaled) {
376 std::array<U, Dim> trasl;
377 for (std::size_t d = 0; d < Dim; ++d) {
378 trasl[d] = vmath::to<U>(this->m_traslation[d]);
379 }
380 return Monomial<Dim, IndexType, U, IsScaled>(
381 exp, vmath::to<U>(m_coefficient), vmath::to<U>(this->m_scaling),
382 trasl);
383 } else {
384 return Monomial<Dim, IndexType, U, IsScaled>(
385 m_exponent, vmath::to<U>(m_coefficient));
386 }
387 }

389 template <std::size_t PreDim, std::size_t PostDim>
390 Monomial<Dim + PreDim + PostDim, IndexType, CoefType, IsScaled>
391 liftDimension() const {
392 constexpr auto OutDim = Dim + PreDim + PostDim;
393 std::array<IndexType, OutDim> exponent = {0};
394 std::copy(exp().cbegin(), exp().cend(), exponent.begin() + PreDim);
395 return {exponent, coef()};
396 }

432 friend constexpr bool operator==(const Monomial& lhs, const Monomial& rhs) {
433 if constexpr (IsScaled) {
434 return lhs.m_coefficient == rhs.m_coefficient &&
435 lhs.m_exponent == rhs.m_exponent &&
436 lhs.m_scaling == rhs.m_scaling &&
437 lhs.m_translation == rhs.m_translation;
438 } else {
439 return lhs.m_coefficient == rhs.m_coefficient &&
440 lhs.m_exponent == rhs.m_exponent;
441 }
442 }
443
450 friend constexpr bool operator<(const Monomial& lhs, const Monomial& rhs) {
451 // Compare exponents lexicographically
452 if (lhs.m_exponent != rhs.m_exponent) {
453 return lhs.m_exponent < rhs.m_exponent;
454 }
455
456 // Compare coefficients
457 if (lhs.m_coefficient != rhs.m_coefficient) {
458 return lhs.m_coefficient < rhs.m_coefficient;
459 }
460
461 if constexpr (IsScaled) {
462 // Compare scaling factors
463 if (lhs.m_scaling != rhs.m_scaling) {
464 return lhs.m_scaling < rhs.m_scaling;
465 }
466
467 // Compare translation factors lexicographically
468 return lhs.m_translation < rhs.m_translation;
469 }
470
471 return false;
472 }

479 CoefType m_coefficient;

481 friend class polynomial::SparsePolynomial<Dim, IndexType, CoefType, IsScaled>;
482};
