70template <typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
71inline std::array<T, 2> getMonomial2D(std::size_t n) {
72 // float32 max safe int should be 16777215, ok for all reasonable polynomials
73 // degrees
74 using real_t = float;
75 // for ease of changing `real_t`
76 constexpr real_t one = 1;
77 constexpr real_t two = 2;
78 constexpr real_t eight = 8;
79 // solve the quadratic
80 const real_t tria_lvl_f =
81 (std::sqrt(one + eight * static_cast<real_t>(n)) - one) / two;
82 // compiler by default should trunc (we need trunc and nothing else!)
83 // possible packing in avx2: https://stackoverflow.com/q/41144668
84 const std::size_t tria_lvl = static_cast<std::size_t>(tria_lvl_f);
85 const std::size_t before =
86 tria_lvl * (tria_lvl + std::size_t(1)) / std::size_t(2);
87 assert(static_cast<std::size_t>(before) <= n);
88 assert(n < static_cast<std::size_t>((tria_lvl + 1) * (tria_lvl + 2) / 2));
89 const std::size_t remain = n - before;
90 return {{static_cast<T>(tria_lvl - remain), static_cast<T>(remain)}};
91}

120template <typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
121inline std::array<T, 3> getMonomial3D(std::size_t n) {
122 if (n == 0) {
123 return {0, 0, 0};
124 }
125 using real_t = float;
126 // for ease of changing `real_t`
127 constexpr real_t one = 1;
128 constexpr real_t two = 2;
129 constexpr real_t three = 3;
130 constexpr real_t six = 6;
131 constexpr real_t cbrt6 = 1.817120592832139658891211756327;
132 const real_t n_f = static_cast<real_t>(n);
133 // the cubic: pyramid_lvl(pyramid_lvl+1)(pyramid_lvl+2)/6 = n
134 // solve the cubic: initial guess
135 const real_t x = cbrt6 * std::cbrt(n_f) - one;
136 // solve the cubic: newton iter
137 const real_t pyramid_lvl_f =
138 (six * n_f + (two * x + three) * x * x) / (three * x * (x + two) + two);
139 // supposing compiler trunc the float
140 const std::size_t pyramid_lvl = static_cast<std::size_t>(pyramid_lvl_f);
141 const std::size_t before =
142 pyramid_lvl * (pyramid_lvl + 1) * (pyramid_lvl + 2) / 6;
143 assert(static_cast<std::size_t>(before) <= n);
144 assert(
145 n < static_cast<std::size_t>(
146 (pyramid_lvl + 1) * (pyramid_lvl + 2) * (pyramid_lvl + 3) / 6));
147 const std::size_t lvl_id = n - before;
148 const auto ij = getMonomial2D<std::size_t>(lvl_id);
149 return {
150 {static_cast<T>(pyramid_lvl - ij[0] - ij[1]), static_cast<T>(ij[0]),
151 static_cast<T>(ij[1])}};
152}

157template <std::size_t Dim, typename T>
158inline std::array<T, Dim> getMonomial(std::size_t n) {
159 if constexpr (Dim == 1) {
160 return {n};
161 } else if constexpr (Dim == 2) {
162 return getMonomial2D<T>(n);
163 } else if constexpr (Dim == 3) {
164 return getMonomial3D<T>(n);
165 } else {
166 static_assert(Dim * 0 && "Not implemented for dimension > 3");
167 }
168}

176template <std::size_t Dim, typename T>
177inline T getLinearIndex(const std::array<T, Dim>& monomial) {
178 if constexpr (Dim == 1) {
179 return monomial[0];
180 } else if constexpr (Dim == 2) {
181 const T ij = monomial[0] + monomial[1];
182 return ij * (ij + 1) / 2 + monomial[1];
183 } else if constexpr (Dim == 3) {
184 const T j_k = monomial[1] + monomial[2];
185 const T i_j_k = monomial[0] + j_k;
186 return i_j_k * (i_j_k + 1) * (i_j_k + 2) / 6 + j_k * (j_k + 1) / 2 +
187 monomial[2];
188 } else {
189 static_assert(Dim * 0 && "Not implemented for dimension > 3");
190 }
191}

211template <std::size_t Dim, typename IndexType, typename CoefType, bool IsScaled>
212class Monomial : public std::conditional_t<
213 IsScaled,
214 internal::ScaledMonomial<Dim, IndexType, CoefType>,
215 internal::NonScaledMonomial> {
216 public:
220 constexpr Monomial() : m_coefficient(0) {
221 m_exponent.fill(0);
222 if constexpr (IsScaled) {
223 this->m_scaling = 0;
224 this->m_translation.fill(0);
225 }
226 }
227
231 explicit constexpr Monomial(const CoefType& coeff) : m_coefficient(coeff) {
232 m_exponent.fill(0);
233 if constexpr (IsScaled) {
234 this->m_scaling = 1;
235 this->m_translation.fill(0);
236 }
237 }
238
242 constexpr Monomial(
243 const std::array<IndexType, Dim>& exponent, const CoefType& coefficient)
244 : m_exponent(exponent), m_coefficient(coefficient) {
245 if constexpr (IsScaled) {
246 this->m_scaling = 1; // Default scaling to 1
247 this->m_translation.fill(0); // Default translation to 0
248 }
249 }
250
254 constexpr Monomial(
255 const std::array<IndexType, Dim>& exponent,
256 const CoefType& coefficient,
257 const CoefType& scaling,
258 const std::array<IndexType, Dim>& translation)
259 : m_exponent(exponent), m_coefficient(coefficient) {
260 if constexpr (IsScaled) {
261 this->m_scaling = scaling;
262 this->m_translation = translation;
263 }
264 }
265
269 constexpr CoefType evaluate(const Eigen::Vector<CoefType, Dim>& point) const {
270 return evaluate(point.data());
271 }
272
276 constexpr CoefType evaluate(const CoefType* point) const {
277 CoefType value = m_coefficient;
278 for (std::size_t i = 0; i < Dim; ++i) {
279 CoefType x = point[i];
280 if constexpr (IsScaled) {
281 x = (x - this->m_translation[i]) * this->m_scaling;
282 }
283 value *= vmath::pow(x, m_exponent[i]);
284 }
285 return value;
286 }
287
294 constexpr Monomial derivative(std::size_t component) const {
295 assert(component < Dim);
296
297 if (m_exponent[component] == 0) {
298 return Monomial{};
299 }
300
301 std::array<IndexType, Dim> new_exponent = m_exponent;
302 new_exponent[component] -= 1;
303
304 CoefType new_coefficient = m_coefficient * m_exponent[component];
305
306 if constexpr (IsScaled) {
307 return Monomial(
308 new_exponent, new_coefficient, this->m_scaling, this->m_translation);
309 } else {
310 return Monomial(new_exponent, new_coefficient);
311 }
312 }
313
317 constexpr bool empty() const { return m_coefficient == CoefType(0); }
318
322 constexpr Eigen::Vector<Monomial, Dim> gradient() const {
323 Eigen::Vector<Monomial, Dim> grad;
324 for (std::size_t d = 0; d < Dim; ++d) {
325 grad[d] = this->derivative(d);
326 }
327 return grad;
328 }

333 template <typename U>
334 Monomial<Dim, IndexType, U, IsScaled> to() const {
335 if constexpr (IsScaled) {
336 std::array<U, Dim> trasl;
337 for (std::size_t d = 0; d < Dim; ++d) {
338 trasl[d] = vmath::to<U>(this->m_traslation[d]);
339 }
340 return Monomial<Dim, IndexType, U, IsScaled>(
341 exp, vmath::to<U>(m_coefficient), vmath::to<U>(this->m_scaling),
342 trasl);
343 } else {
344 return Monomial<Dim, IndexType, U, IsScaled>(
345 m_exponent, vmath::to<U>(m_coefficient));
346 }
347 }

349 template <std::size_t PreDim, std::size_t PostDim>
350 Monomial<Dim + PreDim + PostDim, IndexType, CoefType, IsScaled>
351 liftDimension() const {
352 constexpr auto OutDim = Dim + PreDim + PostDim;
353 std::array<IndexType, OutDim> exponent = {0};
354 std::copy(exp().cbegin(), exp().cend(), exponent.begin() + PreDim);
355 return {exponent, coef()};
356 }

392 friend constexpr bool operator==(const Monomial& lhs, const Monomial& rhs) {
393 if constexpr (IsScaled) {
394 return lhs.m_coefficient == rhs.m_coefficient &&
395 lhs.m_exponent == rhs.m_exponent &&
396 lhs.m_scaling == rhs.m_scaling &&
397 lhs.m_translation == rhs.m_translation;
398 } else {
399 return lhs.m_coefficient == rhs.m_coefficient &&
400 lhs.m_exponent == rhs.m_exponent;
401 }
402 }
403
410 friend constexpr bool operator<(const Monomial& lhs, const Monomial& rhs) {
411 // Compare exponents lexicographically
412 if (lhs.m_exponent != rhs.m_exponent) {
413 return lhs.m_exponent < rhs.m_exponent;
414 }
415
416 // Compare coefficients
417 if (lhs.m_coefficient != rhs.m_coefficient) {
418 return lhs.m_coefficient < rhs.m_coefficient;
419 }
420
421 if constexpr (IsScaled) {
422 // Compare scaling factors
423 if (lhs.m_scaling != rhs.m_scaling) {
424 return lhs.m_scaling < rhs.m_scaling;
425 }
426
427 // Compare translation factors lexicographically
428 return lhs.m_translation < rhs.m_translation;
429 }
430
431 return false;
432 }

439 CoefType m_coefficient;
440
441 friend class polynomial::SparsePolynomial<Dim, IndexType, CoefType, IsScaled>;
442};
