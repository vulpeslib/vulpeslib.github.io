397class Simplexify {
398 public:
406 using SimplexifyEngine =
407 std::conditional_t<Dim == 2, Triangulation, Tetrahedralization>;
408
409 using float_t = SimplexifyEngine::float_t;
410 using index_t = SimplexifyEngine::index_t;
411 static constexpr int32_t dim = SimplexifyEngine::dim;
412 using PointsType = SimplexifyEngine::PointsType;
413 using SimplexesType = SimplexifyEngine::SimplexesType;
414 using PointsViewType = SimplexifyEngine::PointsViewType;
415 using SimplexesViewType = SimplexifyEngine::SimplexesViewType;
416
429 explicit Simplexify(
430 const geometry::Cell<Dim, float_t, index_t>& cell,
431 bool reconstruct_boundaries = false)
432 : m_engine(cell) {
433 if constexpr (Dim == 3) {
434 // for each face of the cell find the triangles that define a
435 // sub-triangulation
436 if (reconstruct_boundaries) {
437 m_engine.reconstructBoundaries();
438 }
439 }
440 }
441
446 auto getSubsimplexesListFacet(index_t face_id) const {
447 if constexpr (Dim == 2) {
448 return Eigen::Vector2i{face_id, (face_id + 1) % m_engine.cell().nfaces()};
449 }
450 if constexpr (Dim == 3) {
451 return getTriangleListFacet(face_id);
452 }
453 };
454
460 index_t getNumPoints() const { return m_engine.getNumPoints(); }
461
471 index_t getNumSimplexes() const {
472 if constexpr (Dim == 2) {
473 return m_engine.getNumTriangles();
474 } else if constexpr (Dim == 3) {
475 return m_engine.getNumTetrahedra();
476 } else {
477 static_assert(Dim * 0, "Not Implemented");
478 return -1;
479 }
480 }
481
490 const PointsViewType& getPointList() const { return m_engine.getPointList(); }
491
504 const SimplexesViewType& getSimplexesList() const {
505 if constexpr (Dim == 2) {
506 return m_engine.getTriangleList();
507 } else if constexpr (Dim == 3) {
508 return m_engine.getTetrahedronList();
509 } else {
510 static_assert(Dim * 0, "Not Implemented");
511 }
512 }
513
514 protected:
515 SimplexifyEngine m_engine;
516};
