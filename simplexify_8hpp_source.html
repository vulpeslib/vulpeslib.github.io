441 template <typename T = void, typename = std::enable_if_t<Dim == 2, T>>

443 const geometry::Cell<2, float_t, index_t>& cell, bool /*unused*/ = false)

458 template <typename T = void, typename = std::enable_if_t<Dim == 3, T>>
459 explicit Simplexify(
460 const geometry::Cell<3, float_t, index_t>& cell,
461 bool reconstruct_boundaries = false)
462 : m_engine(cell, reconstruct_boundaries) {}
463
468 auto getSubsimplexesListFacet(index_t face_id) const {
469 if constexpr (Dim == 2) {
470 // this supposes that the first points in m_out are a copy of the points
471 // in m_in
472 return Eigen::Vector2i{face_id, (face_id + 1) % m_engine.cell().nfaces()};
473 }
474 if constexpr (Dim == 3) {
475 return m_engine.getTriangleListFacet(face_id);
476 }
477 };
478
484 index_t getNumPoints() const { return m_engine.getNumPoints(); }
485
495 index_t getNumSimplexes() const {
496 if constexpr (Dim == 2) {
497 return m_engine.getNumTriangles();
498 } else if constexpr (Dim == 3) {
499 return m_engine.getNumTetrahedra();
500 } else {
501 static_assert(Dim * 0, "Not Implemented");
502 return -1;
503 }
504 }
505
514 const PointsViewType& getPointList() const { return m_engine.getPointList(); }
515
528 const SimplexesViewType& getSimplexesList() const {
529 if constexpr (Dim == 2) {
530 return m_engine.getTriangleList();
531 } else if constexpr (Dim == 3) {
532 return m_engine.getTetrahedronList();
533 } else {
534 static_assert(Dim * 0, "Not Implemented");
535 }
536 }
537
538 protected:
539 SimplexifyEngine m_engine;
540};
