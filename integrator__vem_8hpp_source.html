37// Primary template
38template <typename T, typename U>
39struct concat_tuples;
40
41// Specialization for when both T and U are tuples
42template <typename... Ts, typename... Us>
43struct concat_tuples<std::tuple<Ts...>, std::tuple<Us...>> {
44 using type = std::tuple<Ts..., Us...>;
45};

47// Convenience alias
48template <typename T, typename U>
49using concat_tuples_t = typename concat_tuples<T, U>::type;

52template <
53 typename MeshType,
54 typename FeExpressionTuple,
55 typename FeExpressionFaceTuple = std::tuple<>>
56class WeakFormIntegratorVem {
57 public:
58 // WARNING: assuming all fe spaces are the same
59 using FeType = std::remove_pointer_t<
60 std::tuple_element_t<0, FeExpressionTuple>>::FeSpaceType;
61 // injecting in the FeExpressionTuple the expression needed to compute
62 // local matrices that do not depend only on monomials (e.g. B, D for
63 // the laplacian)
64 using ExtraExpressionTupleType = FeType::ExtraExpressionTuple;
65 using FeExpressionTupleExtended =
66 internal::concat_tuples_t<FeExpressionTuple, ExtraExpressionTupleType>;
67
68 using WeakFormMonomialIntegrator = WeakFormIntegratorBase<
69 MeshType,
70 FeExpressionTupleExtended,
71 FeExpressionFaceTuple>;
72
73 static inline constexpr std::size_t dim = MeshType::self_dim;
74
75 enum Backend { IntegratorSimplexifyBackend, IntegratorQFreeBackend };
76
77 WeakFormIntegratorVem(
78 const MeshType& mesh,
79 CellManager<MeshType, FeType>& cell_manager,
80 const FeExpressionTuple& fe_expression,
81 const FeExpressionFaceTuple& fe_expression_face = FeExpressionFaceTuple{})
82 : m_cell_manager(cell_manager),
83 m_fe_expression(fe_expression),
84 m_fe_expression_face(fe_expression_face) {
85 const auto& fe_space = std::get<0>(fe_expression)->feSpace();
86 const auto extra_expr = fe_space.getExtraExpression();
87 // TODO: lift this variable to be choosen by user
88 Backend backend = IntegratorSimplexifyBackend;
89 if (backend == IntegratorSimplexifyBackend) {
90 using WeakFormMonomialIntegratorConcrete = WeakFormIntegratorSimplexify<
91 MeshType, FeExpressionTupleExtended, FeExpressionFaceTuple>;
92 m_moment_integrator =
93 std::make_unique<WeakFormMonomialIntegratorConcrete>(
94 mesh, cell_manager, std::tuple_cat(fe_expression, extra_expr),
95 fe_expression_face);
96 } else if (backend == IntegratorQFreeBackend) {
97 using WeakFormMonomialIntegratorConcrete = WeakFormIntegratorQFree<
98 MeshType, FeExpressionTupleExtended, FeExpressionFaceTuple>;
99 m_moment_integrator =
100 std::make_unique<WeakFormMonomialIntegratorConcrete>(
101 mesh, cell_manager, std::tuple_cat(fe_expression, extra_expr),
102 fe_expression_face);
103 } else {
104 assert(false && "NotImplemented");
105 std::cout << "NotImplemented" << std::endl;
106 }
107 }
108
109 void integrate(MeshType::index_t cell_id) {
110 m_moment_integrator->integrate(cell_id);
111 // compute the local matrices that are of the FeSpace (e.g. B, D for lapl)
112 const auto& fe_space = std::get<0>(m_fe_expression)->feSpace();
113 const auto& integrals = m_moment_integrator->getIntegrals();
114 const auto dim0 = integrals.dimension(0);
115 const auto dim1 = integrals.dimension(1);
116 constexpr auto dim2 = std::tuple_size<ExtraExpressionTupleType>{};
117 constexpr auto offset = std::tuple_size<FeExpressionTuple>{};
118 const Eigen::TensorMap<const Eigen::Tensor<double, 3>> extra_integrals(
119 &integrals(0, 0, offset), dim0, dim1, dim2);
120 m_extra_matrices =
121 fe_space.template computeExtraMatrices<typename MeshType::index_t>(
122 m_cell_manager.getCell(cell_id), extra_integrals);
123 }
124
125 const auto& getExtraMatrices() const { return m_extra_matrices; }
126 const auto& getIntegrals() const {
127 return m_moment_integrator->getIntegrals();
128 }
129
130 protected:
131 CellManager<MeshType, FeType>& m_cell_manager;
132 std::unique_ptr<WeakFormMonomialIntegrator> m_moment_integrator = nullptr;
133
134 FeExpressionTuple m_fe_expression;
135 FeExpressionFaceTuple m_fe_expression_face;
136
137 std::array<Eigen::MatrixXd, 3> m_extra_matrices;
138};
