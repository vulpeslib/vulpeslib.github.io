67template <
68 typename MeshType,
69 typename FeExpressionTuple,
70 typename FeExpressionFaceTuple = std::tuple<>>
71class WeakFormIntegratorVem {
72 public:
73 // WARNING: assuming all fe spaces are the same
74 using FeType = std::remove_pointer_t<
75 std::tuple_element_t<0, FeExpressionTuple>>::FeSpaceType;
76 // injecting in the FeExpressionTuple the expression needed to compute
77 // local matrices that do not depend only on monomials (e.g. B, D for
78 // the laplacian)
79 using ExtraExpressionTupleType = FeType::ExtraExpressionTuple;
80 using FeExpressionTupleExtended =
81 internal::concat_tuples_t<FeExpressionTuple, ExtraExpressionTupleType>;
82
83 using WeakFormMonomialIntegrator = WeakFormIntegratorBase<
84 MeshType,
85 FeExpressionTupleExtended,
86 FeExpressionFaceTuple>;
87
88 static inline constexpr std::size_t dim = MeshType::self_dim;
89
90 enum Backend { IntegratorSimplexifyBackend, IntegratorQFreeBackend };
91
92 WeakFormIntegratorVem(
93 const MeshType& mesh,
94 CellManager<MeshType, FeType>& cell_manager,
95 const FeExpressionTuple& fe_expression,
96 const FeExpressionFaceTuple& fe_expression_face = FeExpressionFaceTuple{},
97 Backend backend = IntegratorSimplexifyBackend)
98 : m_cell_manager(cell_manager),
99 m_fe_expression(fe_expression),
100 m_fe_expression_face(fe_expression_face) {
101 const auto& fe_space = std::get<0>(fe_expression)->feSpace();
102 const auto extra_expr = fe_space.getExtraExpression();
103 // TODO: lift this variable to be choosen by user
104 if (backend == IntegratorSimplexifyBackend) {
105 using WeakFormMonomialIntegratorConcrete = WeakFormIntegratorSimplexify<
106 MeshType, FeExpressionTupleExtended, FeExpressionFaceTuple>;
107 m_moment_integrator =
108 std::make_unique<WeakFormMonomialIntegratorConcrete>(
109 mesh, cell_manager, std::tuple_cat(fe_expression, extra_expr),
110 fe_expression_face);
111 } else if (backend == IntegratorQFreeBackend) {
112 using WeakFormMonomialIntegratorConcrete = WeakFormIntegratorQFree<
113 MeshType, FeExpressionTupleExtended, FeExpressionFaceTuple>;
114 m_moment_integrator =
115 std::make_unique<WeakFormMonomialIntegratorConcrete>(
116 mesh, cell_manager, std::tuple_cat(fe_expression, extra_expr),
117 fe_expression_face);
118 } else {
119 assert(false && "NotImplemented");
120 std::cout << "NotImplemented" << std::endl;
121 }
122 }
123
124 void integrate(MeshType::index_t cell_id) {
125 m_moment_integrator->integrate(cell_id);
126 // compute the local matrices that are of the FeSpace (e.g. B, D for lapl)
127 const auto& fe_space = std::get<0>(m_fe_expression)->feSpace();
128 const auto& integrals = m_moment_integrator->getIntegrals();
129 const auto dim0 = integrals.dimension(0);
130 const auto dim1 = integrals.dimension(1);
131 constexpr auto dim2 = std::tuple_size<ExtraExpressionTupleType>{};
132 constexpr auto offset = std::tuple_size<FeExpressionTuple>{};
133 const Eigen::TensorMap<const Eigen::Tensor<double, 3>> extra_integrals(
134 &integrals(0, 0, offset), dim0, dim1, dim2);
135 m_extra_matrices =
136 fe_space.template computeExtraMatrices<typename MeshType::index_t>(
137 m_cell_manager.getCell(cell_id), extra_integrals);
138 }
139
140 const auto& getExtraMatrices() const { return m_extra_matrices; }
141 const auto& getIntegrals() const {
142 return m_moment_integrator->getIntegrals();
143 }
144
145 protected:
146 CellManager<MeshType, FeType>& m_cell_manager;
147 std::unique_ptr<WeakFormMonomialIntegrator> m_moment_integrator = nullptr;
148
149 FeExpressionTuple m_fe_expression;
150 FeExpressionFaceTuple m_fe_expression_face;
151
152 std::array<Eigen::MatrixXd, 3> m_extra_matrices;
153};
