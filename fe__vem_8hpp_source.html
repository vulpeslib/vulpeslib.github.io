43template <std::size_t Dim>
44class FeSpaceVEM : public FeSpaceBase<
45 Dim,
46 internal::real_t,
47 DofPolicyFeSpaceVEM<Dim>,
48 FeSpaceVEM<Dim>> {
49 public:
50 using fast_float_t = double;
51 using MyTrialFunction = fe::TrialFunction<FeSpaceVEM>;
52 using MyTestFunction = fe::TestFunction<FeSpaceVEM>;
53 using fast_int_t = uint8_t;
54 using Monomial = monomial::Monomial<Dim, fast_int_t, fast_float_t, false>;
55 // this type is needed to work with the qfree integrator
56 using PolynomialFast =
57 polynomial::SparsePolynomial<Dim, fast_int_t, double, false>;
58 using Polynomial = PolynomialFast;
59 // TODO: we do not actually need the full mass matrix like expressed here
60 // but just the first binomial(order+dim, dim) rows to build D
61 using ExtraExpression0 =
62 fe::FeExpressionProduct<Dim, MyTrialFunction, MyTestFunction>;
63 using ExtraExpressionTuple = std::tuple<ExtraExpression0*>;
64
65 enum ExtraMatrices { MatrixD = 0, MatrixB = 1, MomentProjection = 2 };
66
67 explicit FeSpaceVEM(std::size_t order)
68 : FeSpaceBase<
69 Dim,
70 internal::real_t,
71 DofPolicyFeSpaceVEM<Dim>,
72 FeSpaceVEM<Dim>>(order) {
73 // define the extra expression
74 m_trial = std::make_unique<MyTrialFunction>(*this);
75 m_test = std::make_unique<MyTestFunction>(*this);
76 m_extra_expr_0 = std::make_unique<ExtraExpression0>((*m_trial) * (*m_test));
77 m_extra_expr = std::tuple(m_extra_expr_0.get());
78 // build the basis
79 for (std::size_t k = 0; k < vmath::binomial(order + Dim, Dim); ++k) {
80 const auto exp_int_32_t = monomial::getMonomial<Dim, int32_t>(k);
81 std::array<fast_int_t, Dim> exp_fast_int;
82 for (std::size_t d = 0; d < Dim; ++d) {
83 exp_fast_int[d] = exp_int_32_t[d];
84 }
85 m_monos.emplace_back(exp_fast_int, 1.0);
86 m_polys_fast.emplace_back(m_monos.back());
87 }
88 // store internal Lobatto points
89 m_lobatto_rule = *qrule::QuadratureLibrary<1, fast_float_t>::getRule(
90 qrule::TensorLobatto, order + 1);
91 // compute the laplacian of the monomials for the matrix B
92 for (const auto& poly : m_monos) {
93 m_monos_grad.push_back(poly.gradient());
94 m_monos_lapl.emplace_back();
95 for (std::size_t d = 0; d < Dim; ++d) {
96 m_monos_lapl.back()[d] = m_monos_grad.back()[d].derivative(d);
97 }
98 }
99 }

112 template <typename IndexType>
113 std::array<Eigen::Index, Dim + 1> computeNumDofs(
114 const geometry::Cell<Dim, fast_float_t, IndexType>& cell) const {
115 std::array<Eigen::Index, Dim + 1> n_cum_dofs = {0};
116 n_cum_dofs[0] = cell.vtxs().cols();
117 const auto order = this->m_order;
118 if constexpr (Dim == 2) {
119 // edge dofs
120 n_cum_dofs[1] = n_cum_dofs[0] + cell.nfaces() * (order - 1);
121 } else if constexpr (Dim == 3) {
122 // edge dofs
123 n_cum_dofs[1] = n_cum_dofs[0] + cell.edges().size() * (order - 1);
124 // face dofs
125 n_cum_dofs[2] =
126 n_cum_dofs[1] +
127 cell.nfaces() *
128 ((order > 1) ? vmath::binomial(order + Dim - 3, Dim - 1) : 0);
129 }
130 // volume dofs
131 n_cum_dofs[Dim] = n_cum_dofs[Dim - 1] +
132 ((order > 1) ? vmath::binomial(order + Dim - 2, Dim) : 0);
133 std::cout << "n_cum_dofs: " << n_cum_dofs << std::endl;
134 return n_cum_dofs;
135 }

137 template <typename IndexType>
138 std::array<Eigen::Matrix<fast_float_t, -1, -1>, 3> computeExtraMatrices(
139 const geometry::Cell<Dim, fast_float_t, IndexType>& cell,
140 const Eigen::TensorMap<const Eigen::Tensor<fast_float_t, 3>>& integrals)
141 const {
142 assert(
143 std::abs(cell.measure() - integrals(0, 0, 0)) <
144 1e-14 * std::abs(cell.measure()));
145 const auto n_cum_dofs = computeNumDofs<IndexType>(cell);
146 const auto matrix_D =
147 computeMatrixD<IndexType>(cell, integrals, n_cum_dofs);
148 const auto matrix_B = computeMatrixB<IndexType>(cell, n_cum_dofs);
149 const auto moment_prj =
150 computeMomentProjection(matrix_D, n_cum_dofs[0], integrals);
151 return {matrix_D, matrix_B, moment_prj};
152 }

197template <std::size_t Dim>
198Eigen::Matrix<typename FeSpaceVEM<Dim>::fast_float_t, -1, -1>
199FeSpaceVEM<Dim>::computeMomentProjection(
200 const Eigen::Matrix<fast_float_t, -1, -1>& matrix_D,
201 Eigen::Index n_vtxs,
202 const Eigen::TensorMap<const Eigen::Tensor<fast_float_t, 3>>& integrals)
203 const {
204 const Eigen::Index ndofs = m_monos.size();
205 Eigen::Matrix<fast_float_t, -1, -1> prj(1, ndofs);
206 if (this->m_order > 1) {
207 // integral projection
208 for (Eigen::Index i = 0; i < ndofs; ++i) {
209 prj(i) = integrals(i, 0, 0) / integrals(0, 0, 0);
210 }
211 } else {
212 // vertex average projection
213 prj.setZero();
214 for (Eigen::Index i = 0; i < ndofs; ++i) {
215 for (Eigen::Index j = 0; j < n_vtxs; ++j) {
216 prj(i) += matrix_D(j, i);
217 }
218 prj(i) /= n_vtxs;
219 }
220 }
221 return prj;
222}

224template <std::size_t Dim>
225template <typename IndexType>
226Eigen::Matrix<typename FeSpaceVEM<Dim>::fast_float_t, -1, -1>
227FeSpaceVEM<Dim>::computeMatrixD(
228 const geometry::Cell<Dim, fast_float_t, IndexType>& cell,
229 const Eigen::TensorMap<const Eigen::Tensor<fast_float_t, 3>>& integrals,
230 const std::array<Eigen::Index, Dim + 1>& n_cum_dofs) const {
231 const Eigen::Index n_basis = m_monos.size();
232 const auto n_dofs = n_cum_dofs.back();
233 Eigen::Matrix<fast_float_t, -1, -1> matrix_D(n_dofs, n_basis);
234 // fill vtx dof, just evaluate the monomials on vtxs
235 for (Eigen::Index i = 0; i < n_cum_dofs[0]; ++i) {
236 for (Eigen::Index j = 0; j < n_basis; ++j) {
237 matrix_D(i, j) = m_monos[j].evaluate(cell.vtxs().col(i));
238 }
239 }
240 // fill edge dof
241 if (this->m_order > 1) {
242 const auto n_edges = cell.nedges();
243 for (Eigen::Index i = n_cum_dofs[0]; i < n_cum_dofs[1]; ++i) {
244 // index of the edge dof
245 const auto iloc = i - n_cum_dofs[0];
246 // vtxs are sorted accordingly to the edges in 2D
247 // thus they have the same numbering
248 Eigen::Index pt1_id = iloc / (this->m_order - 1);
249 Eigen::Index pt2_id = (pt1_id + 1) % n_edges;
250 if constexpr (Dim == 3) {
251 // in 3D we pass from edge numbering to vtx
252 pt1_id = cell.edges()[pt1_id][0];
253 pt2_id = cell.edges()[pt1_id][1];
254 }
255 // internal dof index for this edge
256 const auto k = iloc % (this->m_order - 1);
257 // [0, 1] local parametrization of the edge
258 const auto t = m_lobatto_rule.nodes[k + 1].point[0];
259 // recover the point in space as convex combination
260 const Eigen::Vector<fast_float_t, Dim> dof_pt =
261 t * cell.vtxs().col(pt1_id) + (1.0 - t) * cell.vtxs().col(pt2_id);
262 // evaluate on the point on the edge for each basis function
263 for (Eigen::Index j = 0; j < n_basis; ++j) {
264 matrix_D(i, j) = m_monos[j].evaluate(dof_pt);
265 }
266 }
267 // fill face dof
268 // TODO
269 // fill volume dof, use the mass integral injected in the integrator
270 const auto meausure = cell.measure();
271 for (Eigen::Index i = n_cum_dofs[Dim - 1]; i < n_cum_dofs[Dim]; ++i) {
272 for (Eigen::Index j = 0; j < n_basis; ++j) {
273 const auto iloc = i - n_cum_dofs[Dim - 1];
274 matrix_D(i, j) = integrals(iloc, j, 0) / meausure;
275 }
276 }
277 }
278 return matrix_D;
279}

281template <std::size_t Dim>
282template <typename IndexType>
283Eigen::Matrix<typename FeSpaceVEM<Dim>::fast_float_t, -1, -1>
284FeSpaceVEM<Dim>::computeMatrixB(
285 const geometry::Cell<Dim, fast_float_t, IndexType>& cell,
286 const std::array<Eigen::Index, Dim + 1>& n_cum_dofs) const {
287 const Eigen::Index n_basis = m_monos.size();
288 const auto n_dofs = n_cum_dofs.back();
289 Eigen::Matrix<fast_float_t, -1, -1> matrix_B =
290 decltype(matrix_B)::Zero(n_basis, n_dofs);
291 const auto meausure = cell.measure();
292 const auto h = cell.diameter();
293 const auto h2 = h * h;
294 if (this->m_order > 1) {
295 // compute contribution of the laplacian
296 for (Eigen::Index i = 0; i < n_basis; ++i) {
297 // for each monomial in the laplacian
298 for (std::size_t d = 0; d < Dim; ++d) {
299 // find the j-th dof corresponding to this monomial
300 const auto& lapl_mono = m_monos_lapl[i][d];
301 const auto linear_mono_id = monomial::getLinearIndex(lapl_mono.exp());
302 // add the offset of the other dofs
303 const auto j = n_cum_dofs[Dim - 1] + linear_mono_id;
304 // the h2 factor come from the double derivative in the computation
305 // of the coefficients, not from the change of variable
306 matrix_B(i, j) -= meausure * lapl_mono.coef() / h2;
307 }
308 }
309 }
310 // compute the edges contribution
311 if constexpr (Dim == 2) {
312 // we compute the edge integrals **without** a change of variable
313 const Eigen::Matrix<fast_float_t, -1, -1> unit_normals =
314 cell.normals().colwise().normalized();
315 // loop on each vtx of the cell
316 // or to be more precise on each dof associated to a vertex
317 // indeed, j indicates the index of the basis function that is non-zero here
318 for (Eigen::Index j = 0; j < n_cum_dofs[0]; ++j) {
319 // the evaluation of the basis in this point influences
320 // two edges integrals (since the vtx is shared among the two)
321 // find the two edges to which this vertex belongs to
322 const auto edge_id1 = (j + cell.nfaces() - 1) % cell.nfaces();
323 const auto edge_id2 = j;
324 // find the two points that compose the two edges
325 // <pt_prev_id> --- <j> --- <pt_next_id>
326 const auto pt_prev_id = edge_id1;
327 const auto pt_next_id = (j + 1) % cell.nfaces();
328 // compute the jacobian (norm) or each edge and rescale the
329 // lobatto weight accordingly (for the numerical integration)
330 const Eigen::Vector<fast_float_t, Dim> pt = cell.vtxs().col(j);
331 const auto jac1 = (pt - cell.vtxs().col(pt_prev_id)).norm();
332 const auto jac2 = (pt - cell.vtxs().col(pt_next_id)).norm();
333 const auto weight1 = m_lobatto_rule.nodes[0].weight * jac1;
334 const auto weight2 = m_lobatto_rule.nodes[0].weight * jac2;
335 // integrate the dot(grad(u), n)
336 // by definition of the basis func, only on one GL the basis is non-zero
337 for (Eigen::Index i = 1; i < n_basis; ++i) {
338 for (std::size_t d = 0; d < Dim; ++d) {
339 const auto mono_eval = m_monos_grad[i][d].evaluate(pt);
340 matrix_B(i, j) += weight1 * mono_eval * unit_normals(d, edge_id1);
341 matrix_B(i, j) += weight2 * mono_eval * unit_normals(d, edge_id2);
342 }
343 }
344 }
345 // we do the same of the above but now we loop on
346 // nodes internal to the edge, or better the basis functions
347 // associated to them
348 if (this->m_order > 1) {
349 for (Eigen::Index j = n_cum_dofs[0]; j < n_cum_dofs[1]; ++j) {
350 // find edge id, weight and physical extrema defining the edge
351 const auto jloc = j - n_cum_dofs[0];
352 const auto edge_id = jloc / (this->m_order - 1);
353 const auto weight_id = jloc % (this->m_order - 1);
354 const auto pt1_id = edge_id;
355 const auto pt2_id = (edge_id + 1) % cell.nfaces();
356 const Eigen::Vector<fast_float_t, Dim> pt1 = cell.vtxs().col(pt1_id);
357 const Eigen::Vector<fast_float_t, Dim> pt2 = cell.vtxs().col(pt2_id);
358 // find the parametric coordinate of the node
359 const auto t = m_lobatto_rule.nodes[weight_id + 1].point[0];
360 // find the physical point of the node
361 const Eigen::Vector<fast_float_t, Dim> pt = t * pt1 + (1.0 - t) * pt2;
362 // compute the jacobian (norm) of the edge and rescale accordingly
363 const auto jac = (pt1 - pt2).norm();
364 const auto weight = m_lobatto_rule.nodes[weight_id + 1].weight * jac;
365 // integrate the dot(grad(u), n)
366 // by definition of the basis func, only on one GL the basis is non-zero
367 for (Eigen::Index i = 1; i < n_basis; ++i) {
368 for (std::size_t d = 0; d < Dim; ++d) {
369 const auto mono_eval = m_monos_grad[i][d].evaluate(pt);
370 matrix_B(i, j) += weight * mono_eval * unit_normals(d, edge_id);
371 }
372 }
373 }
374 }
375 }
376 // first line is the projector P_0(\phi)
377 if (this->m_order == 1) {
378 const auto n_vtxs = static_cast<fast_float_t>(cell.vtxs().cols());
379 matrix_B.row(0).array() = 1.0 / n_vtxs;
380 } else {
381 matrix_B(0, n_cum_dofs[Dim - 1]) = 1.0;
382 }
383 return matrix_B;
384}
