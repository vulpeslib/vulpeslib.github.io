87 Polynomial3D<T, IsSpaceP> derivative(std::size_t component) const {
88 assert(component < 3);
89 Polynomial3D<T, IsSpaceP> result(dim());
90 if (component == 0) {
91 for (std::size_t i = 1; i < dim(); ++i) {
92 const auto jmax = dim() - IsSpaceP * i;
93 for (std::size_t j = 0; j < jmax; ++j) {
94 const auto kmax = dim() - IsSpaceP * (i + j);
95 for (std::size_t k = 0; k < kmax; ++k) {
96 result.coef(i - 1, j, k) = coef(i, j, k) * T(i);
97 }
98 }
99 }
100
101 } else if (component == 1) {
102 for (std::size_t i = 0; i < dim(); ++i) {
103 const auto jmax = dim() - IsSpaceP * i;
104 for (std::size_t j = 1; j < jmax; ++j) {
105 const auto kmax = dim() - IsSpaceP * (i + j);
106 for (std::size_t k = 0; k < kmax; ++k) {
107 result.coef(i, j - 1, k) = coef(i, j, k) * T(j);
108 }
109 }
110 }
111 } else {
112 for (std::size_t i = 0; i < dim(); ++i) {
113 const auto jmax = dim() - IsSpaceP * i;
114 for (std::size_t j = 0; j < jmax; ++j) {
115 const auto kmax = dim() - IsSpaceP * (i + j);
116 for (std::size_t k = 1; k < kmax; ++k) {
117 result.coef(i, j, k - 1) = coef(i, j, k) * T(k);
118 }
119 }
120 }
121 }
122 return result;
123 }

125 T integrate(
126 const Eigen::Map<Eigen::Vector<T, -1>>& monomial_integrals) const {
127 assert(m_coefficients.size() <= monomial_integrals.size());
128 T integral = 0.0;
129 for (std::size_t i = 0; i < m_coefficients.size(); ++i) {
130 integral += m_coefficients.data()[i] * monomial_integrals[i];
131 }
132 return integral;
133 }

144 T evaluate(const T* x) const {
145 // WARNING: this is not thread safe
146 static std::vector<T> partial_eval_yz;
147 static std::vector<T> partial_eval_x;
148 if (partial_eval_x.size() < dim()) {
149 partial_eval_yz.resize(dim() * dim());
150 partial_eval_x.resize(dim());
151 }
152
153 // Step 1: Evaluate in z for each (i,j)
154 for (std::size_t i = 0; i < dim(); ++i) {
155 const auto jmax = dim() - IsSpaceP * i;
156 for (std::size_t j = 0; j < jmax; ++j) {
157 T Q_ij = T(0);
158 const auto kmax = dim() - IsSpaceP * (i + j);
159 for (std::size_t k = kmax - 1; k > 0; --k) {
160 Q_ij = Q_ij * x[2] + m_coefficients(i, j, k);
161 }
162 partial_eval_yz[i * dim() + j] = Q_ij * x[2] + m_coefficients(i, j, 0);
163 }
164 }
165
166 // Step 2: Evaluate in y for each i
167 for (std::size_t i = 0; i < dim(); ++i) {
168 T Q_i = T(0);
169 const auto jmax = dim() - IsSpaceP * i;
170 for (std::size_t j = jmax - 1; j > 0; --j) {
171 Q_i = Q_i * x[1] + partial_eval_yz[i * dim() + j];
172 }
173 partial_eval_x[i] = Q_i * x[1] + partial_eval_yz[i * dim()];
174 }
175
176 // Step 3: Final evaluation in x
177 T result = T(0);
178 for (std::size_t i = dim() - 1; i > 0; --i) {
179 result = result * x[0] + partial_eval_x[i];
180 }
181 return result * x[0] + partial_eval_x[0];
182 }

206 Polynomial3D<T, IsSpaceP>& operator*=(
207 const Polynomial3D<T, IsSpaceP>& other) {
208 std::size_t new_dim = dim() + other.dim();
209 decltype(m_coefficients) new_coeffs(new_dim);
210
211 for (std::size_t i = 0; i < dim(); ++i) {
212 const auto jmax = dim() - IsSpaceP * i;
213 for (std::size_t j = 0; j < jmax; ++j) {
214 const auto kmax = dim() - IsSpaceP * (i + j);
215 for (std::size_t k = 0; k < kmax; ++k) {
216 for (std::size_t l = 0; l < other.dim(); ++l) {
217 const auto mmax = other.dim() - IsSpaceP * l;
218 for (std::size_t m = 0; m < mmax; ++m) {
219 const auto nmax = other.dim() - IsSpaceP * (l + m);
220 for (std::size_t n = 0; n < nmax; ++n) {
221 new_coeffs(i + l, j + m, k + n) +=
222 coef(i, j, k) * other.coef(l, m, n);
223 }
224 }
225 }
226 }
227 }
228 }
229
230 m_coefficients = std::move(new_coeffs);
231 return *this;
232 }

241 friend std::ostream& operator<<(std::ostream& os, const Polynomial3D& poly) {
242 for (std::size_t i = 0; i < poly.dim(); ++i) {
243 const auto jmax = poly.dim() - IsSpaceP * i;
244 for (std::size_t j = 0; j < jmax; ++j) {
245 const auto kmax = poly.dim() - IsSpaceP * (i + j);
246 for (std::size_t k = 0; k < kmax; ++k) {
247 if (poly.m_coefficients(i, j, k) != T(0)) {
248 if (i > 0 || j > 0 || k > 0) {
249 os << " + ";
250 }
251 os << poly.m_coefficients(i, j, k);
252 if (i > 0) {
253 os << "x^" << i;
254 }
255 if (j > 0) {
256 os << "y^" << j;
257 }
258 if (k > 0) {
259 os << "z^" << k;
260 }
261 }
262 }
263 }
264 }

274 template <bool Add>
275 Polynomial3D<T, IsSpaceP>& add_(const Polynomial3D<T, IsSpaceP>& other) {
276 if (dim() < other.dim()) {
277 auto result_coef = other.m_coefficients;
278 if constexpr (!Add) {
279 // Negate coefficients for subtraction
280 std::transform(
281 result_coef.begin(), result_coef.end(), result_coef.begin(),
282 [](const auto& val) { return -val; });
283 }
284 for (std::size_t i = 0; i < dim(); ++i) {
285 const auto jmax = dim() - IsSpaceP * i;
286 for (std::size_t j = 0; j < jmax; ++j) {
287 const auto kmax = dim() - IsSpaceP * (i + j);
288 for (std::size_t k = 0; k < kmax; ++k) {
289 result_coef(i, j, k) += coef(i, j, k);
290 }
291 }
292 }
293 m_coefficients = std::move(result_coef);
294 } else {
295 for (std::size_t i = 0; i < other.dim(); ++i) {
296 const auto jmax = other.dim() - IsSpaceP * i;
297 for (std::size_t j = 0; j < jmax; ++j) {
298 const auto kmax = other.dim() - IsSpaceP * (i + j);
299 for (std::size_t k = 0; k < kmax; ++k) {
300 if constexpr (Add) {
301 m_coefficients(i, j, k) += other.coef(i, j, k);
302 } else {
303 m_coefficients(i, j, k) -= other.coef(i, j, k);
304 }
305 }
306 }
307 }
308 }
309 return *this;
310 }
