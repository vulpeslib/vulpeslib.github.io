139template <ckdtree_intp_t Dim>
140static ckdtree_intp_t build_ckdtree(
141 ckdtree<Dim>* self,
142 ckdtree_intp_t start_idx,
143 intptr_t end_idx,
144 double* maxes,
145 double* mins,
146 const int _median,
147 const int _compact) {
148 constexpr ckdtree_intp_t m = self->m;
149 const double* data = self->raw_data;
150 ckdtree_intp_t* indices = reinterpret_cast<intptr_t*>(self->raw_indices);
151
152 ckdtreenode new_node = {}, *n, *root;
153 ckdtree_intp_t node_index, less, greater;
154 ckdtree_intp_t i, j, p, d;
155 double size, split, minval, maxval;
156
157 /* put a new node into the node stack */
158 self->tree_buffer->push_back(new_node);
159 node_index = self->tree_buffer->size() - 1;
160 root = tree_buffer_root(self->tree_buffer);
161 n = root + node_index;
162 memset(n, 0, sizeof(n[0]));
163
164 n->start_idx = start_idx;
165 n->end_idx = end_idx;
166 n->children = end_idx - start_idx;
167
168 if (end_idx - start_idx <= self->leafsize) {
169 /* below brute force limit, return leafnode */
170 n->split_dim = -1;
171 return node_index;
172 } else {
173 if (_compact) {
174 /* Recompute hyperrectangle bounds. This should lead to a more
175 * compact kd-tree but comes at the expense of larger construction
176 * time. However, construction time is usually dwarfed by the
177 * query time by orders of magnitude.
178 */
179 const double* tmp_data_point;
180 tmp_data_point = data + indices[start_idx] * m;
181 for (i = 0; i < m; ++i) {
182 maxes[i] = tmp_data_point[i];
183 mins[i] = tmp_data_point[i];
184 }
185 for (j = start_idx + 1; j < end_idx; ++j) {
186 tmp_data_point = data + indices[j] * m;
187 for (i = 0; i < m; ++i) {
188 double tmp = tmp_data_point[i];
189 maxes[i] = maxes[i] > tmp ? maxes[i] : tmp;
190 mins[i] = mins[i] < tmp ? mins[i] : tmp;
191 }
192 }
193 }
194
195 /* split on the dimension with largest spread */
196 d = 0;
197 size = 0;
198 for (i = 0; i < m; ++i) {
199 if (maxes[i] - mins[i] > size) {
200 d = i;
201 size = maxes[i] - mins[i];
202 }
203 }
204 maxval = maxes[d];
205 minval = mins[d];
206 if (maxval == minval) {
207 /* all points are identical; warn user?
208 * return leafnode
209 */
210 n->split_dim = -1;
211 return node_index;
212 }
213
214 /* construct new inner node */
215 auto index_compare = [=](ckdtree_intp_t a, ckdtree_intp_t b) {
216 const double point_a = data[a * m + d];
217 const double point_b = data[b * m + d];
218 return point_a < point_b;
219 };
220
221 auto partition_pivot = [=](ckdtree_intp_t* first, ckdtree_intp_t* last,
222 double pivot) {
223 const auto partition_ptr = std::partition(
224 first, last,
225 [&](ckdtree_intp_t a) { return !(data[a * m + d] >= pivot); });
226 return partition_ptr - indices;
227 };
228
229 if (_median) {
230 /* split on median to create a balanced tree
231 * adopted from scikit-learn
232 */
233 const auto n_points = end_idx - start_idx;
234 auto* node_indices = indices + start_idx;
235 auto mid = node_indices + n_points / 2;
236 std::nth_element(
237 node_indices, mid, node_indices + n_points, index_compare);
238
239 split = data[*mid * m + d];
240 p = partition_pivot(node_indices, mid, split);
241 } else {
242 /* split with the sliding midpoint rule */
243 split = (maxval + minval) / 2;
244
245 p = partition_pivot(indices + start_idx, indices + end_idx, split);
246 }
247
248 /* slide midpoint if necessary */
249 if (p == start_idx) {
250 /* no points less than split */
251 auto min_idx = *std::min_element(
252 indices + start_idx, indices + end_idx, index_compare);
253 split = std::nextafter(data[min_idx * m + d], HUGE_VAL);
254 p = partition_pivot(indices + start_idx, indices + end_idx, split);
255 } else if (p == end_idx) {
256 /* no points greater than split */
257 auto max_idx = *std::max_element(
258 indices + start_idx, indices + end_idx, index_compare);
259 split = data[max_idx * m + d];
260 p = partition_pivot(indices + start_idx, indices + end_idx, split);
261 }
262
263 if (p == start_idx || p == end_idx) {
264 // All children are equal in this dimension, try again with new bounds
265 assert(!_compact);
266 self->tree_buffer->pop_back();
267 std::array<double, 2 * m> tmp_bounds = {0};
268 double* tmp_mins = &tmp_bounds[0];
269 std::copy_n(mins, m, tmp_mins);
270 double* tmp_maxes = &tmp_bounds[m];
271 std::copy_n(maxes, m, tmp_maxes);
272
273 const auto fixed_val = data[indices[start_idx] * m + d];
274 tmp_mins[d] = fixed_val;
275 tmp_maxes[d] = fixed_val;
276
277 return build_ckdtree<Dim>(
278 self, start_idx, end_idx, tmp_maxes, tmp_mins, _median, _compact);
279 }
280
281 if (_compact) {
282 less = build_ckdtree(self, start_idx, p, maxes, mins, _median, _compact);
283 greater = build_ckdtree(self, p, end_idx, maxes, mins, _median, _compact);
284 } else {
285 std::array<double, m> tmp = {0};
286 double* mids = &tmp[0];
287
288 for (i = 0; i < m; ++i) {
289 mids[i] = maxes[i];
290 }
291 mids[d] = split;
292 less = build_ckdtree(self, start_idx, p, mids, mins, _median, _compact);
293
294 for (i = 0; i < m; ++i) {
295 mids[i] = mins[i];
296 }
297 mids[d] = split;
298 greater = build_ckdtree(self, p, end_idx, maxes, mids, _median, _compact);
299 }
300
301 /* recompute n because std::vector can
302 * reallocate its internal buffer
303 */
304 root = tree_buffer_root(self->tree_buffer);
305 n = root + node_index;
306 /* fill in entries */
307 n->less = reinterpret_cast<decltype(n->less)>(less);
308 n->greater = reinterpret_cast<decltype(n->greater)>(greater);
309 n->split_dim = d;
310 n->split = split;
311
312 return node_index;
313 }

329 /* copy array data */

343/*
344 * Rectangle-to-rectangle distance tracker
345 * =======================================
346 *
347 * The logical unit that repeats over and over is to keep track of the
348 * maximum and minimum distances between points in two hyperrectangles
349 * as these rectangles are successively split.
350 *
351 * Example
352 * -------
353 * node1 encloses points in rect1, node2 encloses those in rect2
354 *
355 * cdef RectRectDistanceTracker dist_tracker
356 * dist_tracker = RectRectDistanceTracker(rect1, rect2, p)
357 *
358 * ...
359 *
360 * if dist_tracker.min_distance < ...:
361 * ...
362 *
363 * dist_tracker.push_less_of(1, node1)
364 * do_something(node1.less, dist_tracker)
365 * dist_tracker.pop()
366 *
367 * dist_tracker.push_greater_of(1, node1)
368 * do_something(node1.greater, dist_tracker)
369 * dist_tracker.pop()
370 *
371 * Notice that Point is just a reduced case of Rectangle where
372 * mins == maxes.
373 *
374 */
375
376struct RR_stack_item {
377 ckdtree_intp_t which;
378 ckdtree_intp_t split_dim;
379 double min_along_dim;
380 double max_along_dim;
381 double min_distance;
382 double max_distance;
383};

385const ckdtree_intp_t LESS = 1;
386const ckdtree_intp_t GREATER = 2;

388template <typename MinMaxDist, ckdtree_intp_t Dim>
389struct RectRectDistanceTracker {
390 const ckdtree<Dim>* tree;
391 Rectangle<Dim> rect1;
392 Rectangle<Dim> rect2;
393 double p;
394 double epsfac;
395 double upper_bound;
396 double min_distance;
397 double max_distance;
398
399 ckdtree_intp_t stack_size;
400 ckdtree_intp_t stack_max_size;
401 std::vector<RR_stack_item> stack_arr;
402 RR_stack_item* stack;
403
404 /* if min/max distance / adjustment is less than this,
405 * we believe the incremental tracking is inaccurate */
406 double inaccurate_distance_limit;
407
408 void _resize_stack(const ckdtree_intp_t new_max_size) {
409 stack_arr.resize(new_max_size);
410 stack = &stack_arr[0];
411 stack_max_size = new_max_size;
412 };
413
414 RectRectDistanceTracker(
415 const ckdtree<Dim>* _tree,
416 const Rectangle<Dim>& _rect1,
417 const Rectangle<Dim>& _rect2,
418 const double _p,
419 const double eps,
420 const double _upper_bound)
421 : tree(_tree), rect1(_rect1), rect2(_rect2), stack_arr(8) {
422 p = _p;
423
424 /* internally we represent all distances as distance ** p */
425 if (CKDTREE_LIKELY(p == 2.0)) {
426 upper_bound = _upper_bound * _upper_bound;
427 } else if ((!std::isinf(p)) && (!std::isinf(_upper_bound))) {
428 upper_bound = std::pow(_upper_bound, p);
429 } else {
430 upper_bound = _upper_bound;
431 }
432
433 /* fiddle approximation factor */
434 if (CKDTREE_LIKELY(p == 2.0)) {
435 double tmp = 1. + eps;
436 epsfac = 1. / (tmp * tmp);
437 } else if (eps == 0.) {
438 epsfac = 1.;
439 } else if (std::isinf(p)) {
440 epsfac = 1. / (1. + eps);
441 } else {
442 epsfac = 1. / std::pow((1. + eps), p);
443 }
444
445 stack = &stack_arr[0];
446 stack_max_size = 8;
447 stack_size = 0;
448
449 /* Compute initial min and max distances */
450 MinMaxDist::rect_rect_p(
451 tree, rect1, rect2, p, &min_distance, &max_distance);
452 if (std::isinf(max_distance)) {
453 const char* msg =
454 "Encountering floating point overflow. "
455 "The value of p too large for this dataset; "
456 "For such large p, consider using the special case p=np.inf . ";
457 throw std::invalid_argument(msg); // raises ValueError
458 }
459 inaccurate_distance_limit = max_distance;
460 };
461
462 void push(
463 const ckdtree_intp_t which,
464 const intptr_t direction,
465 const ckdtree_intp_t split_dim,
466 const double split_val) {
467 const double p = this->p;
468 /* subnomial is 1 if round-off is expected to taint the incremental distance
469 * tracking. in that case we always recompute the distances. Recomputing
470 * costs more calls to pow, thus if the round-off error does not seem to
471 * wipe out the value, then we still do the incremental update.
472 * */
473 int subnomial = 0;
474
475 Rectangle<Dim>* rect;
476 if (which == 1) {
477 rect = &rect1;
478 } else {
479 rect = &rect2;
480 }
481
482 /* push onto stack */
483 if (stack_size == stack_max_size) {
484 _resize_stack(stack_max_size * 2);
485 }
486
487 RR_stack_item* item = &stack[stack_size];
488 ++stack_size;
489 item->which = which;
490 item->split_dim = split_dim;
491 item->min_distance = min_distance;
492 item->max_distance = max_distance;
493 item->min_along_dim = rect->mins()[split_dim];
494 item->max_along_dim = rect->maxes()[split_dim];
495
496 /* update min/max distances */
497 double min1, max1;
498 double min2, max2;
499
500 MinMaxDist::interval_interval_p(
501 tree, rect1, rect2, split_dim, p, &min1, &max1);
502
503 if (direction == LESS) {
504 rect->maxes()[split_dim] = split_val;
505 } else {
506 rect->mins()[split_dim] = split_val;
507 }
508
509 MinMaxDist::interval_interval_p(
510 tree, rect1, rect2, split_dim, p, &min2, &max2);
511
512 subnomial = subnomial || (min_distance < inaccurate_distance_limit ||
513 max_distance < inaccurate_distance_limit);
514
515 subnomial = subnomial || ((min1 != 0 && min1 < inaccurate_distance_limit) ||
516 max1 < inaccurate_distance_limit);
517 subnomial = subnomial || ((min2 != 0 && min2 < inaccurate_distance_limit) ||
518 max2 < inaccurate_distance_limit);
519 subnomial = subnomial || (min_distance < inaccurate_distance_limit ||
520 max_distance < inaccurate_distance_limit);
521
522 if (CKDTREE_UNLIKELY(subnomial)) {
523 MinMaxDist::rect_rect_p(
524 tree, rect1, rect2, p, &min_distance, &max_distance);
525 } else {
526 min_distance += (min2 - min1);
527 max_distance += (max2 - max1);
528 }
529 };
530
531 inline void push_less_of(
532 const ckdtree_intp_t which, const ckdtreenode* node) {
533 push(which, LESS, node->split_dim, node->split);
534 };
535
536 inline void push_greater_of(
537 const ckdtree_intp_t which, const ckdtreenode* node) {
538 push(which, GREATER, node->split_dim, node->split);
539 };
540
541 inline void pop() {
542 /* pop from stack */
543 --stack_size;
544
545 /* assert stack_size >= 0 */
546 if (CKDTREE_UNLIKELY(stack_size < 0)) {
547 const char* msg = "Bad stack size. This error should never occur.";
548 throw std::logic_error(msg);
549 }
550
551 RR_stack_item* item = &stack[stack_size];
552 min_distance = item->min_distance;
553 max_distance = item->max_distance;
554
555 if (item->which == 1) {
556 rect1.mins()[item->split_dim] = item->min_along_dim;
557 rect1.maxes()[item->split_dim] = item->max_along_dim;
558 } else {
559 rect2.mins()[item->split_dim] = item->min_along_dim;
560 rect2.maxes()[item->split_dim] = item->max_along_dim;
561 }
562 };
563};

565template <typename Dist1D, ckdtree_intp_t Dim>
566struct BaseMinkowskiDistPp {
567 /* 1-d pieces
568 * These should only be used if p != infinity
569 */
570
571 static inline void interval_interval_p(
572 const ckdtree<Dim>* tree,
573 const Rectangle<Dim>& rect1,
574 const Rectangle<Dim>& rect2,
575 const ckdtree_intp_t k,
576 const double p,
577 double* min,
578 double* max) {
579 /* Compute the minimum/maximum distance along dimension k between points in
580 * two hyperrectangles.
581 */
582 Dist1D::interval_interval(tree, rect1, rect2, k, min, max);
583 *min = std::pow(*min, p);
584 *max = std::pow(*max, p);
585 }
586
587 static inline void rect_rect_p(
588 const ckdtree<Dim>* tree,
589 const Rectangle<Dim>& rect1,
590 const Rectangle<Dim>& rect2,
591 const double p,
592 double* min,
593 double* max) {
594 *min = 0.;
595 *max = 0.;
596 for (ckdtree_intp_t i = 0; i < Dim; ++i) {
597 double min_, max_;
598
599 Dist1D::interval_interval(tree, rect1, rect2, i, &min_, &max_);
600
601 *min += std::pow(min_, p);
602 *max += std::pow(max_, p);
603 }
604 }
605
606 static inline double point_point_p(
607 const ckdtree<Dim>* tree,
608 const double* x,
609 const double* y,
610 const double p,
611 const ckdtree_intp_t k,
612 const double upperbound) {
613 /*
614 * Compute the distance between x and y
615 *
616 * Computes the Minkowski p-distance to the power p between two points.
617 * If the distance**p is larger than upperbound, then any number larger
618 * than upperbound may be returned (the calculation is truncated).
619 */
620
621 ckdtree_intp_t i;
622 double r, r1;
623 r = 0;
624 for (i = 0; i < k; ++i) {
625 r1 = Dist1D::point_point(tree, x, y, i);
626 r += std::pow(r1, p);
627 if (r > upperbound) {
628 return r;
629 }
630 }
631 return r;
632 }
633
634 static inline double distance_p(const double s, const double p) {
635 return std::pow(s, p);
636 }
637};

870/*
871 * Measuring distances
872 * ===================
873 */
874inline double sqeuclidean_distance_double(
875 const double* u, const double* v, ckdtree_intp_t n) {
876 double s;
877 ckdtree_intp_t i = 0;
878 // manually unrolled loop, might be vectorized
879 double acc[4] = {0., 0., 0., 0.};
880 for (; i + 4 <= n; i += 4) {
881 double _u[4] = {u[i], u[i + 1], u[i + 2], u[i + 3]};
882 double _v[4] = {v[i], v[i + 1], v[i + 2], v[i + 3]};
883 double diff[4] = {
884 _u[0] - _v[0], _u[1] - _v[1], _u[2] - _v[2], _u[3] - _v[3]};
885 acc[0] += diff[0] * diff[0];
886 acc[1] += diff[1] * diff[1];
887 acc[2] += diff[2] * diff[2];
888 acc[3] += diff[3] * diff[3];
889 }
890 s = acc[0] + acc[1] + acc[2] + acc[3];
891 if (i < n) {
892 for (; i < n; ++i) {
893 double d = u[i] - v[i];
894 s += d * d;
895 }
896 }
897 return s;
898}

913template <ckdtree_intp_t Dim>
914struct BoxDist1D {
915 static inline void _interval_interval_1d(
916 double min,
917 double max,
918 double* realmin,
919 double* realmax,
920 const double full,
921 const double half) {
922 /* Minimum and maximum distance of two intervals in a periodic box
923 *
924 * min and max is the nonperiodic distance between the near
925 * and far edges.
926 *
927 * full and half are the box size and 0.5 * box size.
928 *
929 * value is returned in realmin and realmax;
930 *
931 * This function is copied from kdcount, and the convention
932 * of is that
933 *
934 * min = rect1.min - rect2.max
935 * max = rect1.max - rect2.min = - (rect2.min - rect1.max)
936 *
937 * We will fix the convention later.
938 * */
939 if (CKDTREE_UNLIKELY(full <= 0)) {
940 /* A non-periodic dimension */
941 /* \/ */
942 if (max <= 0 || min >= 0) {
943 /* do not pass though 0 */
944 min = ckdtree_fabs(min);
945 max = ckdtree_fabs(max);
946 if (min < max) {
947 *realmin = min;
948 *realmax = max;
949 } else {
950 *realmin = max;
951 *realmax = min;
952 }
953 } else {
954 min = ckdtree_fabs(min);
955 max = ckdtree_fabs(max);
956 *realmax = ckdtree_fmax(max, min);
957 *realmin = 0;
958 }
959 /* done with non-periodic dimension */
960 return;
961 }
962 if (max <= 0 || min >= 0) {
963 /* do not pass through 0 */
964 min = ckdtree_fabs(min);
965 max = ckdtree_fabs(max);
966 if (min > max) {
967 double t = min;
968 min = max;
969 max = t;
970 }
971 if (max < half) {
972 /* all below half*/
973 *realmin = min;
974 *realmax = max;
975 } else if (min > half) {
976 /* all above half */
977 *realmax = full - min;
978 *realmin = full - max;
979 } else {
980 /* min below, max above */
981 *realmax = half;
982 *realmin = ckdtree_fmin(min, full - max);
983 }
984 } else {
985 /* pass though 0 */
986 min = -min;
987 if (min > max) {
988 max = min;
989 }
990 if (max > half) {
991 max = half;
992 }
993 *realmax = max;
994 *realmin = 0;
995 }
996 }
997 static inline void interval_interval(
998 const ckdtree<Dim>* tree,
999 const Rectangle<Dim>& rect1,
1000 const Rectangle<Dim>& rect2,
1001 const ckdtree_intp_t k,
1002 double* min,
1003 double* max) {
1004 /* Compute the minimum/maximum distance along dimension k between points in
1005 * two hyperrectangles.
1006 */
1007 _interval_interval_1d(
1008 rect1.mins()[k] - rect2.maxes()[k], rect1.maxes()[k] - rect2.mins()[k],
1009 min, max, tree->raw_boxsize_data[k], tree->raw_boxsize_data[k + Dim]);
1010 }

1012 static inline double point_point(
1013 const ckdtree<Dim>* tree,
1014 const double* x,
1015 const double* y,
1016 const ckdtree_intp_t k) {
1017 double r1;
1018 r1 = wrap_distance(
1019 x[k] - y[k], tree->raw_boxsize_data[k + tree->m],
1020 tree->raw_boxsize_data[k]);
1021 r1 = ckdtree_fabs(r1);
1022 return r1;
1023 }

1025 static inline double wrap_position(const double x, const double boxsize) {
1026 if (boxsize <= 0) {
1027 return x;
1028 }
1029 const double r = std::floor(x / boxsize);
1030 double x1 = x - r * boxsize;
1031 /* ensure result is within the box. */
1032 while (x1 >= boxsize) {
1033 x1 -= boxsize;
1034 }
1035 while (x1 < 0) {
1036 x1 += boxsize;
1037 }
1038 return x1;
1039 }
1040
1041 static inline double side_distance_from_min_max(
1042 const ckdtree<Dim>* tree,
1043 const double x,
1044 const double min,
1045 const double max,
1046 const ckdtree_intp_t k) {
1047 double s, t, tmin, tmax;
1048 double fb = tree->raw_boxsize_data[k];
1049 double hb = tree->raw_boxsize_data[k + tree->m];
1050
1051 if (fb <= 0) {
1052 /* non-periodic dimension */
1053 s = PlainDist1D<Dim>::side_distance_from_min_max(tree, x, min, max, k);
1054 return s;
1055 }
1056
1057 /* periodic */
1058 s = 0;
1059 tmax = x - max;
1060 tmin = x - min;
1061 /* is the test point in this range */
1062 if (CKDTREE_LIKELY(tmax < 0 && tmin > 0)) {
1063 /* yes. min distance is 0 */
1064 return 0;
1065 }
1066
1067 /* no */
1068 tmax = ckdtree_fabs(tmax);
1069 tmin = ckdtree_fabs(tmin);
1070
1071 /* make tmin the closer edge */
1072 if (tmin > tmax) {
1073 t = tmin;
1074 tmin = tmax;
1075 tmax = t;
1076 }
1077
1078 /* both edges are less than half a box. */
1079 /* no wrapping, use the closer edge */
1080 if (tmax < hb) {
1081 return tmin;
1082 }
1083
1084 /* both edge are more than half a box. */
1085 /* wrapping on both edge, use the
1086 * wrapped further edge */
1087 if (tmin > hb) {
1088 return fb - tmax;
1089 }
1090
1091 /* the further side is wrapped */
1092 tmax = fb - tmax;
1093 if (tmin > tmax) {
1094 return tmax;
1095 }
1096 return tmin;
1097 }
1098
1099 private:
1100 static inline double wrap_distance(
1101 const double x, const double hb, const double fb) {
1102 double x1;
1103 if (CKDTREE_UNLIKELY(x < -hb)) {
1104 x1 = fb + x;
1105 } else if (CKDTREE_UNLIKELY(x > hb)) {
1106 x1 = x - fb;
1107 } else {
1108 x1 = x;
1109 }
1110 return x1;
1111 }
1112};

1115 * Priority queue
1116 * ==============
1117 */
1118
1119union heapcontents {
1120 ckdtree_intp_t intdata;
1121 void* ptrdata;
1122};
1123
1124struct heapitem {
1125 double priority;
1126 heapcontents contents;
1127};
1128
1129struct heap {
1130 std::vector<heapitem> _heap;
1131 ckdtree_intp_t n;
1132 ckdtree_intp_t space;
1133
1134 explicit heap(ckdtree_intp_t initial_size) : _heap(initial_size) {
1135 space = initial_size;
1136 n = 0;
1137 }
1138
1139 inline void push(heapitem& item) {
1140 ckdtree_intp_t i;
1141 heapitem t;
1142 n++;
1143
1144 if (n > space) {
1145 _heap.resize(2 * space + 1);
1146 }
1147 space = _heap.size();
1148
1149 i = n - 1;
1150 _heap[i] = item;
1151 while ((i > 0) && (_heap[i].priority < _heap[(i - 1) / 2].priority)) {
1152 t = _heap[(i - 1) / 2];
1153 _heap[(i - 1) / 2] = _heap[i];
1154 _heap[i] = t;
1155 i = (i - 1) / 2;
1156 }
1157 }
1158
1159 inline heapitem peek() { return _heap[0]; }
1160
1161 inline void remove() {
1162 heapitem t;
1163 ckdtree_intp_t i, j, k, l, nn;
1164 _heap[0] = _heap[n - 1];
1165 n--;
1166 /*
1167 * No point in freeing up space as the heap empties.
1168 * The whole heap gets deallocated at the end of any
1169 * query below. Just keep the space to avoid unnecessary
1170 * reallocs.
1171 */
1172 nn = n;
1173 i = 0;
1174 j = 1;
1175 k = 2;
1176 while (((j < nn) && (_heap[i].priority > _heap[j].priority)) ||
1177 ((k < nn) && (_heap[i].priority > _heap[k].priority))) {
1178 if ((k < nn) && (_heap[j].priority > _heap[k].priority)) {
1179 l = k;
1180 } else {
1181 l = j;
1182 }
1183 t = _heap[l];
1184 _heap[l] = _heap[i];
1185 _heap[i] = t;
1186 i = l;
1187 j = 2 * i + 1;
1188 k = 2 * i + 2;
1189 }
1190 }
1191
1192 inline heapitem pop() {
1193 heapitem it = _heap[0];
1194 remove();
1195 return it;
1196 }
1197};

1267/* k-nearest neighbor search for a single point x */
1268template <
1269 typename MinMaxDist,
1270 ckdtree_intp_t Dim,
1271 double p,
1272 bool Periodic = false>
1273static void query_single_point(
1274 const ckdtree<Dim>* self,
1275 double* result_distances,
1276 ckdtree_intp_t* result_indices,
1277 const double* x,
1278 const ckdtree_intp_t* k,
1279 const ckdtree_intp_t nk,
1280 const ckdtree_intp_t kmax,
1281 const double eps,
1282 double distance_upper_bound) {
1283 /* memory pool to allocate and automatically reclaim nodeinfo structs */
1284 nodeinfo_pool<Dim> nipool;
1285
1286 /*
1287 * priority queue for chasing nodes
1288 * entries are:
1289 * - minimum distance between the cell and the target
1290 * - distances between the nearest side of the cell and the target
1291 * the head node of the cell
1292 */
1293 heap q(12);
1294
1295 /*
1296 * priority queue for chasing nodes
1297 * entries are:
1298 * - minimum distance between the cell and the target
1299 * - distances between the nearest side of the cell and the target
1300 * the head node of the cell
1301 */
1302 heap neighbors(kmax);
1303
1304 ckdtree_intp_t i;
1305 constexpr ckdtree_intp_t m = self->m;
1306 nodeinfo<Dim>* ni1;
1307 nodeinfo<Dim>* ni2;
1308 double d;
1309 double epsfac;
1310 heapitem it, it2, neighbor;
1311 const ckdtreenode* node;
1312 const ckdtreenode* inode;
1313
1314 /* set up first nodeifo */
1315 ni1 = nipool.allocate();
1316 ni1->node = self->ctree;
1317
1318 /* initialize first node, update min_distance */
1319 ni1->min_distance = 0;
1320
1321 for (i = 0; i < m; ++i) {
1322 ni1->mins()[i] = self->raw_mins[i];
1323 ni1->maxes()[i] = self->raw_maxes[i];
1324
1325 double side_distance;
1326 if (Periodic) {
1327 side_distance = BoxDist1D<Dim>::side_distance_from_min_max(
1328 self, x[i], self->raw_mins[i], self->raw_maxes[i], i);
1329 } else {
1330 side_distance = PlainDist1D<Dim>::side_distance_from_min_max(
1331 self, x[i], self->raw_mins[i], self->raw_maxes[i], i);
1332 }
1333 side_distance = MinMaxDist::distance_p(side_distance, p);
1334
1335 ni1->side_distances()[i] = 0;
1336 ni1->update_side_distance(i, side_distance, p);
1337 }
1338
1339 /* fiddle approximation factor */
1340 if constexpr (CKDTREE_LIKELY(p == 2.0)) {
1341 double tmp = 1. + eps;
1342 epsfac = 1. / (tmp * tmp);
1343 } else if (eps == 0.) {
1344 epsfac = 1.;
1345 } else if constexpr (std::isinf(p)) {
1346 epsfac = 1. / (1. + eps);
1347 } else {
1348 epsfac = 1. / std::pow((1. + eps), p);
1349 }
1350
1351 /* internally we represent all distances as distance**p */
1352 if constexpr (CKDTREE_LIKELY(p == 2.0)) {
1353 double tmp = distance_upper_bound;
1354 distance_upper_bound = tmp * tmp;
1355 } else if ((!std::isinf(p)) && (!std::isinf(distance_upper_bound))) {
1356 distance_upper_bound = std::pow(distance_upper_bound, p);
1357 }
1358
1359 for (;;) {
1360 if (ni1->node->split_dim == -1) {
1361 node = ni1->node;
1362
1363 /* brute-force */
1364 {
1365 const ckdtree_intp_t start_idx = node->start_idx;
1366 const ckdtree_intp_t end_idx = node->end_idx;
1367 const double* data = self->raw_data;
1368 const ckdtree_intp_t* indices = self->raw_indices;
1369
1370 CKDTREE_PREFETCH(data + indices[start_idx] * m, 0, m);
1371 if (start_idx < end_idx - 1) {
1372 CKDTREE_PREFETCH(data + indices[start_idx + 1] * m, 0, m);
1373 }
1374
1375 for (i = start_idx; i < end_idx; ++i) {
1376 if (i < end_idx - 2) {
1377 CKDTREE_PREFETCH(data + indices[i + 2] * m, 0, m);
1378 }
1379
1380 d = MinMaxDist::point_point_p(
1381 self, data + indices[i] * m, x, p, m, distance_upper_bound);
1382 if (d < distance_upper_bound) {
1383 /* replace furthest neighbor */
1384 if (neighbors.n == kmax) {
1385 neighbors.remove();
1386 }
1387 neighbor.priority = -d;
1388 neighbor.contents.intdata = indices[i];
1389 neighbors.push(neighbor);
1390
1391 /* adjust upper bound for efficiency */
1392 if (neighbors.n == kmax) {
1393 distance_upper_bound = -neighbors.peek().priority;
1394 }
1395 }
1396 }
1397 }
1398 /* done with this node, get another */
1399 if (q.n == 0) {
1400 /* no more nodes to visit */
1401 break;
1402 } else {
1403 it = q.pop();
1404 ni1 = reinterpret_cast<nodeinfo<Dim>*>(it.contents.ptrdata);
1405 }
1406
1407 } else {
1408 inode = ni1->node;
1409 const ckdtree_intp_t split_dim = inode->split_dim;
1410 const double split = inode->split;
1411
1412 /*
1413 * we don't push cells that are too far onto the queue at all,
1414 * but since the distance_upper_bound decreases, we might get
1415 * here even if the cell's too far
1416 */
1417 if (ni1->min_distance > distance_upper_bound * epsfac) {
1418 /* since this is the nearest cell, we're done, bail out */
1419 break;
1420 }
1421 // set up children for searching
1422 // ni2 will be pushed to the queue
1423
1424 ni2 = nipool.allocate();
1425
1426 if constexpr (!Periodic) {
1427 /*
1428 * non periodic : the 'near' node is know from the
1429 * relative distance to the split, and
1430 * has the same distance as the parent node.
1431 *
1432 * we set ni1 to 'near', and set ni2 to 'far'.
1433 * we only recalculate the distance of 'far' later.
1434 *
1435 * This code branch doesn't use min and max.
1436 */
1437 ni2->init_plain(ni1);
1438
1439 double side_distance;
1440
1441 if (x[split_dim] < split) {
1442 ni1->node = inode->less;
1443 ni2->node = inode->greater;
1444 side_distance = split - x[split_dim];
1445 } else {
1446 ni1->node = inode->greater;
1447 ni2->node = inode->less;
1448 side_distance = x[split_dim] - split;
1449 }
1450
1451 side_distance = MinMaxDist::distance_p(side_distance, p);
1452
1453 ni2->update_side_distance(split_dim, side_distance, p);
1454 } else {
1455 /*
1456 * for periodic queries, we do not know which node is closer.
1457 * thus re-calculate ni1 and ni2 both.
1458 *
1459 * this branch we need to keep track of mins and maxes;
1460 */
1461 ni2->init_box(ni1);
1462
1463 double side_distance;
1464
1465 ni1->maxes()[split_dim] = split;
1466 ni1->node = inode->less;
1467
1468 side_distance = BoxDist1D<Dim>::side_distance_from_min_max(
1469 self, x[split_dim], ni1->mins()[split_dim], ni1->maxes()[split_dim],
1470 split_dim);
1471 side_distance = MinMaxDist::distance_p(side_distance, p);
1472
1473 ni1->update_side_distance(split_dim, side_distance, p);
1474
1475 ni2->mins()[split_dim] = split;
1476 ni2->node = inode->greater;
1477
1478 side_distance = BoxDist1D<Dim>::side_distance_from_min_max(
1479 self, x[split_dim], ni2->mins()[split_dim], ni2->maxes()[split_dim],
1480 split_dim);
1481 side_distance = MinMaxDist::distance_p(side_distance, p);
1482
1483 ni2->update_side_distance(split_dim, side_distance, p);
1484 }
1485
1486 /* Ensure ni1 is closer than ni2 */
1487 if (ni1->min_distance > ni2->min_distance) {
1488 {
1489 nodeinfo<Dim>* tmp;
1490 tmp = ni1;
1491 ni1 = ni2;
1492 ni2 = tmp;
1493 }
1494 }
1495
1496 /*
1497 * near child is at the same or closer than the distance as the current
1498 * node we're going here next, so no point pushing it on the queue no need
1499 * to recompute the distance or the side_distances
1500 */
1501
1502 /*
1503 * far child can be further
1504 * push it on the queue if it's near enough
1505 */
1506
1507 if (ni2->min_distance <= distance_upper_bound * epsfac) {
1508 it2.priority = ni2->min_distance;
1509 it2.contents.ptrdata = reinterpret_cast<void*>(ni2);
1510 q.push(it2);
1511 }
1512 }
1513 }
1514
1515 /* heapsort */
1516 std::vector<heapitem> sorted_neighbors(kmax);
1517 ckdtree_intp_t nnb = neighbors.n;
1518 for (i = neighbors.n - 1; i >= 0; --i) {
1519 sorted_neighbors[i] = neighbors.pop();
1520 }
1521
1522 /* fill output arrays with sorted neighbors */
1523 for (i = 0; i < nk; ++i) {
1524 if (CKDTREE_UNLIKELY(k[i] - 1 >= nnb)) {
1525 result_indices[i] = self->n;
1526 result_distances[i] = std::numeric_limits<double>::infinity();
1527 } else {
1528 neighbor = sorted_neighbors[k[i] - 1];
1529 result_indices[i] = neighbor.contents.intdata;
1530 if constexpr (CKDTREE_LIKELY(p == 2.0)) {
1531 result_distances[i] = std::sqrt(-neighbor.priority);
1532 } else if constexpr ((p == 1.) || (std::isinf(p))) {
1533 result_distances[i] = -neighbor.priority;
1534 } else {
1535 result_distances[i] = std::pow((-neighbor.priority), (1. / p));
1536 }
1537 }
1538 }
1539}

1543template <ckdtree_intp_t Dim, double p, bool Periodic = false>
1544int query_knn(
1545 const ckdtree<Dim>* self,
1546 double* dd,
1547 ckdtree_intp_t* ii,
1548 const double* xx,
1549 const ckdtree_intp_t n,
1550 const ckdtree_intp_t* k,
1551 const ckdtree_intp_t nk,
1552 const ckdtree_intp_t kmax,
1553 const double eps,
1554 const double distance_upper_bound) {
1555#define HANDLE_KNN(cond, kls) \
1556 if constexpr (cond) { \
1557 query_single_point<kls, Dim, p, Periodic>( \
1558 self, dd_row, ii_row, xx_row, k, nk, kmax, eps, distance_upper_bound); \
1559 } else
1560
1561 constexpr ckdtree_intp_t m = self->m;
1562 ckdtree_intp_t i;
1563
1564 using MinkowskiDistP2 = BaseMinkowskiDistP2<PlainDist1D<Dim>, Dim>;
1565 using MinkowskiDistP1 = BaseMinkowskiDistP1<PlainDist1D<Dim>, Dim>;
1566 using MinkowskiDistPinf = BaseMinkowskiDistPinf<PlainDist1D<Dim>, Dim>;
1567 using MinkowskiDistPp = BaseMinkowskiDistPp<PlainDist1D<Dim>, Dim>;
1568
1569 using BoxMinkowskiDistP2 = BaseMinkowskiDistP2<BoxDist1D<Dim>, Dim>;
1570 using BoxMinkowskiDistP1 = BaseMinkowskiDistP1<BoxDist1D<Dim>, Dim>;
1571 using BoxMinkowskiDistPinf = BaseMinkowskiDistPinf<BoxDist1D<Dim>, Dim>;
1572 using BoxMinkowskiDistPp = BaseMinkowskiDistPp<BoxDist1D<Dim>, Dim>;
1573
1574 if constexpr (!Periodic) {
1575 for (i = 0; i < n; ++i) {
1576 double* dd_row = dd + (i * nk);
1577 ckdtree_intp_t* ii_row = ii + (i * nk);
1578 const double* xx_row = xx + (i * m);
1579 HANDLE_KNN(p == 2.0, MinkowskiDistP2)
1580 HANDLE_KNN(p == 1.0, MinkowskiDistP1)
1581 HANDLE_KNN(std::isinf(p), MinkowskiDistPinf)
1582 HANDLE_KNN(1, MinkowskiDistPp) {}
1583 }
1584 } else {
1585 std::array<double, m> row = {0};
1586 double* xx_row = &row[0];
1587 int j;
1588 for (i = 0; i < n; ++i) {
1589 double* dd_row = dd + (i * nk);
1590 ckdtree_intp_t* ii_row = ii + (i * nk);
1591 const double* old_xx_row = xx + (i * m);
1592 for (j = 0; j < m; ++j) {
1593 xx_row[j] = BoxDist1D<Dim>::wrap_position(
1594 old_xx_row[j], self->raw_boxsize_data[j]);
1595 }
1596 HANDLE_KNN(p == 2.0, BoxMinkowskiDistP2)
1597 HANDLE_KNN(p == 1.0, BoxMinkowskiDistP1)
1598 HANDLE_KNN(std::isinf(p), BoxMinkowskiDistPinf)
1599 HANDLE_KNN(1, BoxMinkowskiDistPp) {}
1600 }
1601 }
1602 return 0;
1603}

1636template <ckdtree_intp_t Dim, typename MinMaxDist>
1637static void traverse_checking_query_ball_point(
1638 const ckdtree<Dim>* self,
1639 const int return_length,
1640 std::vector<ckdtree_intp_t>& results,
1641 const ckdtreenode* node,
1642 RectRectDistanceTracker<MinMaxDist, Dim>* tracker) {
1643 const ckdtreenode* lnode;
1644 double d;
1645 ckdtree_intp_t i;
1646
1647 if (tracker->min_distance > tracker->upper_bound * tracker->epsfac) {
1648 return;
1649 } else if (tracker->max_distance < tracker->upper_bound / tracker->epsfac) {
1650 traverse_no_checking_query_ball_point(self, return_length, results, node);
1651 } else if (node->split_dim == -1) { /* leaf node */
1652 /* brute-force */
1653 lnode = node;
1654 const double p = tracker->p;
1655 const double tub = tracker->upper_bound;
1656 const double* tpt = tracker->rect1.mins();
1657 const double* data = self->raw_data;
1658 const ckdtree_intp_t* indices = self->raw_indices;
1659 constexpr ckdtree_intp_t m = self->m;
1660 const ckdtree_intp_t start = lnode->start_idx;
1661 const ckdtree_intp_t end = lnode->end_idx;
1662
1663 CKDTREE_PREFETCH(data + indices[start] * m, 0, m);
1664 if (start < end - 1) {
1665 CKDTREE_PREFETCH(data + indices[start + 1] * m, 0, m);
1666 }
1667
1668 for (i = start; i < end; ++i) {
1669 if (i < end - 2) {
1670 CKDTREE_PREFETCH(data + indices[i + 2] * m, 0, m);
1671 }
1672
1673 d = MinMaxDist::point_point_p(
1674 self, data + indices[i] * m, tpt, p, m, tub);
1675
1676 if (d <= tub) {
1677 if (return_length) {
1678 results[0]++;
1679 } else {
1680 results.push_back((ckdtree_intp_t)indices[i]);
1681 }
1682 }
1683 }
1684 } else {
1685 tracker->push_less_of(2, node);
1686 traverse_checking_query_ball_point(
1687 self, return_length, results, node->less, tracker);
1688 tracker->pop();
1689
1690 tracker->push_greater_of(2, node);
1691 traverse_checking_query_ball_point(
1692 self, return_length, results, node->greater, tracker);
1693 tracker->pop();
1694 }
1695}

1697template <ckdtree_intp_t Dim, double p, bool Periodic = false>
1698int query_ball_point(
1699 const ckdtree<Dim>* self,
1700 const double* x,
1701 const double* r,
1702 const double eps,
1703 const ckdtree_intp_t n_queries,
1704 std::vector<ckdtree_intp_t>* results,
1705 const bool return_length,
1706 const bool sort_output) {
1707#define HANDLE_QBP(cond, kls) \
1708 if constexpr (cond) { \
1709 if (return_length) \
1710 results[i].push_back(0); \
1711 RectRectDistanceTracker<kls, Dim> tracker( \
1712 self, point, rect, p, eps, r[i]); \
1713 traverse_checking_query_ball_point( \
1714 self, return_length, results[i], self->ctree, &tracker); \
1715 } else
1716
1717 using MinkowskiDistP2 = BaseMinkowskiDistP2<PlainDist1D<Dim>, Dim>;
1718 using MinkowskiDistP1 = BaseMinkowskiDistP1<PlainDist1D<Dim>, Dim>;
1719 using MinkowskiDistPinf = BaseMinkowskiDistPinf<PlainDist1D<Dim>, Dim>;
1720 using MinkowskiDistPp = BaseMinkowskiDistPp<PlainDist1D<Dim>, Dim>;
1721
1722 using BoxMinkowskiDistP2 = BaseMinkowskiDistP2<BoxDist1D<Dim>, Dim>;
1723 using BoxMinkowskiDistP1 = BaseMinkowskiDistP1<BoxDist1D<Dim>, Dim>;
1724 using BoxMinkowskiDistPinf = BaseMinkowskiDistPinf<BoxDist1D<Dim>, Dim>;
1725 using BoxMinkowskiDistPp = BaseMinkowskiDistPp<BoxDist1D<Dim>, Dim>;
1726
1727 for (ckdtree_intp_t i = 0; i < n_queries; ++i) {
1728 constexpr ckdtree_intp_t m = self->m;
1729 Rectangle<Dim> rect(self->raw_mins, self->raw_maxes);
1730 if constexpr (!Periodic) {
1731 Rectangle<Dim> point(x + i * m, x + i * m);
1732 HANDLE_QBP(p == 2.0, MinkowskiDistP2)
1733 HANDLE_QBP(p == 1.0, MinkowskiDistP1)
1734 HANDLE_QBP(std::isinf(p), MinkowskiDistPinf)
1735 HANDLE_QBP(1, MinkowskiDistPp) {}
1736 } else {
1737 Rectangle<Dim> point(x + i * m, x + i * m);
1738 int j;
1739 for (j = 0; j < m; ++j) {
1740 point.maxes()[j] = point.mins()[j] = BoxDist1D<Dim>::wrap_position(
1741 point.mins()[j], self->raw_boxsize_data[j]);
1742 }
1743 HANDLE_QBP(p == 2.0, BoxMinkowskiDistP2)
1744 HANDLE_QBP(p == 1.0, BoxMinkowskiDistP1)
1745 HANDLE_QBP(std::isinf(p), BoxMinkowskiDistPinf)
1746 HANDLE_QBP(1, BoxMinkowskiDistPp) {}
1747 }
1748
1749 if (!return_length && sort_output) {
1750 std::sort(results[i].begin(), results[i].end());
1751 }
1752 }
1753 return 0;
1754}

1758template <ckdtree_intp_t Dim>
1759static void traverse_no_checking_query_pairs(
1760 const ckdtree<Dim>* self,
1761 std::vector<ordered_pair>* results,
1762 const ckdtreenode* node1,
1763 const ckdtreenode* node2) {
1764 const ckdtreenode* lnode1;
1765 const ckdtreenode* lnode2;
1766 ckdtree_intp_t i, j, min_j;
1767 const ckdtree_intp_t* indices = self->raw_indices;
1768
1769 if (node1->split_dim == -1) { /* leaf node */
1770 lnode1 = node1;
1771
1772 if (node2->split_dim == -1) { /* leaf node */
1773 lnode2 = node2;
1774
1775 const ckdtree_intp_t start1 = lnode1->start_idx;
1776 const ckdtree_intp_t start2 = lnode2->start_idx;
1777 const ckdtree_intp_t end1 = lnode1->end_idx;
1778 const ckdtree_intp_t end2 = lnode2->end_idx;
1779
1780 for (i = start1; i < end1; ++i) {
1781 /* Special care here to avoid duplicate pairs */
1782 if (node1 == node2) {
1783 min_j = i + 1;
1784 } else {
1785 min_j = start2;
1786 }
1787
1788 for (j = min_j; j < end2; ++j) {
1789 add_ordered_pair(results, indices[i], indices[j]);
1790 }
1791 }
1792 } else {
1793 traverse_no_checking_query_pairs(self, results, node1, node2->less);
1794 traverse_no_checking_query_pairs(self, results, node1, node2->greater);
1795 }
1796 } else {
1797 if (node1 == node2) {
1798 /*
1799 * Avoid traversing (node1->less, node2->greater) and
1800 * (node1->greater, node2->less) (it's the same node pair twice
1801 * over, which is the source of the complication in the
1802 * original KDTree.query_pairs)
1803 */
1804 traverse_no_checking_query_pairs(self, results, node1->less, node2->less);
1805 traverse_no_checking_query_pairs(
1806 self, results, node1->less, node2->greater);
1807 traverse_no_checking_query_pairs(
1808 self, results, node1->greater, node2->greater);
1809 } else {
1810 traverse_no_checking_query_pairs(self, results, node1->less, node2);
1811 traverse_no_checking_query_pairs(self, results, node1->greater, node2);
1812 }
1813 }

1816template <ckdtree_intp_t Dim, typename MinMaxDist>
1817static void traverse_checking_query_pairs(
1818 const ckdtree<Dim>* self,
1819 std::vector<ordered_pair>* results,
1820 const ckdtreenode* node1,
1821 const ckdtreenode* node2,
1822 RectRectDistanceTracker<MinMaxDist, Dim>* tracker) {
1823 const ckdtreenode* lnode1;
1824 const ckdtreenode* lnode2;
1825 double d;
1826 ckdtree_intp_t i, j, min_j;
1827
1828 if (tracker->min_distance > tracker->upper_bound * tracker->epsfac) {
1829 return;
1830 } else if (tracker->max_distance < tracker->upper_bound / tracker->epsfac) {
1831 traverse_no_checking_query_pairs(self, results, node1, node2);
1832 } else if (node1->split_dim == -1) { /* 1 is leaf node */
1833 lnode1 = node1;
1834
1835 if (node2->split_dim == -1) { /* 1 & 2 are leaves */
1836 lnode2 = node2;
1837
1838 /* brute-force */
1839 const double p = tracker->p;
1840 const double tub = tracker->upper_bound;
1841 const double* data = self->raw_data;
1842 const double epsfac = tracker->epsfac;
1843 const ckdtree_intp_t* indices = self->raw_indices;
1844 constexpr ckdtree_intp_t m = self->m;
1845 const ckdtree_intp_t start1 = lnode1->start_idx;
1846 const ckdtree_intp_t start2 = lnode2->start_idx;
1847 const ckdtree_intp_t end1 = lnode1->end_idx;
1848 const ckdtree_intp_t end2 = lnode2->end_idx;
1849
1850 CKDTREE_PREFETCH(data + indices[start1] * m, 0, m);
1851 if (start1 < end1 - 1) {
1852 CKDTREE_PREFETCH(data + indices[start1 + 1] * m, 0, m);
1853 }
1854
1855 for (i = start1; i < end1; ++i) {
1856 if (i < end1 - 2) {
1857 CKDTREE_PREFETCH(data + indices[i + 2] * m, 0, m);
1858 }
1859
1860 /* Special care here to avoid duplicate pairs */
1861 if (node1 == node2) {
1862 min_j = i + 1;
1863 } else {
1864 min_j = start2;
1865 }
1866
1867 if (min_j < end2) {
1868 CKDTREE_PREFETCH(data + indices[min_j] * m, 0, m);
1869 }
1870 if (min_j < end2 - 1) {
1871 CKDTREE_PREFETCH(data + indices[min_j + 1] * m, 0, m);
1872 }
1873
1874 for (j = min_j; j < end2; ++j) {
1875 if (j < end2 - 2) {
1876 CKDTREE_PREFETCH(data + indices[j + 2] * m, 0, m);
1877 }
1878
1879 d = MinMaxDist::point_point_p(
1880 self, data + indices[i] * m, data + indices[j] * m, p, m, tub);
1881
1882 if (d <= tub / epsfac) {
1883 add_ordered_pair(results, indices[i], indices[j]);
1884 }
1885 }
1886 }
1887 } else { /* 1 is a leaf node, 2 is inner node */
1888 tracker->push_less_of(2, node2);
1889 traverse_checking_query_pairs(self, results, node1, node2->less, tracker);
1890 tracker->pop();
1891
1892 tracker->push_greater_of(2, node2);
1893 traverse_checking_query_pairs(
1894 self, results, node1, node2->greater, tracker);
1895 tracker->pop();
1896 }
1897 } else { /* 1 is an inner node */
1898 if (node2->split_dim == -1) { /* 1 is an inner node, 2 is a leaf node */
1899 tracker->push_less_of(1, node1);
1900 traverse_checking_query_pairs(self, results, node1->less, node2, tracker);
1901 tracker->pop();
1902
1903 tracker->push_greater_of(1, node1);
1904 traverse_checking_query_pairs(
1905 self, results, node1->greater, node2, tracker);
1906 tracker->pop();
1907 } else { /* 1 and 2 are inner nodes */
1908 tracker->push_less_of(1, node1);
1909 tracker->push_less_of(2, node2);
1910 traverse_checking_query_pairs(
1911 self, results, node1->less, node2->less, tracker);
1912 tracker->pop();
1913
1914 tracker->push_greater_of(2, node2);
1915 traverse_checking_query_pairs(
1916 self, results, node1->less, node2->greater, tracker);
1917 tracker->pop();
1918 tracker->pop();
1919
1920 tracker->push_greater_of(1, node1);
1921 if (node1 != node2) {
1922 /*
1923 * Avoid traversing (node1->less, node2->greater) and
1924 * (node1->greater, node2->less) (it's the same node pair
1925 * twice over, which is the source of the complication in
1926 * the original KDTree.query_pairs)
1927 */
1928 tracker->push_less_of(2, node2);
1929 traverse_checking_query_pairs(
1930 self, results, node1->greater, node2->less, tracker);
1931 tracker->pop();
1932 }
1933 tracker->push_greater_of(2, node2);
1934 traverse_checking_query_pairs(
1935 self, results, node1->greater, node2->greater, tracker);
1936 tracker->pop();
1937 tracker->pop();
1938 }
1939 }
