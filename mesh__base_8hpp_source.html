73template <
74 typename FloatType,
75 typename IndexType,
76 IndexType OuterDim,
77 IndexType SelfDim,
78 typename Derived>
79class MeshBase {
80 static_assert(SelfDim > 0 && SelfDim <= 3);
81 static_assert(OuterDim > 0 && OuterDim <= 3);
82
83 public:
84 static constexpr IndexType outer_dim = OuterDim;
85 static constexpr IndexType self_dim = SelfDim;
86 static constexpr IndexType vtx_per_face = self_dim <= 2 ? self_dim : -1;
87 static constexpr bool is_manifold = !(outer_dim == self_dim);
88 using index_t = IndexType;
89 using SelfPoint = Point<FloatType, SelfDim>;
90 using DualMap = datastructure::CrsMap<IndexType, IndexType>;
91 using CellToFacetMap = datastructure::CrsMap<IndexType, IndexType>;
92 using CellToVtxMap = datastructure::CrsMap<IndexType, IndexType>;
93 using CellToVtxMapUncompressed = datastructure::CrsMapUncompressed<IndexType>;
94 using FacetToCellMap = std::conditional_t<
95 is_manifold,
96 datastructure::CrsMap<IndexType, IndexType>,
97 std::vector<std::array<IndexType, 2>>>;
98
99 using FacetToVtxMap =
100 datastructure::CrsMap<IndexType, IndexType, vtx_per_face>;
101 using VtxToCellMap = datastructure::CrsMap<IndexType, IndexType>;
102
103 using EdgeVtxs = std::array<IndexType, 2>;
104 using EdgeToVtxMap = std::vector<EdgeVtxs>;
105 using CellToEdgeMap = datastructure::CrsMap<IndexType, IndexType>;
106 using EdgeToCellMap = datastructure::CrsMap<IndexType, IndexType>;
107
108 std::vector<SelfPoint>& getVertexes() { return m_vertexes; }
109 const std::vector<SelfPoint>& getVertexes() const { return m_vertexes; }
110
111 DualMap& getDualMap() { return m_dual_graph; }
112 const DualMap& getDualMap() const { return m_dual_graph; }
113
114 CellToFacetMap& getCellToFacetMap() { return m_cell_to_facet; }

168 friend bool operator==(const MeshBase& lhs, const MeshBase& rhs) {
169 return (lhs.m_vertexes == rhs.m_vertexes) &&
170 (lhs.m_dual_graph == rhs.m_dual_graph) &&
171 (lhs.m_cell_to_facet == rhs.m_cell_to_facet) &&
172 (lhs.m_cell_to_vtx == rhs.m_cell_to_vtx) &&
173 (lhs.m_facet_to_cell == rhs.m_facet_to_cell) &&
174 (lhs.m_facet_to_vtx == rhs.m_facet_to_vtx) &&
175 (lhs.m_vtx_to_cell == rhs.m_vtx_to_cell);
176 }
177
178 protected:
179 // necessitates m_cell_to_vtx
180 void buildVtxToCell() {
181 datastructure::CrsMapUncompressed<IndexType> vtx_to_cell;
182 for (IndexType cell_id = 0; cell_id < m_cell_to_vtx.size(); ++cell_id) {
183 const IndexType n_vtxs = m_cell_to_vtx.getRow(cell_id).size();
184 for (IndexType vtx_id = 0; vtx_id < n_vtxs; ++vtx_id) {
185 vtx_to_cell.getRow(m_cell_to_vtx(cell_id, vtx_id)).push_back(cell_id);
186 }
187 }
188 m_vtx_to_cell = vtx_to_cell.template compress<IndexType>();
189 };
190
191 // necessitates m_facet_to_cell
192 void buildCellToFacet() {
193 datastructure::CrsMapUncompressed<IndexType> cell_to_facet;
194 const IndexType n_facets = m_facet_to_cell.size();
195 for (IndexType facet_id = 0; facet_id < n_facets; ++facet_id) {
196 for (const auto cell_id : m_facet_to_cell[facet_id]) {
197 // if not on the boundary
198 if (cell_id != -1) {
199 cell_to_facet.getRow(cell_id).push_back(facet_id);
200 }
201 }
202 }
203 m_cell_to_facet = cell_to_facet.template compress<IndexType>();
204 };
205
206 // necessitates m_cell_to_facet and m_facet_to_cell
207 void buildDualGraph() {
208 datastructure::CrsMapUncompressed<IndexType> dual_graph;
209 for (IndexType cell_id = 0; cell_id < m_cell_to_facet.size(); ++cell_id) {
210 for (const auto facet_id : m_cell_to_facet.getRow(cell_id)) {
211 for (const auto cell_neigh : m_facet_to_cell[facet_id]) {
212 if (cell_neigh != static_cast<IndexType>(-1) &&
213 cell_neigh != cell_id) {
214 dual_graph.getRow(cell_id).push_back(cell_neigh);
215 }
216 }
217 }
218 }
219 m_dual_graph = dual_graph.template compress<IndexType>();
220 }
221
222 std::vector<SelfPoint> m_vertexes;
223 // strong dual graph (a facet must be in common)
224 // the graph is stored as the list of neighbors for each vertex
225 DualMap m_dual_graph;
226 CellToFacetMap m_cell_to_facet;
227 CellToVtxMap m_cell_to_vtx;
228 FacetToCellMap m_facet_to_cell;
229 // in 1D facets are isomophic to vertices
230 FacetToVtxMap m_facet_to_vtx;
231 VtxToCellMap m_vtx_to_cell;
232};
