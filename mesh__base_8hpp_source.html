124template <typename VecType>
125void checkEmptySets(const VecType& labels) {
126 std::vector<bool> class_found;
127 for (size_t i = 0; i < labels.size(); i++) {
128 if (class_found.size() <= labels[i]) {
129 class_found.resize(labels[i] + 1, false);
130 }
131 class_found[labels[i]] = true;
132 }
133 for (size_t i = 0; i < class_found.size(); i++) {
134 if (!class_found[i]) {
135 std::cerr << "WARNING: The class " << i << " of " << class_found.size()
136 << " is empty!" << std::endl;
137 }
138 }
139}

156template <typename index_t>
157datastructure::CrsMap<index_t, index_t> buildSubDual(
158 const datastructure::CrsMap<index_t, index_t>& dual_map,
159 const std::vector<index_t>& subset_idxs) {
160 using IndMap = datastructure::CrsMap<index_t, index_t>;
161 using IndVec = std::vector<index_t>;
162
163 // sort subset_idxs
164 IndVec tmp;
165 const IndVec* subset_idxs_sorted = &subset_idxs;
166 if (!std::is_sorted(subset_idxs.begin(), subset_idxs.end())) {
167 tmp = subset_idxs;
168 std::sort(tmp.begin(), tmp.end());
169 subset_idxs_sorted = &tmp;
170 }
171
172 // compute the old to new cell id map
173 index_t n_nodes = dual_map.size();
174 IndVec old_to_new(n_nodes, -1);
175 const index_t n_subset_idxs = subset_idxs.size();
176 for (index_t new_id = 0; new_id < n_subset_idxs; ++new_id) {
177 old_to_new[(*subset_idxs_sorted)[new_id]] = new_id;
178 }
179
180 // extract the dual map of the subset directly in compressed form!
181 IndMap sub_dual_map;
182 sub_dual_map.data().resize(0);
183 sub_dual_map.indices().resize(0);
184 sub_dual_map.indices().push_back(0);
185 for (index_t ncid = 0; ncid < n_subset_idxs; ++ncid) {
186 const auto ocid = (*subset_idxs_sorted)[ncid];
187 for (const auto neigh_ocid : dual_map[ocid]) {
188 const auto it = std::lower_bound(
189 (*subset_idxs_sorted).begin(), (*subset_idxs_sorted).end(),
190 neigh_ocid);
191 if (it != (*subset_idxs_sorted).end() && (*it) == neigh_ocid) {
192 sub_dual_map.data().push_back(old_to_new[neigh_ocid]);
193 }
194 }
195 sub_dual_map.indices().push_back(sub_dual_map.data().size());
196 }
197 return sub_dual_map;
198};

215template <typename index_t>
216index_t removeEmptySets(std::vector<index_t>& labels) {
217 constexpr auto npos = static_cast<index_t>(-1);
218 if (labels.empty()) {
219 return 0;
220 }
221 // NOTE: we need much less memory than labels.size()
222 // we assume that few labels are missing
223 const index_t max_label_id = *std::max_element(labels.begin(), labels.end());
224 std::vector<index_t> old_to_new(max_label_id + 1, npos);
225 index_t new_id = 0;
226 for (std::size_t i = 0; i < labels.size(); i++) {
227 if (old_to_new[labels[i]] == npos) {
228 old_to_new[labels[i]] = new_id++;
229 }
230 labels[i] = old_to_new[labels[i]];
231 }
232 return new_id;
233}

659datastructure::CrsMap<IndexType, IndexType>
660MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
661 invertCellAgglomeration(
662 const std::vector<IndexType>& old_to_new_cell) const {
663 datastructure::CrsMapUncompressed<IndexType, 64> new_to_old_cell_uncompr;
664 for (std::size_t cell_id = 0; cell_id < old_to_new_cell.size(); ++cell_id) {
665 const auto cell_id_new = old_to_new_cell[cell_id];
666 if (cell_id_new == UNDEF) {
667 continue;
668 }
669 new_to_old_cell_uncompr.getRow(cell_id_new).push_back(cell_id);
670 }
671 return new_to_old_cell_uncompr.template compress<IndexType>();
672}

674template <
675 typename FloatType,
676 typename IndexType,
677 IndexType OuterDim,
678 IndexType SelfDim,
679 typename Derived>
680std::vector<IndexType>
681MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
682 buildNewToOldFacetAgglomerationAndCellToFacet(
683 const auto& new_to_old_cell, Derived& agg_mesh) const {
684 std::vector<IndexType> new_to_old_facet;
685 std::vector<IndexType> old_to_new_facet(m_facet_to_cell.size(), -1);
686 datastructure::CrsMapUncompressed<IndexType, 64> new_cell_to_facets_uncpr;
687 new_cell_to_facets_uncpr.resize(new_to_old_cell.size());
688 // NOTE: a flat vector of booleans here performed much worse when tested
689 boost::unordered_flat_map<IndexType, IndexType> facet_table;
690 for (IndexType new_cid = 0; new_cid < new_to_old_cell.size(); ++new_cid) {
691 for (const auto old_cid : new_to_old_cell[new_cid]) {
692 for (const auto old_fid : m_cell_to_facet[old_cid]) {
693 // we only care if the count is only exactly 1
694 // it can be only 0 [face does not belong to cell]
695 // 1 [face is on boundary]
696 // 2 [face is internal] -> not(not(0)) == 0
697 facet_table[old_fid] = !facet_table[old_fid];
698 }
699 }
700 for (auto it = facet_table.begin(); it != facet_table.end(); ++it) {
701 const auto count = it->second;
702 const auto old_fid = it->first;
703 if (count) {
704 IndexType new_fid = old_to_new_facet[old_fid];
705 if (new_fid == UNDEF) {
706 new_fid = new_to_old_facet.size();
707 old_to_new_facet[old_fid] = new_fid;
708 new_to_old_facet.push_back(old_fid);
709 }
710 new_cell_to_facets_uncpr[new_cid].push_back(new_fid);
711 }
712 }
713 facet_table.clear();
714 }
715 agg_mesh.m_cell_to_facet =
716 new_cell_to_facets_uncpr.template compress<IndexType>();
717 return new_to_old_facet;
718}

720template <
721 typename FloatType,
722 typename IndexType,
723 IndexType OuterDim,
724 IndexType SelfDim,
725 typename Derived>
726auto MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
727 buildFacetToCellAndFacetToVtxAgglomeration(
728 const std::vector<IndexType>& new_to_old_facet,
729 const std::vector<IndexType>& old_to_new_cell,
730 Derived& agg_mesh) const {
731 std::vector<IndexType> new_to_old_vtx;
732 std::vector<IndexType> old_to_new_vtx(m_vertexes.size(), UNDEF);
733
734 agg_mesh.m_facet_to_cell.resize(new_to_old_facet.size(), {UNDEF, UNDEF});
735 agg_mesh.m_facet_to_vtx.indices().push_back(0);
736
737 for (std::size_t new_fid = 0; new_fid < new_to_old_facet.size(); ++new_fid) {
738 const auto old_fid = new_to_old_facet[new_fid];
739 // face to cell
740 for (std::size_t d = 0; d < 2; ++d) {
741 const auto old_cid = m_facet_to_cell[old_fid][d];
742 if (old_cid != UNDEF) {
743 const auto new_cid = old_to_new_cell[old_cid];
744 assert(new_cid != UNDEF);
745 agg_mesh.m_facet_to_cell[new_fid][d] = new_cid;
746 }
747 }
748
749 // facet to vertex and vertex to vertex (old/new)
750 for (const auto old_vid : m_facet_to_vtx[old_fid]) {
751 IndexType new_vid = old_to_new_vtx[old_vid];
752 if (new_vid == UNDEF) {
753 new_vid = new_to_old_vtx.size();
754 new_to_old_vtx.push_back(old_vid);
755 old_to_new_vtx[old_vid] = new_vid;
756 }
757 agg_mesh.m_facet_to_vtx.data().push_back(new_vid);
758 }
759 if constexpr (vtx_per_face == -1) {
760 agg_mesh.m_facet_to_vtx.indices().push_back(
761 agg_mesh.m_facet_to_vtx.data().size());
762 }
763 }
764
765 // add vtxs
766 for (std::size_t ovid = 0; ovid < old_to_new_vtx.size(); ++ovid) {
767 const auto nvid = old_to_new_vtx[ovid];
768 if (nvid != UNDEF) {
769 if (agg_mesh.m_vertexes.size() < static_cast<std::size_t>(nvid + 1)) {
770 agg_mesh.m_vertexes.resize(nvid + 1);
771 }
772 agg_mesh.m_vertexes[nvid] = m_vertexes[ovid];
773 }
774 }
775
776 return old_to_new_vtx;
777}

779template <
780 typename FloatType,
781 typename IndexType,
782 IndexType OuterDim,
783 IndexType SelfDim,
784 typename Derived>
785void MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
786 buildCellToVtxAgglomeration(
787 const auto& new_to_old_cell,
788 const auto& old_to_new_vtx,
789 Derived& agg_mesh) const {
790 agg_mesh.m_cell_to_vtx.indices().push_back(0);
791 std::vector<bool> is_vtx_present(old_to_new_vtx.size(), false);
792
793 for (IndexType new_cid = 0; new_cid < new_to_old_cell.size(); ++new_cid) {
794 for (const auto old_cid : new_to_old_cell[new_cid]) {
795 for (const auto old_vid : m_cell_to_vtx[old_cid]) {
796 const auto new_vid = old_to_new_vtx[old_vid];
797 if (new_vid != UNDEF && (!is_vtx_present[old_vid])) {
798 agg_mesh.m_cell_to_vtx.data().push_back(new_vid);
799 is_vtx_present[old_vid] = true;
800 }
801 }
802 }
803 std::fill(is_vtx_present.begin(), is_vtx_present.end(), false);
804 agg_mesh.m_cell_to_vtx.indices().push_back(
805 agg_mesh.m_cell_to_vtx.data().size());
806 }

809template <
810 typename FloatType,
811 typename IndexType,
812 IndexType OuterDim,
813 IndexType SelfDim,
814 typename Derived>
815void MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
816 agglomerateCheckConnected(const datastructure::CrsMap<IndexType, IndexType>&
817 new_to_old_cell) const {
818 // safety check: check each new cell is made by connected components
819 std::vector<IndexType> glob_to_loc(numCells(), UNDEF);
820 IndexType prev_new_cid = UNDEF;
821 for (IndexType new_cid = 0; new_cid < new_to_old_cell.size(); ++new_cid) {
822 if (new_to_old_cell[new_cid].size() <= 1) {
823 continue;
824 }
825 std::vector<IndexType> old_cids(
826 new_to_old_cell[new_cid].begin(), new_to_old_cell[new_cid].end());
827 // overwrite old map entries
828 if (VULPES_UNLIKELY(prev_new_cid != UNDEF)) {
829 for (auto id : new_to_old_cell[prev_new_cid]) {
830 glob_to_loc[id] = UNDEF;
831 }
832 }
833 prev_new_cid = new_cid;
834 // update map
835 for (std::size_t i = 0; i < old_cids.size(); ++i) {
836 glob_to_loc[old_cids[i]] = i;
837 }
838 // dfs to check it's connected
839 std::vector<bool> visited(old_cids.size(), false);
840 std::vector<IndexType> to_visit = {old_cids[0]};
841 while (!to_visit.empty()) {
842 const auto curr = to_visit.back();
843 visited[glob_to_loc[curr]] = true;
844 to_visit.pop_back();
845 for (const auto neigh : m_dual_graph[curr]) {
846 if (glob_to_loc[neigh] == UNDEF) {
847 continue;
848 }
849 if (!visited[glob_to_loc[neigh]]) {
850 to_visit.push_back(neigh);
851 }
852 }
853 }
854 // check all are visited
855 for (std::size_t i = 0; i < visited.size(); ++i) {
856 if (!visited[i]) {
857 std::cout << "ERROR: in new cell " + std::to_string(new_cid) +
858 ", starting from old cell " +
859 std::to_string(old_cids[0]) +
860 " is not possible to reach the old cell " +
861 std::to_string(old_cids[i])
862 << std::endl;
863
864 for (auto cid : {old_cids[0], old_cids[i]}) {
865 SelfPoint avg = SelfPoint::Zero();
866 for (const auto id : m_cell_to_vtx[cid]) {
867 avg += m_vertexes[id];
868 }
869 avg /= m_cell_to_vtx[cid].size();
870 std::cout << "Old cell " << cid << " avg: " << avg.transpose()
871 << std::endl;
872 }
873 std::exit(-1);
874 }
875 }
876 }

887 const std::vector<IndexType>& old_to_new_cell_input,
888 bool check_connected) const {
889 auto old_to_new_cell = old_to_new_cell_input;
890 internal::removeEmptySets(old_to_new_cell);
891 assert(old_to_new_cell.size() == static_cast<std::size_t>(numCells()));
892 Derived agg_mesh;
893
894 // invert old to new cell
895 Profiler::start("invertCellAgglomeration");
896 const auto new_to_old_cell = invertCellAgglomeration(old_to_new_cell);
897 Profiler::stop("invertCellAgglomeration");
898
899 if (check_connected) {
900 Profiler::start("agglomerateCheckConnected");
901 agglomerateCheckConnected(new_to_old_cell);
902 Profiler::stop("agglomerateCheckConnected");
903 }
904 // cell to faces and facet to facet (old/new)
905 Profiler::start("buildNewToOldFacetAgglomerationAndCellToFacet");
906 const auto new_to_old_facet =
907 buildNewToOldFacetAgglomerationAndCellToFacet(new_to_old_cell, agg_mesh);
908 Profiler::stop("buildNewToOldFacetAgglomerationAndCellToFacet");
909
910 // face to cell && facet to vertex and vertex to vertex (old/new)
911 Profiler::start("buildFacetToCellAndFacetToVtxAgglomeration");
912 const auto old_to_new_vtx = buildFacetToCellAndFacetToVtxAgglomeration(
913 new_to_old_facet, old_to_new_cell, agg_mesh);
914 Profiler::stop("buildFacetToCellAndFacetToVtxAgglomeration");
915
916 // cell to vtx
917 Profiler::start("buildCellToVtxAgglomeration");
918 buildCellToVtxAgglomeration(new_to_old_cell, old_to_new_vtx, agg_mesh);
919 Profiler::stop("buildCellToVtxAgglomeration");
920
921 Profiler::start("agglomeration::buildDualGraph");
922 agg_mesh.buildDualGraph();
923 Profiler::stop("agglomeration::buildDualGraph");
924
925 Profiler::start("agglomeration::buildVtxToCell");
926 agg_mesh.buildVtxToCell();
927 Profiler::stop("agglomeration::buildVtxToCell");
928
929 // transport tag
930 if (!m_mesh_tags.tags[CellEnty].empty()) {
931 for (const auto& [tag_name, tag] : m_mesh_tags.tags[CellEnty]) {
932 agg_mesh.m_mesh_tags.tags[CellEnty][tag_name].resize(
933 new_to_old_cell.size());
934 for (index_t ncid = 0; ncid < new_to_old_cell.size(); ++ncid) {
935 const auto row = new_to_old_cell[ncid];
936 agg_mesh.m_mesh_tags.tags[CellEnty][tag_name][ncid] = tag[row[0]];
937#ifndef NDEBUG
938 for (const auto old_cid : row) {
939 std::cout << "old_cid" << old_cid << " tag " << tag[old_cid]
940 << " new_cid " << ncid << " tag " << tag[row[0]]
941 << std::endl;
942 assert(tag[old_cid] == tag[row[0]]);
943 }
944#endif
945 }
946 }
947 }
948 return agg_mesh;
949}

991 typename IndexType,
992 IndexType OuterDim,
993 IndexType SelfDim,
994 typename Derived>
995void MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
996 buildDualGraph() {
997 datastructure::CrsMapUncompressed<IndexType> dual_graph;
998 dual_graph.resize(m_cell_to_facet.size());
999 for (IndexType cell_id = 0; cell_id < m_cell_to_facet.size(); ++cell_id) {
1000 auto& cell_neighs = dual_graph[cell_id];
1001 for (const auto facet_id : m_cell_to_facet.getRow(cell_id)) {
1002 for (const auto cell_neigh : m_facet_to_cell[facet_id]) {
1003 if ((cell_neigh != UNDEF) && (cell_neigh != cell_id)) {
1004 const auto begin = cell_neighs.begin();
1005 const auto end = cell_neighs.end();
1006 if (std::find(begin, end, cell_neigh) == end) {
1007 cell_neighs.push_back(cell_neigh);
1008 }
1009 }
1010 }
1011 }
1012 assert(
1013 ((m_cell_to_facet.size() == 1) ||
1014 (dual_graph.getRow(cell_id).size() > 0)) &&
1015 "Cell is unconnected");
1016 }
1017 m_dual_graph = dual_graph.template compress<IndexType>();
1018
1019 if (m_cell_to_facet.size() == 1) {
1020 if (m_dual_graph.size() == 0) {
1021 m_dual_graph.indices().resize(2, 0);
1022 }
1023 }
1024}
