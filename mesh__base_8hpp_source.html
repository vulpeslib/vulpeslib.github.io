488template <
489 typename FloatType,
490 typename IndexType,
491 IndexType OuterDim,
492 IndexType SelfDim,
493 typename Derived>
494datastructure::CrsMap<IndexType, IndexType>
495MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
496 invertCellAgglomeration(
497 const std::vector<IndexType>& old_to_new_cell) const {
498 datastructure::CrsMapUncompressed<IndexType, 64> new_to_old_cell_uncompr;
499 for (std::size_t cell_id = 0; cell_id < old_to_new_cell.size(); ++cell_id) {
500 const auto cell_id_new = old_to_new_cell[cell_id];
501 if (cell_id_new == UNDEF) {
502 continue;
503 }
504 new_to_old_cell_uncompr.getRow(cell_id_new).push_back(cell_id);
505 }
506 return new_to_old_cell_uncompr.template compress<IndexType>();
507}

509template <
510 typename FloatType,
511 typename IndexType,
512 IndexType OuterDim,
513 IndexType SelfDim,
514 typename Derived>
515std::vector<IndexType>
516MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
517 buildNewToOldFacetAgglomerationAndCellToFacet(
518 const auto& new_to_old_cell, Derived& agg_mesh) const {
519 std::vector<IndexType> new_to_old_facet;
520 std::vector<IndexType> old_to_new_facet(m_facet_to_cell.size(), -1);
521 datastructure::CrsMapUncompressed<IndexType, 64> new_cell_to_facets_uncpr;
522 new_cell_to_facets_uncpr.resize(new_to_old_cell.size());
523 // NOTE: a flat vector of booleans here performed much worse when tested
524 boost::unordered_flat_map<IndexType, IndexType> facet_table;
525 for (IndexType new_cid = 0; new_cid < new_to_old_cell.size(); ++new_cid) {
526 for (const auto old_cid : new_to_old_cell[new_cid]) {
527 for (const auto old_fid : m_cell_to_facet[old_cid]) {
528 // we only care if the count is only exactly 1
529 // it can be only 0 [face does not belong to cell]
530 // 1 [face is on boundary]
531 // 2 [face is internal] -> not(not(0)) == 0
532 facet_table[old_fid] = !facet_table[old_fid];
533 }
534 }
535 for (auto it = facet_table.begin(); it != facet_table.end(); ++it) {
536 const auto count = it->second;
537 const auto old_fid = it->first;
538 if (count) {
539 IndexType new_fid = old_to_new_facet[old_fid];
540 if (new_fid == UNDEF) {
541 new_fid = new_to_old_facet.size();
542 old_to_new_facet[old_fid] = new_fid;
543 new_to_old_facet.push_back(old_fid);
544 }
545 new_cell_to_facets_uncpr[new_cid].push_back(new_fid);
546 }
547 }
548 facet_table.clear();
549 }
550 agg_mesh.m_cell_to_facet =
551 new_cell_to_facets_uncpr.template compress<IndexType>();
552 return new_to_old_facet;
553}

555template <
556 typename FloatType,
557 typename IndexType,
558 IndexType OuterDim,
559 IndexType SelfDim,
560 typename Derived>
561auto MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
562 buildFacetToCellAndFacetToVtxAgglomeration(
563 const std::vector<IndexType>& new_to_old_facet,
564 const std::vector<IndexType>& old_to_new_cell,
565 Derived& agg_mesh) const {
566 std::vector<IndexType> new_to_old_vtx;
567 std::vector<IndexType> old_to_new_vtx(m_vertexes.size(), UNDEF);
568
569 agg_mesh.m_facet_to_cell.resize(new_to_old_facet.size(), {UNDEF, UNDEF});
570 agg_mesh.m_facet_to_vtx.indices().push_back(0);
571
572 for (std::size_t new_fid = 0; new_fid < new_to_old_facet.size(); ++new_fid) {
573 const auto old_fid = new_to_old_facet[new_fid];
574 // face to cell
575 for (std::size_t d = 0; d < 2; ++d) {
576 const auto old_cid = m_facet_to_cell[old_fid][d];
577 if (old_cid != UNDEF) {
578 const auto new_cid = old_to_new_cell[old_cid];
579 assert(new_cid != UNDEF);
580 agg_mesh.m_facet_to_cell[new_fid][d] = new_cid;
581 }
582 }
583
584 // facet to vertex and vertex to vertex (old/new)
585 for (const auto old_vid : m_facet_to_vtx[old_fid]) {
586 IndexType new_vid = old_to_new_vtx[old_vid];
587 if (new_vid == UNDEF) {
588 new_vid = new_to_old_vtx.size();
589 new_to_old_vtx.push_back(old_vid);
590 old_to_new_vtx[old_vid] = new_vid;
591 }
592 agg_mesh.m_facet_to_vtx.data().push_back(new_vid);
593 }
594 if constexpr (vtx_per_face == -1) {
595 agg_mesh.m_facet_to_vtx.indices().push_back(
596 agg_mesh.m_facet_to_vtx.data().size());
597 }
598 }
599
600 // add vtxs
601 for (std::size_t ovid = 0; ovid < old_to_new_vtx.size(); ++ovid) {
602 const auto nvid = old_to_new_vtx[ovid];
603 if (nvid != UNDEF) {
604 if (agg_mesh.m_vertexes.size() < static_cast<std::size_t>(nvid + 1)) {
605 agg_mesh.m_vertexes.resize(nvid + 1);
606 }
607 agg_mesh.m_vertexes[nvid] = m_vertexes[ovid];
608 }
609 }
610
611 return old_to_new_vtx;
612}

614template <
615 typename FloatType,
616 typename IndexType,
617 IndexType OuterDim,
618 IndexType SelfDim,
619 typename Derived>
620void MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
621 buildCellToVtxAgglomeration(
622 const auto& new_to_old_cell,
623 const auto& old_to_new_vtx,
624 Derived& agg_mesh) const {
625 agg_mesh.m_cell_to_vtx.indices().push_back(0);
626 std::vector<bool> is_vtx_present(old_to_new_vtx.size(), false);
627
628 for (IndexType new_cid = 0; new_cid < new_to_old_cell.size(); ++new_cid) {
629 for (const auto old_cid : new_to_old_cell[new_cid]) {
630 for (const auto old_vid : m_cell_to_vtx[old_cid]) {
631 const auto new_vid = old_to_new_vtx[old_vid];
632 if (new_vid != UNDEF && (!is_vtx_present[old_vid])) {
633 agg_mesh.m_cell_to_vtx.data().push_back(new_vid);
634 is_vtx_present[old_vid] = true;
635 }
636 }
637 }
638 std::fill(is_vtx_present.begin(), is_vtx_present.end(), false);
639 agg_mesh.m_cell_to_vtx.indices().push_back(
640 agg_mesh.m_cell_to_vtx.data().size());
641 }
642}

644template <
645 typename FloatType,
646 typename IndexType,
647 IndexType OuterDim,
648 IndexType SelfDim,
649 typename Derived>
650void MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
651 agglomerateCheckConnected(
652 const datastructure::CrsMap<IndexType, IndexType>& new_to_old_cell)
653 const {
654 // safety check: check each new cell is made by connected components
655 std::vector<IndexType> glob_to_loc(numCells(), UNDEF);
656 IndexType prev_new_cid = UNDEF;
657 for (IndexType new_cid = 0; new_cid < new_to_old_cell.size(); ++new_cid) {
658 if (new_to_old_cell[new_cid].size() <= 1) {
659 continue;
660 }
661 std::vector<IndexType> old_cids(
662 new_to_old_cell[new_cid].begin(), new_to_old_cell[new_cid].end());
663 // overwrite old map entries
664 if (VULPES_UNLIKELY(prev_new_cid != UNDEF)) {
665 for (auto id : new_to_old_cell[prev_new_cid]) {
666 glob_to_loc[id] = UNDEF;
667 }
668 }
669 prev_new_cid = new_cid;
670 // update map
671 for (std::size_t i = 0; i < old_cids.size(); ++i) {
672 glob_to_loc[old_cids[i]] = i;
673 }
674 // dfs to check it's connected
675 std::vector<bool> visited(old_cids.size(), false);
676 std::vector<IndexType> to_visit = {old_cids[0]};
677 while (!to_visit.empty()) {
678 const auto curr = to_visit.back();
679 visited[glob_to_loc[curr]] = true;
680 to_visit.pop_back();
681 for (const auto neigh : m_dual_graph[curr]) {
682 if (glob_to_loc[neigh] == UNDEF) {
683 continue;
684 }
685 if (!visited[glob_to_loc[neigh]]) {
686 to_visit.push_back(neigh);
687 }
688 }
689 }
690 // check all are visited
691 for (std::size_t i = 0; i < visited.size(); ++i) {
692 if (!visited[i]) {
693 std::cout << "ERROR: in new cell " + std::to_string(new_cid) +
694 ", starting from old cell " +
695 std::to_string(old_cids[0]) +
696 " is not possible to reach the old cell " +
697 std::to_string(old_cids[i])
698 << std::endl;
699
700 for (auto cid : {old_cids[0], old_cids[i]}) {
701 SelfPoint avg = SelfPoint::Zero();
702 for (const auto id : m_cell_to_vtx[cid]) {
703 avg += m_vertexes[id];
704 }
705 avg /= m_cell_to_vtx[cid].size();
706 std::cout << "Old cell " << cid << " avg: " << avg.transpose()
707 << std::endl;
708 }
709 std::exit(-1);
710 }
711 }
712 }
713}

715template <
716 typename FloatType,
717 typename IndexType,
718 IndexType OuterDim,
719 IndexType SelfDim,
720 typename Derived>
721Derived
722MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::agglomerateBase(
723 const std::vector<IndexType>& old_to_new_cell, bool check_connected) const {
724 assert(old_to_new_cell.size() == static_cast<std::size_t>(numCells()));
725 Derived agg_mesh;
726
727 // invert old to new cell
728 Profiler::start("invertCellAgglomeration");
729 const auto new_to_old_cell = invertCellAgglomeration(old_to_new_cell);
730 Profiler::stop("invertCellAgglomeration");
731
732 if (check_connected) {
733 Profiler::start("agglomerateCheckConnected");
734 agglomerateCheckConnected(new_to_old_cell);
735 Profiler::stop("agglomerateCheckConnected");
736 }
737 // cell to faces and facet to facet (old/new)
738 Profiler::start("buildNewToOldFacetAgglomerationAndCellToFacet");
739 const auto new_to_old_facet =
740 buildNewToOldFacetAgglomerationAndCellToFacet(new_to_old_cell, agg_mesh);
741 Profiler::stop("buildNewToOldFacetAgglomerationAndCellToFacet");
742
743 // face to cell && facet to vertex and vertex to vertex (old/new)
744 Profiler::start("buildFacetToCellAndFacetToVtxAgglomeration");
745 const auto old_to_new_vtx = buildFacetToCellAndFacetToVtxAgglomeration(
746 new_to_old_facet, old_to_new_cell, agg_mesh);
747 Profiler::stop("buildFacetToCellAndFacetToVtxAgglomeration");
748
749 // cell to vtx
750 Profiler::start("buildCellToVtxAgglomeration");
751 buildCellToVtxAgglomeration(new_to_old_cell, old_to_new_vtx, agg_mesh);
752 Profiler::stop("buildCellToVtxAgglomeration");
753
754 Profiler::start("AGG::buildDualGraph");
755 agg_mesh.buildDualGraph();
756 Profiler::stop("AGG::buildDualGraph");
757
758 Profiler::start("AGG::buildVtxToCell");
759 agg_mesh.buildVtxToCell();
760 Profiler::stop("AGG::buildVtxToCell");
761
762 // transport tag
763 if (!m_mesh_tags.tags[CellEnty].empty()) {
764 for (const auto& [tag_name, tag] : m_mesh_tags.tags[CellEnty]) {
765 agg_mesh.m_mesh_tags.tags[CellEnty][tag_name].resize(
766 new_to_old_cell.size());
767 for (index_t ncid = 0; ncid < new_to_old_cell.size(); ++ncid) {
768 const auto row = new_to_old_cell[ncid];
769 agg_mesh.m_mesh_tags.tags[CellEnty][tag_name][ncid] = tag[row[0]];
770#ifndef NDEBUG
771 for (const auto old_cid : row) {
772 assert(tag[old_cid] == tag[row[0]]);
773 }
774#endif
775 }
776 }
777 }
778 return agg_mesh;
779}

787void MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
788 buildVtxToCell() {
789 datastructure::CrsMapUncompressed<IndexType> vtx_to_cell;
790 for (IndexType cell_id = 0; cell_id < m_cell_to_vtx.size(); ++cell_id) {

798template <
799 typename FloatType,
800 typename IndexType,
801 IndexType OuterDim,
802 IndexType SelfDim,
803 typename Derived>
804void MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
805 buildCellToFacet() {
806 datastructure::CrsMapUncompressed<IndexType> cell_to_facet;
807 const IndexType n_facets = m_facet_to_cell.size();
808 for (IndexType facet_id = 0; facet_id < n_facets; ++facet_id) {
809 for (const auto cell_id : m_facet_to_cell[facet_id]) {
810 // if not on the boundary
811 if (cell_id != -1) {
812 cell_to_facet.getRow(cell_id).push_back(facet_id);
813 }
814 }
815 }
816 m_cell_to_facet = cell_to_facet.template compress<IndexType>();

819template <
820 typename FloatType,
821 typename IndexType,
822 IndexType OuterDim,
823 IndexType SelfDim,
824 typename Derived>
825void MeshBase<FloatType, IndexType, OuterDim, SelfDim, Derived>::
826 buildDualGraph() {
827 datastructure::CrsMapUncompressed<IndexType> dual_graph;
828 dual_graph.resize(m_cell_to_facet.size());
829 for (IndexType cell_id = 0; cell_id < m_cell_to_facet.size(); ++cell_id) {
830 auto& cell_neighs = dual_graph[cell_id];
831 for (const auto facet_id : m_cell_to_facet.getRow(cell_id)) {
832 for (const auto cell_neigh : m_facet_to_cell[facet_id]) {
833 if ((cell_neigh != UNDEF) && (cell_neigh != cell_id)) {
834 const auto begin = cell_neighs.begin();
835 const auto end = cell_neighs.end();
836 if (std::find(begin, end, cell_neigh) == end) {
837 cell_neighs.push_back(cell_neigh);
838 }
839 }
840 }
841 }
842 assert(
843 ((m_cell_to_facet.size() == 1) ||
844 (dual_graph.getRow(cell_id).size() > 0)) &&
845 "Cell is unconnected");
846 }
847 m_dual_graph = dual_graph.template compress<IndexType>();
848
849 if (m_cell_to_facet.size() == 1) {
850 if (m_dual_graph.size() == 0) {
851 m_dual_graph.indices().resize(2, 0);
852 }
853 }
854}

941template <typename index_t>
942datastructure::CrsMap<index_t, index_t> buildSubDual(
943 const datastructure::CrsMap<index_t, index_t>& dual_map,
944 const std::vector<index_t>& subset_idxs) {
945 using IndMap = datastructure::CrsMap<index_t, index_t>;
946 using IndVec = std::vector<index_t>;
947
948 // sort subset_idxs
949 IndVec tmp;
950 const IndVec* subset_idxs_sorted = &subset_idxs;
951 if (!std::is_sorted(subset_idxs.begin(), subset_idxs.end())) {
952 tmp = subset_idxs;
953 std::sort(tmp.begin(), tmp.end());
954 subset_idxs_sorted = &tmp;
955 }
956
957 // compute the old to new cell id map
958 index_t n_nodes = dual_map.size();
959 IndVec old_to_new(n_nodes, -1);
960 const index_t n_subset_idxs = subset_idxs.size();
961 for (index_t new_id = 0; new_id < n_subset_idxs; ++new_id) {
962 old_to_new[(*subset_idxs_sorted)[new_id]] = new_id;
963 }
964
965 // extract the dual map of the subset directly in compressed form!
966 IndMap sub_dual_map;
967 sub_dual_map.data().resize(0);
968 sub_dual_map.indices().resize(0);
969 sub_dual_map.indices().push_back(0);
970 for (index_t ncid = 0; ncid < n_subset_idxs; ++ncid) {
971 const auto ocid = (*subset_idxs_sorted)[ncid];
972 for (const auto neigh_ocid : dual_map[ocid]) {
973 const auto it = std::lower_bound(
974 (*subset_idxs_sorted).begin(), (*subset_idxs_sorted).end(),
975 neigh_ocid);
976 if (it != (*subset_idxs_sorted).end() && (*it) == neigh_ocid) {
977 sub_dual_map.data().push_back(old_to_new[neigh_ocid]);
978 }
979 }
980 sub_dual_map.indices().push_back(sub_dual_map.data().size());
981 }
982 return sub_dual_map;
983};

1000template <typename index_t>
1001index_t removeEmptySets(std::vector<index_t>& labels) {
1002 constexpr auto npos = static_cast<index_t>(-1);
1003 // NOTE: we need much less memory than labels.size()
1004 // we assume that few labels are missing
1005 const index_t max_label_id = *std::max_element(labels.begin(), labels.end());
1006 std::vector<index_t> old_to_new(max_label_id + 1, npos);
1007 index_t new_id = 0;
1008 for (std::size_t i = 0; i < labels.size(); i++) {
1009 if (old_to_new[labels[i]] == npos) {
1010 old_to_new[labels[i]] = new_id++;
1011 }
1012 labels[i] = old_to_new[labels[i]];
1013 }
1014 return new_id;
1015}
