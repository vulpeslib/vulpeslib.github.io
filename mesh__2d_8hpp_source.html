138template <typename FloatType, typename IndexType>
139void Mesh2D<FloatType, IndexType>::
140 buildFacetToCellAndCellToFacetAndFacetToVertex() {
141 // the two vertices id the identify an edge
142 using EdgeToCellKey = std::array<IndexType, 2>;
143 // the two cell ids to which the cell belongs
144 // -1 if the edge belongs to one cell only (is on boundary)
145 // first cell is the one where the edge is passed with increasing vertex id
146 // last element is the edge id
147 using EdgeToCellVal = std::array<IndexType, 3>;
148 datastructure::CrsMapUncompressed<IndexType> cell_to_facet;
149 std::map<EdgeToCellKey, EdgeToCellVal> edge_to_cell_map;
150 cell_to_facet.resize(this->m_cell_to_vtx.size());
151 // compute the edge to cell map, each edge belong to two elements if
152 // internal one if on the boundary
153 for (IndexType cell_id = 0; cell_id < this->m_cell_to_vtx.size(); ++cell_id) {
154 const auto cell = this->m_cell_to_vtx.getRow(cell_id);
155 for (IndexType v_id = 0; v_id < cell.size(); ++v_id) {
156 // vertex id of the points in the edge
157 const IndexType edge_left = cell[v_id];
158 const IndexType edge_right = cell[(v_id + 1) % cell.size()];
159 // the vertex id in the edge must be increasing
160 // moreover, the one with the "correct" orientation is in the first
161 // place
162 EdgeToCellVal edge_to_cell_val = {
163 static_cast<IndexType>(-1), static_cast<IndexType>(-1),
164 static_cast<IndexType>(-1)};
165 std::size_t wrong_orientation = !(edge_left < edge_right);
166 EdgeToCellKey edge_to_cell_key =
167 wrong_orientation ? EdgeToCellKey{edge_right, edge_left}
168 : EdgeToCellKey{edge_left, edge_right};
169 edge_to_cell_val[wrong_orientation] = cell_id;
170 auto [it, inserted] =
171 edge_to_cell_map.emplace(edge_to_cell_key, edge_to_cell_val);
172 if (!inserted) {
173 it->second[wrong_orientation] = cell_id;
174 } else {
175 it->second[2] = edge_to_cell_map.size() - 1;
176 }
177 cell_to_facet[cell_id].push_back(it->second[2]);
178 }
179 }
180 // fill the data structures
181 std::vector<IndexType> edge_vtx_data(edge_to_cell_map.size() * 2);
182 this->m_facet_to_cell.resize(edge_to_cell_map.size());
183 for (const auto& [edge_vtxs, cells] : edge_to_cell_map) {
184 edge_vtx_data[2 * cells[2]] = edge_vtxs[0];
185 edge_vtx_data[2 * cells[2] + 1] = edge_vtxs[1];
186 this->m_facet_to_cell[cells[2]] = {cells[0], cells[1]};
187 }
188 using FacetToVtxMap = decltype(this->m_facet_to_vtx);
189 this->m_facet_to_vtx = FacetToVtxMap(std::move(edge_vtx_data), {});
190 this->m_cell_to_facet = cell_to_facet.template compress<IndexType>();
191}

194inline std::string getFileExtension(const std::string& filename) {
195 std::size_t dotPos = filename.find_last_of('.');
196
197 if (dotPos == std::string::npos || dotPos == 0 ||
198 dotPos == filename.length() - 1) {
199 return "";
200 }
201
202 return filename.substr(dotPos + 1);
203}

205inline std::vector<std::array<double, 3>> readpts(const std::string& filename) {
206 std::ifstream file(filename);
207 if (!file.is_open()) {
208 std::cerr << "Error: Could not open file " << filename << std::endl;
209 return {};
210 }
211
212 std::vector<std::array<double, 3>> points;
213 std::string line;
214 bool readingPoints = false;
215 int num_points = 0;
216
217 while (std::getline(file, line)) {
218 std::istringstream iss(line);
219 if (line.find("POINTS") != std::string::npos) {
220 std::string keyword;
221 iss >> keyword;
222 iss >> num_points;
223 readingPoints = true;
224 continue;
225 }
226
227 if (readingPoints) {
228 std::array<double, 3> point = {};
229 while (iss >> point[0] >> point[1] >> point[2]) {
230 points.push_back(point);
231 }
232 }
233
234 if (readingPoints && points.size() >= static_cast<size_t>(num_points)) {
235 break;
236 }
237 }
238
239 file.close();
240 return points;
241}

246template <typename FloatType, typename IndexType>
247Mesh2D<FloatType, IndexType> read(
248 const std::string& filename, bool check_vtxs_unique = true) {
249 vtkSmartPointer<vtkPointSet> unstructured_grid;
250 bool read_pts = false;
251 // read file
252 const auto ext = internal::getFileExtension(filename);
253 if (ext == "vtu") {
254 vtkNew<vtkXMLUnstructuredGridReader> reader;
255 reader->SetFileName(filename.c_str());
256 reader->Update();
257 unstructured_grid = reader->GetOutput();
258 } else if (ext == "vtk") {
259 vtkNew<vtkPolyDataReader> reader;
260 reader->SetFileName(filename.c_str());
261 reader->Update();
262 unstructured_grid = reader->GetOutput();
263 static_assert(
264 std::is_same_v<
265 decltype(unstructured_grid->GetPoints()->GetPoint(0)), double*>);
266 read_pts = true;
267 } else {
268 assert(false && "File with not supported extension");
269 }
270
271 std::vector<std::array<double, 3>> precise_pts;
272 if (read_pts) {
273 precise_pts = internal::readpts(filename);
274 }
275 // get points
276 std::vector<Point<FloatType, 2>> mesh_vtxs;
277 vtkPoints* points = unstructured_grid->GetPoints();
278 mesh_vtxs.resize(points->GetNumberOfPoints());
279 for (vtkIdType i = 0; i < points->GetNumberOfPoints(); ++i) {
280 const double* point = read_pts ? &precise_pts[i][0] : points->GetPoint(i);
281 mesh_vtxs[i][0] = point[0];
282 mesh_vtxs[i][1] = point[1];
283 assert(point[2] == 0);
284 }
285
286 // TODO: should this become a method of mesh to check consistency?
287 // TODO: implement query radius instead of looking for the 2 nearest
288 if (check_vtxs_unique) {
289 datastructure::KDTree<2> tree_vtx(&mesh_vtxs[0][0], mesh_vtxs.size() * 2);
290 tree_vtx.build(16, true, true);
291 // TODO: loop over each point
292 // TODO: normalize distance by point magnitude
293 // TODO: put 1e-14 as constant outside
294 // TODO: add for 3D
295 const auto [dists, indxs] = tree_vtx.queryKnn<2.0>(&mesh_vtxs[0][0], 1, 2);
296 if (dists(1, 0) < 1e-14) {
297 std::cout << "ERROR: Vertex collision" << std::endl;
298 std::exit(-1);
299 }
300 }
301
302 // get cells
303 datastructure::CrsMapUncompressed<IndexType> mesh_cells;
304 const vtkIdType n_cells = unstructured_grid->GetNumberOfCells();
305 for (vtkIdType i = 0; i < n_cells; ++i) {
306 vtkCell* cell = unstructured_grid->GetCell(i);
307 const int32_t n_points = static_cast<int32_t>(cell->GetNumberOfPoints());
308 for (int32_t j = 0; j < n_points; ++j) {
309 mesh_cells.getRow(i).push_back(cell->GetPointId(j));
310 }
311 }
312
313 return Mesh2D<FloatType, IndexType>{mesh_vtxs, mesh_cells};
314}

316template <typename FloatType, typename IndexType>
317Mesh2D<FloatType, IndexType> rectangle(
318 const Point<FloatType, 2>& corner_a,
319 const Point<FloatType, 2>& corner_b,
320 const Point<IndexType, 2>& n_divisions,
321 bool tria_elements = false) {
322 datastructure::CrsMapUncompressed<IndexType> cell_indices;
323 std::vector<Point<FloatType, 2>> vertices;
324 // Calculate the step size in x and y direction
325 const FloatType x_step = (corner_b[0] - corner_a[0]) / n_divisions[0];
326 const FloatType y_step = (corner_b[1] - corner_a[1]) / n_divisions[1];
327
328 // Generate the vertices
329 for (IndexType i = 0; i <= n_divisions[0]; ++i) {
330 for (IndexType j = 0; j <= n_divisions[1]; ++j) {
331 // Compute the x and y coordinates of the vertex
332 const FloatType x = corner_a[0] + i * x_step;
333 const FloatType y = corner_a[1] + j * y_step;
334 vertices.push_back({x, y});
335 }
336 }
337
338 // Generate the cell indices
339 for (IndexType i = 0; i < n_divisions[0]; ++i) {
340 for (IndexType j = 0; j < n_divisions[1]; ++j) {
341 // Index of the bottom-left vertex of the cell
342 const IndexType bottom_left = i * (n_divisions[1] + 1) + j;
343 const IndexType bottom_right = (i + 1) * (n_divisions[1] + 1) + j;
344 const IndexType top_left = i * (n_divisions[1] + 1) + (j + 1);
345 const IndexType top_right = (i + 1) * (n_divisions[1] + 1) + (j + 1);
346 // Store the cell as four vertex indices
347 if (tria_elements) {
348 cell_indices.push_back({bottom_left, bottom_right, top_left});
349 cell_indices.push_back({bottom_right, top_right, top_left});
350 } else {
351 cell_indices.push_back(
352 {bottom_left, bottom_right, top_right, top_left});
353 }
354 }
355 }
356
357 return Mesh2D<FloatType, IndexType>{vertices, cell_indices};
358}
