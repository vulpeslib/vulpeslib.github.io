302template <typename FloatType, typename IndexType>
303std::array<std::vector<FloatType>, 2> buildCartesianBbox(
304 const Mesh2D<FloatType, IndexType>& mesh,
305 IndexType n_cells_x,
306 FloatType stretch) {
307 constexpr auto MAX = std::numeric_limits<FloatType>::max();
308 Point<FloatType, 2> min_corner = Point<FloatType, 2>::Ones() * MAX;
309 Point<FloatType, 2> max_corner = -Point<FloatType, 2>::Ones() * MAX;
310 for (const auto& point : mesh.getVertexes()) {
311 min_corner = min_corner.cwiseMin(point);
312 max_corner = max_corner.cwiseMax(point);
313 }
314 // avoid exact match of cartesian grid with mesh by adding little extra space
315 const Point<FloatType, 2> bbox_size = max_corner - min_corner;
316 constexpr auto DELTA = 1e-2;
317 min_corner -= bbox_size * DELTA;
318 max_corner(0) += bbox_size(0) * DELTA;
319 using Vectord = Eigen::Vector<FloatType, -1>;
320 const Vectord x_grid =
321 Vectord::LinSpaced(n_cells_x + 1, min_corner(0), max_corner(0));
322 std::array<FloatType, 2> h = {(max_corner(0) - min_corner(0)) / n_cells_x, 0};
323 h[1] = h[0] * stretch;
324 std::array<std::vector<FloatType>, 2> grids = {
325 std::vector<FloatType>{min_corner[0]},
326 std::vector<FloatType>{min_corner[1]}};
327 for (std::size_t d = 0; d < 2; ++d) {
328 std::size_t i = 1;
329 do {
330 grids[d].push_back(grids[d][0] + i * h[d]);
331 i++;
332 } while (grids[d].back() < max_corner(d));
333 }
334 return grids;
335}
