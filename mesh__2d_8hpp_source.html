301template <typename FloatType, typename IndexType>
302std::array<std::vector<FloatType>, 2> buildCartesianBbox(
303 const Mesh2D<FloatType, IndexType>& mesh,
304 IndexType n_cells_x,
305 FloatType stretch) {
306 constexpr auto MAX = std::numeric_limits<FloatType>::max();
307 Point<FloatType, 2> min_corner = Point<FloatType, 2>::Ones() * MAX;
308 Point<FloatType, 2> max_corner = -Point<FloatType, 2>::Ones() * MAX;
309 for (const auto& point : mesh.getVertexes()) {
310 min_corner = min_corner.cwiseMin(point);
311 max_corner = max_corner.cwiseMax(point);
312 }
313 // avoid exact match of cartesian grid with mesh by adding little extra space
314 const Point<FloatType, 2> bbox_size = max_corner - min_corner;
315 constexpr auto DELTA = 1e-2;
316 min_corner -= bbox_size * DELTA;
317 max_corner(0) += bbox_size(0) * DELTA;
318 using Vectord = Eigen::Vector<FloatType, -1>;
319 const Vectord x_grid =
320 Vectord::LinSpaced(n_cells_x + 1, min_corner(0), max_corner(0));
321 std::array<FloatType, 2> h = {(max_corner(0) - min_corner(0)) / n_cells_x, 0};
322 h[1] = h[0] * stretch;
323 std::array<std::vector<FloatType>, 2> grids = {
324 std::vector<FloatType>{min_corner[0]},
325 std::vector<FloatType>{min_corner[1]}};
326 for (std::size_t d = 0; d < 2; ++d) {
327 std::size_t i = 1;
328 do {
329 grids[d].push_back(grids[d][0] + i * h[d]);
330 i++;
331 } while (grids[d].back() < max_corner(d));
332 }
333 return grids;
334}
