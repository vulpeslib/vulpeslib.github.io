33template <
34 typename MeshType,
35 typename FeExpressionTuple,
36 typename FeExpressionFaceTuple = std::tuple<>>
37class WeakFormIntegratorVemMoment : public WeakFormIntegratorBase<
38 MeshType,
39 FeExpressionTuple,
40 FeExpressionFaceTuple> {
41 public:
42 using WeakFormIntegratorBaseType = WeakFormIntegratorBase<
43 MeshType,
44 FeExpressionTuple,
45 FeExpressionFaceTuple>;
46 static inline constexpr std::size_t dim = MeshType::self_dim;
47
48 WeakFormIntegratorVemMoment(
49 const MeshType& mesh,
50 const dofs::DofHandler<MeshType>& dof_handler,
51 const FeExpressionTuple& fe_expression,
52 const FeExpressionFaceTuple& fe_expression_face = FeExpressionFaceTuple{})
53 : WeakFormIntegratorBaseType(
54 mesh, dof_handler, fe_expression, fe_expression_face) {
55 const auto n_dof_loc = fe_expression.feSpace().getBasis().size();
56 m_poly_table.resize(n_dof_loc, n_dof_loc);
57
58 for (std::size_t i = 0; i < n_dof_loc; ++i) {
59 const auto poly = fe_expression.evaluatePoly(m_fe_values, i, -1);
60 for (std::size_t j = 0; j < n_dof_loc; ++j) {
61 // poly is made by many monomials, however, fixed j (and the basis func
62 // phi_j) only one of the monomial might be non-zero
63
64 // find the dof corresponding to j
65 m_poly_table(i, j) = findCoefDof(poly, j);
66 }
67 }
68 }
69
70 void integrate(MeshType::index_t cell_id) override {
71 m_integrals = abs(E) * m_poly_table /
72 fe_expression.evaluatePolyCoef(Mapping{1 / diam});
73 }
74};
