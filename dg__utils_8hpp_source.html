75template <typename MeshType, typename FeSpaceType, typename ExprType>
76double integrateFeExpression(
77 const MeshType& mesh,
78 assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
79 const ExprType& expression) {
80 // Create and use the scalar expression integrator
81 // Quadrature order 5 is sufficient for squared expressions of typical orders
82 assemble::ScalarExpressionIntegrator<MeshType, FeSpaceType, ExprType>
83 integrator(mesh, cell_manager, expression, 5);
84 return integrator.integrate();
85}

104template <
105 typename MeshType,
106 typename FeSpaceType,
107 typename DofHandlerType,
108 typename ExprType>
109double computeL2Error(
110 const MeshType& mesh,
111 assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
112 const DofHandlerType& dof_handler,
113 const la::Vector& u_h,
114 const ExprType& u_exact_expr) {
115 constexpr std::size_t Dim = MeshType::self_dim;
116 // Create the numerical solution as an FE expression
117 fe::FeExactExpressionFeProjection<Dim, FeSpaceType, DofHandlerType> u_h_expr(
118 u_h, cell_manager.getFeSpace(), dof_handler);
119
120 // Form the difference expression: (u_h - u_exact)
121 auto diff_expr = u_h_expr - u_exact_expr;
122
123 // Form the squared difference: (u_h - u_exact)^2
124 auto diff_squared = diff_expr * diff_expr;
125
126 // Integrate the squared difference over the domain
127 double l2_error_squared =
128 integrateFeExpression(mesh, cell_manager, diff_squared);
129
130 // Return the L2 error (square root of the integral)
131 return std::sqrt(std::max(0.0, l2_error_squared));
132}

149template <
150 typename MeshType,
151 typename FeSpaceType,
152 typename DofHandlerType,
153 typename ExprType>
154double computeH1SemiError(
155 const MeshType& mesh,
156 assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
157 const DofHandlerType& dof_handler,
158 const la::Vector& u_h,
159 const ExprType& grad_u_exact_expr) {
160 constexpr std::size_t Dim = MeshType::self_dim;
161
162 fe::FeGradProjectionExpr<Dim, FeSpaceType, DofHandlerType> grad_uh_expr(
163 u_h, cell_manager.getFeSpace(), dof_handler);
164
165 auto diff_expr = grad_uh_expr - grad_u_exact_expr;
166 auto diff_squared = fe::inner(diff_expr, diff_expr);
167
168 double h1_semi_error_squared =
169 integrateFeExpression(mesh, cell_manager, diff_squared);
170
171 return std::sqrt(std::max(0.0, h1_semi_error_squared));
172}

184template <typename MeshType>
185double computeJumpError(
186 const dofs::DofHandler<MeshType>& dof_handler,
187 const la::Vector& u_h,
188 const la::SparseMatrix& matrix) {
189 la::Vector temp(dof_handler, u_h.getBlockSize());
190 matrix.vmult(u_h, temp); // temp = matrix * u_h
191 return std::sqrt(la::dot(u_h, temp));

203template <std::size_t Dim>
204auto createPenaltyExpression(std::size_t order, double gamma) {
205 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
206 [order, gamma](
207 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
208 const auto& cell = *cells[fe_values.getCellId()];
209 const double penalty = gamma * order * order;
210 const double p_this = penalty / cell.diameter();
211 if (fe_values.hasFaceNeigh()) {
212 assert(&fe_values.neighValues());
213 const auto neigh_cell_id = fe_values.neighValues().getCellId();
214 const auto& neigh_cell = *cells[neigh_cell_id];
215 const double p_neigh = penalty / neigh_cell.diameter();
216 return std::max(p_this, p_neigh);
217 } else {
218 return p_this;
219 }
220 });
221}

233template <std::size_t Dim>
234auto createPenaltyExpressionKb(std::size_t order, double gamma) {
235 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
236 [order, gamma](
237 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
238 const auto& cell = *cells[fe_values.getCellId()];
239 const auto face_id = fe_values.getFaceId();
240 const double penalty = gamma * order * order;
241 const double Cinv = cell.measure() / cell.faceKb()[face_id];
242 const double facet_diam = cell.faceDiam()[face_id];
243 const double p_this = penalty * Cinv * (facet_diam / cell.measure());
244 if (fe_values.hasFaceNeigh()) {
245 assert(&fe_values.neighValues());
246 const auto neigh_cell_id = fe_values.neighValues().getCellId();
247 const auto& neigh_cell = *cells[neigh_cell_id];
248 const auto neigh_face_id = fe_values.neighValues().getFaceId();
249 const double Cinv_neigh =
250 neigh_cell.measure() / neigh_cell.faceKb()[neigh_face_id];
251 const double p_neigh =
252 penalty * Cinv_neigh * (facet_diam / neigh_cell.measure());
253 return std::max(p_this, p_neigh);
254 } else {
255 return p_this;
256 }
257 });
258}

273template <std::size_t Dim>
274auto createPenaltyExpressionHarmonic(std::size_t order, double gamma) {
275 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
276 [order, gamma](
277 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
278 const auto& cell = *cells[fe_values.getCellId()];
279 const auto face_id = fe_values.getFaceId();
280 const double facet_diam = cell.faceDiam()(face_id);
281 const double penalty = gamma * order * order;
282 const double Cinv = cell.measure() / cell.faceKb()[face_id];
283 const double p_this = penalty * Cinv * (facet_diam / cell.measure());
284
285 if (fe_values.hasFaceNeigh()) {
286 const auto neigh_cell_id = fe_values.neighValues().getCellId();
287 const auto& neigh_cell = *cells[neigh_cell_id];
288 const auto neigh_face_id = fe_values.neighValues().getFaceId();
289 const double Cinv_neigh =
290 neigh_cell.measure() / neigh_cell.faceKb()[neigh_face_id];
291 const double p_neigh =
292 penalty * Cinv_neigh * (facet_diam / neigh_cell.measure());
293 // Media armonica tra p_this e p_neigh
294 return 2.0 * p_this * p_neigh / (p_this + p_neigh);
295 } else {
296 return p_this;
297 }
298 });
299}

307template <std::size_t Dim, typename VelocityFunc>
308class FeExpressionUpwindFromFunctionWrapper
309 : public fe::FeExpressionBase<
310 Dim,
311 FeExpressionUpwindFromFunctionWrapper<Dim, VelocityFunc>> {
312 public:
313 static constexpr bool is_leaf = true;
314 static constexpr uint8_t is_bilinear = fe::IsNothing;
315
316 explicit constexpr FeExpressionUpwindFromFunctionWrapper(
317 const VelocityFunc& velocity_func)
318 : m_velocity_func(velocity_func) {}
319 FeExpressionUpwindFromFunctionWrapper(
320 const FeExpressionUpwindFromFunctionWrapper&) = delete;
321 FeExpressionUpwindFromFunctionWrapper& operator=(
322 const FeExpressionUpwindFromFunctionWrapper&) = delete;
323 FeExpressionUpwindFromFunctionWrapper(
324 FeExpressionUpwindFromFunctionWrapper&&) = delete;
325 FeExpressionUpwindFromFunctionWrapper& operator=(
326 FeExpressionUpwindFromFunctionWrapper&&) = delete;

328 template <std::size_t I = 0, std::size_t J = 0>
329 decltype(auto) evaluate(
330 const fe::FeValues<Dim>&, std::size_t, std::size_t, std::size_t q) const {
331 return m_evals[q];
332 }
333
334 constexpr fe::UpdateFlagsType computeUpdateFlags() const {
335 return fe::UpdateQuadraturePts | fe::UpdateNormals;
336 }
337 constexpr std::size_t computeOrder() const { return 0; }
338
339 constexpr bool precompute(
340 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>&) const {
341 const Eigen::Index nqpts = fe_values.weights().size();
342 assert(fe_values.qpts().cols() == nqpts);
343 m_evals.resize(nqpts);
344 const auto& normal = fe_values.getFaceNormal();
345 for (Eigen::Index q = 0; q < nqpts; ++q) {
346 const auto vel = m_velocity_func(fe_values.qpts().col(q));
347 const double vel_dot_n = vel.dot(normal);
348 m_evals[q] = 0.5 * (std::abs(vel_dot_n) + vel_dot_n);
349 }
350 return true;
351 }
352
353 protected:
354 const VelocityFunc m_velocity_func;
355 mutable std::vector<double> m_evals;
356};

388template <std::size_t Dim>
389auto createUpwindFluxConstant(double vx, double vy = 0.0, double vz = 0.0) {
390 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
391 [vx, vy, vz](
392 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
393 const auto& cell = *cells[fe_values.getCellId()];
394 const auto face_id = fe_values.getFaceId();
395 const auto& normal = cell.normals().col(face_id);
396
397 // Compute velocity . n
398 double vel_dot_n = vx * normal(0);
399 if constexpr (Dim >= 2) {
400 vel_dot_n += vy * normal(1);
401 }
402 if constexpr (Dim >= 3) {
403 vel_dot_n += vz * normal(2);
404 }
405
406 // Upwind flux: 0.5 * (|velocity . n| + velocity . n)
407 return 0.5 * (std::abs(vel_dot_n) + vel_dot_n);
408 });
409}

426template <std::size_t Dim, typename VelocityFunc>
427auto createUpwindFluxVariable(VelocityFunc velocity_func) {
428 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
429 [velocity_func](
430 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
431 const auto& cell = *cells[fe_values.getCellId()];
432 const auto face_id = fe_values.getFaceId();
433 const auto& normal = cell.normals().col(face_id);
434
435 // Face centroid: average of the two edge vertices for 2D polygons.
436 const Eigen::Index nvtxs = cell.vtxs().cols();
437 const Eigen::Vector<double, Dim> face_center =
438 0.5 *
439 (cell.vtxs().col(face_id) + cell.vtxs().col((face_id + 1) % nvtxs));
440
441 // Evaluate velocity at face center and compute vel . n
442 const auto vel = velocity_func(face_center);
443 const double vel_dot_n = vel.dot(normal);
444
445 // Upwind flux: 0.5 * (|velocity . n| + velocity . n)
446 return 0.5 * (std::abs(vel_dot_n) + vel_dot_n);
447 });
448}

461template <typename MeshType>
462double computeError(
463 const dofs::DofHandler<MeshType>& dof_handler,
464 const la::Vector& u_h,
465 const la::Vector& u_exact,
466 const la::SparseMatrix& matrix) {
467 // Compute the residual: error_u = u_exact - u_h
468 la::Vector error_u(dof_handler, u_h.getBlockSize());
469 error_u.add(-1.0, u_h);
470 error_u.add(1.0, u_exact);
471 error_u.compress();
472
473 // Compute the error norm using the provided matrix
474 la::Vector temp(dof_handler, u_h.getBlockSize());
475 // std::cout << "Error Calc with matrix: " << std::endl;
476 matrix.vmult(error_u, temp); // temp = matrix * error_u
477
478 if (la::dot(error_u, temp) < 0) {
479 std::cout
480 << "Warning: Negative value in error computation. Setting error to 0."
481 << std::endl;
482 return 0.0;
483 } else {
484 return std::sqrt(la::dot(error_u, temp));
485 }
486}

518template <typename MeshType>
519la::Vector projectConstantOntoModalSpace(
520 const dofs::DofHandler<MeshType>& dof_handler,
521 double constant_value,
522 const la::SparseMatrix& mass_matrix,
523 const la::Vector& rhs_const) {
524 la::Vector scaled_rhs(rhs_const);
525 scaled_rhs.scale(constant_value);
526
527 la::Vector c_proj(dof_handler);
528 la::LinearSolver solver(la::LinearSolver::PrecondOnly, la::LinearSolver::LU);
529 solver.setOperators(mass_matrix);
530 solver.solve(scaled_rhs, c_proj);
531
532 return c_proj;
533}

548template <typename MeshType>
549double forceEqualMean(
550 const dofs::DofHandler<MeshType>& dof_handler,
551 la::Vector& u,
552 const la::Vector& u_reference,
553 const la::SparseMatrix& mass_matrix,
554 const la::Vector& rhs_const,
555 double domain_measure) {
556 double avg_u = computeAverage(u, rhs_const, domain_measure);
557 double avg_reference = computeAverage(u_reference, rhs_const, domain_measure);
558
559 double shift_physical = avg_reference - avg_u;
560
561 la::Vector c_proj = projectConstantOntoModalSpace(
562 dof_handler, shift_physical, mass_matrix, rhs_const);
563
564 u.add(1.0, c_proj);
565 u.compress();
566
567 return shift_physical;
568}
