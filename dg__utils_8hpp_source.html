63template <typename MeshType, typename FeSpaceType, typename ExprType>
64double integrateFeExpression(
65 const MeshType& mesh,
66 assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
67 const ExprType& expression) {
68 // Create and use the scalar expression integrator
69 // Quadrature order 5 is sufficient for squared expressions of typical orders
70 assemble::ScalarExpressionIntegrator<MeshType, FeSpaceType, ExprType>
71 integrator(mesh, cell_manager, expression, 5);
72 return integrator.integrate();
73}

92template <
93 typename MeshType,
94 typename FeSpaceType,
95 typename DofHandlerType,
96 typename ExprType>
97double computeL2Error(
98 const MeshType& mesh,
99 assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
100 const DofHandlerType& dof_handler,
101 const la::Vector& u_h,
102 const ExprType& u_exact_expr) {
103 constexpr std::size_t Dim = MeshType::self_dim;
104 // Create the numerical solution as an FE expression
105 fe::FeExactExpressionFeProjection<Dim, FeSpaceType, DofHandlerType> u_h_expr(
106 u_h, cell_manager.getFeSpace(), dof_handler);
107
108 // Form the difference expression: (u_h - u_exact)
109 auto diff_expr = u_h_expr - u_exact_expr;
110
111 // Form the squared difference: (u_h - u_exact)^2
112 auto diff_squared = diff_expr * diff_expr;
113
114 // Integrate the squared difference over the domain
115 double l2_error_squared =
116 integrateFeExpression(mesh, cell_manager, diff_squared);
117
118 // Return the L2 error (square root of the integral)
119 return std::sqrt(std::max(0.0, l2_error_squared));
120}

131template <std::size_t Dim>
132auto createPenaltyExpression(std::size_t order, double gamma) {
133 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
134 [order, gamma](
135 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
136 const auto& cell = *cells[fe_values.getCellId()];
137 const double penalty = gamma * order * order;
138 const double p_this = penalty / cell.diameter();
139 if (fe_values.hasFaceNeigh()) {
140 assert(&fe_values.neighValues());
141 const auto neigh_cell_id = fe_values.neighValues().getCellId();
142 const auto& neigh_cell = *cells[neigh_cell_id];
143 const double p_neigh = penalty / neigh_cell.diameter();
144 return std::max(p_this, p_neigh);
145 } else {
146 return p_this;
147 }
148 });
149}

161template <std::size_t Dim>
162auto createPenaltyExpressionKb(std::size_t order, double gamma) {
163 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
164 [order, gamma](
165 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
166 const auto& cell = *cells[fe_values.getCellId()];
167 const auto face_id = fe_values.getFaceId();
168 const double penalty = gamma * order * order;
169 const double Cinv = cell.measure() / cell.faceKb()[face_id];
170 const double facet_diam = cell.faceDiam()[face_id];
171 const double p_this = penalty * Cinv * (facet_diam / cell.measure());
172 if (fe_values.hasFaceNeigh()) {
173 assert(&fe_values.neighValues());
174 const auto neigh_cell_id = fe_values.neighValues().getCellId();
175 const auto& neigh_cell = *cells[neigh_cell_id];
176 const auto neigh_face_id = fe_values.neighValues().getFaceId();
177 const double Cinv_neigh =
178 neigh_cell.measure() / neigh_cell.faceKb()[neigh_face_id];
179 const double p_neigh =
180 penalty * Cinv_neigh * (facet_diam / neigh_cell.measure());
181 return std::max(p_this, p_neigh);
182 } else {
183 return p_this;
184 }
185 });
186}

201template <std::size_t Dim>
202auto createPenaltyExpressionHarmonic(std::size_t order, double gamma) {
203 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
204 [order, gamma](
205 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
206 const auto& cell = *cells[fe_values.getCellId()];
207 const auto face_id = fe_values.getFaceId();
208 const double facet_diam = cell.faceDiam()(face_id);
209 const double penalty = gamma * order * order;
210 const double Cinv = cell.measure() / cell.faceKb()[face_id];
211 const double p_this = penalty * Cinv * (facet_diam / cell.measure());
212
213 if (fe_values.hasFaceNeigh()) {
214 const auto neigh_cell_id = fe_values.neighValues().getCellId();
215 const auto& neigh_cell = *cells[neigh_cell_id];
216 const auto neigh_face_id = fe_values.neighValues().getFaceId();
217 const double Cinv_neigh =
218 neigh_cell.measure() / neigh_cell.faceKb()[neigh_face_id];
219 const double p_neigh =
220 penalty * Cinv_neigh * (facet_diam / neigh_cell.measure());
221 // Media armonica tra p_this e p_neigh
222 return 2.0 * p_this * p_neigh / (p_this + p_neigh);
223 } else {
224 return p_this;
225 }
226 });
227}

235template <std::size_t Dim, typename VelocityFunc>
236class FeExpressionUpwindFromFunctionWrapper
237 : public fe::FeExpressionBase<
238 Dim,
239 FeExpressionUpwindFromFunctionWrapper<Dim, VelocityFunc>> {
240 public:
241 static constexpr bool is_leaf = true;
242 static constexpr uint8_t is_bilinear = fe::IsNothing;
243
244 explicit constexpr FeExpressionUpwindFromFunctionWrapper(
245 const VelocityFunc& velocity_func)
246 : m_velocity_func(velocity_func) {}
247 FeExpressionUpwindFromFunctionWrapper(
248 const FeExpressionUpwindFromFunctionWrapper&) = delete;
249 FeExpressionUpwindFromFunctionWrapper& operator=(
250 const FeExpressionUpwindFromFunctionWrapper&) = delete;
251 FeExpressionUpwindFromFunctionWrapper(
252 FeExpressionUpwindFromFunctionWrapper&&) = delete;
253 FeExpressionUpwindFromFunctionWrapper& operator=(
254 FeExpressionUpwindFromFunctionWrapper&&) = delete;

256 template <std::size_t I = 0, std::size_t J = 0>
257 decltype(auto) evaluate(
258 const fe::FeValues<Dim>&, std::size_t, std::size_t, std::size_t q) const {
259 return m_evals[q];
260 }
261
262 constexpr fe::UpdateFlagsType computeUpdateFlags() const {
263 return fe::UpdateQuadraturePts | fe::UpdateNormals;
264 }
265 constexpr std::size_t computeOrder() const { return 0; }
266
267 constexpr bool precompute(
268 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>&) const {
269 const Eigen::Index nqpts = fe_values.weights().size();
270 assert(fe_values.qpts().cols() == nqpts);
271 m_evals.resize(nqpts);
272 const auto& normal = fe_values.getFaceNormal();
273 for (Eigen::Index q = 0; q < nqpts; ++q) {
274 const auto vel = m_velocity_func(fe_values.qpts().col(q));
275 const double vel_dot_n = vel.dot(normal);
276 m_evals[q] = 0.5 * (std::abs(vel_dot_n) + vel_dot_n);
277 }
278 return true;
279 }
280
281 protected:
282 const VelocityFunc m_velocity_func;
283 mutable std::vector<double> m_evals;
284};

316template <std::size_t Dim>
317auto createUpwindFluxConstant(double vx, double vy = 0.0, double vz = 0.0) {
318 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
319 [vx, vy, vz](
320 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
321 const auto& cell = *cells[fe_values.getCellId()];
322 const auto face_id = fe_values.getFaceId();
323 const auto& normal = cell.normals().col(face_id);
324
325 // Compute velocity . n
326 double vel_dot_n = vx * normal(0);
327 if constexpr (Dim >= 2) {
328 vel_dot_n += vy * normal(1);
329 }
330 if constexpr (Dim >= 3) {
331 vel_dot_n += vz * normal(2);
332 }
333
334 // Upwind flux: 0.5 * (|velocity . n| + velocity . n)
335 return 0.5 * (std::abs(vel_dot_n) + vel_dot_n);
336 });
337}

354template <std::size_t Dim, typename VelocityFunc>
355auto createUpwindFluxVariable(VelocityFunc velocity_func) {
356 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
357 [velocity_func](
358 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
359 const auto& cell = *cells[fe_values.getCellId()];
360 const auto face_id = fe_values.getFaceId();
361 const auto& normal = cell.normals().col(face_id);
362
363 // Face centroid: average of the two edge vertices for 2D polygons.
364 const Eigen::Index nvtxs = cell.vtxs().cols();
365 const Eigen::Vector<double, Dim> face_center =
366 0.5 *
367 (cell.vtxs().col(face_id) + cell.vtxs().col((face_id + 1) % nvtxs));
368
369 // Evaluate velocity at face center and compute vel . n
370 const auto vel = velocity_func(face_center);
371 const double vel_dot_n = vel.dot(normal);
372
373 // Upwind flux: 0.5 * (|velocity . n| + velocity . n)
374 return 0.5 * (std::abs(vel_dot_n) + vel_dot_n);
375 });
376}

389template <typename MeshType>
390double computeError(
391 const dofs::DofHandler<MeshType>& dof_handler,
392 const la::Vector& u_h,
393 const la::Vector& u_exact,
394 const la::SparseMatrix& matrix) {
395 // Compute the residual: error_u = u_exact - u_h
396 la::Vector error_u(dof_handler, u_h.getBlockSize());
397 error_u.add(-1.0, u_h);
398 error_u.add(1.0, u_exact);
399 error_u.compress();
400
401 // Compute the error norm using the provided matrix
402 la::Vector temp(dof_handler, u_h.getBlockSize());
403 // std::cout << "Error Calc with matrix: " << std::endl;
404 matrix.vmult(error_u, temp); // temp = matrix * error_u
405
406 if (la::dot(error_u, temp) < 0) {
407 std::cout
408 << "Warning: Negative value in error computation. Setting error to 0."
409 << std::endl;
410 return 0.0;
411 } else {
412 return std::sqrt(la::dot(error_u, temp));
413 }
414}

446template <typename MeshType>
447la::Vector projectConstantOntoModalSpace(
448 const dofs::DofHandler<MeshType>& dof_handler,
449 double constant_value,
450 const la::SparseMatrix& mass_matrix,
451 const la::Vector& rhs_const) {
452 la::Vector scaled_rhs(rhs_const);
453 scaled_rhs.scale(constant_value);
454
455 la::Vector c_proj(dof_handler);
456 la::LinearSolver solver(la::LinearSolver::PrecondOnly, la::LinearSolver::LU);
457 solver.setOperators(mass_matrix);
458 solver.solve(scaled_rhs, c_proj);
459
460 return c_proj;
461}

476template <typename MeshType>
477double forceEqualMean(
478 const dofs::DofHandler<MeshType>& dof_handler,
479 la::Vector& u,
480 const la::Vector& u_reference,
481 const la::SparseMatrix& mass_matrix,
482 const la::Vector& rhs_const,
483 double domain_measure) {
484 double avg_u = computeAverage(u, rhs_const, domain_measure);
485 double avg_reference = computeAverage(u_reference, rhs_const, domain_measure);
486
487 double shift_physical = avg_reference - avg_u;
488
489 la::Vector c_proj = projectConstantOntoModalSpace(
490 dof_handler, shift_physical, mass_matrix, rhs_const);
491
492 u.add(1.0, c_proj);
493 u.compress();
494
495 return shift_physical;
496}
