78template <typename MeshType, typename FeSpaceType, typename ExprType>
79double integrateFeExpression(
80 const MeshType& mesh,
81 assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
82 const ExprType& expression) {
83 // Create and use the scalar expression integrator
84 // Quadrature order 5 is sufficient for squared expressions of typical orders
85 assemble::ScalarExpressionIntegrator<MeshType, FeSpaceType, ExprType>
86 integrator(mesh, cell_manager, expression, 5);
87 return integrator.integrate();
88}

107template <
108 typename MeshType,
109 typename FeSpaceType,
110 typename DofHandlerType,
111 typename ExprType>
112double computeL2Error(
113 const MeshType& mesh,
114 assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
115 const DofHandlerType& dof_handler,
116 const la::Vector& u_h,
117 const ExprType& u_exact_expr) {
118 constexpr std::size_t Dim = MeshType::self_dim;
119 // Create the numerical solution as an FE expression
120 fe::FeExactExpressionFeProjection<Dim, FeSpaceType, DofHandlerType> u_h_expr(
121 u_h, cell_manager.getFeSpace(), dof_handler);
122
123 // Form the difference expression: (u_h - u_exact)
124 auto diff_expr = u_h_expr - u_exact_expr;
125
126 // Form the squared difference: (u_h - u_exact)^2
127 auto diff_squared = diff_expr * diff_expr;
128
129 // Integrate the squared difference over the domain
130 double l2_error_squared =
131 integrateFeExpression(mesh, cell_manager, diff_squared);
132
133 // Return the L2 error (square root of the integral)
134 return std::sqrt(std::max(0.0, l2_error_squared));
135}

152template <
153 typename MeshType,
154 typename FeSpaceType,
155 typename DofHandlerType,
156 typename ExprType>
157double computeH1SemiError(
158 const MeshType& mesh,
159 assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
160 const DofHandlerType& dof_handler,
161 const la::Vector& u_h,
162 const ExprType& grad_u_exact_expr) {
163 constexpr std::size_t Dim = MeshType::self_dim;
164
165 fe::FeGradProjectionExpr<Dim, FeSpaceType, DofHandlerType> grad_uh_expr(
166 u_h, cell_manager.getFeSpace(), dof_handler);
167
168 auto diff_expr = grad_uh_expr - grad_u_exact_expr;
169 auto diff_squared = fe::inner(diff_expr, diff_expr);
170
171 double h1_semi_error_squared =
172 integrateFeExpression(mesh, cell_manager, diff_squared);
173
174 return std::sqrt(std::max(0.0, h1_semi_error_squared));
175}

206template <std::size_t Dim>
207auto createPenaltyExpression(std::size_t order, double gamma) {
208 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
209 [order, gamma](
210 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
211 const auto& cell = *cells[fe_values.getCellId()];
212 const double penalty = gamma * order * order;
213 const double p_this = penalty / cell.diameter();
214 if (fe_values.hasFaceNeigh()) {
215 assert(&fe_values.neighValues());
216 const auto neigh_cell_id = fe_values.neighValues().getCellId();
217 const auto& neigh_cell = *cells[neigh_cell_id];
218 const double p_neigh = penalty / neigh_cell.diameter();
219 return std::max(p_this, p_neigh);
220 } else {
221 return p_this;
222 }
223 });
224}

236template <std::size_t Dim>
237auto createPenaltyExpressionKb(std::size_t order, double gamma) {
238 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
239 [order, gamma](
240 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
241 const auto& cell = *cells[fe_values.getCellId()];
242 const auto face_id = fe_values.getFaceId();
243 const double penalty = gamma * order * order;
244 const double Cinv = cell.measure() / cell.faceKb()[face_id];
245 const double facet_diam = cell.faceDiam()[face_id];
246 const double p_this = penalty * Cinv * (facet_diam / cell.measure());
247 if (fe_values.hasFaceNeigh()) {
248 assert(&fe_values.neighValues());
249 const auto neigh_cell_id = fe_values.neighValues().getCellId();
250 const auto& neigh_cell = *cells[neigh_cell_id];
251 const auto neigh_face_id = fe_values.neighValues().getFaceId();
252 const double Cinv_neigh =
253 neigh_cell.measure() / neigh_cell.faceKb()[neigh_face_id];
254 const double p_neigh =
255 penalty * Cinv_neigh * (facet_diam / neigh_cell.measure());
256 return std::max(p_this, p_neigh);
257 } else {
258 return p_this;
259 }
260 });
261}

276template <std::size_t Dim>
277auto createPenaltyExpressionHarmonic(std::size_t order, double gamma) {
278 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
279 [order, gamma](
280 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
281 const auto& cell = *cells[fe_values.getCellId()];
282 const auto face_id = fe_values.getFaceId();
283 const double facet_diam = cell.faceDiam()(face_id);
284 const double penalty = gamma * order * order;
285 const double Cinv = cell.measure() / cell.faceKb()[face_id];
286 const double p_this = penalty * Cinv * (facet_diam / cell.measure());
287
288 if (fe_values.hasFaceNeigh()) {
289 const auto neigh_cell_id = fe_values.neighValues().getCellId();
290 const auto& neigh_cell = *cells[neigh_cell_id];
291 const auto neigh_face_id = fe_values.neighValues().getFaceId();
292 const double Cinv_neigh =
293 neigh_cell.measure() / neigh_cell.faceKb()[neigh_face_id];
294 const double p_neigh =
295 penalty * Cinv_neigh * (facet_diam / neigh_cell.measure());
296 // Media armonica tra p_this e p_neigh
297 return 2.0 * p_this * p_neigh / (p_this + p_neigh);
298 } else {
299 return p_this;
300 }
301 });
302}

310template <std::size_t Dim, typename VelocityFunc>
311class FeExpressionUpwindFromFunctionWrapper
312 : public fe::FeExpressionBase<
313 Dim,
314 FeExpressionUpwindFromFunctionWrapper<Dim, VelocityFunc>> {
315 public:
316 static constexpr bool is_leaf = true;
317 static constexpr uint8_t is_bilinear = fe::IsNothing;
318
319 explicit constexpr FeExpressionUpwindFromFunctionWrapper(
320 const VelocityFunc& velocity_func)
321 : m_velocity_func(velocity_func) {}
322 FeExpressionUpwindFromFunctionWrapper(
323 const FeExpressionUpwindFromFunctionWrapper&) = delete;
324 FeExpressionUpwindFromFunctionWrapper& operator=(
325 const FeExpressionUpwindFromFunctionWrapper&) = delete;
326 FeExpressionUpwindFromFunctionWrapper(
327 FeExpressionUpwindFromFunctionWrapper&&) = delete;
328 FeExpressionUpwindFromFunctionWrapper& operator=(
329 FeExpressionUpwindFromFunctionWrapper&&) = delete;
330
331 template <std::size_t I = 0, std::size_t J = 0>
332 decltype(auto) evaluate(
333 const fe::FeValues<Dim>&, std::size_t, std::size_t, std::size_t q) const {
334 return m_evals[q];
335 }
336
337 constexpr fe::UpdateFlagsType computeUpdateFlags() const {
338 return fe::UpdateQuadraturePts | fe::UpdateNormals;
339 }
340 constexpr std::size_t computeOrder() const { return 0; }
341
342 constexpr bool precompute(
343 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>&) const {
344 const Eigen::Index nqpts = fe_values.weights().size();
345 assert(fe_values.qpts().cols() == nqpts);
346 m_evals.resize(nqpts);
347 const auto& normal = fe_values.getFaceNormal();
348 for (Eigen::Index q = 0; q < nqpts; ++q) {
349 const auto vel = m_velocity_func(fe_values.qpts().col(q));
350 const double vel_dot_n = vel.dot(normal);
351 m_evals[q] = 0.5 * (std::abs(vel_dot_n) + vel_dot_n);
352 }
353 return true;
354 }
355
356 protected:
357 const VelocityFunc m_velocity_func;
358 mutable std::vector<double> m_evals;
359};

391template <std::size_t Dim>
392auto createUpwindFluxConstant(double vx, double vy = 0.0, double vz = 0.0) {
393 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
394 [vx, vy, vz](
395 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
396 const auto& cell = *cells[fe_values.getCellId()];
397 const auto face_id = fe_values.getFaceId();
398 const auto& normal = cell.normals().col(face_id);
399
400 // Compute velocity . n
401 double vel_dot_n = vx * normal(0);
402 if constexpr (Dim >= 2) {
403 vel_dot_n += vy * normal(1);
404 }
405 if constexpr (Dim >= 3) {
406 vel_dot_n += vz * normal(2);
407 }
408
409 // Upwind flux: 0.5 * (|velocity . n| + velocity . n)
410 return 0.5 * (std::abs(vel_dot_n) + vel_dot_n);
411 });
412}

429template <std::size_t Dim, typename VelocityFunc>
430auto createUpwindFluxVariable(VelocityFunc velocity_func) {
431 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
432 [velocity_func](
433 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
434 const auto& cell = *cells[fe_values.getCellId()];
435 const auto face_id = fe_values.getFaceId();
436 const auto& normal = cell.normals().col(face_id);
437
438 // Face centroid: average of the two edge vertices for 2D polygons.
439 const Eigen::Index nvtxs = cell.vtxs().cols();
440 const Eigen::Vector<double, Dim> face_center =
441 0.5 *
442 (cell.vtxs().col(face_id) + cell.vtxs().col((face_id + 1) % nvtxs));
443
444 // Evaluate velocity at face center and compute vel . n
445 const auto vel = velocity_func(face_center);
446 const double vel_dot_n = vel.dot(normal);
447
448 // Upwind flux: 0.5 * (|velocity . n| + velocity . n)
449 return 0.5 * (std::abs(vel_dot_n) + vel_dot_n);
450 });
451}

464template <typename MeshType>
465double computeError(
466 const dofs::DofHandler<MeshType>& dof_handler,
467 const la::Vector& u_h,
468 const la::Vector& u_exact,
469 const la::SparseMatrix& matrix) {
470 // Compute the residual: error_u = u_exact - u_h
471 la::Vector error_u(dof_handler, u_h.getBlockSize());
472 error_u.add(-1.0, u_h);
473 error_u.add(1.0, u_exact);
474 error_u.compress();
475
476 // Compute the error norm using the provided matrix
477 la::Vector temp(dof_handler, u_h.getBlockSize());
478 // std::cout << "Error Calc with matrix: " << std::endl;
479 matrix.vmult(error_u, temp); // temp = matrix * error_u
480
481 if (la::dot(error_u, temp) < 0) {
482 std::cout
483 << "Warning: Negative value in error computation. Setting error to 0."
484 << std::endl;
485 return 0.0;
486 } else {
487 return std::sqrt(la::dot(error_u, temp));
488 }
489}

521template <typename MeshType>
522la::Vector projectConstantOntoModalSpace(
523 const dofs::DofHandler<MeshType>& dof_handler,
524 double constant_value,
525 const la::SparseMatrix& mass_matrix,
526 const la::Vector& rhs_const) {
527 la::Vector scaled_rhs(rhs_const);
528 scaled_rhs.scale(constant_value);
529
530 la::Vector c_proj(dof_handler);
531 la::LinearSolver solver(la::LinearSolver::PrecondOnly, la::LinearSolver::LU);
532 solver.setOperators(mass_matrix);
533 solver.solve(scaled_rhs, c_proj);
534
535 return c_proj;
536}

551template <typename MeshType>
552double forceEqualMean(
553 const dofs::DofHandler<MeshType>& dof_handler,
554 la::Vector& u,
555 const la::Vector& u_reference,
556 const la::SparseMatrix& mass_matrix,
557 const la::Vector& rhs_const,
558 double domain_measure) {
559 double avg_u = computeAverage(u, rhs_const, domain_measure);
560 double avg_reference = computeAverage(u_reference, rhs_const, domain_measure);
561
562 double shift_physical = avg_reference - avg_u;
563
564 la::Vector c_proj = projectConstantOntoModalSpace(
565 dof_handler, shift_physical, mass_matrix, rhs_const);
566
567 u.add(1.0, c_proj);
568 u.compress();
569
570 return shift_physical;
571}

581template <
582 typename MeshType,
583 typename CellManagerType,
584 typename DofHandlerType,
585 typename FeSpaceType,
586 typename Expression>
602la::Vector l2ProjectExpressionScalar(
603 const MeshType& mesh,
604 CellManagerType& cell_manager,
605 const DofHandlerType& dof_handler,
606 const FeSpaceType& fe_space,
607 const Expression& expr,
608 const la::SparseMatrix& mass_scalar) {
609 namespace fe = vulpes::fe;
610
611 fe::TestFunction v(fe_space);
612 auto wf_projection = expr * v;
613
614 assemble::WeakFormIntegratorSimplexify int_smplx_projection(
615 mesh, cell_manager, std::tuple(&wf_projection), std::tuple());
616
617 la::Vector rhs(dof_handler);
618 const std::size_t n_cells = static_cast<std::size_t>(mesh.numCells());
619 for (std::size_t cell_id = 0; cell_id < n_cells; ++cell_id) {
620 int_smplx_projection.integrate(cell_id);
621 rhs.insertValues(
622 int_smplx_projection.getIntegral(&wf_projection),
623 dof_handler.getCellToDof()[cell_id]);
624 }
625 rhs.compress();
626
627 la::Vector projected(dof_handler);
628 la::LinearSolver solver(la::LinearSolver::PrecondOnly, la::LinearSolver::LU);
629 solver.setOperators(mass_scalar);
630 solver.solve(rhs, projected);
631 return projected;
632}

671template <
672 std::size_t VecSize,
673 std::size_t NFields,
674 typename MeshType,
675 typename CellManagerType,
676 typename DofHandlerType,
677 typename FeSpaceType,
678 typename Expression>
700la::Vector l2ProjectExpressionBlockedSubset(
701 const MeshType& mesh,
702 CellManagerType& cell_manager,
703 const DofHandlerType& dof_handler,
704 const FeSpaceType& fe_space,
705 const Expression& expr,
706 std::size_t component_offset,
707 const la::SparseMatrix& mass_blocked) {
708 namespace fe = vulpes::fe;
709
710 assert(component_offset + VecSize <= NFields);
711
712 fe::TestFunction<FeSpaceType, 0, VecSize> v(fe_space);
713 auto wf_projection = fe::inner(expr, v);
714
715 assemble::WeakFormIntegratorSimplexify int_smplx_projection(
716 mesh, cell_manager, std::tuple(&wf_projection), std::tuple());
717
718 la::Vector rhs(dof_handler, NFields);
719 const auto& cell_to_dof = dof_handler.getCellToDof();
720 const Eigen::Index ndof_comp = fe_space.getBasis().size();
721 const std::size_t n_cells = static_cast<std::size_t>(mesh.numCells());
722
723 std::array<PetscScalar, NFields> block_values{};
724 for (std::size_t cell_id = 0; cell_id < n_cells; ++cell_id) {
725 int_smplx_projection.integrate(cell_id);
726 const auto local = int_smplx_projection.getIntegral(&wf_projection);
727
728 for (Eigen::Index i = 0; i < ndof_comp; ++i) {
729 block_values.fill(0.0);
730 for (std::size_t c = 0; c < VecSize; ++c) {
731 block_values[component_offset + c] = local(
732 i * static_cast<Eigen::Index>(VecSize) +
733 static_cast<Eigen::Index>(c),
734 0);
735 }
736 const PetscInt block_index = cell_to_dof[cell_id][i];
737 rhs.insertValuesBlocked(block_values.data(), &block_index, 1, ADD_VALUES);
738 }
739 }
740 rhs.compress();
741
742 la::Vector projected(dof_handler, NFields);
743 la::LinearSolver solver(la::LinearSolver::PrecondOnly, la::LinearSolver::LU);
744 solver.setOperators(mass_blocked);
745 solver.solve(rhs, projected);
746 return projected;
747}
