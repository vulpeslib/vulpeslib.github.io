52template <std::size_t Dim>
53auto createPenaltyExpression(std::size_t order, double gamma) {
54 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
55 [order, gamma](
56 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
57 const auto& cell = *cells[fe_values.getCellId()];
58 const double penalty = gamma * order * order;
59 const double p_this = penalty / cell.diameter();
60 if (fe_values.hasFaceNeigh()) {
61 assert(&fe_values.neighValues());
62 const auto neigh_cell_id = fe_values.neighValues().getCellId();
63 const auto& neigh_cell = *cells[neigh_cell_id];
64 const double p_neigh = penalty / neigh_cell.diameter();
65 return std::max(p_this, p_neigh);
66 } else {
67 return p_this;
68 }
69 });
70}

82template <std::size_t Dim>
83auto createPenaltyExpressionKb(std::size_t order, double gamma) {
84 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
85 [order, gamma](
86 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
87 const auto& cell = *cells[fe_values.getCellId()];
88 const auto face_id = fe_values.getFaceId();
89 const double penalty = gamma * order * order;
90 const double Cinv = cell.measure() / cell.faceKb()[face_id];
91 const double facet_diam = cell.faceDiam()[face_id];
92 const double p_this = penalty * Cinv * (facet_diam / cell.measure());
93 if (fe_values.hasFaceNeigh()) {
94 assert(&fe_values.neighValues());
95 const auto neigh_cell_id = fe_values.neighValues().getCellId();
96 const auto& neigh_cell = *cells[neigh_cell_id];
97 const auto neigh_face_id = fe_values.neighValues().getFaceId();
98 const double Cinv_neigh =
99 neigh_cell.measure() / neigh_cell.faceKb()[neigh_face_id];
100 const double p_neigh =
101 penalty * Cinv_neigh * (facet_diam / neigh_cell.measure());
102 return std::max(p_this, p_neigh);
103 } else {
104 return p_this;
105 }
106 });
107}

122template <std::size_t Dim>
123auto createPenaltyExpressionHarmonic(std::size_t order, double gamma) {
124 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
125 [order, gamma](
126 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
127 const auto& cell = *cells[fe_values.getCellId()];
128 const auto face_id = fe_values.getFaceId();
129 const double facet_diam = cell.faceDiam()(face_id);
130 const double penalty = gamma * order * order;
131 const double Cinv = cell.measure() / cell.faceKb()[face_id];
132 const double p_this = penalty * Cinv * (facet_diam / cell.measure());
133
134 if (fe_values.hasFaceNeigh()) {
135 const auto neigh_cell_id = fe_values.neighValues().getCellId();
136 const auto& neigh_cell = *cells[neigh_cell_id];
137 const auto neigh_face_id = fe_values.neighValues().getFaceId();
138 const double Cinv_neigh =
139 neigh_cell.measure() / neigh_cell.faceKb()[neigh_face_id];
140 const double p_neigh =
141 penalty * Cinv_neigh * (facet_diam / neigh_cell.measure());
142 // Media armonica tra p_this e p_neigh
143 return 2.0 * p_this * p_neigh / (p_this + p_neigh);
144 } else {
145 return p_this;
146 }
147 });
148}

161template <typename MeshType>
162double computeError(
163 const dofs::DofHandler<MeshType>& dof_handler,
164 const la::Vector& u_h,
165 const la::Vector& u_exact,
166 const la::SparseMatrix& matrix) {
167 // Compute the residual: error_u = u_exact - u_h
168 la::Vector error_u(dof_handler, u_h.getBlockSize());
169 error_u.add(-1.0, u_h);
170 error_u.add(1.0, u_exact);
171 error_u.compress();
172
173 // Compute the error norm using the provided matrix
174 la::Vector temp(dof_handler, u_h.getBlockSize());
175 matrix.vmult(error_u, temp); // temp = matrix * error_u
176 return std::sqrt(la::dot(error_u, temp));
177}

209template <typename MeshType>
210la::Vector projectConstantOntoModalSpace(
211 const dofs::DofHandler<MeshType>& dof_handler,
212 double constant_value,
213 const la::SparseMatrix& mass_matrix,
214 const la::Vector& rhs_const) {
215 la::Vector scaled_rhs(rhs_const);
216 scaled_rhs.scale(constant_value);
217
218 la::Vector c_proj(dof_handler);
219 la::LinearSolver solver(la::LinearSolver::PrecondOnly, la::LinearSolver::LU);
220 solver.setOperators(mass_matrix);
221 solver.solve(scaled_rhs, c_proj);
222
223 return c_proj;
224}

239template <typename MeshType>
240double forceEqualMean(
241 const dofs::DofHandler<MeshType>& dof_handler,
242 la::Vector& u,
243 const la::Vector& u_reference,
244 const la::SparseMatrix& mass_matrix,
245 const la::Vector& rhs_const,
246 double domain_measure) {
247 double avg_u = computeAverage(u, rhs_const, domain_measure);
248 double avg_reference = computeAverage(u_reference, rhs_const, domain_measure);
249
250 double shift_physical = avg_reference - avg_u;
251
252 la::Vector c_proj = projectConstantOntoModalSpace(
253 dof_handler, shift_physical, mass_matrix, rhs_const);
254
255 u.add(1.0, c_proj);
256 u.compress();
257
258 return shift_physical;
259}
