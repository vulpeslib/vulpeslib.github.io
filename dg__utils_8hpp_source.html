60template <typename MeshType, typename FeSpaceType, typename ExprType>
61double integrateFeExpression(
62 const MeshType& mesh,
63 assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
64 const ExprType& expression) {
65 // Create and use the scalar expression integrator
66 // Quadrature order 5 is sufficient for squared expressions of typical orders
67 assemble::ScalarExpressionIntegrator<MeshType, FeSpaceType, ExprType>
68 integrator(mesh, cell_manager, expression, 5);
69 return integrator.integrate();
70}

89template <
90 typename MeshType,
91 typename FeSpaceType,
92 typename DofHandlerType,
93 typename ExprType>
94double computeL2Error(
95 const MeshType& mesh,
96 assemble::CellManager<MeshType, FeSpaceType>& cell_manager,
97 const DofHandlerType& dof_handler,
98 const la::Vector& u_h,
99 const ExprType& u_exact_expr) {
100 constexpr std::size_t Dim = MeshType::self_dim;
101 // Create the numerical solution as an FE expression
102 fe::FeExactExpressionFeProjection<Dim, FeSpaceType, DofHandlerType> u_h_expr(
103 u_h, cell_manager.getFeSpace(), dof_handler);
104
105 // Form the difference expression: (u_h - u_exact)
106 auto diff_expr = u_h_expr - u_exact_expr;
107
108 // Form the squared difference: (u_h - u_exact)^2
109 auto diff_squared = diff_expr * diff_expr;
110
111 // Integrate the squared difference over the domain
112 double l2_error_squared =
113 integrateFeExpression(mesh, cell_manager, diff_squared);
114
115 // Return the L2 error (square root of the integral)
116 return std::sqrt(std::max(0.0, l2_error_squared));
117}

128template <std::size_t Dim>
129auto createPenaltyExpression(std::size_t order, double gamma) {
130 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
131 [order, gamma](
132 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
133 const auto& cell = *cells[fe_values.getCellId()];
134 const double penalty = gamma * order * order;
135 const double p_this = penalty / cell.diameter();
136 if (fe_values.hasFaceNeigh()) {
137 assert(&fe_values.neighValues());
138 const auto neigh_cell_id = fe_values.neighValues().getCellId();
139 const auto& neigh_cell = *cells[neigh_cell_id];
140 const double p_neigh = penalty / neigh_cell.diameter();
141 return std::max(p_this, p_neigh);
142 } else {
143 return p_this;
144 }
145 });
146}

158template <std::size_t Dim>
159auto createPenaltyExpressionKb(std::size_t order, double gamma) {
160 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
161 [order, gamma](
162 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
163 const auto& cell = *cells[fe_values.getCellId()];
164 const auto face_id = fe_values.getFaceId();
165 const double penalty = gamma * order * order;
166 const double Cinv = cell.measure() / cell.faceKb()[face_id];
167 const double facet_diam = cell.faceDiam()[face_id];
168 const double p_this = penalty * Cinv * (facet_diam / cell.measure());
169 if (fe_values.hasFaceNeigh()) {
170 assert(&fe_values.neighValues());
171 const auto neigh_cell_id = fe_values.neighValues().getCellId();
172 const auto& neigh_cell = *cells[neigh_cell_id];
173 const auto neigh_face_id = fe_values.neighValues().getFaceId();
174 const double Cinv_neigh =
175 neigh_cell.measure() / neigh_cell.faceKb()[neigh_face_id];
176 const double p_neigh =
177 penalty * Cinv_neigh * (facet_diam / neigh_cell.measure());
178 return std::max(p_this, p_neigh);
179 } else {
180 return p_this;
181 }
182 });
183}

198template <std::size_t Dim>
199auto createPenaltyExpressionHarmonic(std::size_t order, double gamma) {
200 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
201 [order, gamma](
202 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
203 const auto& cell = *cells[fe_values.getCellId()];
204 const auto face_id = fe_values.getFaceId();
205 const double facet_diam = cell.faceDiam()(face_id);
206 const double penalty = gamma * order * order;
207 const double Cinv = cell.measure() / cell.faceKb()[face_id];
208 const double p_this = penalty * Cinv * (facet_diam / cell.measure());
209
210 if (fe_values.hasFaceNeigh()) {
211 const auto neigh_cell_id = fe_values.neighValues().getCellId();
212 const auto& neigh_cell = *cells[neigh_cell_id];
213 const auto neigh_face_id = fe_values.neighValues().getFaceId();
214 const double Cinv_neigh =
215 neigh_cell.measure() / neigh_cell.faceKb()[neigh_face_id];
216 const double p_neigh =
217 penalty * Cinv_neigh * (facet_diam / neigh_cell.measure());
218 // Media armonica tra p_this e p_neigh
219 return 2.0 * p_this * p_neigh / (p_this + p_neigh);
220 } else {
221 return p_this;
222 }
223 });
224}

237template <typename MeshType>
238double computeError(
239 const dofs::DofHandler<MeshType>& dof_handler,
240 const la::Vector& u_h,
241 const la::Vector& u_exact,
242 const la::SparseMatrix& matrix) {
243 // Compute the residual: error_u = u_exact - u_h
244 la::Vector error_u(dof_handler, u_h.getBlockSize());
245 error_u.add(-1.0, u_h);
246 error_u.add(1.0, u_exact);
247 error_u.compress();
248
249 // Compute the error norm using the provided matrix
250 la::Vector temp(dof_handler, u_h.getBlockSize());
251 matrix.vmult(error_u, temp); // temp = matrix * error_u
252 return std::sqrt(la::dot(error_u, temp));
253}

285template <typename MeshType>
286la::Vector projectConstantOntoModalSpace(
287 const dofs::DofHandler<MeshType>& dof_handler,
288 double constant_value,
289 const la::SparseMatrix& mass_matrix,
290 const la::Vector& rhs_const) {
291 la::Vector scaled_rhs(rhs_const);
292 scaled_rhs.scale(constant_value);
293
294 la::Vector c_proj(dof_handler);
295 la::LinearSolver solver(la::LinearSolver::PrecondOnly, la::LinearSolver::LU);
296 solver.setOperators(mass_matrix);
297 solver.solve(scaled_rhs, c_proj);
298
299 return c_proj;
300}

315template <typename MeshType>
316double forceEqualMean(
317 const dofs::DofHandler<MeshType>& dof_handler,
318 la::Vector& u,
319 const la::Vector& u_reference,
320 const la::SparseMatrix& mass_matrix,
321 const la::Vector& rhs_const,
322 double domain_measure) {
323 double avg_u = computeAverage(u, rhs_const, domain_measure);
324 double avg_reference = computeAverage(u_reference, rhs_const, domain_measure);
325
326 double shift_physical = avg_reference - avg_u;
327
328 la::Vector c_proj = projectConstantOntoModalSpace(
329 dof_handler, shift_physical, mass_matrix, rhs_const);
330
331 u.add(1.0, c_proj);
332 u.compress();
333
334 return shift_physical;
335}
