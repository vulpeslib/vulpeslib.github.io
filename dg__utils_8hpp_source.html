50template <std::size_t Dim>
51auto createPenaltyExpression(std::size_t order, double gamma) {
52 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
53 [order, gamma](
54 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
55 const auto& cell = *cells[fe_values.getCellId()];
56 const double penalty = gamma * order * order;
57 const double p_this = penalty / cell.diameter();
58 if (fe_values.hasFaceNeigh()) {
59 assert(&fe_values.neighValues());
60 const auto neigh_cell_id = fe_values.neighValues().getCellId();
61 const auto& neigh_cell = *cells[neigh_cell_id];
62 const double p_neigh = penalty / neigh_cell.diameter();
63 return std::max(p_this, p_neigh);
64 } else {
65 return p_this;
66 }
67 });
68}

80template <std::size_t Dim>
81auto createPenaltyExpressionKb(std::size_t order, double gamma) {
82 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
83 [order, gamma](
84 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
85 const auto& cell = *cells[fe_values.getCellId()];
86 const auto face_id = fe_values.getFaceId();
87 const double penalty = gamma * order * order;
88 const double Cinv = cell.measure() / cell.faceKb()[face_id];
89 const double facet_diam = cell.faceDiam()[face_id];
90 const double p_this = penalty * Cinv * (facet_diam / cell.measure());
91 if (fe_values.hasFaceNeigh()) {
92 assert(&fe_values.neighValues());
93 const auto neigh_cell_id = fe_values.neighValues().getCellId();
94 const auto& neigh_cell = *cells[neigh_cell_id];
95 const auto neigh_face_id = fe_values.neighValues().getFaceId();
96 const double Cinv_neigh =
97 neigh_cell.measure() / neigh_cell.faceKb()[neigh_face_id];
98 const double p_neigh =
99 penalty * Cinv_neigh * (facet_diam / neigh_cell.measure());
100 return std::max(p_this, p_neigh);
101 } else {
102 return p_this;
103 }
104 });
105}

120template <std::size_t Dim>
121auto createPenaltyExpressionHarmonic(std::size_t order, double gamma) {
122 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
123 [order, gamma](
124 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
125 const auto& cell = *cells[fe_values.getCellId()];
126 const auto face_id = fe_values.getFaceId();
127 const double facet_diam = cell.faceDiam()(face_id);
128 const double penalty = gamma * order * order;
129 const double Cinv = cell.measure() / cell.faceKb()[face_id];
130 const double p_this = penalty * Cinv * (facet_diam / cell.measure());
131
132 if (fe_values.hasFaceNeigh()) {
133 const auto neigh_cell_id = fe_values.neighValues().getCellId();
134 const auto& neigh_cell = *cells[neigh_cell_id];
135 const auto neigh_face_id = fe_values.neighValues().getFaceId();
136 const double Cinv_neigh =
137 neigh_cell.measure() / neigh_cell.faceKb()[neigh_face_id];
138 const double p_neigh =
139 penalty * Cinv_neigh * (facet_diam / neigh_cell.measure());
140 // Media armonica tra p_this e p_neigh
141 return 2.0 * p_this * p_neigh / (p_this + p_neigh);
142 } else {
143 return p_this;
144 }
145 });
146}

159template <typename MeshType>
160double computeError(
161 const dofs::DofHandler<MeshType>& dof_handler,
162 const la::Vector& u_h,
163 const la::Vector& u_exact,
164 const la::SparseMatrix& matrix) {
165 // Compute the residual: error_u = u_exact - u_h
166 la::Vector error_u(dof_handler, u_h.getBlockSize());
167 error_u.add(-1.0, u_h);
168 error_u.add(1.0, u_exact);
169 error_u.compress();
170
171 // Compute the error norm using the provided matrix
172 la::Vector temp(dof_handler, u_h.getBlockSize());
173 matrix.vmult(error_u, temp); // temp = matrix * error_u
174 return std::sqrt(la::dot(error_u, temp));
175}
