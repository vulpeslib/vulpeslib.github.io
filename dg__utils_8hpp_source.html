41template <std::size_t Dim>
42auto createPenaltyExpression(std::size_t order, double gamma) {
43 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
44 [order, gamma](
45 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
46 const auto& cell = *cells[fe_values.getCellId()];
47 const double penalty = gamma * order * order;
48 const double p_this = penalty / cell.diameter();
49 if (fe_values.hasFaceNeigh()) {
50 assert(&fe_values.neighValues());
51 const auto neigh_cell_id = fe_values.neighValues().getCellId();
52 const auto& neigh_cell = *cells[neigh_cell_id];
53 const double p_neigh = penalty / neigh_cell.diameter();
54 return std::max(p_this, p_neigh);
55 } else {
56 return p_this;
57 }
58 });
59}

61template <std::size_t Dim>
62auto createPenaltyExpressionKb(std::size_t order, double gamma) {
63 return fe::FeExpressionCellConstantNeedNeigh<Dim>(
64 [order, gamma](
65 const fe::FeValues<Dim>& fe_values, const fe::CellsType<Dim>& cells) {
66 const auto& cell = *cells[fe_values.getCellId()];
67 const auto face_id = fe_values.getFaceId();
68 const double penalty = gamma * order * order;
69 const double Cinv = cell.measure() / cell.faceKb()[face_id];
70 const double facet_diam = cell.faceDiam()[face_id];
71 const double p_this = penalty * Cinv * (facet_diam / cell.measure());
72 if (fe_values.hasFaceNeigh()) {
73 assert(&fe_values.neighValues());
74 const auto neigh_cell_id = fe_values.neighValues().getCellId();
75 const auto& neigh_cell = *cells[neigh_cell_id];
76 const auto neigh_face_id = fe_values.neighValues().getFaceId();
77 const double Cinv_neigh =
78 neigh_cell.measure() / neigh_cell.faceKb()[neigh_face_id];
79 const double p_neigh =
80 penalty * Cinv_neigh * (facet_diam / neigh_cell.measure());
81 return std::max(p_this, p_neigh);
82 } else {
83 return p_this;
84 }
85 });
86}

88template <typename MeshType>
89double computeError(
90 const dofs::DofHandler<MeshType>& dof_handler,
91 const la::Vector& u_h,
92 const la::Vector& u_exact,
93 const la::SparseMatrix& matrix) {
94 // Compute the residual: error_u = u_exact - u_h
95 la::Vector error_u(dof_handler);
96 error_u.add(-1.0, u_h);
97 error_u.add(1.0, u_exact);
98 error_u.compress();
99
100 // Compute the error norm using the provided matrix
101 la::Vector temp(dof_handler);
102 matrix.vmult(error_u, temp); // temp = matrix * error_u
103 return std::sqrt(la::dot(error_u, temp));
104}
