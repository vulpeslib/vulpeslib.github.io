104template <std::size_t dim, typename T>
105QuadratureRule<T, dim> buildTensorQuadrature(
106 const QuadratureRule1D<T>& rule1d) {
107 if constexpr (dim == 1) {
108 return rule1d;
109 }
110 QuadratureRule<T, dim> rule;
111 const std::size_t size_1d = rule1d.nodes.size();
112 const std::size_t size_nd = vmath::upow(size_1d, dim);
113 rule.nodes.resize(size_nd);
114 for (std::size_t i = 0; i < size_nd; ++i) {
115 rule.nodes[i].weight = T(1);
116 // this is like writing `i` in radix `size_1d`
117 std::size_t idx = i;
118 for (std::size_t d = 0; d < dim; ++d) {
119 std::size_t digit = idx % size_1d;
120 idx /= size_1d;
121 rule.nodes[i].weight *= rule1d.nodes[digit].weight;
122 rule.nodes[i].point[d] = rule1d.nodes[digit].point[0];
123 }
124 }
125 return rule;
126}
