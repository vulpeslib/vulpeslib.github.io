22template <std::floating_point T>
23T EvaluateLegendre(T x, size_t n) {
24 if (n == 0) return 1.0;
25 T p1 = 1.0;
26 T p2 = x;
27 T p3 = x;
28 for (size_t i = 1; i < n; ++i) {
29 p3 = (T(2ull * i + 1ull) * x * p2 - T(i) * p1) / T(i + 1ull);
30 p1 = p2;
31 p2 = p3;
32 }
33 return p3;
34}

47template <std::floating_point T>
48std::array<T, 2> EvaluateLegendreDer(T x, size_t n) {
49 if (n == 0) return {T(1.0), T(0.0)};

51 T p2 = x;
52 T p3 = x;

113template <std::floating_point T>
114std::vector<std::array<T, 2>> GaussLegendreQuadrature(size_t n) {
115 if (n == 0) {
116 return {};
117 }
118 if (n == 1) {
119 return {{T(0), T(2)}};
120 }
121 std::vector<std::array<T, 2>> xw(n);
122 for (size_t i = 0; i < n / 2; ++i) {
123 // initial approximation
124 xw[i][0] = std::cos(std::numbers::pi_v<T> * (i + T(0.75)) / (n + T(0.5)));
125 constexpr auto ITERMAX = 10 * sizeof(T);
126 for (int iter = 0; iter < ITERMAX; ++iter) {
127 const auto [px, pdx] = EvaluateLegendreDer(xw[i][0], n);
128 // Newton method
129 const auto delta = px / pdx;
130 xw[i][0] -= delta;
131 if (std::abs(delta) < std::numeric_limits<T>::epsilon()) break;
132 // TODO: handle output in a more gracefull way
133 if (iter == ITERMAX - 1)
134 std::cout << "WARNING: reached max iterations" << std::endl;
135 }
136 xw[n - i - 1][0] = -xw[i][0];
137 const auto [px, pdx] = EvaluateLegendreDer(xw[i][0], n);
138 const auto w = T(2) / ((T(1) - xw[i][0] * xw[i][0]) * pdx * pdx);
139 xw[i][1] = w;
140 xw[n - i - 1][1] = w;
141 }
142 const auto [px, pdx] = EvaluateLegendreDer(T(0), n);
143 xw[n / 2][1] = T(2) / (pdx * pdx);
144 return xw;
