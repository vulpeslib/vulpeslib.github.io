61template <typename FloatType, typename IndexType>
62class Mesh3D : public MeshBase<
63 FloatType,
64 IndexType,
65 3,
66 3,
67 Mesh3D<FloatType, IndexType>> {
68 public:
69 using SelfPoint = Point<FloatType, 3>;
70 using BaseType =
71 MeshBase<FloatType, IndexType, 3, 3, Mesh3D<FloatType, IndexType>>;
72 using typename BaseType::CellToVtxMap;
73 using typename BaseType::CellToVtxMapUncompressed;
74 // for each cell a vector with the vertex indices of each face of that cell
75 // (like in VTU and voro++)
76 using CellsFaceVtxsUncompressed = std::vector<std::vector<IndexType>>;
77 using typename BaseType::CellToEdgeMap;
78 using typename BaseType::EdgeToCellMap;
79 using typename BaseType::EdgeToVtxMap;
80 using typename BaseType::EdgeVtxs;
81
82 Mesh3D() = default;
83
84 template <typename MeshType>
85 static Mesh3D extrude(
86 const MeshBase<FloatType, IndexType, 2, 2, MeshType>& mesh_2d,
87 const Eigen::Vector<FloatType, 2> z_range,
88 IndexType n_elems);
89
90 explicit Mesh3D(
91 const std::vector<SelfPoint>& vertexes,
92 const CellToVtxMapUncompressed& cell_vertices,
93 const CellsFaceVtxsUncompressed& cells_face_vtxs) {
94 this->m_vertexes = vertexes;
95 this->m_cell_to_vtx = cell_vertices.template compress<IndexType>();
96 buildFacetToCellAndFacetToVertex(cells_face_vtxs);
97 this->buildInternalMaps();
98 buildEdgeMaps();
99 }
100
101 template <class Archive>
102 void serialize(Archive& ar, const unsigned int) {
103 ar& static_cast<BaseType&>(*this);
104 ar & m_cell_to_edge;
105 ar & m_edge_to_cell;
106 ar & m_edge_to_vtx;
107 }
108
109 friend bool operator==(const Mesh3D& lhs, const Mesh3D& rhs) {
110 return (static_cast<const BaseType&>(lhs) ==
111 static_cast<const BaseType&>(rhs)) &&
112 (lhs.m_cell_to_edge == rhs.m_cell_to_edge) &&
113 (lhs.m_edge_to_cell == rhs.m_edge_to_cell) &&
114 (lhs.m_edge_to_vtx == rhs.m_edge_to_vtx);
115 }
116
117 EdgeToVtxMap& getEdgeToVtxMap() { return m_edge_to_vtx; }
118 const EdgeToVtxMap& getEdgeToVtxMap() const { return m_edge_to_vtx; }
119
120 CellToEdgeMap& getCellToEdgeMap() { return m_cell_to_edge; }
121 const CellToEdgeMap& getCellToEdgeMap() const { return m_cell_to_edge; }
122
123 EdgeToCellMap& getEdgeToCellMap() { return m_edge_to_cell; }
124 const EdgeToCellMap& getEdgeToCellMap() const { return m_edge_to_cell; }
125
126 constexpr IndexType nEntity(std::size_t dim) const {
127 if (dim == 0) {
128 return this->getVertexes().size();
129 } else if (dim == 1) {
130 return this->getEdgeToCellMap().size();
131 } else if (dim == 2) {
132 return this->getFacetToCellMap().size();
133 } else if (dim == 3) {
134 assert(this->m_cell_to_facet.size() == this->m_dual_graph.size());
135 return this->getDualMap().size();
136 } else {
137 assert(false);
138 return -1;
139 }
140 }
141
142 template <std::size_t dim>
143 constexpr auto& getEntityToCellMap() const {
144 if constexpr (dim == 0) {
145 return this->getVtxToCellMap();
146 } else if constexpr (dim == 1) {
147 return this->getEdgeToCellMap();
148 } else if constexpr (dim == 2) {
149 return this->getFacetToCellMap();
150 } else {
151 static_assert(dim * 0);
152 }
153 }
154
155 protected:
156 // necessitates also m_cell_to_vtx
157 void buildFacetToCellAndFacetToVertex(
158 const CellsFaceVtxsUncompressed& cells_face_vtxs);
159 void buildEdgeMaps();
160
161 // handy for building the C2E and E2C maps
162 EdgeToVtxMap m_edge_to_vtx;
163 // needed by the dof handler
164 CellToEdgeMap m_cell_to_edge;
165 EdgeToCellMap m_edge_to_cell;
166};

168template <typename FloatType, typename IndexType>
169template <typename MeshType>
170Mesh3D<FloatType, IndexType> Mesh3D<FloatType, IndexType>::extrude(
171 const MeshBase<FloatType, IndexType, 2, 2, MeshType>& mesh_2d,
172 const Eigen::Vector<FloatType, 2> z_range,
173 IndexType n_elems) {
174 assert(n_elems > 0);
175 Mesh3D mesh;
176 const Eigen::Vector<FloatType, -1> zz =
177 Eigen::Vector<FloatType, -1>::LinSpaced(
178 n_elems + 1, z_range[0], z_range[1]);
179 // 1. build vertexes
180 for (IndexType k = 0; k < n_elems + 1; ++k) {
181 for (const auto& pt_2d : mesh_2d.getVertexes()) {
182 mesh.m_vertexes.push_back({pt_2d[0], pt_2d[1], zz[k]});
183 }
184 }
185 // 2. build cell to vtx
186 const auto n_cells_2d = mesh_2d.numCells();
187 mesh.m_cell_to_vtx.indices().push_back(0);
188 const auto n_vtxs_2d = mesh_2d.getVertexes().size();
189 for (IndexType cell_id = 0; cell_id < n_cells_2d; ++cell_id) {
190 for (IndexType k = 0; k < n_elems; ++k) {
191 const auto cell_vtxs = mesh_2d.getCellToVtxMap()[cell_id];
192 for (IndexType lvid = 0; lvid < cell_vtxs.size(); ++lvid) {
193 const auto vid = cell_vtxs[lvid];
194 mesh.m_cell_to_vtx.data().push_back(vid + n_vtxs_2d * k);
195 mesh.m_cell_to_vtx.data().push_back(vid + n_vtxs_2d * (k + 1));
196 }
197 mesh.m_cell_to_vtx.indices().push_back(mesh.m_cell_to_vtx.data().size());
198 }
199 }
200 // 3. build facet to cell & facet to vertex
201 // 3.1 facets that in 2D are cells
202 mesh.m_facet_to_vtx.indices().push_back(0);
203 for (IndexType cell_id = 0; cell_id < n_cells_2d; ++cell_id) {
204 // build facet to cell
205 // first and last slice are the only one that belong to one cell
206 mesh.m_facet_to_cell.push_back({cell_id * n_elems, -1});
207 for (IndexType k = 1; k < n_elems; ++k) {
208 mesh.m_facet_to_cell.push_back(
209 {cell_id * n_elems + k - 1, cell_id * n_elems + k});
210 }
211 mesh.m_facet_to_cell.push_back({-1, (cell_id + 1) * n_elems - 1});
212 // build facet to vertex, the same map of cell to vtx for mesh_2d
213 const auto cell_vtxs = mesh_2d.getCellToVtxMap()[cell_id];
214 const auto n_cell_vtxs = cell_vtxs.size();
215 for (IndexType k = 0; k < n_elems + 1; ++k) {
216 for (IndexType lvid = 0; lvid < n_cell_vtxs; ++lvid) {
217 const auto vid = cell_vtxs[lvid];
218 mesh.m_facet_to_vtx.data().push_back(vid + n_vtxs_2d * k);
219 }
220 mesh.m_facet_to_vtx.indices().push_back(
221 mesh.m_facet_to_vtx.indices().back() + n_cell_vtxs);
222 }
223 }
224 // 3.2 facets that in 2D are edges (i.e. facets for the mesh_2d)
225 const auto n_facets_2d =
226 static_cast<IndexType>(mesh_2d.getFacetToCellMap().size());
227 for (IndexType face_id = 0; face_id < n_facets_2d; ++face_id) {
228 const auto facet_to_cell = mesh_2d.getFacetToCellMap()[face_id];
229 // facet to cell, the same of 2d facet to cell but
230 // repeated for each slice and reversed order to maintain ccw
231 if (facet_to_cell[0] == -1) {
232 for (IndexType k = 0; k < n_elems; ++k) {
233 mesh.m_facet_to_cell.push_back({facet_to_cell[1] * n_elems + k, -1});
234 }
235 } else if (facet_to_cell[1] == -1) {
236 for (IndexType k = 0; k < n_elems; ++k) {
237 mesh.m_facet_to_cell.push_back({-1, facet_to_cell[0] * n_elems + k});
238 }
239 } else {
240 for (IndexType k = 0; k < n_elems; ++k) {
241 mesh.m_facet_to_cell.push_back(
242 {facet_to_cell[1] * n_elems + k, facet_to_cell[0] * n_elems + k});
243 }
244 }
245 // facet to vertex, each facet has 4 vtxs (side of the prism)
246 const auto facet_to_vtx = mesh_2d.getFacetToVtxMap()[face_id];
247 const std::array<IndexType, 2> edge_vtxs{facet_to_vtx[0], facet_to_vtx[1]};
248 for (IndexType k = 0; k < n_elems; ++k) {
249 mesh.m_facet_to_vtx.data().push_back(edge_vtxs[0] + k * n_vtxs_2d);
250 mesh.m_facet_to_vtx.data().push_back(edge_vtxs[1] + k * n_vtxs_2d);
251 mesh.m_facet_to_vtx.data().push_back(edge_vtxs[1] + (k + 1) * n_vtxs_2d);
252 mesh.m_facet_to_vtx.data().push_back(edge_vtxs[0] + (k + 1) * n_vtxs_2d);
253 mesh.m_facet_to_vtx.indices().push_back(
254 mesh.m_facet_to_vtx.indices().back() + 4);
255 }
256 }
257 // 4. build other maps
258 mesh.buildInternalMaps();
259 mesh.buildEdgeMaps();
260 return mesh;
261}

263template <typename FloatType, typename IndexType>
264void Mesh3D<FloatType, IndexType>::buildFacetToCellAndFacetToVertex(
265 const CellsFaceVtxsUncompressed& cells_face_vtxs) {
266 // the three vtxs in the face with smallest id, define uniquely a face
267 using FaceKey = std::array<IndexType, 3>;
268 std::map<FaceKey, IndexType> unique_faces;
269 datastructure::CrsMapUncompressed<IndexType> face_to_vtxs;
270 this->m_facet_to_cell.clear();
271 for (size_t cell_id = 0; cell_id < cells_face_vtxs.size(); ++cell_id) {
272 // make a copy that can be modified to partial sort
273 auto face_vtxs = cells_face_vtxs[cell_id];
274 size_t count = 0;
275 while (count < face_vtxs.size()) {
276 //
277 size_t face_size = face_vtxs[count];
278 count++;
279 //
280 assert(count + face_size <= face_vtxs.size());
281 const auto face_begin = face_vtxs.begin() + count;
282 std::partial_sort(face_begin, face_begin + 3, face_begin + face_size);
283 const auto face_key =
284 FaceKey{face_vtxs[count], face_vtxs[count + 1], face_vtxs[count + 2]};
285 const auto [it, inserted] =
286 unique_faces.emplace(face_key, this->m_facet_to_cell.size());
287 const auto face_id = it->second;
288 //
289 if (face_id >= (IndexType)this->m_facet_to_cell.size()) {
290 this->m_facet_to_cell.resize(face_id + 1, {-1, -1});
291 }
292 auto& facet_cells = this->m_facet_to_cell[face_id];
293 if (facet_cells[0] == -1) {
294 facet_cells[0] = cell_id;
295 } else {
296 facet_cells[1] = cell_id;
297 // if the input data is consistent, at most each face is visited twice
298 // thus we can remove the element from the map (reducing the size)
299 unique_faces.erase(it);
300 }
301 if (inserted) {
302 for (size_t i = 0; i < face_size; ++i) {
303 face_to_vtxs.getRow(face_id).push_back(
304 cells_face_vtxs[cell_id][count + i]);
305 }
306 }
307 count += face_size;
308 }
309 }
310 this->m_facet_to_vtx = face_to_vtxs.template compress<IndexType>();
311}

313template <typename FloatType, typename IndexType>
314void Mesh3D<FloatType, IndexType>::buildEdgeMaps() {
315 // compute edge to vertex by loop on all the faces vertices
316 std::map<EdgeVtxs, IndexType> vtx_to_edge;
317 datastructure::CrsMapUncompressed<IndexType> cell_to_edge;
318 const auto n_faces = this->m_facet_to_vtx.size();
319 for (IndexType face_id = 0; face_id < n_faces; ++face_id) {
320 const auto face_vtxs = this->m_facet_to_vtx[face_id];
321 for (IndexType i = 0; i < face_vtxs.size(); ++i) {
322 const IndexType v1 = face_vtxs[i];
323 const IndexType v2 = face_vtxs[(i + 1) % face_vtxs.size()];
324 const IndexType vmin = std::min(v1, v2);
325 const IndexType vmax = std::max(v1, v2);
326 const auto [it, inserted] = vtx_to_edge.emplace(
327 EdgeVtxs{vmin, vmax}, static_cast<IndexType>(vtx_to_edge.size()));
328 const IndexType edge_id = it->second;
329 // update cell to edge
330 if (inserted) {
331 for (const auto cell_id : this->m_facet_to_cell[face_id]) {
332 if (cell_id != -1) {
333 cell_to_edge.getRow(cell_id).push_back(edge_id);
334 }
335 }
336 }
337 }
338 }
339
340 // compress edge to vertex and cell to edge
341 m_edge_to_vtx.resize(vtx_to_edge.size());
342 for (const auto& [edge_vtx, edge_id] : vtx_to_edge) {
343 m_edge_to_vtx[edge_id] = edge_vtx;
344 }
345 m_cell_to_edge = cell_to_edge.template compress<IndexType>();
346
347 // invert cell to edge
348 datastructure::CrsMapUncompressed<IndexType> edge_to_cell;
349 for (IndexType cell_id = 0; cell_id < m_cell_to_edge.size(); ++cell_id) {
350 const IndexType n_edges = m_cell_to_edge.getRow(cell_id).size();
351 for (IndexType edge_id = 0; edge_id < n_edges; ++edge_id) {
352 edge_to_cell.getRow(m_cell_to_edge(cell_id, edge_id)).push_back(cell_id);
353 }
354 }
355 m_edge_to_cell = edge_to_cell.template compress<IndexType>();
356}

359template <typename FloatType, typename IndexType>
360Mesh3D<FloatType, IndexType> readVTU(
361 const std::string& filename, bool /*check_vtxs_unique*/ = true) {
362 // read file
363 vtkNew<vtkXMLUnstructuredGridReader> reader;
364 reader->SetFileName(filename.c_str());
365 reader->Update();
366 const vtkSmartPointer<vtkUnstructuredGrid> unstructured_grid =
367 reader->GetOutput();
368
369 // get points
370 std::vector<Point<FloatType, 3>> mesh_vtxs;
371 vtkPoints* points = unstructured_grid->GetPoints();
372 mesh_vtxs.resize(points->GetNumberOfPoints());
373 for (vtkIdType i = 0; i < points->GetNumberOfPoints(); ++i) {
374 const double* point = points->GetPoint(i);
375 mesh_vtxs[i][0] = point[0];
376 mesh_vtxs[i][1] = point[1];
377 mesh_vtxs[i][2] = point[2];
378 }
379
380 // get cells
381 datastructure::CrsMapUncompressed<IndexType> cell_vtxs;
382 std::vector<std::vector<IndexType>> cells_face_idxs;
383 const vtkIdType n_cells = unstructured_grid->GetNumberOfCells();
384 for (vtkIdType i = 0; i < n_cells; ++i) {
385 vtkCell* cell = unstructured_grid->GetCell(i);
386
387 // build cell vertexes
388 vtkIdList* point_ids = cell->GetPointIds();
389 for (vtkIdType j = 0; j < cell->GetNumberOfPoints(); ++j) {
390 cell_vtxs.getRow(i).push_back(point_ids->GetId(j));
391 }
392
393 // build faces connectivity
394 if (cell->GetCellType() == VTK_POLYHEDRON) {
395 cells_face_idxs.emplace_back();
396 auto& cell_faces = cells_face_idxs.back();
397 const vtkIdType n_faces = cell->GetNumberOfFaces();
398 const vtkIdType* face_vtxs_stream = cell->GetFaces();
399 assert(face_vtxs_stream[0] == n_faces);
400 size_t ptr = 1;
401 for (vtkIdType k = 0; k < n_faces; ++k) {
402 const vtkIdType n_face_vtxs = face_vtxs_stream[ptr++];
403 cell_faces.push_back(n_face_vtxs);
404 for (vtkIdType j = 0; j < n_face_vtxs; ++j) {
405 const vtkIdType vtx_id = face_vtxs_stream[ptr++];
406 cell_faces.push_back(vtx_id);
407 }
408 }
409 } else if (cell->GetCellType() == VTK_TETRA) {
410 // The VTK tetrahedron is defined by the four points (0-3); where (0,1,2)
411 // is the base of the tetrahedron which, using the right hand rule, forms
412 // a triangle whose normal points in the direction of the fourth point.
413 const auto& vtxs = cell_vtxs.getRow(i);
414 cells_face_idxs.push_back({
415 3,
416 vtxs[0],
417 vtxs[2],
418 vtxs[1],
419 3,
420 vtxs[1],
421 vtxs[2],
422 vtxs[3],
423 3,
424 vtxs[0],
425 vtxs[3],
426 vtxs[2],
427 3,
428 vtxs[0],
429 vtxs[1],
430 vtxs[3],
431 });
432 } else {
433 assert(false && "NotImplemented");
434 std::cout << "ERROR: Unhandled VTK CellType" << std::endl;
435 }
436 }
437 return Mesh3D<FloatType, IndexType>{mesh_vtxs, cell_vtxs, cells_face_idxs};
438}

440template <typename FloatType, typename IndexType>
441Mesh3D<FloatType, IndexType> rect_cuboid(
442 const Point<FloatType, 3>& corner_a,
443 const Point<FloatType, 3>& corner_b,
444 const Point<IndexType, 3>& n_divisions,
445 bool tetra_elements = false) {
446 datastructure::CrsMapUncompressed<IndexType> cell_indices;
447 std::vector<Point<FloatType, 3>> vertices;
448 std::vector<std::vector<IndexType>> cells_face_vtxs;
449
450 // Calculate the step size in x and y direction
451 const FloatType x_step = (corner_b[0] - corner_a[0]) / n_divisions[0];
452 const FloatType y_step = (corner_b[1] - corner_a[1]) / n_divisions[1];
453 const FloatType z_step = (corner_b[2] - corner_a[2]) / n_divisions[2];
454
455 // Generate the vertices
456 for (IndexType i = 0; i <= n_divisions[0]; ++i) {
457 for (IndexType j = 0; j <= n_divisions[1]; ++j) {
458 for (IndexType k = 0; k <= n_divisions[2]; ++k) {
459 // Compute the x and y coordinates of the vertex
460 const FloatType x = corner_a[0] + i * x_step;
461 const FloatType y = corner_a[1] + j * y_step;
462 const FloatType z = corner_a[2] + k * z_step;
463 vertices.push_back({x, y, z});
464 }
465 }
466 }
467 // Generate the cell indices
468 for (int i = 0; i < n_divisions[0]; ++i) {
469 for (int j = 0; j < n_divisions[1]; ++j) {
470 for (int k = 0; k < n_divisions[2]; ++k) {
471 // Index of the bottom-front-left vertex of the cell
472 const IndexType /*bottom_front_left */ v0 =
473 i * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
474 j * (n_divisions[2] + 1) + k;
475 const IndexType /*bottom_front_right */ v1 =
476 (i + 1) * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
477 j * (n_divisions[2] + 1) + k;
478 const IndexType /*bottom_back_right */ v2 =
479 (i + 1) * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
480 j * (n_divisions[2] + 1) + (k + 1);
481 const IndexType /*bottom_back_left */ v3 =
482 i * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
483 j * (n_divisions[2] + 1) + (k + 1);
484 const IndexType /*top_front_left */ v4 =
485 i * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
486 (j + 1) * (n_divisions[2] + 1) + k;
487 const IndexType /*top_front_right */ v5 =
488 (i + 1) * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
489 (j + 1) * (n_divisions[2] + 1) + k;
490 const IndexType /*top_back_right */ v6 =
491 (i + 1) * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
492 (j + 1) * (n_divisions[2] + 1) + (k + 1);
493 const IndexType /*top_back_left */ v7 =
494 i * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
495 (j + 1) * (n_divisions[2] + 1) + (k + 1);
496
497 if (tetra_elements) {
498 cell_indices.push_back({v0, v1, v5, v6});
499 cells_face_vtxs.push_back(
500 {3, v0, v5, v1, 3, v0, v1, v6, 3, v0, v6, v5, 3, v1, v5, v6});
501 cell_indices.push_back({v0, v5, v4, v6});
502 cells_face_vtxs.push_back(
503 {3, v0, v4, v5, 3, v0, v5, v6, 3, v0, v6, v4, 3, v5, v4, v6});
504 cell_indices.push_back({v0, v1, v2, v6});
505 cells_face_vtxs.push_back(
506 {3, v0, v1, v2, 3, v0, v6, v1, 3, v0, v2, v6, 3, v1, v6, v2});
507 cell_indices.push_back({v0, v3, v2, v6});
508 cells_face_vtxs.push_back(
509 {3, v0, v2, v3, 3, v0, v3, v6, 3, v0, v6, v2, 3, v3, v2, v6});
510 cell_indices.push_back({v0, v3, v7, v6});
511 cells_face_vtxs.push_back(
512 {3, v0, v3, v7, 3, v0, v6, v3, 3, v0, v7, v6, 3, v3, v6, v7});
513 cell_indices.push_back({v0, v4, v7, v6});
