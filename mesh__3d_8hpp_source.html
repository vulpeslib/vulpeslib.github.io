164template <typename FloatType, typename IndexType>
165void Mesh3D<FloatType, IndexType>::buildFacetToCellAndFacetToVertex(
166 const CellsFaceVtxsUncompressed& cells_face_vtxs) {
167 // the three vtxs in the face with smallest id, define uniquely a face
168 using FaceKey = std::array<IndexType, 3>;
169 std::map<FaceKey, IndexType> unique_faces;
170 datastructure::CrsMapUncompressed<IndexType> face_to_vtxs;
171 this->m_facet_to_cell.clear();
172 for (size_t cell_id = 0; cell_id < cells_face_vtxs.size(); ++cell_id) {
173 // make a copy that can be modified to partial sort
174 auto face_vtxs = cells_face_vtxs[cell_id];
175 size_t count = 0;
176 while (count < face_vtxs.size()) {
177 //
178 size_t face_size = face_vtxs[count];
179 count++;
180 //
181 assert(count + face_size <= face_vtxs.size());
182 std::partial_sort(
183 &face_vtxs[count], &face_vtxs[count + 3],
184 &face_vtxs[count + face_size]);
185 const auto face_key =
186 FaceKey{face_vtxs[count], face_vtxs[count + 1], face_vtxs[count + 2]};
187 const auto [it, inserted] =
188 unique_faces.emplace(face_key, this->m_facet_to_cell.size());
189 const auto face_id = it->second;
190 //
191 if (face_id >= (IndexType)this->m_facet_to_cell.size()) {
192 this->m_facet_to_cell.resize(face_id + 1, {-1, -1});
193 }
194 auto& facet_cells = this->m_facet_to_cell[face_id];
195 if (facet_cells[0] == -1) {
196 facet_cells[0] = cell_id;
197 } else {
198 facet_cells[1] = cell_id;
199 // if the input data is consistent, at most each face is visited twice
200 // thus we can remove the element from the map (reducing the size)
201 unique_faces.erase(it);
202 }
203 if (inserted) {
204 for (size_t i = 0; i < face_size; ++i) {
205 face_to_vtxs.getRow(face_id).push_back(
206 cells_face_vtxs[cell_id][count + i]);
207 }
208 }
209 count += face_size;
210 }
211 }
212 this->m_facet_to_vtx = face_to_vtxs.template compress<IndexType>();
213}

215template <typename FloatType, typename IndexType>
216void Mesh3D<FloatType, IndexType>::buildEdgeMaps() {
217 // compute edge to vertex by loop on all the faces vertices
218
219 std::unordered_map<EdgeVtxs, IndexType, decltype([](const auto& key) {
220 return key[0];
221 })>
222 vtx_to_edge;
223 datastructure::CrsMapUncompressed<IndexType> cell_to_edge;
224 const auto n_faces = this->m_facet_to_vtx.size();
225 for (IndexType face_id = 0; face_id < n_faces; ++face_id) {
226 const auto face_vtxs = this->m_facet_to_vtx[face_id];
227 for (IndexType i = 0; i < face_vtxs.size(); ++i) {
228 const IndexType v1 = face_vtxs[i];
229 const IndexType v2 = face_vtxs[(i + 1) % face_vtxs.size()];
230 const IndexType vmin = std::min(v1, v2);
231 const IndexType vmax = std::max(v1, v2);
232 const auto [it, inserted] = vtx_to_edge.emplace(
233 EdgeVtxs{vmin, vmax}, static_cast<IndexType>(vtx_to_edge.size()));
234 const IndexType edge_id = it->second;
235
236 // update cell to edge
237 if (inserted) {
238 for (const auto cell_id : this->m_facet_to_cell[face_id]) {
239 if (cell_id != -1) {
240 cell_to_edge.getRow(cell_id).push_back(edge_id);
241 }
242 }
243 }
244 }
245 }
246
247 // compress edge to vertex and cell to edge
248 m_edge_to_vtx.resize(vtx_to_edge.size());
249 for (const auto& [edge_vtx, edge_id] : vtx_to_edge) {
250 m_edge_to_vtx[edge_id] = edge_vtx;
251 }
252 m_cell_to_edge = cell_to_edge.template compress<IndexType>();
253
254 // invert cell to edge
255 datastructure::CrsMapUncompressed<IndexType> edge_to_cell;
256 for (IndexType cell_id = 0; cell_id < m_cell_to_edge.size(); ++cell_id) {
257 const IndexType n_edges = m_cell_to_edge.getRow(cell_id).size();
258 for (IndexType edge_id = 0; edge_id < n_edges; ++edge_id) {
259 edge_to_cell.getRow(m_cell_to_edge(cell_id, edge_id)).push_back(cell_id);
260 }
261 }
262 m_edge_to_cell = edge_to_cell.template compress<IndexType>();
263}

265template <typename FloatType, typename IndexType>
266void Mesh3D<FloatType, IndexType>::outputVTU(
267 const std::string& filename, const VtuOptions& options) const {
268 vtkNew<vtkUnstructuredGrid> unstructured_grid;
269 // set points
270 vtkNew<vtkPoints> points;
271 for (const auto& vtx : this->m_vertexes) {
272 points->InsertNextPoint(vtx.data());
273 }
274 unstructured_grid->SetPoints(points);
275
276 // set cells
277 for (IndexType cell_id = 0; cell_id < this->m_cell_to_vtx.size(); ++cell_id) {
278 const auto cell = this->m_cell_to_vtx.getRow(cell_id);
279 std::vector<vtkIdType> vtk_vtx_ids;
280 for (IndexType i = 0; i < cell.size(); ++i) {
281 vtk_vtx_ids.push_back(cell[i]);
282 }
283
284 vtkNew<vtkIdList> faces;
285 const IndexType n_faces = this->m_cell_to_facet[cell_id].size();
286
287 for (IndexType i = 0; i < n_faces; ++i) {
288 const IndexType face_id = this->m_cell_to_facet[cell_id][i];
289 const IndexType face_size = this->m_facet_to_vtx[face_id].size();
290 faces->InsertNextId(face_size);
291
292 for (IndexType j = 0; j < face_size; ++j) {
293 faces->InsertNextId(this->m_facet_to_vtx[face_id][j]);
294 }
295 }
296 unstructured_grid->InsertNextCell(
297 VTK_POLYHEDRON, cell.size(), vtk_vtx_ids.data(), n_faces,
298 faces->GetPointer(0));
299 }
300
301 // save
302 vtkNew<vtkXMLUnstructuredGridWriter> writer;
303 writer->SetFileName(filename.c_str());
304 writer->SetInputData(unstructured_grid);
305 writer->SetCompressorType(options.compressor_type);
306 writer->SetDataMode(options.data_mode);
307 writer->Write();
308}

311template <typename FloatType, typename IndexType>
312Mesh3D<FloatType, IndexType> readVTU(
313 const std::string& filename, bool /*check_vtxs_unique = true*/) {
314 // read file
315 vtkNew<vtkXMLUnstructuredGridReader> reader;
316 reader->SetFileName(filename.c_str());
317 reader->Update();
318 const vtkSmartPointer<vtkUnstructuredGrid> unstructured_grid =
319 reader->GetOutput();
320
321 // get points
322 std::vector<Point<FloatType, 3>> mesh_vtxs;
323 vtkPoints* points = unstructured_grid->GetPoints();
324 mesh_vtxs.resize(points->GetNumberOfPoints());
325 for (vtkIdType i = 0; i < points->GetNumberOfPoints(); ++i) {
326 const double* point = points->GetPoint(i);
327 mesh_vtxs[i][0] = point[0];
328 mesh_vtxs[i][1] = point[1];
329 mesh_vtxs[i][2] = point[2];
330 }
331
332 // get cells
333 datastructure::CrsMapUncompressed<IndexType> cell_vtxs;
334 std::vector<std::vector<IndexType>> cells_face_idxs;
335 const vtkIdType n_cells = unstructured_grid->GetNumberOfCells();
336 for (vtkIdType i = 0; i < n_cells; ++i) {
337 vtkCell* cell = unstructured_grid->GetCell(i);
338 assert(cell->GetCellType() == VTK_POLYHEDRON);
339
340 // build cell vertexes
341 vtkIdList* point_ids = cell->GetPointIds();
342 for (vtkIdType j = 0; j < cell->GetNumberOfPoints(); ++j) {
343 cell_vtxs.getRow(i).push_back(point_ids->GetId(j));
344 }
345
346 // build faces connectivity
347 cells_face_idxs.emplace_back();
348 const vtkIdType n_faces = cell->GetNumberOfFaces();
349 const vtkIdType* face_vtxs_stream = cell->GetFaces();
350 assert(face_vtxs_stream[0] == n_faces);
351 size_t ptr = 1;
352 for (vtkIdType k = 0; k < n_faces; ++k) {
353 const vtkIdType n_face_vtxs = face_vtxs_stream[ptr++];
354 cells_face_idxs.back().push_back(n_face_vtxs);
355 for (vtkIdType j = 0; j < n_face_vtxs; ++j) {
356 const vtkIdType vtx_id = face_vtxs_stream[ptr++];
357 cells_face_idxs.back().push_back(vtx_id);
358 }
359 }
360 }
361 return Mesh3D<FloatType, IndexType>{mesh_vtxs, cell_vtxs, cells_face_idxs};
362}

364template <typename FloatType, typename IndexType>
365Mesh3D<FloatType, IndexType> rect_cuboid(
366 const Point<FloatType, 3>& corner_a,
367 const Point<FloatType, 3>& corner_b,
368 const Point<IndexType, 3>& n_divisions,
369 bool tetra_elements = false) {
370 datastructure::CrsMapUncompressed<IndexType> cell_indices;
371 std::vector<Point<FloatType, 3>> vertices;
372 std::vector<std::vector<IndexType>> cells_face_vtxs;
373
374 // Calculate the step size in x and y direction
375 const FloatType x_step = (corner_b[0] - corner_a[0]) / n_divisions[0];
376 const FloatType y_step = (corner_b[1] - corner_a[1]) / n_divisions[1];
377 const FloatType z_step = (corner_b[2] - corner_a[2]) / n_divisions[2];
378
379 // Generate the vertices
380 for (IndexType i = 0; i <= n_divisions[0]; ++i) {
381 for (IndexType j = 0; j <= n_divisions[1]; ++j) {
382 for (IndexType k = 0; k <= n_divisions[2]; ++k) {
383 // Compute the x and y coordinates of the vertex
384 const FloatType x = corner_a[0] + i * x_step;
385 const FloatType y = corner_a[1] + j * y_step;
386 const FloatType z = corner_a[2] + k * z_step;
387 vertices.push_back({x, y, z});
388 }
389 }
390 }
391 // Generate the cell indices
392 for (int i = 0; i < n_divisions[0]; ++i) {
393 for (int j = 0; j < n_divisions[1]; ++j) {
394 for (int k = 0; k < n_divisions[2]; ++k) {
395 // Index of the bottom-front-left vertex of the cell
396 const IndexType /*bottom_front_left */ v0 =
397 i * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
398 j * (n_divisions[2] + 1) + k;
399 const IndexType /*bottom_front_right */ v1 =
400 (i + 1) * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
401 j * (n_divisions[2] + 1) + k;
402 const IndexType /*bottom_back_right */ v2 =
403 (i + 1) * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
404 j * (n_divisions[2] + 1) + (k + 1);
405 const IndexType /*bottom_back_left */ v3 =
406 i * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
407 j * (n_divisions[2] + 1) + (k + 1);
408 const IndexType /*top_front_left */ v4 =
409 i * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
410 (j + 1) * (n_divisions[2] + 1) + k;
411 const IndexType /*top_front_right */ v5 =
412 (i + 1) * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
413 (j + 1) * (n_divisions[2] + 1) + k;
414 const IndexType /*top_back_right */ v6 =
415 (i + 1) * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
416 (j + 1) * (n_divisions[2] + 1) + (k + 1);
417 const IndexType /*top_back_left */ v7 =
418 i * (n_divisions[1] + 1) * (n_divisions[2] + 1) +
419 (j + 1) * (n_divisions[2] + 1) + (k + 1);
420
421 if (tetra_elements) {
422 cell_indices.push_back({v0, v1, v5, v6});
423 cells_face_vtxs.push_back(
424 {3, v0, v1, v5, 3, v0, v1, v6, 3, v0, v5, v6, 3, v1, v5, v6});
425 cell_indices.push_back({v0, v5, v4, v6});
426 cells_face_vtxs.push_back(
427 {3, v0, v5, v4, 3, v0, v5, v6, 3, v0, v4, v6, 3, v5, v4, v6});
428 cell_indices.push_back({v0, v1, v2, v6});
429 cells_face_vtxs.push_back(
430 {3, v0, v1, v2, 3, v0, v1, v6, 3, v0, v2, v6, 3, v1, v2, v6});
431 cell_indices.push_back({v0, v3, v2, v6});
432 cells_face_vtxs.push_back(
433 {3, v0, v3, v2, 3, v0, v3, v6, 3, v0, v2, v6, 3, v3, v2, v6});
434 cell_indices.push_back({v0, v3, v7, v6});
435 cells_face_vtxs.push_back(
436 {3, v0, v3, v7, 3, v0, v3, v6, 3, v0, v7, v6, 3, v3, v7, v6});
437 cell_indices.push_back({v0, v4, v7, v6});
438 cells_face_vtxs.push_back(
439 {3, v0, v4, v7, 3, v0, v4, v6, 3, v0, v7, v6, 3, v4, v7, v6});
440
441 } else {
442 // Store the cell as eight vertex indices (defining a hexahedron)
443 cell_indices.push_back({v0, v1, v2, v3, v4, v5, v6, v7});
444 cells_face_vtxs.push_back({4, v0, v1, v5, v4, // Front
445 4, v3, v2, v6, v7, // Back
446 4, v0, v3, v7, v4, // Left
447 4, v1, v2, v6, v5, // Right
448 4, v0, v1, v2, v3, // Bottom
449 4, v4, v5, v6, v7}); // Top
450 }
451 }
452 }
453 }
454
455 auto mesh =
456 Mesh3D<FloatType, IndexType>{vertices, cell_indices, cells_face_vtxs};
457 return mesh;
458}
