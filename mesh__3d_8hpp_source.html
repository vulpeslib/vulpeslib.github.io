253 constexpr IndexType nEntity(std::size_t dim) const {
254 if (dim == 0) {
255 return this->getVertexes().size();
256 } else if (dim == 1) {
257 return this->getEdgeToCellMap().size();
258 } else if (dim == 2) {
259 return this->getFacetToCellMap().size();
260 } else if (dim == 3) {
261 assert(this->m_cell_to_facet.size() == this->m_dual_graph.size());
262 return this->getDualMap().size();
263 } else {
264 assert(false);
265 return -1;
266 }
267 }

276 template <std::size_t dim>
277 constexpr auto& getEntityToCellMap() const {
278 if constexpr (dim == 0) {
279 return this->getVtxToCellMap();
280 } else if constexpr (dim == 1) {
281 return this->getEdgeToCellMap();
282 } else if constexpr (dim == 2) {
283 return this->getFacetToCellMap();
284 } else {
285 static_assert(dim * 0);
286 }
287 }
