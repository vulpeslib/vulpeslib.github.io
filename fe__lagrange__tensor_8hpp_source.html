34template <std::size_t Dim, typename DofPolicy, typename Poly1DPolicy>
35class FeSpaceLagrangeQ : public FeSpaceBase<
36 Dim,
37 internal::real_t,
38 DofPolicy,
39 FeSpaceLagrangeQ<Dim, DofPolicy, Poly1DPolicy>> {
40 public:
41 using Polynomial = polynomial::NdPolynomial<internal::real_t, Dim, false>;
42 explicit FeSpaceLagrangeQ(std::size_t order)
43 : FeSpaceBase<
44 Dim,
45 internal::real_t,
46 DofPolicy,
47 FeSpaceLagrangeQ<Dim, DofPolicy, Poly1DPolicy>>(order) {
48 const auto nodes_1d = Poly1DPolicy::computeNodes(order + 1);
49 const auto coef_1d = internal::computeLagrangianPolyCoef<1>(nodes_1d);
50 const auto size = order + 1;
51 const auto size_nd = vmath::pow(size, Dim);
52 m_nodes.resize(size_nd);
53 // loop on each basis function, that is each element of the cartesian
54 // product of 1d polynomials
55 for (std::size_t i = 0; i < size_nd; ++i) {
56 // this is like writing `i` in radix
57 std::size_t idx = i;
58 m_polys.emplace_back(size);
59 std::fill(m_polys.back().data().begin(), m_polys.back().data().end(), 1);
60 for (std::size_t d = 0; d < Dim; ++d) {
61 const auto basis_id_cart_d = idx % size;
62 const auto node_d = nodes_1d[basis_id_cart_d][0];
63 idx /= size;
64 // loop on each coef of the nd-polynomial
65 for (std::size_t j = 0; j < size_nd; ++j) {
66 std::array<std::size_t, Dim> coef_id_cart;
67 std::size_t coef_id = j;
68 for (std::size_t d2 = 0; d2 < Dim; ++d2) {
69 coef_id_cart[d2] = coef_id % size;
70 coef_id /= size;
71 }
72 // multiply by the 1d coef
73 std::apply(
74 [&](auto... args) {
75 m_polys.back().coef(args...) *=
76 coef_1d(coef_id_cart[d], basis_id_cart_d);
77 },
78 coef_id_cart);
79 }
80 m_nodes[i][d] = node_d;
81 }
82 }
83 }
84
85 const auto& getNodes() const { return m_nodes; }
86 const auto& getBasis() const { return m_polys; }
87 inline static constexpr FeType elemType() { //
88 return FeType::Tensor;
89 };
90
91 protected:
92 std::vector<Polynomial> m_polys;
93 std::vector<Point<internal::real_t, Dim>> m_nodes;
94};
