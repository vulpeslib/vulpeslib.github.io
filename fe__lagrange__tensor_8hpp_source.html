38template <std::size_t Dim, typename DofPolicy, typename Poly1DPolicy>
39class FeSpaceLagrangeQ : public FeSpaceBase<
40 Dim,
41 internal::real_t,
42 DofPolicy,
43 FeSpaceLagrangeQ<Dim, DofPolicy, Poly1DPolicy>> {
44 public:
45 using Polynomial = polynomial::NdPolynomial<internal::real_t, Dim, false>;
46 explicit FeSpaceLagrangeQ(std::size_t order)
47 : FeSpaceBase<
48 Dim,
49 internal::real_t,
50 DofPolicy,
51 FeSpaceLagrangeQ<Dim, DofPolicy, Poly1DPolicy>>(order) {
52 const auto nodes_1d = Poly1DPolicy::computeNodes(order + 1);
53 const auto coef_1d = internal::computeLagrangianPolyCoef<1>(nodes_1d);
54 const auto size = order + 1;
55 const auto size_nd = vmath::pow(size, Dim);
56 m_nodes.resize(size_nd);
57 // loop on each basis function, that is each element of the cartesian
58 // product of 1d polynomials
59 for (std::size_t i = 0; i < size_nd; ++i) {
60 // this is like writing `i` in radix
61 std::size_t idx = i;
62 m_polys.emplace_back(size);
63 std::fill(m_polys.back().data().begin(), m_polys.back().data().end(), 1);
64 for (std::size_t d = 0; d < Dim; ++d) {
65 const auto basis_id_cart_d = idx % size;
66 const auto node_d = nodes_1d[basis_id_cart_d][0];
67 idx /= size;
68 // loop on each coef of the nd-polynomial
69 for (std::size_t j = 0; j < size_nd; ++j) {
70 std::array<std::size_t, Dim> coef_id_cart;
71 std::size_t coef_id = j;
72 for (std::size_t d2 = 0; d2 < Dim; ++d2) {
73 coef_id_cart[d2] = coef_id % size;
74 coef_id /= size;
75 }
76 // multiply by the 1d coef
77 std::apply(
78 [&](auto... args) {
79 m_polys.back().coef(args...) *=
80 coef_1d(coef_id_cart[d], basis_id_cart_d);
81 },
82 coef_id_cart);
83 }
84 m_nodes[i][d] = node_d;
85 }
86 }
87 }
88
89 const auto& getNodes() const { return m_nodes; }
90 const auto& getBasis() const { return m_polys; }
91 inline static constexpr FeType elemType() { return FeType::Tensor; };
92
93 protected:
94 std::vector<Polynomial> m_polys;
95 std::vector<Point<internal::real_t, Dim>> m_nodes;
96};
