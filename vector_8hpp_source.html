73 template <typename DofHandler>
74 void init(const DofHandler& dof_handler, PetscInt block_size) {
75 createVector();
76 if (block_size > 1) {
77 PetscCallAbortWorld(VecSetBlockSize(m_petsc_vector, block_size));
78 }
79 const PetscInt n_glob = dof_handler.ndofsGlobal();
80 PetscInt dof_begin = 0;
81 PetscInt dof_end = n_glob;
82 if constexpr (DofHandler::parallel) {
83 dof_begin = dof_handler.dofOffset()[mpi::rank()];
84 dof_end = dof_handler.dofOffset()[mpi::rank() + 1];
85 }
86 const PetscInt n_loc = dof_end - dof_begin;
87 setVectorSizes(n_loc * block_size, n_glob * block_size);
88 setupVector();
89 }
