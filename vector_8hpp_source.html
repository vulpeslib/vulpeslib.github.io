64 template <typename DofHandler>
65 void init(const DofHandler& dof_handler, PetscInt block_size) {
66 createVector();
67 if (block_size > 1) {
68 PetscCallAbortWorld(VecSetBlockSize(m_petsc_vector, block_size));
69 }
70 const PetscInt n_glob = dof_handler.ndofsGlobal();
71 PetscInt dof_begin = 0;
72 PetscInt dof_end = n_glob;
73 if constexpr (DofHandler::parallel) {
74 dof_begin = dof_handler.dofOffset()[mpi::rank()];
75 dof_end = dof_handler.dofOffset()[mpi::rank() + 1];
76 }
77 const PetscInt n_loc = dof_end - dof_begin;
78 setVectorSizes(n_loc * block_size, n_glob * block_size);
79 setupVector();
80 }
