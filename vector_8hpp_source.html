72 template <typename DofHandler>
73 void init(const DofHandler& dof_handler, PetscInt block_size) {
74 createVector();
75 if (block_size > 1) {
76 PetscCallAbortWorld(VecSetBlockSize(m_petsc_vector, block_size));
77 }
78 const PetscInt n_glob = dof_handler.ndofsGlobal();
79 PetscInt dof_begin = 0;
80 PetscInt dof_end = n_glob;
81 if constexpr (DofHandler::parallel) {
82 dof_begin = dof_handler.dofOffset()[mpi::rank()];
83 dof_end = dof_handler.dofOffset()[mpi::rank() + 1];
84 }
85 const PetscInt n_loc = dof_end - dof_begin;
86 setVectorSizes(n_loc * block_size, n_glob * block_size);
87 setupVector();
88 }
