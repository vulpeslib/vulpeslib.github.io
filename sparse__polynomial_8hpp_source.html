86 explicit SparsePolynomial(
87 const std::array<const Polynomial<CoefType>*, Dim>& polynomials) {
88 std::array<std::size_t, Dim> sizes;
89 std::size_t size_nd = 1;
90 for (std::size_t d = 0; d < Dim; ++d) {
91 sizes[d] = polynomials[d]->size();
92 size_nd *= sizes[d];
93 }
94 // loop on each element of the cartesian product
95 for (std::size_t i = 0; i < size_nd; ++i) {
96 // get the list of indices
97 std::array<std::size_t, Dim> idxs;
98 // this is like writing `i` in radix
99 std::size_t idx = i;
100 CoefType monomial_coef = CoefType(1);
101 for (std::size_t d = 0; d < Dim; ++d) {
102 idxs[d] = idx % sizes[d];
103 monomial_coef *= (*polynomials[d])[idxs[d]];
104 idx /= sizes[d];
105 }
106 if (monomial_coef != CoefType(0)) {
107 m_monomials.emplace_back(idxs, monomial_coef);
108 }
109 }
110 }

112 CoefType evaluate(const Eigen::Vector<CoefType, Dim>& point) const {
113 return evaluate(point.data());
114 }

116 CoefType evaluate(const CoefType* point) const {
117 CoefType result = CoefType(0);
118 for (const auto& monomial : m_monomials) {
119 result += monomial.evaluate(point);
120 }
121 return result;
122 }
123
124 CoefType integrate(
125 const datastructure::SimplexDataView<Dim, true, CoefType, std::size_t>&
126 monomial_integrals) const {
127 assert(m_monomials.size() <= monomial_integrals.numel());
128 CoefType integral = 0.0;
129 for (const auto& monomial : m_monomials) {
130 integral +=
131 monomial.m_coefficient *
132 monomial_integrals(monomial.m_exponent[0], monomial.m_exponent[1]);
133 }
134 return integral;
135 }

137 SparsePolynomial derivative(std::size_t component) const {
138 SparsePolynomial result;
139 for (const auto& monomial : m_monomials) {
140 const auto monomial_der = monomial.derivative(component);
141 if (!monomial_der.empty()) {
142 result.m_monomials.push_back(monomial_der);
143 }
144 }
145 return result;
146 }

185 SparsePolynomial& operator*=(const SparsePolynomial& other) {
186 decltype(m_monomials) result_monomials;
187 for (const auto& monomial1 : m_monomials) {
188 for (const auto& monomial2 : other.m_monomials) {
189 // Multiply the coefficients
190 CoefType new_coef = monomial1.m_coefficient * monomial2.m_coefficient;
191 std::array<IndexType, Dim> new_exponent;
192 for (std::size_t i = 0; i < Dim; ++i) {
193 new_exponent[i] = monomial1.m_exponent[i] + monomial2.m_exponent[i];
194 }
195 // We need to check if the monomial already exists (if not scaled)
196 if constexpr (IsScaled) {
197 result_monomials.emplace_back(new_exponent, new_coef);
198 } else {
199 // WARNING
200 // Linear search for merging, worth only if polynomial has few values
201 bool found = false;
202 for (auto& existing_monomial : result_monomials) {
203 if (existing_monomial.m_exponent == new_exponent) {
204 existing_monomial.m_coefficient += new_coef;
205 found = true;
206 break;
207 }
208 }
209 // If no existing monomial was found, add the new monomial to the
210 // result
211 if (!found) {
212 result_monomials.emplace_back(new_exponent, new_coef);
213 }
214 }
215 }
216 }
217
218 m_monomials = std::move(result_monomials);
219 return *this;
220 }

222 SparsePolynomial& operator*=(const CoefType& scalar) {
223 for (auto& monomial : m_monomials) {
224 monomial.m_coefficient *= scalar;
225 }
226 return *this;
227 }

253 protected:
257 template <bool IsAddition>
258 SparsePolynomial& add(const SparsePolynomial& other) {
259 for (const auto& other_monomial : other.m_monomials) {
260 if constexpr (IsScaled) {
261 m_monomials.push_back(other_monomial);
262 } else {
263 bool found = false;
264 for (auto& monomial : m_monomials) {
265 if (monomial.m_exponent == other_monomial.m_exponent) {
266 if constexpr (IsAddition) {
267 monomial.m_coefficient += other_monomial.m_coefficient;
268 } else {
269 monomial.m_coefficient -= other_monomial.m_coefficient;
270 }
271 found = true;
272 break;
273 }
274 }
275 if (!found) {
276 if constexpr (IsAddition) {
277 m_monomials.push_back(other_monomial);
278 } else {
279 auto negated_monomial = other_monomial;
280 negated_monomial.m_coefficient = -negated_monomial.m_coefficient;
281 m_monomials.push_back(negated_monomial);
282 }
283 }
284 }
285 }
286 return *this;
287 }
