110 template <typename FloatType, typename Polynomial>
111 void evalOnReferenceFrame(
112 const std::vector<Polynomial>& basis,
113 const qrule::QuadratureRule<FloatType, Dim>& qrule) {
114 const int64_t basis_size = basis.size();
115 const int64_t nodes_size = qrule.nodes.size();
116 if ((m_scalar_shapes_ref.rows() != basis_size) ||
117 (m_scalar_shapes_ref.cols() != nodes_size)) {
118 m_scalar_shapes_ref.resize(basis_size, nodes_size);
119 m_grads_shapes_ref.resize(basis_size, nodes_size);
120 m_scalar_shapes.resize(basis_size, nodes_size);
121 m_grads_shapes.resize(basis_size, nodes_size);
122 m_physical_qpts_ref.resize(Dim, nodes_size);
123 m_physical_qpts.resize(Dim, nodes_size);
124 m_quadrature_weights_ref.resize(nodes_size);
125 m_quadrature_weights.resize(nodes_size);
126 }
127
128 // TODO: choose carefully memory alignment
129 // TODO: update only what needed
130 for (int64_t i = 0; i < basis_size; ++i) {
131 const auto& basis_func = basis[i];
132 const auto grad = basis_func.gradient();
133 for (int64_t j = 0; j < nodes_size; ++j) {
134 m_scalar_shapes_ref(i, j) =
135 vmath::to<real_t>(basis_func.evaluate(qrule.nodes[j].point.data()));
136 for (std::size_t d = 0; d < Dim; ++d) {
137 m_grads_shapes_ref(i, j)(d) =
138 vmath::to<real_t>(grad[d].evaluate(qrule.nodes[j].point.data()));
139 }
140 }
141 }
142
143 for (int64_t j = 0; j < nodes_size; ++j) {
144 m_quadrature_weights_ref(j) = vmath::to<real_t>(qrule.nodes[j].weight);
145 for (std::size_t d = 0; d < Dim; ++d) {
146 m_physical_qpts_ref(d, j) = vmath::to<real_t>(qrule.nodes[j].point[d]);
147 }
148 }
149 }

151 // TODO: index type should be template parameter
152 template <typename MappingType>
153 void evalAndApplyMapping(const MappingType& mapping, UpdateFlagsType flags) {
154 if (flags & UpdateWeights) {
155 for (int64_t i = 0; i < m_quadrature_weights.size(); ++i) {
156 m_quadrature_weights[i] =
157 m_quadrature_weights_ref[i] * mapping.jacobianDet({});
158 }
159 }
160
161 if (flags & UpdateGradients) {
162 for (int64_t i = 0; i < m_grads_shapes.rows(); ++i) {
163 for (int64_t j = 0; j < m_grads_shapes.cols(); ++j) {
164 m_grads_shapes(i, j) = mapping.jacobianInv({}).array() *
165 m_grads_shapes_ref(i, j).array();
166 }
167 }
168 }
169
170 if (flags & UpdateValues) {
171 m_scalar_shapes = m_scalar_shapes_ref;
172 }
173
174 if (flags & UpdateQuadraturePts) {
175 mapping.transform(m_physical_qpts_ref, m_physical_qpts);
176 }
177 }
