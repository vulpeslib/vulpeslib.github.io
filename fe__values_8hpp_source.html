93 template <typename FloatType, typename Polynomial>
94 void evalOnReferenceFrame(
95 const std::vector<Polynomial>& basis,
96 const qrule::QuadratureRule<FloatType, Dim>& qrule) {
97 const int64_t basis_size = basis.size();
98 const int64_t nodes_size = qrule.nodes.size();
99 if ((m_scalar_shapes_ref.rows() != basis_size) ||
100 (m_scalar_shapes_ref.cols() != nodes_size)) {
101 m_scalar_shapes_ref.resize(basis_size, nodes_size);
102 m_grads_shapes_ref.resize(basis_size, nodes_size);
103 m_scalar_shapes.resize(basis_size, nodes_size);
104 m_grads_shapes.resize(basis_size, nodes_size);
105 m_physical_qpts_ref.resize(Dim, nodes_size);
106 m_quadrature_weights_ref.resize(nodes_size);
107 m_quadrature_weights.resize(nodes_size);
108 }
109
110 // TODO: choose carefully memory alignment
111 // TODO: update only what needed
112 for (int64_t i = 0; i < basis_size; ++i) {
113 const auto& basis_func = basis[i];
114 const auto grad = basis_func.gradient();
115 for (int64_t j = 0; j < nodes_size; ++j) {
116 m_scalar_shapes_ref(i, j) =
117 vmath::to<real_t>(basis_func.evaluate(qrule.nodes[j].point.data()));
118 for (std::size_t d = 0; d < Dim; ++d) {
119 m_grads_shapes_ref(i, j)(d) =
120 vmath::to<real_t>(grad[d].evaluate(qrule.nodes[j].point.data()));
121 }
122 }
123 }
124
125 for (int64_t j = 0; j < nodes_size; ++j) {
126 m_quadrature_weights_ref(j) = vmath::to<real_t>(qrule.nodes[j].weight);
127 for (std::size_t d = 0; d < Dim; ++d) {
128 m_physical_qpts_ref(d, j) = vmath::to<real_t>(qrule.nodes[j].point[d]);
129 }
130 }
131 }

133 // TODO: index type should be template parameter
134 template <typename MappingType>
135 void evalAndApplyMapping(const MappingType& mapping, UpdateFlagsType flags) {
136 if (flags & UpdateWeights) {
137 for (int64_t i = 0; i < m_quadrature_weights.size(); ++i) {
138 m_quadrature_weights[i] =
139 m_quadrature_weights_ref[i] * mapping.jacobianDet({});
140 }
141 }
142
143 if (flags & UpdateGradients) {
144 for (int64_t i = 0; i < m_grads_shapes.rows(); ++i) {
145 for (int64_t j = 0; j < m_grads_shapes.cols(); ++j) {
146 m_grads_shapes(i, j) = mapping.jacobianInv({}).array() *
147 m_grads_shapes_ref(i, j).array();
148 }
149 }
150 }
151
152 if (flags & UpdateValues) {
153 m_scalar_shapes = m_scalar_shapes_ref;
154 }
155 }
