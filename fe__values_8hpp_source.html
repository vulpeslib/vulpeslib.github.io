116 template <typename FloatType, typename Polynomial>
117 void evalOnReferenceFrame(
118 const std::vector<Polynomial>& basis,
119 const qrule::QuadratureRule<FloatType, Dim>& qrule) {
120 const int64_t basis_size = basis.size();
121 const int64_t nodes_size = qrule.nodes.size();
122 if ((m_scalar_shapes_ref.rows() != basis_size) ||
123 (m_scalar_shapes_ref.cols() != nodes_size)) {
124 m_scalar_shapes_ref.resize(basis_size, nodes_size);
125 m_grads_shapes_ref.resize(basis_size, nodes_size);
126 m_scalar_shapes.resize(basis_size, nodes_size);
127 m_grads_shapes.resize(basis_size, nodes_size);
128 m_physical_qpts_ref.resize(Dim, nodes_size);
129 m_physical_qpts.resize(Dim, nodes_size);
130 m_quadrature_weights_ref.resize(nodes_size);
131 m_quadrature_weights.resize(nodes_size);
132 }
133
134 // TODO: choose carefully memory alignment
135 // TODO: update only what needed
136 for (int64_t i = 0; i < basis_size; ++i) {
137 const auto& basis_func = basis[i];
138 const auto grad = basis_func.gradient();
139 for (int64_t j = 0; j < nodes_size; ++j) {
140 m_scalar_shapes_ref(i, j) =
141 vmath::to<real_t>(basis_func.evaluate(qrule.nodes[j].point.data()));
142 for (std::size_t d = 0; d < Dim; ++d) {
143 m_grads_shapes_ref(i, j)(d) =
144 vmath::to<real_t>(grad[d].evaluate(qrule.nodes[j].point.data()));
145 }
146 }
147 }
148
149 for (int64_t j = 0; j < nodes_size; ++j) {
150 m_quadrature_weights_ref(j) = vmath::to<real_t>(qrule.nodes[j].weight);
151 for (std::size_t d = 0; d < Dim; ++d) {
152 m_physical_qpts_ref(d, j) = vmath::to<real_t>(qrule.nodes[j].point[d]);
153 }
154 }
155 }

162 template <typename MappingType>
163 void evalAndApplyMapping(const MappingType& mapping, UpdateFlagsType flags) {
164 if (flags & UpdateWeights) {
165 for (int64_t i = 0; i < m_quadrature_weights.size(); ++i) {
166 m_quadrature_weights[i] =
167 m_quadrature_weights_ref[i] * mapping.jacobianDet({});
168 }
169 }
170
171 if (flags & UpdateGradients) {
172 for (int64_t i = 0; i < m_grads_shapes.rows(); ++i) {
173 for (int64_t j = 0; j < m_grads_shapes.cols(); ++j) {
174 m_grads_shapes(i, j) = mapping.jacobianInv({}).array() *
175 m_grads_shapes_ref(i, j).array();
176 }
177 }
178 }
179
180 if (flags & UpdateValues) {
181 m_scalar_shapes = m_scalar_shapes_ref;
182 }
183
184 if (flags & UpdateQuadraturePts) {
185 mapping.transform(m_physical_qpts_ref, m_physical_qpts);
186 }
187 }
