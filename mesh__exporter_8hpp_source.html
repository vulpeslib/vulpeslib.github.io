169template <typename MeshDerived>
170void MeshExporterBase<MeshDerived>::save(
171 const std::string& filename, const VtuOptions& options) const {
172 vtkNew<vtkAppendFilter> merged_grids;
173 for (const auto& grid : m_vtk_grids) {
174 merged_grids->AddInputData(grid);
175 }
176 merged_grids->MergePointsOff();
177 merged_grids->Update();
178 //
179 if (filename.ends_with(".vtu")) {
180 vtkNew<vtkXMLUnstructuredGridWriter> writer;
181 writer->SetFileName(filename.c_str());
182 writer->SetInputConnection(merged_grids->GetOutputPort());
183 writer->SetCompressorType(options.compressor_type);
184 writer->SetDataMode(options.data_mode);
185 writer->Write();
186 } else if (filename.ends_with(".vtk")) {
187 vtkNew<vtkUnstructuredGridWriter> writer;
188 writer->SetFileName(filename.c_str());
189 writer->SetInputConnection(merged_grids->GetOutputPort());
190 writer->SetFileTypeToASCII();
191 writer->Write();
192 } else {
193 assert(false && "Unhandled output format");
194 std::cout << "Unhandled output format" << std::endl;
195 }
196}

341void MeshExporter<MeshDerived>::attachVertexData(
342 const std::string& label,
343 const Eigen::Map<const Eigen::Matrix<float_t, -1, -1>> data_view) const {
344 assert(this->m_vtk_grids.size() == 1);
345 auto vtk_unstructured_grid = this->m_vtk_grids[0];
346 assert(
347 data_view.cols() ==
348 static_cast<Eigen::Index>(this->m_mesh.getVertexes().size()));
349 for (Eigen::Index c = 0; c < data_view.rows(); ++c) {
350 vtkNew<vtkFloatArray> scalar;
351 const std::string curr_label = label + "_" + std::to_string(c);
352 scalar->SetName(curr_label.c_str());
353 for (Eigen::Index i = 0; i < data_view.cols(); ++i) {
354 scalar->InsertNextValue(data_view(c, i));
355 }
356 vtk_unstructured_grid->GetPointData()->AddArray(scalar);
357 }

376template <typename MeshDerived>
377void MeshExporter<MeshDerived>::buildVtkMesh2D() {
378 this->m_vtk_grids.resize(1);
379 this->m_vtk_grids[0] = vtkSmartPointer<vtkUnstructuredGrid>::New();
380 auto vtk_unstructured_grid = this->m_vtk_grids[0];
381 // set vertices
382 vtkNew<vtkPoints> points;
383 for (const auto& point : this->m_mesh.getVertexes()) {
384 points->InsertNextPoint(point[0], point[1], 0.0);
385 }
386 vtk_unstructured_grid->SetPoints(points);
387
388 // set cells
389 vtkNew<vtkCellArray> cells;
390 for (index_t cell_id = 0; cell_id < this->m_mesh.numCells(); ++cell_id) {
391 const auto cell = this->m_mesh.getCellToVtxMap().getRow(cell_id);
392 vtkNew<vtkPolygon> vtk_cell;
393 vtk_cell->GetPointIds()->SetNumberOfIds(cell.size());
394 for (index_t i = 0; i < cell.size(); ++i) {
395 vtk_cell->GetPointIds()->SetId(i, cell[i]);
396 }
397 cells->InsertNextCell(vtk_cell);
398 }
399 vtk_unstructured_grid->SetCells(VTK_POLYGON, cells);
400}

402template <typename MeshDerived>
403void MeshExporter<MeshDerived>::buildVtkMesh3D() {
404 this->m_vtk_grids.resize(1);
405 this->m_vtk_grids[0] = vtkSmartPointer<vtkUnstructuredGrid>::New();
406 auto vtk_unstructured_grid = this->m_vtk_grids[0];
407 // set points
408 vtkNew<vtkPoints> points;
409 for (const auto& vtx : this->m_mesh.getVertexes()) {
410 points->InsertNextPoint(vtx.data());
411 }
412 vtk_unstructured_grid->SetPoints(points);
413
414 // set cells
415 for (index_t cell_id = 0; cell_id < this->m_mesh.numCells(); ++cell_id) {
416 const auto cell = this->m_mesh.getCellToVtxMap().getRow(cell_id);
417 std::vector<vtkIdType> vtk_vtx_ids;
418 for (index_t i = 0; i < cell.size(); ++i) {
419 vtk_vtx_ids.push_back(cell[i]);
420 }
421
422 vtkNew<vtkIdList> faces;
423 const index_t n_faces = this->m_mesh.getCellToFacetMap()[cell_id].size();
424
425 for (index_t i = 0; i < n_faces; ++i) {
426 const index_t face_id = this->m_mesh.getCellToFacetMap()[cell_id][i];
427 const auto face_vtxs = this->m_mesh.getFacetToVtxMap()[face_id];
428 const index_t face_size = face_vtxs.size();
429 faces->InsertNextId(face_size);
430
431 const auto fw_order =
432 (this->m_mesh.getFacetToCellMap()[face_id][0] == cell_id);
433 if (fw_order) {
434 for (auto vid : face_vtxs) {
435 faces->InsertNextId(vid);
436 }
437 } else {
438 for (auto it = face_vtxs.end(); it != face_vtxs.begin();) {
439 faces->InsertNextId(*(--it));
440 }
441 }
442 }
443 vtk_unstructured_grid->InsertNextCell(
444 VTK_POLYHEDRON, cell.size(), vtk_vtx_ids.data(), n_faces,
445 faces->GetPointer(0));
446 }
447}

571template <typename MeshDerived>
572void MeshExporterSubmesh<MeshDerived>::attachVertexData(
573 const std::string& label,
574 const datastructure::CrsMap<float_t, index_t>& data) const {
575 auto& grids = this->m_vtk_grids;
576
577 assert(data.size() == static_cast<index_t>(grids.size()));
578 for (std::size_t grid_id = 0; grid_id < grids.size(); ++grid_id) {
579 auto grid = grids[grid_id];
580 const auto cell_data = data[grid_id];
581 assert(cell_data.size() % grid->GetNumberOfPoints() == 0);
582 const auto block_size = cell_data.size() / grid->GetNumberOfPoints();
583
584 for (Eigen::Index b = 0; b < block_size; ++b) {
585 vtkNew<vtkFloatArray> scalar;
586 const std::string curr_label = label + "_" + std::to_string(b);
587 scalar->SetName(curr_label.c_str());
588 for (index_t i = b; i < cell_data.size(); i += block_size) {
589 scalar->InsertNextValue(cell_data[i]);
590 }
591 grid->GetPointData()->AddArray(scalar);
592 }
593 }
594}

614template <typename MeshDerived>
615void MeshExporterSubmesh<MeshDerived>::buildVtkMesh2D(
616 std::size_t min_n_submesh_elem) {
617 constexpr auto dim = MeshDerived::self_dim;
618 if constexpr (dim == 2) {
619 for (index_t cell_id = 0; cell_id < this->m_mesh.numCells(); ++cell_id) {
620 this->m_vtk_grids.push_back(vtkSmartPointer<vtkUnstructuredGrid>::New());
621 auto vtk_unstructured_grid = this->m_vtk_grids.back();
622
623 Cell2D<double> cell(
624 this->m_mesh, cell_id, CellInitFaces | CellInitMeasure);
625 const auto max_area =
626 cell.measure() / static_cast<double>(min_n_submesh_elem);
627 auto options = Triangulation::s_triangle_default_options;
628 options.maxarea = max_area;
629 options.fixedarea = 1;
630 Triangulation triangulation(cell, options);
631
632 const auto pts = triangulation.getPointList();
633 const auto tria = triangulation.getTriangleList();
634
635 // set vertices
636 vtkNew<vtkPoints> points;
637 for (Eigen::Index i = 0; i < pts.cols(); ++i) {
638 points->InsertNextPoint(pts(0, i), pts(1, i), 0.0);
639 }
640 vtk_unstructured_grid->SetPoints(points);
641
642 // set cells
643 vtkNew<vtkCellArray> cells;
644 for (index_t tri_id = 0; tri_id < tria.cols(); ++tri_id) {
645 vtkNew<vtkTriangle> vtk_cell;
646 for (int32_t d = 0; d < 3; ++d) {
647 vtk_cell->GetPointIds()->SetId(d, tria(d, tri_id));
648 }
649 cells->InsertNextCell(vtk_cell);
650 }
651 vtk_unstructured_grid->SetCells(VTK_TRIANGLE, cells);
652 }
653 }

656template <typename MeshDerived>
657void MeshExporterSubmesh<MeshDerived>::buildVtkMesh3D(
658 std::size_t min_n_submesh_elem) {
659 constexpr auto dim = MeshDerived::self_dim;
660 if constexpr (dim == 3) {
661 for (index_t cell_id = 0; cell_id < this->m_mesh.numCells(); ++cell_id) {
662 this->m_vtk_grids.push_back(vtkSmartPointer<vtkUnstructuredGrid>::New());
663 auto vtk_unstructured_grid = this->m_vtk_grids.back();
664 const auto cell_options = CellInitFaces | CellInitMeasure;
665 Cell3D<double, index_t> cell(this->m_mesh, cell_id, cell_options);
666 const auto max_volume =
667 cell.measure() / static_cast<double>(min_n_submesh_elem);
668 assert(max_volume > 0);
669 auto options = Tetrahedralization::s_tetgen_default_options;
670 options.maxvolume = max_volume;
671 options.fixedvolume = 1;
672 options.maxvolume_length = std::pow(max_volume, 1. / 3.) / 3.;
673 Tetrahedralization tetrahedralization(cell, false, options);
674
675 const auto pts = tetrahedralization.getPointList();
676 const auto tets = tetrahedralization.getTetrahedronList();
677
678 // set points
679 vtkNew<vtkPoints> points;
680 for (Eigen::Index i = 0; i < pts.cols(); ++i) {
681 points->InsertNextPoint(pts(0, i), pts(1, i), pts(2, i));
682 }
683 vtk_unstructured_grid->SetPoints(points);
684
685 // set cells
686 for (index_t tet_id = 0; tet_id < tets.cols(); ++tet_id) {
687 vtkNew<vtkTetra> tetra;
688 for (int32_t d = 0; d < 4; ++d) {
689 tetra->GetPointIds()->SetId(d, tets(d, tet_id));
690 }
691 vtk_unstructured_grid->InsertNextCell(
692 tetra->GetCellType(), tetra->GetPointIds());
693 }
694 }
695 }
