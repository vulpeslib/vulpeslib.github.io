71template <std::size_t Dim, typename FeSpace, bool IsTest = false>
72class FeExpressionFunc
73 : public FeExpressionBase<Dim, FeExpressionFunc<Dim, FeSpace, IsTest>> {
74 public:
75 static constexpr bool is_leaf = true;
76 using FeSpaceType = FeSpace;
77
78 explicit constexpr FeExpressionFunc(const FeSpace& fe_space)
79 : m_fe_space(fe_space) {};
80 FeExpressionFunc(const FeExpressionFunc&) = delete;
81 FeExpressionFunc& operator=(const FeExpressionFunc&) = delete;
82 FeExpressionFunc(FeExpressionFunc&&) = delete;
83 FeExpressionFunc& operator=(FeExpressionFunc&&) = delete;
84
85 constexpr const FeSpace& feSpace() const { return m_fe_space; }
86
87 decltype(auto) evaluate(
88 const FeValues<Dim>& fe_values,
89 std::size_t i,
90 std::size_t j,
91 std::size_t q) const {
92 if constexpr (IsTest) {
93 return fe_values.getShapeEval()(j, q);
94 } else {
95 return fe_values.getShapeEval()(i, q);
96 }
97 }
98
99 constexpr UpdateFlags computeUpdateFlags() const { return UpdateValues; }
100 constexpr std::size_t computeOrder() const { return m_fe_space.order(); }
101
102 decltype(auto) evaluatePoly(
103 const FeValues<Dim>&, std::size_t i, std::size_t j) const {
104 if constexpr (IsTest) {
105 return m_fe_space.getBasis()[j];
106 } else {
107 return m_fe_space.getBasis()[i];
108 }
109 }
110
111 decltype(auto) evaluatePolyCoef(const auto&) const { return 1.0; }
112 constexpr std::size_t nPolyComponents() const { return 1; }
113
114 protected:
115 const FeSpace& m_fe_space;
116};

133class FeExpressionGradFunc
134 : public FeExpressionBase<Dim, FeExpressionGradFunc<Dim, FeSpace, IsTest>> {
135 public:
136 static constexpr bool is_leaf = false;
137 using FeSpaceType = FeSpace;
138
139 explicit FeExpressionGradFunc(const FeExpressionFunc<Dim, FeSpace, IsTest>& w)
140 : m_w(w) {}
141
142 constexpr const auto& feSpace() const { return m_w.feSpace(); }
143
144 decltype(auto) evaluate(
145 const FeValues<Dim>& fe_values,
146 std::size_t i,
147 std::size_t j,
148 std::size_t q) const {
149 if constexpr (IsTest) {
150 return fe_values.getGradEval()(j, q);
151 } else {
152 return fe_values.getGradEval()(i, q);
153 }
154 }
155
156 constexpr UpdateFlags computeUpdateFlags() const { return UpdateGradients; }
157 constexpr std::size_t computeOrder() const {
158 return m_w.computeOrder() > 0 ? m_w.computeOrder() - 1 : 0;
159 }
160
161 decltype(auto) evaluatePoly(
162 const FeValues<Dim>&, std::size_t i, std::size_t j) const {
163 // WARNING, TODO
164 // we are not caching the gradient, however the polynomial expression
165 // evaluation should be cached in a dictionary, thus, this overhead
166 // should be seldomly encountered
167 if constexpr (IsTest) {
168 return feSpace().getBasis()[j].gradient();
169 } else {
170 return feSpace().getBasis()[i].gradient();
171 }
172 }
173
174 decltype(auto) evaluatePolyCoef(const auto& mapping) const {
175 return mapping.jacobianInv({});
176 }
177
178 constexpr std::size_t nPolyComponents() const { return Dim; }
179
180 protected:
181 const FeExpressionFunc<Dim, FeSpace, IsTest>& m_w;
182};

431template <std::size_t Dim, typename E1, typename E2>
432class FeExpressionInner
433 : public FeExpressionBase<Dim, FeExpressionInner<Dim, E1, E2>> {
434 public:
435 static constexpr bool is_leaf = false;
436 using FeSpaceType = typename E2::FeSpaceType;
437
438 FeExpressionInner(E1 const& u, E2 const& v) : m_u(u), m_v(v) {}
439
440 // TODO: handle case where different
441 constexpr const auto& feSpace() const { return m_v.feSpace(); }
442
443 decltype(auto) evaluate(
444 const FeValues<Dim>& fe_values,
445 std::size_t i,
446 std::size_t j,
447 std::size_t q) const {
448 return m_u.evaluate(fe_values, i, j, q)
449 .dot(m_v.evaluate(fe_values, i, j, q));
450 }
451
452 constexpr UpdateFlags computeUpdateFlags() const {
453 return UpdateFlags(m_u.computeUpdateFlags() | m_v.computeUpdateFlags());
454 }
455 constexpr std::size_t computeOrder() const {
456 return m_u.computeOrder() + m_v.computeOrder();
457 }
458
459 decltype(auto) evaluatePoly(
460 const FeValues<Dim>& fe_values, std::size_t i, std::size_t j) const {
461 using ArrayType = std::decay_t<decltype(m_u.evaluatePoly(fe_values, i, j))>;
462 return ArrayType{
463 m_u.evaluatePoly(fe_values, i, j).array() *
464 m_v.evaluatePoly(fe_values, i, j).array()};
465 }
466
467 decltype(auto) evaluatePolyCoef(const auto& mapping) const {
468 using ArrayType = std::decay_t<decltype(m_u.evaluatePolyCoef(mapping))>;
469 return ArrayType{
470 m_u.evaluatePolyCoef(mapping).array() *
471 m_v.evaluatePolyCoef(mapping).array()};
472 }
473
474 constexpr std::size_t nPolyComponents() const {
475 assert(m_u.nPolyComponents() == m_v.nPolyComponents());
476 return m_u.nPolyComponents();
477 }
478
479 protected:
480 typename std::conditional_t<E1::is_leaf, const E1&, const E1> m_u;
481 typename std::conditional_t<E2::is_leaf, const E2&, const E2> m_v;
482};
