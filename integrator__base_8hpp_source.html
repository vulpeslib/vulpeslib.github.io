122 WeakFormIntegratorBase(
123 const MeshType& mesh,
124 CellManager<MeshType, FeType>& cell_manager,
125 const FeExpressionTuple& fe_expression,
126 const FeExpressionFaceTuple& fe_expression_face)
127 : m_mesh(mesh),
128 m_cell_manager(cell_manager),
129 m_fe_expression(fe_expression),
130 m_fe_expression_face(fe_expression_face),
131 m_fe_space(std::get<0>(fe_expression)->feSpace()) {
132 const uint32_t n_dof_loc = m_fe_space.getBasis().size();
133 m_integrals.resize(
134 n_dof_loc * nblocks, n_dof_loc * nblocks, s_n_expr_bilinear);
135 m_rhs.resize(n_dof_loc * nblocks, s_n_expr_linear);
136 m_face_integrals.resize(
137 n_dof_loc * nblocks, n_dof_loc * nblocks, s_n_face_expr_bilinear);
138 m_face_rhs.resize(n_dof_loc * nblocks, s_n_face_expr_linear);
139 using index_t = typename MeshType::index_t;
140 index_t n_max_neigh = 0;
141 for (index_t cid = 0; cid < mesh.getDualMap().size(); ++cid) {
142 n_max_neigh = std::max(n_max_neigh, mesh.getDualMap()[cid].size());
143 }
144 m_face_integrals_neigh.resize(
145 n_dof_loc * nblocks, n_dof_loc * nblocks, s_n_face_expr, n_max_neigh);
146 computeMaxOrder();
147 computeUpdateFlags();
148 buildWeakFormPointerTable();
149 }

163 void computeUpdateFlags() {
164 if constexpr (s_n_expr) {
165 m_update_flags |= std::apply(
166 [](const auto&... expr) {
167 return (expr->computeUpdateFlags() | ...);
168 },
169 m_fe_expression);
170 }
171 if constexpr (s_n_face_expr) {
172 m_update_flags_face |= std::apply(
173 [](const auto&... expr) {
174 return (expr->computeUpdateFlags() | ...);
175 },
176 m_fe_expression_face);
177 }
178 }

187 void computeMaxOrder() {
188 m_order = 1;
189 m_order_face = 1;
190 if constexpr (s_n_expr) {
191 m_order += std::apply(
192 [](const auto&... expr) {
193 return std::max({expr->computeOrder()...});
194 },
195 m_fe_expression);
196 }
197 if constexpr (s_n_face_expr) {
198 m_order_face += std::apply(
199 [](const auto&... expr) {
200 return std::max({expr->computeOrder()...});
201 },
202 m_fe_expression_face);
203 }

228 void preallocate(index_t cell_id) {
229 m_integrals.setZero();
230 m_rhs.setZero();
231 m_need_neigh_cell = (m_update_flags_face & fe::UpdateNeighborCell) ||
232 (m_update_flags_face & fe::UpdateNeighborValues);
233 m_need_neigh_values = m_update_flags_face & fe::UpdateNeighborValues;
234 if (m_need_neigh_cell) {
235 m_fe_values.createNeighValues();
236 }
237 m_cell_id = cell_id;
238 m_fe_values.setCellId(m_cell_id);
239 if constexpr (s_n_face_expr) {
240 m_face_integrals.setZero();
241 m_face_rhs.setZero();
242 }
243 m_face_integrals_neigh.setZero();
244 }

250 void findCellNeigh() {
251 m_fe_values.setFaceId(m_face_id);
252 // `findCellNeigh` builds the neighbor if we ask for it
253 // so we call it only if we need it
254 if (m_need_neigh_cell) {
255 m_cell_neigh_info = m_cell_manager.findCellNeigh(m_cell_id, m_face_id);
256 if (m_cell_neigh_info.have_neigh) {
257 auto& neigh_fe_values = m_fe_values.neighValues();
258 neigh_fe_values.setCellId(m_cell_neigh_info.neigh_cell_id);
259 neigh_fe_values.setFaceId(m_cell_neigh_info.loc_neigh_face_id);
260 getCell(m_cell_neigh_info.neigh_cell_id);
261 }
262 } else {
263 m_cell_neigh_info = decltype(m_cell_neigh_info){};
264 }
265 m_fe_values.setHasFaceNeigh(m_cell_neigh_info.have_neigh);
266 }

277 void applyBasisScaling() {
278 if constexpr (FeType::needBasisScaling()) {
279 const auto& mapping = m_cell_manager.getMapping(m_cell_id);
280 const double scaling2 = mapping->jacobianDet({});
281 const double scaling = std::sqrt(scaling2);
282
283 m_rhs /= scaling;
284 Eigen::Map<Eigen::Vector<double, -1>> integrals_map(
285 m_integrals.data(), m_integrals.size());
286 integrals_map /= scaling2;
287
288 m_face_rhs /= scaling;
289 Eigen::Map<Eigen::Vector<double, -1>> face_integrals_map(
290 m_face_integrals.data(), m_face_integrals.size());
291 face_integrals_map /= scaling2;
292
293 const auto n_neigh = m_mesh.getDualMap()[m_cell_id].size();
294 for (index_t ni = 0; ni < n_neigh; ++ni) {
295 const auto neigh_cell_id = m_mesh.getDualMap()[m_cell_id][ni];
296 const auto& neigh_map = m_cell_manager.getMapping(neigh_cell_id);
297 const double neigh_scaling = std::sqrt(neigh_map->jacobianDet({}));
298 Eigen::Map<Eigen::Vector<double, -1>> face_integrals_neigh_map(
299 &m_face_integrals_neigh(0, 0, 0, ni), m_face_integrals.size());
300 face_integrals_neigh_map /= (scaling * neigh_scaling);
301 }
302 }

323 Eigen::Map<Eigen::MatrixXd> getCurrFaceIntegral(
324 const auto& expr, Eigen::Index expr_id) {
325 const bool expr_need_neigh =
326 expr->computeUpdateFlags() & fe::UpdateNeighborValues;
327 if (expr_need_neigh && (!m_cell_neigh_info.have_neigh)) {
328 return Eigen::Map<Eigen::MatrixXd>{nullptr, 0, 0};
329 }
330 // the fe expression might want to skip a face
331 // e.g. for Dirichlet boundary conditions
332 if (!expr->precompute(m_fe_values, m_cell_manager.getCells())) {
333 return Eigen::Map<Eigen::MatrixXd>{nullptr, 0, 0};
334 }
335 double* data_ptr =
336 expr_need_neigh
337 ? &m_face_integrals_neigh(
338 0, 0, expr_id, m_cell_neigh_info.loc_neigh_cell_id)
339 : &m_face_integrals(0, 0, expr_id);
340 const Eigen::Index n_dof_loc = m_fe_space.getBasis().size();
341 return Eigen::Map<Eigen::MatrixXd>{
342 data_ptr, n_dof_loc * nblocks, n_dof_loc * nblocks};
343 }

356 void buildWeakFormPointerTable() {
357 const auto check = [](auto it_inserted) {
358 const auto [it, inserted] = it_inserted;
359 if (!inserted) {
360 std::cout << "ERROR: you are using the same weak form twice!"
361 << std::endl;
362 std::exit(-1);
363 }
364 };
365 auto [bilinear_fe_expression, non_bilinear_fe_expression] =
366 internal::split_expression_tuple(m_fe_expression);
367 vutils::for_each_idxed(
368 bilinear_fe_expression,
369 [this, &check](auto expr_ptr, Eigen::Index expr_id) {
370 check(m_wf_ptr_table.emplace(
371 static_cast<void*>(expr_ptr), &m_integrals(0, 0, expr_id)));
372 });
373 vutils::for_each_idxed(
374 non_bilinear_fe_expression,
375 [this, &check](auto expr_ptr, Eigen::Index expr_id) {
376 check(m_wf_ptr_table.emplace(
377 static_cast<void*>(expr_ptr), &m_rhs(0, expr_id)));
378 });
379 auto [bilinear_fe_expression_face, non_bilinear_fe_expression_face] =
380 internal::split_expression_tuple(m_fe_expression_face);
381 vutils::for_each_idxed(
382 bilinear_fe_expression_face,
383 [this, &check](auto expr_ptr, Eigen::Index expr_id) {
384 const bool expr_need_neigh =
385 expr_ptr->computeUpdateFlags() & fe::UpdateNeighborValues;
386 if (expr_need_neigh) {
387 check(m_wf_ptr_table.emplace(
388 static_cast<void*>(expr_ptr),
389 &m_face_integrals_neigh(0, 0, expr_id, 0)));
390 } else {
391 check(m_wf_ptr_table.emplace(
392 static_cast<void*>(expr_ptr),
393 &m_face_integrals(0, 0, expr_id)));
394 }
395 });
396 vutils::for_each_idxed(
397 non_bilinear_fe_expression_face,
398 [this, &check](auto expr_ptr, Eigen::Index expr_id) {
399 check(m_wf_ptr_table.emplace(
400 static_cast<void*>(expr_ptr), &m_face_rhs(0, expr_id)));
401 });
402 }

428 const auto getIntegral(auto wf_ptr, index_t loc_neigh_id = 0) const {
429 assert(
430 (wf_ptr->computeUpdateFlags() & fe::UpdateNeighborValues) ||
431 (loc_neigh_id == 0));
432 constexpr bool is_bilinear =
433 std::remove_pointer_t<decltype(wf_ptr)>::is_bilinear == fe::IsBilinear;
434 const Eigen::Index n_dof_loc = m_fe_space.getBasis().size();
435 const auto nrows = n_dof_loc * nblocks;
436 const auto ncols = is_bilinear ? nrows : 1;
437 const auto slice_neigh_size = nrows * ncols * s_n_face_expr;
438 const double* data = m_wf_ptr_table.at(wf_ptr);
439 return Eigen::Map<const Eigen::MatrixXd>{
440 data + loc_neigh_id * slice_neigh_size, nrows, ncols};
441 }
