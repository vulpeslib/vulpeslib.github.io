41template <
42 typename MeshType,
43 typename FeExpressionTuple,
44 typename FeExpressionFaceTuple = std::tuple<>>
45class WeakFormIntegratorBase {
46 static_assert(
47 internal::is_tuple_of_fe_expr_pointers_v<FeExpressionTuple>,
48 "fe_expression must be a tuple of pointer to FeExpression");
49 static_assert(
50 internal::is_tuple_of_fe_expr_pointers_v<FeExpressionFaceTuple>,
51 "fe_expression_face must be a tuple of pointer to FeExpression");
52
53 public:
54 using index_t = typename MeshType::index_t;
55 using FeType = std::remove_pointer_t<
56 std::tuple_element_t<0, FeExpressionTuple>>::FeSpaceType;
58 static inline constexpr index_t s_n_expr =
59 std::tuple_size<FeExpressionTuple>{};
61 static inline constexpr index_t s_n_face_expr =
62 std::tuple_size<FeExpressionFaceTuple>{};
63
64 using FeExpressionTupleBilinear =
65 decltype(internal::split_expression_tuple(FeExpressionTuple{}).first);
66 using FeExpressionTupleLinear =
67 decltype(internal::split_expression_tuple(FeExpressionTuple{}).second);
69 static inline constexpr index_t s_n_expr_bilinear =
70 std::tuple_size<FeExpressionTupleBilinear>{};
72 static inline constexpr index_t s_n_expr_linear =
73 std::tuple_size<FeExpressionTupleLinear>{};
74
75 using FeExpressionTupleFaceBilinear =
76 decltype(internal::split_expression_tuple(FeExpressionFaceTuple{}).first);
77 using FeExpressionTupleFaceLinear =
78 decltype(internal::split_expression_tuple(FeExpressionFaceTuple{})
79 .second);
81 static inline constexpr index_t s_n_face_expr_bilinear =
82 std::tuple_size<FeExpressionTupleFaceBilinear>{};
84 static inline constexpr index_t s_n_face_expr_linear =
85 std::tuple_size<FeExpressionTupleFaceLinear>{};
86
87 static inline constexpr std::size_t dim = MeshType::self_dim;
88 using CellType = geometry::Cell<dim, double, index_t>;
89 using MappingType = std::decay_t<decltype(FeType::getMapping(CellType{}))>;
90
91 static inline constexpr uint32_t nblocks =
92 std::remove_pointer_t<
93 std::tuple_element_t<0, FeExpressionTuple>>::k_index_end -
94 std::remove_pointer_t<
95 std::tuple_element_t<0, FeExpressionTuple>>::k_index_begin;

97 WeakFormIntegratorBase(
98 const MeshType& mesh,
99 CellManager<MeshType, FeType>& cell_manager,
100 const FeExpressionTuple& fe_expression,
101 const FeExpressionFaceTuple& fe_expression_face)
102 : m_mesh(mesh),
103 m_cell_manager(cell_manager),
104 m_fe_expression(fe_expression),
105 m_fe_expression_face(fe_expression_face),
106 m_fe_space(std::get<0>(fe_expression)->feSpace()) {
107 const uint32_t n_dof_loc = m_fe_space.getBasis().size();
108 m_integrals.resize(
109 n_dof_loc * nblocks, n_dof_loc * nblocks, s_n_expr_bilinear);
110 m_rhs.resize(n_dof_loc, s_n_expr_linear);
111 m_face_integrals.resize(n_dof_loc, n_dof_loc, s_n_face_expr_bilinear);
112 m_face_rhs.resize(n_dof_loc, s_n_face_expr_linear);
113 computeMaxOrder();
114 computeUpdateFlags();
115 }
116 virtual ~WeakFormIntegratorBase() = default;
117
118 virtual void integrate(index_t cell_id) = 0;
119
120 void computeUpdateFlags() {
121 if constexpr (s_n_expr) {
122 m_update_flags |= std::apply(
123 [](const auto&... expr) {
124 return (expr->computeUpdateFlags() | ...);
125 },
126 m_fe_expression);
127 }
128 if constexpr (s_n_face_expr) {
129 m_update_flags_face |= std::apply(
130 [](const auto&... expr) {
131 return (expr->computeUpdateFlags() | ...);
132 },
133 m_fe_expression_face);
134 }
135 }
136
137 void computeMaxOrder() {
138 m_order = 1;
139 m_order_face = 1;
140 if constexpr (s_n_expr) {
141 m_order += std::apply(
142 [](const auto&... expr) {
143 return std::max({expr->computeOrder()...});
144 },
145 m_fe_expression);
146 }
147 if constexpr (s_n_face_expr) {
148 m_order_face += std::apply(
149 [](const auto&... expr) {
150 return std::max({expr->computeOrder()...});
151 },
152 m_fe_expression_face);
153 }
154 }

156 const CellType& getCell(index_t cell_id, bool update_mapping = false) {
157 const auto& cell = m_cell_manager.getCell(cell_id);
158 if (update_mapping) {
159 m_mapping = m_cell_manager.getMapping(cell_id).get();
160 }
161 return cell;
162 }
163
164 void preallocate(index_t cell_id) {
165 m_integrals.setZero();
166 m_rhs.setZero();
167 const uint32_t n_dof_loc = m_fe_space.getBasis().size();
168 m_need_neigh_cell = (m_update_flags_face & fe::UpdateNeighborCell) ||
169 (m_update_flags_face & fe::UpdateNeighborValues);
170 m_need_neigh_values = m_update_flags_face & fe::UpdateNeighborValues;
171 if (m_need_neigh_cell) {
172 m_fe_values.createNeighValues();
173 }
174 m_cell_id = cell_id;
175 m_fe_values.setCellId(m_cell_id);
176 if constexpr (s_n_face_expr) {
177 m_face_integrals.setZero();
178 m_face_rhs.setZero();
179 }
180 const auto n_neigh = m_mesh.getDualMap()[m_cell_id].size();
181 if (m_face_integrals_neigh.dimension(3) < n_neigh) {
182 m_face_integrals_neigh.resize(
183 n_dof_loc, n_dof_loc, s_n_face_expr, n_neigh);
184 }
185 m_face_integrals_neigh.setZero();
186 }
187
188 void findCellNeigh() {
189 m_fe_values.setFaceId(m_face_id);
190 if (m_need_neigh_cell) {
191 m_cell_neigh_info = m_cell_manager.findCellNeigh(m_cell_id, m_face_id);
192 if (m_cell_neigh_info.have_neigh) {
193 auto& neigh_fe_values = m_fe_values.neighValues();
194 neigh_fe_values.setCellId(m_cell_neigh_info.neigh_cell_id);
195 neigh_fe_values.setFaceId(m_cell_neigh_info.loc_neigh_face_id);
196 getCell(m_cell_neigh_info.neigh_cell_id);
197 }
198 } else {
199 m_cell_neigh_info = decltype(m_cell_neigh_info){};
200 }
201 m_fe_values.setHasFaceNeigh(m_cell_neigh_info.have_neigh);
202 }
203
204 Eigen::Map<Eigen::MatrixXd> getCurrFaceIntegral(
205 const auto& expr, Eigen::Index expr_id) {
206 const bool expr_need_neigh =
207 expr->computeUpdateFlags() & fe::UpdateNeighborValues;
208 if (expr_need_neigh && (!m_cell_neigh_info.have_neigh)) {
209 return Eigen::Map<Eigen::MatrixXd>{nullptr, 0, 0};
210 }
211 // the fe expression might want to skip a face
212 // e.g. for Dirichlet boundary conditions
213 if (!expr->precompute(m_fe_values, m_cell_manager.getCells())) {
214 return Eigen::Map<Eigen::MatrixXd>{nullptr, 0, 0};
215 }
216 double* data_ptr =
217 expr_need_neigh
218 ? &m_face_integrals_neigh(
219 0, 0, expr_id, m_cell_neigh_info.loc_neigh_cell_id)
220 : &m_face_integrals(0, 0, expr_id);
221 const Eigen::Index n_dof_loc = m_fe_space.getBasis().size();
222 return Eigen::Map<Eigen::MatrixXd>{data_ptr, n_dof_loc, n_dof_loc};
223 }
224
225 const auto& getIntegrals() const { return m_integrals; }
226 const auto& getRhsIntegrals() const { return m_rhs; }
227 const auto& getFaceIntegrals() const { return m_face_integrals; }
228 const auto& getRhsFaceIntegrals() const { return m_face_rhs; }
229 const auto& getFaceIntegralsNeigh() const { return m_face_integrals_neigh; }
230
231 protected:
232 const MeshType& m_mesh;
233 CellManager<MeshType, FeType>& m_cell_manager;
234 FeExpressionTuple m_fe_expression;
235 FeExpressionFaceTuple m_fe_expression_face;
236 const FeType& m_fe_space;
237 std::size_t m_order = -1;
238 std::size_t m_order_face = -1;
239
240 fe::UpdateFlagsType m_update_flags = 0;
241 fe::UpdateFlagsType m_update_flags_face = 0;
242 Eigen::Tensor<double, 3> m_integrals;
243 Eigen::MatrixXd m_rhs;
244 Eigen::Tensor<double, 3> m_face_integrals;
245 Eigen::MatrixXd m_face_rhs;
246 Eigen::Tensor<double, 4> m_face_integrals_neigh;
247
248 fe::FeValues<dim> m_fe_values;
249
250 bool m_need_neigh_cell = -1;
251 // subset of the above, if this true, than the above is true
252 bool m_need_neigh_values = -1;
253
254 using CellNeighInfo = CellManager<MeshType, FeType>::NeighInfo;
255 index_t m_cell_id = -1;
256 index_t m_face_id = -1;
257 CellNeighInfo m_cell_neigh_info;
258 const MappingType* m_mapping = nullptr;
259};
