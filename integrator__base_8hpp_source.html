99 WeakFormIntegratorBase(
100 const MeshType& mesh,
101 CellManager<MeshType, FeType>& cell_manager,
102 const FeExpressionTuple& fe_expression,
103 const FeExpressionFaceTuple& fe_expression_face)
104 : m_mesh(mesh),
105 m_cell_manager(cell_manager),
106 m_fe_expression(fe_expression),
107 m_fe_expression_face(fe_expression_face),
108 m_fe_space(std::get<0>(fe_expression)->feSpace()) {
109 const uint32_t n_dof_loc = m_fe_space.getBasis().size();
110 m_integrals.resize(
111 n_dof_loc * nblocks, n_dof_loc * nblocks, s_n_expr_bilinear);
112 m_rhs.resize(n_dof_loc * nblocks, s_n_expr_linear);
113 m_face_integrals.resize(
114 n_dof_loc * nblocks, n_dof_loc * nblocks, s_n_face_expr_bilinear);
115 m_face_rhs.resize(n_dof_loc * nblocks, s_n_face_expr_linear);
116 using index_t = typename MeshType::index_t;
117 index_t n_max_neigh = 0;
118 for (index_t cid = 0; cid < mesh.getDualMap().size(); ++cid) {
119 n_max_neigh = std::max(n_max_neigh, mesh.getDualMap()[cid].size());
120 }
121 m_face_integrals_neigh.resize(
122 n_dof_loc * nblocks, n_dof_loc * nblocks, s_n_face_expr, n_max_neigh);
123 computeMaxOrder();
124 computeUpdateFlags();
125 buildWeakFormPointerTable();
126 }

131 void computeUpdateFlags() {
132 if constexpr (s_n_expr) {
133 m_update_flags |= std::apply(
134 [](const auto&... expr) {
135 return (expr->computeUpdateFlags() | ...);
136 },
137 m_fe_expression);
138 }
139 if constexpr (s_n_face_expr) {
140 m_update_flags_face |= std::apply(
141 [](const auto&... expr) {
142 return (expr->computeUpdateFlags() | ...);
143 },
144 m_fe_expression_face);
145 }
146 }
147
148 void computeMaxOrder() {
149 m_order = 1;
150 m_order_face = 1;
151 if constexpr (s_n_expr) {
152 m_order += std::apply(
153 [](const auto&... expr) {
154 return std::max({expr->computeOrder()...});
155 },
156 m_fe_expression);
157 }
158 if constexpr (s_n_face_expr) {
159 m_order_face += std::apply(
160 [](const auto&... expr) {
161 return std::max({expr->computeOrder()...});
162 },
163 m_fe_expression_face);
164 }
165 }

175 void preallocate(index_t cell_id) {
176 m_integrals.setZero();
177 m_rhs.setZero();
178 m_need_neigh_cell = (m_update_flags_face & fe::UpdateNeighborCell) ||
179 (m_update_flags_face & fe::UpdateNeighborValues);
180 m_need_neigh_values = m_update_flags_face & fe::UpdateNeighborValues;
181 if (m_need_neigh_cell) {
182 m_fe_values.createNeighValues();
183 }
184 m_cell_id = cell_id;
185 m_fe_values.setCellId(m_cell_id);
186 if constexpr (s_n_face_expr) {
187 m_face_integrals.setZero();
188 m_face_rhs.setZero();
189 }
190 m_face_integrals_neigh.setZero();
191 }
192
193 void findCellNeigh() {
194 m_fe_values.setFaceId(m_face_id);
195 if (m_need_neigh_cell) {
196 m_cell_neigh_info = m_cell_manager.findCellNeigh(m_cell_id, m_face_id);
197 if (m_cell_neigh_info.have_neigh) {
198 auto& neigh_fe_values = m_fe_values.neighValues();
199 neigh_fe_values.setCellId(m_cell_neigh_info.neigh_cell_id);
200 neigh_fe_values.setFaceId(m_cell_neigh_info.loc_neigh_face_id);
201 getCell(m_cell_neigh_info.neigh_cell_id);
202 }
203 } else {
204 m_cell_neigh_info = decltype(m_cell_neigh_info){};
205 }
206 m_fe_values.setHasFaceNeigh(m_cell_neigh_info.have_neigh);
207 }

209 void applyBasisScaling() {
210 if constexpr (FeType::needBasisScaling()) {
211 const auto& mapping = m_cell_manager.getMapping(m_cell_id);
212 const double scaling2 = mapping->jacobianDet({});
213 const double scaling = std::sqrt(scaling2);
214
215 m_rhs /= scaling;
216 Eigen::Map<Eigen::Vector<double, -1>> integrals_map(
217 m_integrals.data(), m_integrals.size());
218 integrals_map /= scaling2;
219
220 m_face_rhs /= scaling;
221 Eigen::Map<Eigen::Vector<double, -1>> face_integrals_map(
222 m_face_integrals.data(), m_face_integrals.size());
223 face_integrals_map /= scaling2;
224
225 const auto n_neigh = m_mesh.getDualMap()[m_cell_id].size();
226 for (index_t ni = 0; ni < n_neigh; ++ni) {
227 const auto neigh_cell_id = m_mesh.getDualMap()[m_cell_id][ni];
228 const auto& neigh_map = m_cell_manager.getMapping(neigh_cell_id);
229 const double neigh_scaling = std::sqrt(neigh_map->jacobianDet({}));
230 Eigen::Map<Eigen::Vector<double, -1>> face_integrals_neigh_map(
231 &m_face_integrals_neigh(0, 0, 0, ni), m_face_integrals.size());
232 face_integrals_neigh_map /= (scaling * neigh_scaling);
233 }
234 }

237 Eigen::Map<Eigen::MatrixXd> getCurrFaceIntegral(
238 const auto& expr, Eigen::Index expr_id) {
239 const bool expr_need_neigh =
240 expr->computeUpdateFlags() & fe::UpdateNeighborValues;
241 if (expr_need_neigh && (!m_cell_neigh_info.have_neigh)) {
242 return Eigen::Map<Eigen::MatrixXd>{nullptr, 0, 0};
243 }
244 // the fe expression might want to skip a face
245 // e.g. for Dirichlet boundary conditions
246 if (!expr->precompute(m_fe_values, m_cell_manager.getCells())) {
247 return Eigen::Map<Eigen::MatrixXd>{nullptr, 0, 0};
248 }
249 double* data_ptr =
250 expr_need_neigh
251 ? &m_face_integrals_neigh(
252 0, 0, expr_id, m_cell_neigh_info.loc_neigh_cell_id)
253 : &m_face_integrals(0, 0, expr_id);
254 const Eigen::Index n_dof_loc = m_fe_space.getBasis().size();
255 return Eigen::Map<Eigen::MatrixXd>{
256 data_ptr, n_dof_loc * nblocks, n_dof_loc * nblocks};
257 }

259 void buildWeakFormPointerTable() {
260 const auto check = [](auto it_inserted) {
261 const auto [it, inserted] = it_inserted;
262 if (!inserted) {
263 std::cout << "ERROR: you are using the same weak form twice!"
264 << std::endl;
265 std::exit(-1);
266 }
267 };
268 auto [bilinear_fe_expression, non_bilinear_fe_expression] =
269 internal::split_expression_tuple(m_fe_expression);
270 vutils::for_each_idxed(
271 bilinear_fe_expression,
272 [this, &check](auto expr_ptr, Eigen::Index expr_id) {
273 check(m_wf_ptr_table.emplace(
274 static_cast<void*>(expr_ptr), &m_integrals(0, 0, expr_id)));
275 });
276 vutils::for_each_idxed(
277 non_bilinear_fe_expression,
278 [this, &check](auto expr_ptr, Eigen::Index expr_id) {
279 check(m_wf_ptr_table.emplace(
280 static_cast<void*>(expr_ptr), &m_rhs(0, expr_id)));
281 });
282 auto [bilinear_fe_expression_face, non_bilinear_fe_expression_face] =
283 internal::split_expression_tuple(m_fe_expression_face);
284 vutils::for_each_idxed(
285 bilinear_fe_expression_face,
286 [this, &check](auto expr_ptr, Eigen::Index expr_id) {
287 const bool expr_need_neigh =
288 expr_ptr->computeUpdateFlags() & fe::UpdateNeighborValues;
289 if (expr_need_neigh) {
290 check(m_wf_ptr_table.emplace(
291 static_cast<void*>(expr_ptr),
292 &m_face_integrals_neigh(0, 0, expr_id, 0)));
293 } else {
294 check(m_wf_ptr_table.emplace(
295 static_cast<void*>(expr_ptr),
296 &m_face_integrals(0, 0, expr_id)));
297 }
298 });
299 vutils::for_each_idxed(
300 non_bilinear_fe_expression_face,
301 [this, &check](auto expr_ptr, Eigen::Index expr_id) {
302 check(m_wf_ptr_table.emplace(
303 static_cast<void*>(expr_ptr), &m_face_rhs(0, expr_id)));
304 });
305 }

313 const auto getIntegral(auto wf_ptr, index_t loc_neigh_id = 0) const {
314 assert(
315 (wf_ptr->computeUpdateFlags() & fe::UpdateNeighborValues) ||
316 (loc_neigh_id == 0));
317 constexpr bool is_bilinear =
318 std::remove_pointer_t<decltype(wf_ptr)>::is_bilinear == fe::IsBilinear;
319 const Eigen::Index n_dof_loc = m_fe_space.getBasis().size();
320 const auto nrows = n_dof_loc * nblocks;
321 const auto ncols = is_bilinear ? nrows : 1;
322 const auto slice_neigh_size = nrows * ncols * s_n_face_expr;
323 const double* data = m_wf_ptr_table.at(wf_ptr);
324 return Eigen::Map<const Eigen::MatrixXd>{
325 data + loc_neigh_id * slice_neigh_size, nrows, ncols};
326 }
